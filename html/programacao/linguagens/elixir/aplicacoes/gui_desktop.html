Elixir — GUI e Desktop

Elixir não é a primeira escolha para interfaces gráficas tradicionais, mas existem soluções maduras: Scenic é um framework 2D puro-Elixir com GPU rendering (OpenGL), projetado para interfaces embarcadas e desktop. wxErlang expõe a API nativa wx (wxWidgets) para aplicações de sistema. O padrão emergente é usar Phoenix LiveView como GUI local via Tauri ou Elixir Desktop.

Aplicações desktop com Elixir se beneficiam do mesmo supervisor tree: se a janela crasha, o processo reinicia sem derrubar a lógica de negócio. Perfeito para dashboards industriais, ferramentas de monitoramento e kiosks.

Scenic — Framework 2D Declarativo

Scene com componentes, eventos e navegação
# mix.exs
{:scenic, "~> 0.11"},
{:scenic_driver_glfw, "~> 0.11", targets: :host}

# config/config.exs
config :minha_app, :viewport,
  name: :main_viewport,
  size: {800, 600},
  theme: :dark,
  drivers: [
    %{module: Scenic.Driver.Glfw, opts: [title: "Minha App"]}
  ],
  default_scene: {MinhaApp.Scene.Home, nil}

# lib/minha_app/scene/home.ex
defmodule MinhaApp.Scene.Home do
  use Scenic.Scene
  import Scenic.Primitives
  import Scenic.Components
  alias Scenic.Graph

  @graph Graph.build(font: :roboto, font_size: 24)
          |> add_specs_to_graph([
            text_spec("Bem-vindo ao Scenic!", translate: {100, 100}),
            rect_spec({200, 50}, fill: :blue, translate: {100, 200}, id: :btn_ok),
            text_spec("Clique aqui", translate: {130, 230}, fill: :white),
            circle_spec(50, fill: :green, translate: {400, 300}, id: :circulo)
          ])

  @impl Scenic.Scene
  def init(scene, _params, _opts) do
    scene = push_graph(scene, @graph)
    {:ok, scene}
  end

  @impl Scenic.Scene
  def handle_event({:click, :btn_ok}, _from, scene) do
    graph = Graph.modify(@graph, :circulo, &update_opts(&1, fill: :red))
    scene = push_graph(scene, graph)
    {:noreply, scene}
  end
end

wxErlang — Widgets Nativos

Janela com botão e text input via :wx
# wxErlang vem incluso na instalação OTP — sem dependência extra

defmodule MinhaApp.Janela do
  def start do
    wx = :wx.new()
    frame = :wxFrame.new(wx, -1, ~c"Minha Janela",
      size: {500, 400})

    panel = :wxPanel.new(frame)
    sizer = :wxBoxSizer.new(:wx_const.wx_vertical())

    # Label
    label = :wxStaticText.new(panel, -1, ~c"Nome:")
    input = :wxTextCtrl.new(panel, 101)
    btn   = :wxButton.new(panel, 102, label: ~c"Saudar")

    :wxSizer.add(sizer, label, proportion: 0)
    :wxSizer.add(sizer, input, proportion: 0, flag: :wx_const.wx_expand())
    :wxSizer.add(sizer, btn,   proportion: 0)
    :wxPanel.setSizer(panel, sizer)

    # Evento de botão
    :wxButton.connect(btn, :command_button_clicked)
    :wxFrame.show(frame)

    loop(input)
  end

  defp loop(input) do
    receive do
      {:wx, 102, _, _, _} ->
        nome = :wxTextCtrl.getValue(input)
        :wx_misc.messageBox(~c"Olá, #{nome}!", ~c"Mensagem")
        loop(input)
      _ -> loop(input)
    end
  end
end

LiveView como GUI Local

Elixir Desktop — app Phoenix em janela nativa
# mix.exs — Elixir Desktop (wrapper Webview)
{:desktop, "~> 0.2"}

# Roda um servidor Phoenix normal + abre janela nativa
defmodule MinhaApp.Application do
  use Application

  @impl true
  def start(_type, _args) do
    children = [
      MinhaAppWeb.Endpoint,
      {Desktop.Window, [
        app: :minha_app,
        id: MainWindow,
        title: "Minha App Desktop",
        size: {1200, 800},
        url: &MinhaAppWeb.Endpoint.url/0
      ]}
    ]
    Supervisor.start_link(children, strategy: :one_for_one)
  end
end

# O resultado: interface HTML/CSS/LiveView full-featured
