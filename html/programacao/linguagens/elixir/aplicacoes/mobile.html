Elixir — Mobile

Elixir não compila para iOS ou Android diretamente — mas é uma escolha de alta produtividade para o backend de aplicativos mobile: API REST com Phoenix, subscriptions GraphQL com Absinthe, push notifications com Pigeon, autenticação OAuth2, e sincronização em tempo real via Phoenix Channels. Aplicativos Flutter, React Native e SwiftUI consomem esse backend.

A combinação Flutter + Phoenix é especialmente poderosa: Flutter fornece UI nativa cross-platform, Phoenix fornece backend escalável com WebSocket nativo. Push notifications, sincronização offline-first e autenticação JWT integram sem friction.

API REST para Mobile — Paginação e Versionamento

Endpoint otimizado para mobile com cursor pagination
# router.ex — versionamento de API
scope "/api" do
  pipe_through :api

  scope "/v1", MeuAppWeb.V1 do
    pipe_through :authenticate
    resources "/feed", FeedController, only: [:index]
    post "/push/token", PushController, :register
  end
end

# feed_controller.ex — cursor-based pagination (mais eficiente que offset)
defmodule MeuAppWeb.V1.FeedController do
  use MeuAppWeb, :controller

  def index(conn, %{"cursor" => cursor, "limit" => limit}) do
    limit = min(String.to_integer(limit), 50)  # cap em 50
    {posts, next_cursor} = Feed.listar_cursor(cursor, limit)

    render(conn, :index,
      posts: posts,
      meta: %{
        next_cursor: next_cursor,
        has_more: !is_nil(next_cursor)
      })
  end
  def index(conn, params), do: index(conn, Map.merge(%{"cursor" => nil, "limit" => "20"}, params))
end

# feed.ex — query com cursor
def listar_cursor(nil, limit) do
  posts = from(p in Post,
    order_by: [desc: p.inserted_at],
    limit: ^(limit + 1))  # pegar 1 a mais p/ saber se há próxima página
  |> Repo.all()
  cursor_result(posts, limit)
end

def listar_cursor(cursor_encoded, limit) do
  {ts, id} = decodificar_cursor(cursor_encoded)
  posts = from(p in Post,
    where: p.inserted_at < ^ts or (p.inserted_at == ^ts and p.id < ^id),
    order_by: [desc: p.inserted_at, desc: p.id],
    limit: ^(limit + 1))
  |> Repo.all()
  cursor_result(posts, limit)
end

Push Notifications — Pigeon

Envio para FCM (Android) e APNs (iOS)
# mix.exs
{:pigeon, "~> 2.0"},
{:kadabra, "~> 0.6"}   # HTTP/2 para APNs

# config/config.exs
config :pigeon, :fcm,
  adapter: Pigeon.FCM,
  project_id: System.get_env("FIREBASE_PROJECT_ID"),
  service_account_json: System.get_env("FIREBASE_CREDENTIALS_JSON")

config :pigeon, :apns,
  adapter: Pigeon.APNS,
  key: System.get_env("APNS_KEY"),
  key_identifier: System.get_env("APNS_KEY_ID"),
  team_id:  System.get_env("APPLE_TEAM_ID")

# Enviar para Android (FCM v1)
notif = Pigeon.FCM.Notification.new(token,
  %{title: "Nova mensagem", body: "João: Oi!"},
  %{tipo: "chat", sala_id: "123"})

case Pigeon.FCM.push(notif) do
  %{status: :success} -> :ok
  %{status: :unregistered} -> remover_token(token)
  _ -> :erro
end

# Enviar para iOS (APNs)
notif_ios = Pigeon.APNS.Notification.new(
  %{alert: %{title: "Nova mensagem", body: "João: Oi!"}, badge: 1},
  token, "com.meuapp.bundle")
Pigeon.APNS.push(notif_ios)

Configurar Phoenix para PWA com service worker

# assets/static/manifest.json
{
  "name": "Meu App",
  "short_name": "App",
  "start_url": "/",

# endpoint.ex — cachear assets para PWA
plug Plug.Static,
  at: "/", from: :meu_app, gzip: false,
  headers: %{"service-worker-allowed" => "/"},
  only: ~w(assets fonts images favicon.ico robots.txt sw.js manifest.json)

# Reconexão automática no LiveView (mobile pode desconectar)
# assets/app.js
let liveSocket = new LiveSocket("/live", Socket, {
  reconnectAfterMs: (tries) => [50, 100, 500, 1000, 3000][tries - 1] || 5000,
  params: {_csrf_token: csrfToken}
})

# LiveView com detecção de plataforma mobile
defmodule MeuAppWeb.HomeLive do
  use MeuAppWeb, :live_view

  @impl true
  def mount(_params, session, socket) do
    mobile? = get_connect_params(socket)["mobile"] == true
    {:ok, assign(socket, mobile: mobile?, user: session["user_id"])}
  end
end

A estratégia de cursor pagination é essencial para feeds mobile: OFFSET ficam cada vez mais lentos à medida que o usuário pagina — o banco precisa varrer e descartar N registros. Com cursor baseado em timestamp+id, cada página é sempre uma query O(log n) no índice, independente de quantas páginas já foram carregadas.
