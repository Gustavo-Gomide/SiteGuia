Elixir — Dados e IA

O ecossistema Nx (Numerical Elixir) traz computação numérica de alta performance para a plataforma BEAM: tensors com aceleração por GPU (EXLA/XLA, CUDA, Metal), redes neurais com Axon, modelos pré-treinados com Bumblebee e DataFrames com Explorer. Toda a cadeia de ML pode rodar dentro de um servidor Phoenix — inferência em tempo real via LiveView.

O que torna o stack Nx único: você pode fazer inferência de ML em processos Elixir supervisionados. Se o processo de inferência crasha, o Supervisor reinicia. Se a demanda aumenta, você escala processos. Nenhum "modelo server" separado necessário.

Nx — Computação Numérica com Tensors

Nx.tensor, operações e defn para JIT compilation
# mix.exs
{:nx, "~> 0.9"},
{:exla, "~> 0.9"}  # backend XLA (CPU/GPU/TPU)

# Configurar backend
Nx.default_backend(EXLA.Backend)

# Criar tensors
a = Nx.tensor([[1, 2], [3, 4]], type: :f32)
b = Nx.tensor([[5, 6], [7, 8]], type: :f32)

# Operações elementares e álgebra linear
soma   = Nx.add(a, b)              # [[6,8],[10,12]]
produto = Nx.dot(a, b)             # multiplicação de matrizes
norma  = Nx.LinAlg.norm(a)
inv    = Nx.LinAlg.invert(a)
softmax = Nx.softmax(a)

# defn — compilado por JIT para CPU/GPU
defmodule MinhasOps do
  import Nx.Defn

  defn sigmoid(x) do
    Nx.divide(1.0, Nx.add(1.0, Nx.exp(Nx.negate(x))))
  end

  defn relu(x), do: Nx.max(0.0, x)

  defn mse_loss(y_pred, y_true) do
    diff = Nx.subtract(y_pred, y_true)
    diff |> Nx.pow(2) |> Nx.mean()
  end

  defn grad_descida(params, grads, lr) do
    Nx.subtract(params, Nx.multiply(lr, grads))
  end
end

# Grad automático
grad_fn = Nx.Defn.jit(fn x -> Nx.grad(x, fn x -> Nx.sum(Nx.pow(x, 2)) end) end)
grad_fn.(Nx.tensor([1.0, 2.0, 3.0]))  # [2.0, 4.0, 6.0]

Axon — Redes Neurais

Definição, treinamento e predição com Axon
# mix.exs
{:axon, "~> 0.7"}

# Definir modelo MLP para classificação binária
model = Axon.input("input", shape: {nil, 784})
        |> Axon.dense(256, activation: :relu)
        |> Axon.dropout(rate: 0.3)
        |> Axon.dense(128, activation: :relu)
        |> Axon.dropout(rate: 0.2)
        |> Axon.dense(10, activation: :softmax)

# CNN para imagens
cnn = Axon.input("imagem", shape: {nil, 28, 28, 1})
      |> Axon.conv(32, kernel_size: {3, 3}, activation: :relu)
      |> Axon.max_pool(kernel_size: {2, 2})
      |> Axon.conv(64, kernel_size: {3, 3}, activation: :relu)
      |> Axon.flatten()
      |> Axon.dense(10, activation: :softmax)

# Treinar modelo

Bumblebee — Modelos Pré-treinados (BERT, GPT-2)

Inferência com modelos HuggingFace via Serving
# mix.exs
{:bumblebee, "~> 0.6"},
{:nx, "~> 0.9"},
{:exla, "~> 0.9"}

# Classificação de sentimentos com BERT
{:ok, model_info} = Bumblebee.load_model({:hf, "distilbert-base-uncased-finetuned-sst-2-english"})
{:ok, tokenizer}   = Bumblebee.load_tokenizer({:hf, "distilbert-base-uncased-finetuned-sst-2-english"})

serving = Bumblebee.Text.text_classification(model_info, tokenizer,
            compile: [batch_size: 4, sequence_length: 128],
            defn_options: [compiler: EXLA])

resultado = Nx.Serving.run(serving, "Este produto é incrível!")
# %{predictions: [%{label: "POSITIVE", score: 0.998}]}

# Integrar como GenServer supervisionado na aplicação
defmodule MeuApp.IAServer do
  use Supervisor

  def start_link(_), do: Supervisor.start_link(__MODULE__, [], name: __MODULE__)

  @impl true
  def init(_) do
    {:ok, model} = Bumblebee.load_model({:hf, "distilbert-base-uncased-finetuned-sst-2-english"})
    {:ok, tokenizer} = Bumblebee.load_tokenizer({:hf, "distilbert-base-uncased-finetuned-sst-2-english"})
    serving = Bumblebee.Text.text_classification(model, tokenizer)

    children = [
      {Nx.Serving, serving: serving, name: MeuApp.Serving, batch_size: 8, batch_timeout: 100}
    ]
    Supervisor.init(children, strategy: :one_for_one)
  end
end

# Chamar de qualquer processo
Nx.Serving.batched_run(MeuApp.Serving, texto)

Explorer — DataFrames

Manipulação de dados com Explorer.DataFrame
# mix.exs
{:explorer, "~> 0.10"}

alias Explorer.DataFrame, as: DF
alias Explorer.Series, as: S

# Carregar dados
df = DF.from_csv!("vendas.csv")
df_parquet = DF.from_parquet!("dados.parquet")

# Explorar
DF.shape(df)       # {linhas, colunas}
DF.dtypes(df)      # %{"preco" => :float, "produto" => :string, ...}
DF.describe(df)    # estatísticas: mean, std, min, max

# Filtrar, selecionar e transformar
resultado = df
  |> DF.filter(col("preco") > 100)
  |> DF.select(["produto", "preco", "categoria"])
  |> DF.mutate(preco_com_iva: col("preco") * 1.17)
  |> DF.sort_by([{:desc, col("preco")}])

# Agrupamento e agregação
por_categoria = df
  |> DF.group_by("categoria")
  |> DF.summarise(
      total_vendas: sum(col("preco")),
      qtd: count(col("id")),
      media: mean(col("preco")))

# Join entre DataFrames
df_joined = DF.join(df_vendas, df_clientes,
  on: [{"cliente_id", "id"}], how: :left)

# Exportar
DF.to_csv!(resultado, "resultado.csv")
DF.to_parquet!(resultado, "resultado.parquet")
