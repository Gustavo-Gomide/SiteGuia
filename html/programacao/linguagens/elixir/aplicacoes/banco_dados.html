Elixir — Banco de Dados

Ecto é a biblioteca de acesso a dados do ecossistema Elixir: não é um ORM clássico, mas um toolkit composável de queries, changesets e migrations. A separação entre schema (mapeamento), changeset (validação) e Repo (execução) permite testar cada camada isoladamente e evitar uma classe inteira de bugs existentes em ORMs convencionais.

Em Ecto, os dados nunca são salvos implicitamente. Todo write passa explicitamente por Repo.insert/update/delete. Valide antes com changeset, compose com Multi, execute com Repo — tudo auditável e reversível.

defmodule MeuApp.Repo.Migrations.CriaTabelaUsuarios do
  use Ecto.Migration

  def change do
    create table(:usuarios, primary_key: false) do
      add :id, :binary_id, primary_key: true, default: fragment("gen_random_uuid()")
      add :nome, :string, null: false
      add :email, :string, null: false
      add :senha_hash, :string, null: false
      add :role, :string, default: "user"
      add :ativo, :boolean, default: true
      add :ultimo_login, :utc_datetime
      timestamps(type: :utc_datetime)
    end

    create unique_index(:usuarios, [:email])
    create index(:usuarios, [:role, :ativo])

    # Migration de constraint com check
    create constraint(:usuarios, :role_valido,
      check: "role IN ('admin', 'user', 'moderador')")
  end
end

Schemas e Changesets

Schema com associações e changesets para create/update
defmodule MeuApp.Contas.Usuario do
  use Ecto.Schema
  import Ecto.Changeset

  @primary_key {:id, :binary_id, autogenerate: true}
  @foreign_key_type :binary_id

  schema "usuarios" do
    field :nome, :string
    field :email, :string
    field :senha, :string, virtual: true  # não persiste
    field :senha_hash, :string
    field :role, :string, default: "user"
    has_many :posts, MeuApp.Blog.Post, on_delete: :delete_all
    has_one :perfil, MeuApp.Contas.Perfil
    timestamps(type: :utc_datetime)
  end

  @permitidos_create ~w(nome email senha role)a
  @obrigatorios ~w(nome email senha)a

  def changeset_create(usuario, attrs) do
    usuario
    |> cast(attrs, @permitidos_create)
    |> validate_required(@obrigatorios)
    |> validate_length(:nome, min: 2, max: 100)
    |> validate_format(:email, ~r/^[^\s]+@[^\s]+\.[^\s]+$/,
        message: "formato inválido de email")
    |> validate_length(:senha, min: 8)
    |> validate_inclusion(:role, ~w(admin user moderador guest))
    |> unique_constraint(:email)
    |> hash_senha()
  end

  def changeset_update(usuario, attrs) do
    usuario
    |> cast(attrs, ~w(nome bio avatar_url)a)
    |> validate_required(~w(nome)a)
    |> validate_length(:nome, min: 2, max: 100)
  end

  defp hash_senha(changeset) do
    case changeset do
      %Ecto.Changeset{valid?: true, changes: %{senha: senha}} ->
        put_change(changeset, :senha_hash, Bcrypt.hash_pwd_salt(senha))
      _ -> changeset
    end
  end
end

Queries Composíveis e Prevenção de N+1

Ecto.Query — composição, preload e agregações
import Ecto.Query

# Query base reutilizável
defmodule MeuApp.Blog.PostQuery do
  def base, do: from(p in Post, where: p.publicado == true)
  def por_autor(q, id), do: where(q, [p], p.usuario_id == ^id)
  def recentes(q, dias), do: where(q, [p], p.inserted_at > ago(^dias, "day"))
  def com_comentarios(q), do: preload(q, [p], :comentarios)
end

# Compondo a query dinamicamente
posts = PostQuery.base()
        |> PostQuery.por_autor(user_id)
        |> PostQuery.recentes(30)
        |> PostQuery.com_comentarios()
        |> Repo.all()

# N+1: ERRADO — cada post gera 1 query para autor
posts = Repo.all(Post)
Enum.each(posts, fn p -> IO.inspect(p.usuario.nome) end)  # N queries! BUG!

# N+1: CORRETO — preload em 1 query JOIN
posts = Post |> preload([:usuario, comentarios: :autor]) |> Repo.all()

# Preload condicional com query customizada
posts = Repo.preload(posts,
  comentarios: from(c in Comentario, order_by: [desc: c.inserted_at], limit: 5))

# JOIN manual + select customizado
from(u in Usuario,
  join: p in assoc(u, :posts),
  group_by: u.id,
  select: %{usuario: u, total_posts: count(p.id), views: sum(p.views)})
|> Repo.all()

# Subquery
posts_populares = from(p in Post, where: p.views > 1000, select: p.usuario_id)
from(u in Usuario, where: u.id in subquery(posts_populares)) |> Repo.all()

Transações com Ecto.Multi

Multi — operações atômicas composíveis
alias Ecto.Multi

def transferir_creditos(de_id, para_id, valor) do
  Multi.new()
  |> Multi.run(:de, fn repo, _ ->
    case repo.get(Usuario, de_id) do
      nil -> {:error, :usuario_nao_encontrado}
      u when u.creditos < valor -> {:error, :saldo_insuficiente}
      u -> {:ok, u}
    end
  end)
  |> Multi.run(:para, fn repo, _ ->
    case repo.get(Usuario, para_id) do
      nil -> {:error, :destinatario_nao_encontrado}
      u -> {:ok, u}
    end
  end)
  |> Multi.update(:debitar, fn %{de: u} ->
    change(u, creditos: u.creditos - valor)
  end)
  |> Multi.update(:creditar, fn %{para: u} ->
    change(u, creditos: u.creditos + valor)
  end)
  |> Multi.insert(:historico, fn %{de: de, para: para} ->
    Transferencia.changeset(%Transferencia{},
      %{de_id: de.id, para_id: para.id, valor: valor})
  end)
  |> Repo.transaction()
  |> case do
    {:ok, %{historico: h}} -> {:ok, h}
    {:error, _op, reason, _changes} -> {:error, reason}
  end
end

Raw SQL e Multi-tenant

Repo.query!/2 e prefixos de schema
# Raw SQL com parâmetros seguros (sem interpolação)
{:ok, %{rows: rows}} = Repo.query(
  "SELECT id, nome, SUM(views) FROM posts GROUP BY 1,2 HAVING SUM(views) > $1",
  [1000]
)

# Mapear resultado manualmente
usuarios = Repo.query!("SELECT id, email FROM usuarios WHERE ativo = $1", [true])
|> Map.get(:rows)
|> Enum.map(fn [id, email] -> %{id: id, email: email} end)

# Multi-tenant com prefixos de schema PostgreSQL
# Cada cliente tem seu schema: cliente_abc.usuarios, cliente_xyz.usuarios
opts_tenant = [prefix: "cliente_#{tenant_id}"]

Repo.all(Usuario, opts_tenant)
Repo.insert(changeset, opts_tenant)

# Criar schema de tenant em migration
def criar_tenant(tenant_id) do
  Repo.query!("CREATE SCHEMA IF NOT EXISTS \"cliente_#{tenant_id}\"")
  Ecto.Migrator.run(Repo, migrations_path(), :up,
    all: true, prefix: "cliente_#{tenant_id}")
end

Ecto.Multi não apenas agrupa operações em uma transação — ele acumula o resultado de cada passo, permitindo que operações subsequentes usem dados das anteriores. Se qualquer operação retornar {:error, reason}, toda a transação é revertida e o Multi retorna qual operação falhou — sem try/catch, sem surpresas.
