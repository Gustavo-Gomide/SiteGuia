Elixir — Jogos

Elixir não renderiza jogos diretamente — mas é uma escolha excelente para game servers multiplayer: cada sala de jogo é um GenServer, cada jogador é um processo, o estado do jogo é imutável e auditável, e a tolerância a falhas garante que um crash em uma sala não derruba as demais. Usado em jogos como Spellbreak (backend) e similiares.

O modelo de atores da BEAM é perfeitamente alinhado com jogos multiplayer: cada entidade (sala, jogador, NPC) é um processo independente. Comunicação por mensagens elimina race conditions. Supervisores garantem que o servidor de jogo nunca para completamente.

Servidor de Sala de Jogo

GenServer para estado de partida multiplayer
defmodule Jogo.Sala do
  use GenServer

  @max_jogadores 4
  @tick_ms 50  # 20 ticks/segundo

  defstruct [
    :id, jogadores: %{}, estado: :aguardando,
    turno: 0, objetos: [], tick_ref: nil
  ]

  def start_link(sala_id) do
    GenServer.start_link(__MODULE__, sala_id,
      name: {:via, Registry, {Jogo.Registry, sala_id}})
  end

  def entrar(sala_id, jogador_id, nick) do
    GenServer.call(via(sala_id), {:entrar, jogador_id, nick})
  end
  def acao(sala_id, jogador_id, acao) do
    GenServer.cast(via(sala_id), {:acao, jogador_id, acao})
  end

  @impl true
  def init(sala_id) do
    {:ok, %__MODULE__{id: sala_id}}
  end

  @impl true
  def handle_call({:entrar, jogador_id, nick}, _from, sala) do
    cond do
      map_size(sala.jogadores) >= @max_jogadores ->
        {:reply, {:error, :sala_cheia}, sala}
      sala.estado != :aguardando ->
        {:reply, {:error, :partida_em_andamento}, sala}
      true ->
        novo_jogador = %{id: jogador_id, nick: nick, pontos: 0, pos: {0, 0}}
        jogadores = Map.put(sala.jogadores, jogador_id, novo_jogador)
        nova_sala = %{sala | jogadores: jogadores}
        nova_sala = if map_size(jogadores) == @max_jogadores, do: iniciar_partida(nova_sala), else: nova_sala
        {:reply, :ok, nova_sala}
    end
  end

  @impl true
  def handle_cast({:acao, jogador_id, acao}, sala) do
    nova_sala = processar_acao(sala, jogador_id, acao)
    broadcast(sala.id, {:estado_atualizado, nova_sala})
    {:noreply, nova_sala}
  end

  @impl true
  def handle_info(:tick, sala) do
    nova_sala = sala |> atualizar_fisica() |> verificar_colisoes()
    Process.send_after(self(), :tick, @tick_ms)
    {:noreply, %{nova_sala | turno: sala.turno + 1}}
  end

  defp iniciar_partida(sala) do
    Process.send_after(self(), :tick, @tick_ms)
    %{sala | estado: :em_andamento}
  end

  defp via(id), do: {:via, Registry, {Jogo.Registry, id}}
end

Lobby e DynamicSupervisor para Salas

Criar e destruir salas dinamicamente
defmodule Jogo.Lobby do
  use DynamicSupervisor

  def start_link(_), do: DynamicSupervisor.start_link(__MODULE__, [], name: __MODULE__)

  @impl true
  def init(_), do: DynamicSupervisor.init(strategy: :one_for_one)

  def criar_sala(sala_id) do
    DynamicSupervisor.start_child(__MODULE__, {Jogo.Sala, sala_id})
  end

  def listar_salas do
    Registry.select(Jogo.Registry, [{{:"$1", :"$2", :"$3"}, [], [{{:"$1", :"$2"}}]}])
  end

  def fechar_sala(sala_id) do
    case Registry.lookup(Jogo.Registry, sala_id) do
      [{pid, _}] -> DynamicSupervisor.terminate_child(__MODULE__, pid)
      [] -> {:error, :sala_nao_encontrada}
    end
  end
end

Broadcast de Estado via PubSub

Sincronizar estado de jogo em tempo real
# Cada client conecta a um Channel da sala
defmodule MeuJogoWeb.SalaChannel do
  use MeuJogoWeb, :channel

  def join("jogo:" <> sala_id, %{"jogador_id" => jid, "nick" => nick}, socket) do
    case Jogo.Sala.entrar(sala_id, jid, nick) do
      :ok ->
        socket = assign(socket, sala_id: sala_id, jogador_id: jid)
        Phoenix.PubSub.subscribe(MeuJogo.PubSub, "sala:#{sala_id}")
        {:ok, socket}
      {:error, motivo} -> {:error, %{reason: motivo}}
    end
  end

  def handle_in("acao", %{"tipo" => tipo, "dados" => dados}, socket) do
    Jogo.Sala.acao(socket.assigns.sala_id, socket.assigns.jogador_id, {tipo, dados})
    {:noreply, socket}
  end

  def handle_info({:estado_atualizado, estado}, socket) do
    push(socket, "estado", JogoJSON.serialize(estado))
    {:noreply, socket}
  end
end

# Na Sala.GenServer — ao mudar estado, publica para todos
defp broadcast(sala_id, evento) do
  Phoenix.PubSub.broadcast(MeuJogo.PubSub, "sala:#{sala_id}", evento)
end
