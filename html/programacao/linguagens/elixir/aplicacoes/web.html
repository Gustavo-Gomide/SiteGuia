Elixir — Web

Phoenix é o framework web mais performático baseado em linguagem de alto nível, capaz de suportar 2 milhões de conexões WebSocket simultâneas em uma única máquina — demonstrado pelo time do Discord. Para APIs REST, suporta centenas de milhares de requisições por segundo. Para UIs em tempo real, o Phoenix Channels e LiveView eliminam a necessidade de servidores de websocket separados.

O benchmark de 2 milhões de conexões do Phoenix Channels não é marketing — é resultado direto de processos leves da BEAM (não threads): cada conexão WebSocket é um processo Elixir consumindo ~2-4 KB de RAM.

REST API com Phoenix

API JSON completa — controller, params e resposta
# router.ex — definir API
scope "/api/v1", MeuAppWeb do
  pipe_through [:api, :authenticate]  # pipeline com autenticação

  resources "/produtos", ProdutoController,
    only: [:index, :show, :create, :update, :delete]
  get "/produtos/:id/estoque", ProdutoController, :estoque
end

# produto_controller.ex
defmodule MeuAppWeb.ProdutoController do
  use MeuAppWeb, :controller
  alias MeuApp.Loja

  def index(conn, params) do
    produtos = Loja.listar_produtos(params)
    render(conn, :index, produtos: produtos)
  end

  def create(conn, %{"produto" => produto_params}) do
    case Loja.criar_produto(produto_params) do
      {:ok, produto} ->
        conn
        |> put_status(:created)
        |> put_resp_header("location", ~p"/api/v1/produtos/#{produto.id}")
        |> render(:show, produto: produto)
      {:error, %Ecto.Changeset{} = cs} ->
        conn
        |> put_status(:unprocessable_entity)
        |> render(:error, changeset: cs)
    end
  end

  def delete(conn, %{"id" => id}) do
    produto = Loja.get_produto!(id)
    with {:ok, _} <- Loja.deletar_produto(produto) do
      send_resp(conn, :no_content, "")
    end
  end
end

# produto_json.ex — serialização (Phoenix 1.7+)
defmodule MeuAppWeb.ProdutoJSON do
  def index(%{produtos: produtos}), do: %{data: Enum.map(produtos, &data/1)}
  def show(%{produto: produto}), do: %{data: data(produto)}

  defp data(produto) do
    %{id: produto.id, nome: produto.nome,
      preco: produto.preco, estoque: produto.estoque}
  end
end

Autenticação — Guardian JWT

Guardian — token-based auth
# mix.exs
{:guardian, "~> 2.0"},
{:bcrypt_elixir, "~> 3.0"}

# lib/meu_app_web/auth/guardian.ex
defmodule MeuAppWeb.Auth.Guardian do
  use Guardian, otp_app: :meu_app

  def subject_for_token(%Usuario{id: id}, _claims), do: {:ok, to_string(id)}
  def resource_from_claims(%{"sub" => id}) do
    usuario = Repo.get(Usuario, id)
    {:ok, usuario}
  end
end

# Plug de autenticação
defmodule MeuAppWeb.Auth.Pipeline do
  use Guardian.Plug.Pipeline,
    otp_app: :meu_app,
    module: MeuAppWeb.Auth.Guardian,
    error_handler: MeuAppWeb.Auth.ErrorHandler

  plug Guardian.Plug.VerifyHeader, scheme: "Bearer"
  plug Guardian.Plug.EnsureAuthenticated
  plug Guardian.Plug.LoadResource
end

# Login — gerar token
def login(conn, %{"email" => email, "senha" => senha}) do
  with {:ok, usuario} <- Contas.autenticar(email, senha),
       {:ok, token, _claims} <- Guardian.encode_and_sign(usuario) do
    render(conn, :token, token: token, usuario: usuario)
  else
    {:error, :invalido} ->
      conn |> put_status(:unauthorized) |> render(:error, msg: "Credenciais inválidas")
  end
end

Phoenix Channels — WebSockets

Channel para chat em tempo real
# user_socket.ex
defmodule MeuAppWeb.UserSocket do
  use Phoenix.Socket
  channel "sala:*", MeuAppWeb.SalaChannel

  def connect(%{"token" => token}, socket, _info) do
    case Guardian.decode_and_verify(token) do
      {:ok, claims} -> {:ok, assign(socket, :user_id, claims["sub"])}
      _ -> :error
    end
  end
  def id(socket), do: "users_socket:#{socket.assigns.user_id}"
end

# sala_channel.ex
defmodule MeuAppWeb.SalaChannel do
  use MeuAppWeb, :channel

  def join("sala:" <> sala_id, _payload, socket) do
    mensagens = Chat.listar_mensagens_recentes(sala_id)
    {:ok, %{mensagens: mensagens}, assign(socket, :sala_id, sala_id)}
  end

  def handle_in("nova_mensagem", %{"corpo" => corpo}, socket) do
    user_id = socket.assigns.user_id
    sala_id = socket.assigns.sala_id

    case Chat.criar_mensagem(sala_id, user_id, corpo) do
      {:ok, msg} ->
        broadcast!(socket, "nova_mensagem", MensagemJSON.serialize(msg))
        {:reply, :ok, socket}
      {:error, _} ->
        {:reply, {:error, %{msg: "falha ao enviar"}}, socket}
    end
  end
end

Deploy — Mix Release e Docker

Build release e Dockerfile multi-stage
# Dockerfile multi-stage para produção
FROM elixir:1.17-otp-27-alpine AS build

RUN apk add --no-cache build-base git nodejs npm

WORKDIR /app
ENV MIX_ENV=prod

COPY mix.exs mix.lock ./
RUN mix deps.get --only $MIX_ENV
RUN mix deps.compile

COPY assets assets
RUN mix assets.deploy

COPY lib lib
COPY config config
COPY priv priv

RUN mix compile
RUN mix release

# Imagem de runtime minimal
FROM alpine:3.19 AS app

RUN apk add --no-cache libstdc++ openssl ncurses-libs

WORKDIR /app
ENV HOME=/app MIX_ENV=prod

COPY --from=build /app/_build/prod/rel/meu_app ./

EXPOSE 4000
CMD ["bin/meu_app", "start"]

# runtime.exs — configuração em runtime (env vars)
# config/runtime.exs
import Config

if config_env() == :prod do
  config :meu_app, MeuApp.Repo,
    url: System.fetch_env!("DATABASE_URL"),
    pool_size: String.to_integer(System.get_env("POOL_SIZE", "10"))

  config :meu_app_web, MeuAppWeb.Endpoint,
    url: [host: System.fetch_env!("HOST")],
    secret_key_base: System.fetch_env!("SECRET_KEY_BASE")
end

O Discord usou Phoenix Channels para atingir 5 milhões de usuários online simultâneos antes de migrar partes para Rust por razões de latência de GC. Mas durante anos, um único cluster Elixir sustentou esse tráfego — porque na BEAM, cada conexão WebSocket não é uma thread do sistema operacional (cara), mas um processo leve com near-zero overhead de alocação e GC por-processo.
