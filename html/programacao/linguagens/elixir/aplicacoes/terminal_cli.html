Elixir — Terminal e CLI

Elixir oferece múltiplas formas de criar ferramentas de linha de comando: escript gera um executável standalone (não requer Elixir instalado na máquina alvo), Mix tasks criam comandos no contexto do projeto, e bibliotecas como Owl e Prompt permitem criar TUIs (Text User Interfaces) ricas com tabelas, barras de progresso, listas interativas e input colorido.

Um escript carrega apenas o ERTS (Erlang Runtime) compactado no binário — o executável resultante tem ~1-3 MB, roda em qualquer Linux/macOS sem dependências externas, e inicializa em milissegundos graças ao pré-carregamento do bytecode.

escript — Executável Standalone

Configurar mix.exs e implementar main/1
# mix.exs
def project do
  [
    app: :minha_cli,
    escript: escript_config()
  ]
end

defp escript_config do
  [main_module: MinhaCli]
end

# lib/minha_cli.ex — módulo principal
defmodule MinhaCli do
  def main(argv) do
    {opts, args, invalid} = OptionParser.parse(argv,
      strict: [
        help: :boolean,
        verbose: :boolean,
        output: :string,
        format: :string
      ],
      aliases: [h: :help, v: :verbose, o: :output])

    cond do
      Keyword.get(opts, :help) -> mostrar_ajuda()
      invalid != [] -> erro_e_sair("Opções inválidas: #{inspect(invalid)}")
      args == [] -> erro_e_sair("Arquivo de entrada obrigatório")
      true -> executar(args, opts)
    end
  end

  defp executar([arquivo | _], opts) do
    verbose = Keyword.get(opts, :verbose, false)
    saida   = Keyword.get(opts, :output, "resultado.txt")

    if verbose, do: log("Processando #{arquivo}...")

    arquivo
    |> File.stream!()
    |> Stream.map(&String.trim/1)
    |> Stream.filter(&(&1 != ""))
    |> Enum.into(File.stream!(saida))

    log("Concluído → #{saida}")
  end

  defp mostrar_ajuda do
    IO.puts(~S"""
    minha_cli — processador de arquivos

    USO: minha_cli [opções] arquivo

    OPÇÕES:
      -h, --help        Exibir esta ajuda
      -v, --verbose     Saída detalhada
      -o, --output ARQ  Arquivo de saída (padrão: resultado.txt)
    """)
  end

  defp erro_e_sair(msg) do
    IO.puts(:stderr, "ERRO: #{msg}")
    System.halt(1)
  end

  defp log(msg), do: IO.puts(IO.ANSI.cyan() <> "[info] " <> IO.ANSI.reset() <> msg)
end

# Compilar e executar
$ mix escript.build
$ ./minha_cli --verbose dados.txt --output saida.txt

IO.ANSI — Terminal Colorido

Cores, formatação e progress bar manual
alias IO.ANSI

# Texto colorido e formatado
IO.puts(ANSI.green() <> ANSI.bright() <> "✓ Sucesso!" <> ANSI.reset())
IO.puts(ANSI.red() <> "✗ Erro: arquivo não encontrado" <> ANSI.reset())
IO.puts(ANSI.yellow() <> "⚠ Aviso: campo opcional ausente" <> ANSI.reset())

# Tabela formatada no terminal
defmodule Tabela do
  def print(cabecalhos, linhas) do
    larguras = Enum.map(0..length(cabecalhos)-1, fn i ->
      max_col = Enum.map(linhas, fn l -> String.length(Enum.at(l, i, "")) end)
      Enum.max([String.length(Enum.at(cabecalhos, i)) | max_col])
    end)

    sep = larguras |> Enum.map(&String.duplicate("-", &1 + 2)) |> Enum.join("+")
    IO.puts("+#{sep}+")
    print_linha(cabecalhos, larguras, bold: true)
    IO.puts("+#{sep}+")
    Enum.each(linhas, &print_linha(&1, larguras))
    IO.puts("+#{sep}+")
  end

  defp print_linha(cols, larguras, opts \\ []) do
    bold = Keyword.get(opts, :bold, false)
    linha = cols |> Enum.zip(larguras)
      |> Enum.map(fn {c, l} -> " #{String.pad_trailing(c, l)} " end)
      |> Enum.join("|")
    cor = if bold, do: ANSI.bright() <> ANSI.cyan(), else: ""
    IO.puts("#{cor}|#{linha}|#{ANSI.reset()}")
  end
end

# Uso:
Tabela.print(
  ["Nome", "Versão", "Status"],
  [["Phoenix", "1.7.14", "ok"], ["Ecto", "3.12.3", "ok"]]
)

Mix Tasks Personalizadas

mix meu_app.seed e mix meu_app.relatorio
# lib/mix/tasks/meu_app.seed.ex
defmodule Mix.Tasks.MeuApp.Seed do
  use Mix.Task

  @shortdoc "Popula o banco com dados de teste"
  @moduledoc """
  Insere dados de seed no banco de dados.

  ## Uso

      mix meu_app.seed              # insere dados padrão
      mix meu_app.seed --env prod   # force em produção
  """

  @impl Mix.Task
  def run(args) do
    {opts, _, _} = OptionParser.parse(args, switches: [env: :string])

    if Mix.env() == :prod and Keyword.get(opts, :env) != "prod" do
      Mix.raise("Use --env prod para executar em produção")
    end

    Mix.Task.run("app.start")   # iniciar aplicação (banco disponível)

    IO.puts(IO.ANSI.yellow() <> "Inserindo seeds..." <> IO.ANSI.reset())

    Enum.each(1..10, fn i ->
      {:ok, u} = MeuApp.Contas.criar_usuario(%{
        nome: "Usuário #{i}",
        email: "user#{i}@seed.test",
        senha: "senha12345"
      })
      IO.puts("  ✓ #{u.email}")
    end)

    IO.puts(IO.ANSI.green() <> "Seed concluído!" <> IO.ANSI.reset())
  end
end

# Executar
$ mix meu_app.seed
$ mix meu_app.seed --env prod
