Elixir — Embedded e IoT

Nerves transforma Elixir em uma stack completa para dispositivos embarcados: você escreve código Elixir, o Nerves compila para uma imagem Linux minimalista (sem systemd, sem apt, sem bash) que roda diretamente em Raspberry Pi, Beaglebone, ou placas personalizadas. O supervisor da BEAM substitui o init do sistema operacional.

Um dispositivo Nerves reinicia em less than 5 segundos, consome menos de 64 MB de RAM e usa atualizações atômicas via A/B partition — se a nova firmware falhar, o dispositivo reverte automaticamente para a versão anterior.

Nerves — Setup e Estrutura de Projeto

Instalação, targets e flashing em Raspberry Pi
# Instalar toolchains e bootstrapper
$ mix archive.install hex nerves_bootstrap
$ mix nerves.new meu_device --target rpi4

# Estrutura gerada:
# meu_device/
#   config/target.exs    — config específica do target
#   lib/meu_device/      — sua aplicação
#   rootfs_overlay/      — arquivos extras na imagem
#   mix.exs

# mix.exs — dependências
def deps(_target) do
  [
    {:nerves, "~> 1.11", runtime: false},
    {:nerves_system_rpi4, "~> 1.25", runtime: false, targets: :rpi4},
    {:nerves_runtime, "~> 0.13"},
    {:nerves_pack, "~> 0.7"},      # Wi-Fi, SSH, mDNS
    {:circuits_gpio, "~> 2.0"},
    {:circuits_i2c, "~> 2.0"},
    {:circuits_uart, "~> 1.5"}
  ]
end

# Compilar e flashear
$ export MIX_TARGET=rpi4
$ mix deps.get
$ mix firmware          # gera meu_device.fw
$ mix firmware.burn     # grava no cartão SD
$ mix upload            # atualiza via rede (OTA)
$ mix ssh               # acesso IEx remoto

Circuits — GPIO, I2C e UART

Controle de hardware com Circuits.GPIO e Circuits.I2C
# GPIO — ler botão e acionfar LED
alias Circuits.GPIO

{:ok, led}  = GPIO.open(17, :output)
{:ok, btn}  = GPIO.open(27, :input, pull_mode: :pullup)

GPIO.write(led, 1)              # ligar LED
GPIO.read(btn)                  # 0 = pressionado (pullup)

# Interrupcão via mensagem ao processo
GPIO.set_interrupts(btn, :both)
receive do
  {:circuits_gpio, 27, _ts, valor} ->
    IO.puts("GPIO 27 mudou para #{valor}")
end

# I2C — ler sensor de temperatura BME280
alias Circuits.I2C

{:ok, bus} = I2C.open("i2c-1")
# Endereço do BME280 = 0x76
{:ok, <>} = I2C.write_read(bus, 0x76, <<0xD0>>, 1)
# chip_id deve ser 0x60 (BME280)

# UART — comunicação serial com sensor
alias Circuits.UART

{:ok, pid} = UART.start_link()
UART.open(pid, "/dev/ttyAMA0",
  speed: 115200, data_bits: 8, stop_bits: 1, parity: :none,
  active: true)  # ativa modo evento → mensagens ao processo

UART.write(pid, "AT\r\n")  # enviar comando AT para modem
receive do
  {:circuits_uart, "/dev/ttyAMA0", dados} -> IO.inspect(dados)
end

GenServer para Gerenciar Hardware

Sensor supervisionado com leitura periódica
defmodule MeuDevice.SensorTemperatura do
  use GenServer
  alias Circuits.I2C
  @intervalo_ms 5_000
  @endereco 0x76

  def start_link(_), do: GenServer.start_link(__MODULE__, [], name: __MODULE__)
  def leitura, do: GenServer.call(__MODULE__, :leitura)

  @impl true
  def init(_) do
    {:ok, bus} = I2C.open("i2c-1")
    schedule_leitura()
    {:ok, %{bus: bus, ultima_leitura: nil}}
  end

  @impl true
  def handle_call(:leitura, _from, state) do
    {:reply, state.ultima_leitura, state}
  end

  @impl true
  def handle_info(:ler_sensor, %{bus: bus} = state) do
    leitura = ler_bme280(bus)
    schedule_leitura()
    {:noreply, %{state | ultima_leitura: leitura}}
  end

  defp schedule_leitura(), do: Process.send_after(self(), :ler_sensor, @intervalo_ms)

  defp ler_bme280(bus) do
    # leitura simplificada — implementação real lê registradores de compensação
    {:ok, dados} = I2C.write_read(bus, @endereco, <<0xF7>>, 8)
    %{temperatura: decodificar_temp(dados), ts: DateTime.utc_now()}
  end
end

NervesHub — OTA e Gestão de Frota

Deploy de firmware via NervesHub Cloud
# mix.exs
{:nerves_hub_link, "~> 2.4"}

# config/target.exs — credenciais do dispositivo
config :nerves_hub_link,
  org: "minha-empresa",
  product: "meu-device"

# Assinar firmware antes de publicar (chave RSA)
$ mix nerves.artifact          # empacotar
$ nerves_hub product create meu-device
$ nerves_hub firmware publish  # enviar para a nuvem
$ nerves_hub deployment create --name v1.2 \
    --firmware meu_device.fw --version >=1.2.0

# No dispositivo — verificar atualizações automaticamente
# NervesHubLink gerencia o upgrade A/B partition:
# 1. Baixa nova firmware em /dev/mmcblk0p3 (partição inativa)
# 2. Verifica hash SHA256
# 3. Marca nova partição como "next boot"
# 4. Reinicia — se boot falhar 3x → volta para partição anterior

# Rollback manual
Nerves.Runtime.revert()
