Fundamentos de Elixir

Fundamentos do Elixir

Tipos primitivos, coleções, pattern matching, módulos, funções, pipe operator, estruturas de controle e recursão — todos os pilares da programação funcional em Elixir.

Dominar os fundamentos do Elixir significa pensar em transformações de dados imutáveis, correspondência de padrões e composição de funções — uma mudança de paradigma que recompensa com código mais previsível e conciso.

Sintaxe Base no Formato Site Guia

Pattern matching, pipe e Enum em bloco padrao
entrada = [1, 2, 3, 4]

resultado =
  entrada
  |> Enum.map(fn n -> n * 2 end)
  |> Enum.filter(fn n -> rem(n, 4) == 0 end)

{:ok, pares} = {:ok, resultado}
IO.inspect(pares)

# Inteiros
x = 42

# Strings
s = "Elixir"

String.starts_with?(s, "El") # → true

String.replace(s, "x", "X") # → "EliXir"

"Olá" <> " Mundo"        # concatenação → "Olá Mundo"

# Atoms
:ok
:error

Coleções

Listas — Lista Ligada Imutável

# Listas são listas ligadas (linked lists)
lista = [1, 2, 3, 4, 5]

# head e tail (cabeça e cauda)
[head | tail] = lista
# head = 1, tail = [2, 3, 4, 5]

# Prepend é O(1); append é O(n)
nova = [0 | lista]          # → [0, 1, 2, 3, 4, 5]

# Operações de lista
List.first(lista)           # → 1
List.last(lista)            # → 5
length(lista)              # → 5
lista ++ [6, 7]            # → [1, 2, 3, 4, 5, 6, 7]
lista -- [2, 4]            # → [1, 3, 5]
Enum.reverse(lista)        # → [5, 4, 3, 2, 1]
Enum.sort([3, 1, 2])       # → [1, 2, 3]
Enum.uniq([1, 2, 1, 3])   # → [1, 2, 3]
Enum.member?(lista, 3)    # → true
3 in lista                 # → true (equivalente)

Tuplas — Estrutura de Tamanho Fixo

# Tuplas: acesso O(1), tamanho fixo, armazenadas contiguamente
tupla = {:ok, 42, "hello"}

elem(tupla, 0)             # → :ok (índice 0-based)
elem(tupla, 1)             # → 42
tuple_size(tupla)          # → 3
put_elem(tupla, 1, 100)   # → {:ok, 100, "hello"} (nova tupla)

# Padrão comum: {:ok, valor} ou {:error, motivo}
case File.read("arquivo.txt") do
  {:ok, conteudo}  -> IO.puts(conteudo)
  {:error, motivo} -> IO.puts("Erro: #{motivo}")
end

Mapas — Dicionários Imutáveis

# Map literal
pessoa = %{nome: "Ana", idade: 30, cidade: "SP"}

# Acesso por atom key (sintaxe ponto)
pessoa.nome                # → "Ana"
pessoa[:idade]             # → 30 (não levanta erro se key ausente, retorna nil)
Map.get(pessoa, :nome)    # → "Ana"
Map.get(pessoa, :peso, 0) # → 0 (default)
Map.has_key?(pessoa, :nome) # → true

# Atualização imutável: %{mapa | key: novo_val}
atualizado = %{pessoa | idade: 31}
# pessoa não foi modificado; atualizado é nova struct

# Adicionar/remover chaves
Map.put(pessoa, :email, "ana@ex.com")
Map.delete(pessoa, :cidade)
Map.merge(pessoa, %{pais: "Brasil"})

# Keys string
config = %{"host" => "localhost", "port" => 5432}
config["host"]              # → "localhost"

# Pattern matching em mapas
%{nome: nome, idade: idade} = pessoa
# nome = "Ana", idade = 30

Keyword Lists e MapSets

# Keyword list: lista de tuplas {:atom, valor} — permite duplicate keys
opts = [timeout: 5000, retries: 3, timeout: 3000]
Keyword.get(opts, :timeout)        # → 5000 (primeiro match)
Keyword.get_values(opts, :timeout)  # → [5000, 3000]
Keyword.put(opts, :debug, true)    # adiciona chave

# Padrão comum em APIs Elixir: último argumento é keyword list de opts
MyMod.connect(url, timeout: 5000, ssl: true)

# MapSet — conjunto de valores únicos
s = MapSet.new([1, 2, 3, 2, 1])     # → MapSet{1, 2, 3}
MapSet.member?(s, 2)                 # → true
MapSet.put(s, 4)                     # → MapSet{1, 2, 3, 4}
s2 = MapSet.new([2, 3, 5])
MapSet.intersection(s, s2)           # → MapSet{2, 3}
MapSet.union(s, s2)                  # → MapSet{1, 2, 3, 4, 5}
MapSet.difference(s, s2)             # → MapSet{1}

Pattern Matching

O Operador = é Matching, não Atribuição

# = é operador de match (unificação)
x = 1         # match: x recebe 1
1 = x         # match com literal: OK (x é 1)
# 2 = x      → MatchError: no match of right hand side value: 1

# Destructuring em tuplas
{a, b, c} = {1, 2, 3}   # a=1, b=2, c=3
{:ok, resultado} = {:ok, 42}   # resultado=42

# Destructuring em listas
[primeiro | resto] = [1, 2, 3]   # primeiro=1, resto=[2,3]
[a, b | _] = [10, 20, 30, 40]   # a=10, b=20, _ ignora resto

# Pin operator ^ — evita rebinding, força match com valor atual
x = 5
^x = 5        # OK
# ^x = 6     → MatchError

# Pattern matching em mapas (match parcial)
%{nome: nome} = %{nome: "Ana", idade: 30}
# nome = "Ana" (campos extras ignorados)

# Matching em strings
"hello " <> nome = "hello mundo"
# nome = "mundo"

# Ignoring com _
{_, segundo, _} = {1, 2, 3}
# segundo = 2

Módulos e Funções

Definindo Módulos e Funções Nomeadas

defmodule Matematica do
  @moduledoc "Operações matemáticas básicas"

  # Função pública — def
  def soma(a, b) do
    a + b
  end

  # Forma compacta (one-liner)
  def quadrado(x), do: x * x

  # Função privada — defp
  defp helper(x), do: x * 2

  # Múltiplas cláusulas (pattern matching em parâmetros)
  def fatorial(0), do: 1
  def fatorial(n) when n > 0 do
    n * fatorial(n - 1)
  end

  # Valor padrão com \\
  def saudar(nome, sufixo \\ "!") do
    "Olá, #{nome}#{sufixo}"
  end
end

Matematica.soma(3, 4)         # → 7
Matematica.fatorial(5)       # → 120
Matematica.saudar("Ana")     # → "Olá, Ana!"
Matematica.saudar("Ana", ".") # → "Olá, Ana."

Funções Anônimas e Closures

# fn ... end
soma = fn(a, b) -> a + b end
soma.(3, 4)   # → 7 (note o ponto antes dos parênteses)

# Sintaxe curta &
dobro = &(&1 * 2)    # &1 = primeiro argumento
dobro.(5)             # → 10

soma_curta = &(&1 + &2)
soma_curta.(3, 4)     # → 7

# Captura de função nomeada: &Modulo.funcao/aridade
dobrar = &Matematica.quadrado/1
dobrar.(4)             # → 16

# Closure captura variáveis do escopo léxico
multiplicador = 3
triplicar = fn x -> x * multiplicador end
triplicar.(5)          # → 15

# Múltiplas cláusulas em fn
descrever = fn
  0  -> "zero"
  1  -> "um"
  n  -> "outro: #{n}"
end

descrever.(0)   # → "zero"
descrever.(7)   # → "outro: 7"

Guard Clauses — when

defmodule Validador do
  # Guards em cláusulas de função
  def classificar(n) when is_integer(n) and n < 0, do: :negativo
  def classificar(0), do: :zero
  def classificar(n) when is_integer(n) and n > 0, do: :positivo
  def classificar(_), do: :nao_inteiro

  # Guards permitidos: is_integer/is_float/is_atom/is_binary/is_list/is_map
  # is_nil/is_boolean/is_function/is_pid/is_reference/is_tuple
  # comparações: ==, !=, <, >, <=, >=, ===, !==
  # aritmética: +, -, *, div, rem, abs, length, size, etc.
  # lógica: and, or, not
end

Validador.classificar(-5)    # → :negativo
Validador.classificar(0)     # → :zero
Validador.classificar(7)     # → :positivo
Validador.classificar("x")   # → :nao_inteiro

Pipe Operator |>

Encadeamento de Transformações

# Sem pipe: difícil de ler (inside-out)
resultado = Enum.sum(Enum.filter(Enum.map([1,2,3,4,5], &(&1 * 2)), &(&1 > 4)))

# Com pipe: legível de cima para baixo
resultado =
  [1, 2, 3, 4, 5]
  |> Enum.map(&(&1 * 2))    # [2, 4, 6, 8, 10]
  |> Enum.filter(&(&1 > 4)) # [6, 8, 10]
  |> Enum.sum()             # 24

# |> passa o resultado como PRIMEIRO argumento da próxima função
# "hello" |> String.upcase() equivale a String.upcase("hello")

# Exemplo prático: processamento de texto
texto_limpo =
  "  Olá, MUNDO! Como vai você?  "
  |> String.trim()
  |> String.downcase()
  |> String.replace(~r/[!?,]/, "")   # regex: remover pontuação
  |> String.split()
  |> Enum.uniq()
  |> Enum.join(" ")

# → "olá mundo como vai você"

Estruturas de Controle

if, unless, cond, case, with

# if / else (retornam valor — são expressões)
resultado = if x > 0 do
  "positivo"
else
  "não positivo"
end

# unless = if not
unless is_nil(valor) do
  IO.puts(valor)
end

# cond: múltiplas condições (like elsif-chain)
descricao = cond do
  nota >= 9 -> "Excelente"
  nota >= 7 -> "Bom"
  nota >= 5 -> "Regular"
  true      -> "Reprovado"   # else default
end

# case: pattern matching com múltiplos padrões
case Map.fetch(mapa, :chave) do
  {:ok, valor}    -> "Encontrado: #{valor}"
  :error          -> "Não encontrado"
end

# with: encadeia pattern matches, propaga erros
with {:ok, usuario} <- buscar_usuario(id),
     {:ok, perfil}  <- buscar_perfil(usuario),
     {:ok, logo}    <- buscar_logo(perfil) do
  {:ok, %{usuario: usuario, logo: logo}}
else
  {:error, motivo} -> {:error, "Falha: #{motivo}"}
end
# with curto-circuita no primeiro match falho e vai para else

Recursão e Tail Call

Recursão com Acumulador (TCO)

defmodule Recursao do
  # Recursão simples (não tail-call: pode estourar stack)
  def soma_lista([]), do: 0
  def soma_lista([head | tail]) do
    head + soma_lista(tail)
  end

  # Tail-call (TCO): acumulador elimina frames de stack
  def soma_tc(lista, acc \\ 0)
  def soma_tc([], acc), do: acc
  def soma_tc([head | tail], acc) do
    soma_tc(tail, acc + head)   # chamada final = tail call
  end

  # Fibonacci com cache implícito via acumulador
  def fib(n), do: fib(n, 0, 1)
  defp fib(0, a, _), do: a
  defp fib(n, a, b), do: fib(n - 1, b, a + b)

  # Processar lista com map recursivo
  def duplicar([]), do: []
  def duplicar([head | tail]) do
    [head * 2 | duplicar(tail)]
  end
end

Recursao.soma_tc([1, 2, 3, 4, 5])   # → 15
Recursao.fib(10)                    # → 55
Recursao.duplicar([1, 2, 3])       # → [2, 4, 6]

Enum e Stream

Enum — Operações Eager sobre Coleções

lista = [1, 2, 3, 4, 5, 6]

# map: transforma cada elemento
Enum.map(lista, fn x -> x * x end)    # [1, 4, 9, 16, 25, 36]

# filter: seleciona elementos
Enum.filter(lista, &(rem(&1, 2) == 0)) # [2, 4, 6]

# reduce: acumula resultado
Enum.reduce(lista, 0, &(&1 + &2))     # 21

# group_by: agrupa por função chave
Enum.group_by(lista, &(rem(&1, 2)))
# %{0 => [2, 4, 6], 1 => [1, 3, 5]}

# flat_map: map + flatten
Enum.flat_map([1, 2, 3], fn x -> [x, x * 10] end)
# [1, 10, 2, 20, 3, 30]

# find / any? / all? / none? / count
Enum.find(lista, &(&1 > 3))         # → 4
Enum.any?(lista, &(&1 > 5))         # → true
Enum.all?(lista, &(&1 > 0))         # → true
Enum.count(lista, &(&1 > 3))        # → 3
Enum.min(lista)                      # → 1
Enum.max(lista)                      # → 6
Enum.sum(lista)                      # → 21
Enum.take(lista, 3)                 # → [1, 2, 3]
Enum.drop(lista, 3)                 # → [4, 5, 6]
Enum.chunk_every(lista, 2)          # → [[1,2], [3,4], [5,6]]
Enum.zip([1,2], ["a","b"])          # → [{1,"a"}, {2,"b"}]
Enum.with_index(lista)              # [{1,0}, {2,1}, ...]

Stream — Avaliação Lazy

# Stream é lazy: não processa até ser materializado
# Ideal para coleções grandes ou infinitas

# Sequência infinita
Stream.iterate(1, &(&1 + 1))         # 1, 2, 3, 4, ... (infinita)
|> Stream.filter(&(rem(&1, 2) == 0)) # números pares
|> Stream.take(5)                    # pega os primeiros 5
|> Enum.to_list()                    # materializa → [2, 4, 6, 8, 10]

# Leitura lazy de arquivo linha por linha
File.stream!("grande_arquivo.log")
|> Stream.filter(& String.contains?(&1, "ERROR"))
|> Stream.map(& String.trim(&1))
|> Enum.take(100)

# Stream.resource: cria stream de recurso externo
Stream.resource(
  fn -> File.open!("data.csv") end,
  fn file ->
    case IO.read(file, :line) do
      :eof  -> {:halt, file}
      linha -> {[linha], file}
    end
  end,
  &File.close/1
)

Structs

defstruct — Mapas com Schema Fixo

defmodule Usuario do
  # defstruct define campos com valores padrão
  defstruct [
    :nome,          # obrigatório (default nil)
    :email,         # obrigatório (default nil)
    idade: 0,       # default 0
    ativo: true    # default true
  ]

  # @enforce_keys garante que campos sejam fornecidos
  @enforce_keys [:nome, :email]

  def novo(nome, email), do: %Usuario{nome: nome, email: email}

  def desativar(%Usuario{} = u), do: %{u | ativo: false}
end

u = Usuario.novo("Ana", "ana@ex.com")
# %Usuario{nome: "Ana", email: "ana@ex.com", idade: 0, ativo: true}

u.nome             # → "Ana"
u.ativo            # → true

atualizado = %{u | idade: 25}
# %Usuario{nome: "Ana", ..., idade: 25}

# Pattern matching em structs
def processar(%Usuario{ativo: false} = u), do: "Usuário #{u.nome} está inativo"
def processar(%Usuario{nome: nome}), do: "Processando #{nome}"

# is_struct/2 verifica tipo de struct
is_struct(u, Usuario)   # → true
is_struct(u, OutroMod)  # → false
