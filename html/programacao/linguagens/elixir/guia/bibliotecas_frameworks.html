Elixir — Bibliotecas e Frameworks

O ecossistema Elixir é centrado no Phoenix Framework para web, Ecto para persistência, Oban para background jobs e LiveView para interfaces reativas sem JavaScript. O gerenciador de pacotes Hex hospeda milhares de bibliotecas, e o próprio Erlang/OTP fornece dezenas de módulos robustos sem dependências externas.

Phoenix LiveView revolucionou o desenvolvimento web com Elixir: interfaces reativas com updates em tempo real via WebSocket, zero JavaScript customizado, e a mesma performance e tolerância a falhas dos processos OTP.

defmodule MeuAppWeb.Router do
  use MeuAppWeb, :router

  pipeline :browser do
    plug :accepts, ["html"]
    plug :fetch_session
    plug :fetch_live_flash
    plug :protect_from_forgery
    plug :put_secure_browser_headers
  end

  pipeline :api do
    plug :accepts, ["json"]
  end

  scope "/", MeuAppWeb do
    pipe_through :browser
    get "/", PageController, :home
    live "/produtos", ProdutoLive.Index
    live "/produtos/:id", ProdutoLive.Show
  end

  scope "/api", MeuAppWeb do
    pipe_through :api
    resources "/usuarios", UsuarioController, only: [:index, :show, :create]
  end
end

# Controller
defmodule MeuAppWeb.UsuarioController do
  use MeuAppWeb, :controller
  alias MeuApp.Contas

  def index(conn, _params) do
    usuarios = Contas.listar_usuarios()
    render(conn, :index, usuarios: usuarios)
  end

  def create(conn, %{"usuario" => params}) do
    case Contas.criar_usuario(params) do
      {:ok, usuario} ->
        conn
        |> put_status(:created)
        |> render(:show, usuario: usuario)
      {:error, changeset} ->
        conn
        |> put_status(:unprocessable_entity)
        |> render(:error, changeset: changeset)
    end
  end
end

Ecto — Banco de Dados

Ecto Schema, Changeset e Query
# Schema — mapeamento para tabela
defmodule MeuApp.Contas.Usuario do
  use Ecto.Schema
  import Ecto.Changeset

  schema "usuarios" do
    field :nome, :string
    field :email, :string
    field :senha_hash, :string
    field :senha, :string, virtual: true  # não persiste
    field :ativo, :boolean, default: true
    has_many :pedidos, MeuApp.Loja.Pedido
    timestamps()  # inserted_at, updated_at
  end

  def changeset(usuario, attrs) do
    usuario
    |> cast(attrs, [:nome, :email, :senha])
    |> validate_required([:nome, :email, :senha])
    |> validate_format(:email, ~r/^[^\s]+@[^\s]+$/)
    |> validate_length(:senha, min: 8)
    |> unique_constraint(:email)
    |> hash_senha()
  end

  defp hash_senha(%{valid?: true, changes: %{senha: senha}} = cs) do
    put_change(cs, :senha_hash, Bcrypt.hash_pwd_salt(senha))
  end
  defp hash_senha(cs), do: cs
end

import Ecto.Query

# Buscar por email
from(u in Usuario, where: u.email == ^email, select: u)
|> Repo.one()

# JOIN com pedidos
from(u in Usuario,
  join: p in assoc(u, :pedidos),
  where: p.status == "pendente",
  where: u.ativo == true,
  order_by: [desc: p.inserted_at],
  limit: 50,
  select: %{usuario: u.nome, pedido_id: p.id, total: p.total}
)
|> Repo.all()

Phoenix LiveView — UI Reativa

LiveView — contador em tempo real
defmodule MeuAppWeb.ContadorLive do
  use MeuAppWeb, :live_view

  @impl true
  def mount(_params, _session, socket) do
    {:ok, assign(socket, contador: 0, historico: [])}
  end

  @impl true
  def handle_event("incrementar", _params, socket) do
    novo = socket.assigns.contador + 1
    hist = [novo | socket.assigns.historico] |> Enum.take(5)
    {:noreply, assign(socket, contador: novo, historico: hist)}
  end

  @impl true
  def handle_event("resetar", _, socket) do
    {:noreply, assign(socket, contador: 0, historico: [])}
  end

  @impl true
  def render(assigns) do
    ~H"""
    <div class="contador">
      <h1>Contador: <%= @contador %></h1>
      <button phx-click="incrementar">+1</button>
      <button phx-click="resetar">Reset</button>
      <ul>
        <%= for n <- @historico do %>
          <li><%= n %></li>
        <% end %>
      </ul>
    </div>
    """
  end
end

# LiveView com subscribe para atualizações em broadcast
def mount(_params, _session, socket) do
  if connected?(socket), do: MeuAppWeb.Endpoint.subscribe("pedidos:novos")
  pedidos = Loja.listar_pedidos_recentes()
  {:ok, assign(socket, pedidos: pedidos)}
end

def handle_info({:novo_pedido, pedido}, socket) do
  pedidos = [pedido | socket.assigns.pedidos] |> Enum.take(20)
  {:noreply, assign(socket, pedidos: pedidos)}
end

Phoenix LiveView demonstrou que é possível ter interfaces reativas de tempo real com a simplicidade do server-side rendering: o estado vive no servidor em um processo Elixir, atualizações são enviadas como diffs mínimos via WebSocket, e o desenvolvedor escreve apenas HTML + lógica Elixir — sem Redux, sem hooks, sem re-renders desnecessários.
