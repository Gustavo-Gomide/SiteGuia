Elixir — Avançado

O diferencial de Elixir está em seu modelo de concorrência baseado em processos leves da BEAM — não threads do SO, mas atores isolados com caixas de mensagem. Junto com o framework OTP (Open Telecom Platform), o Elixir oferece primitivas para construir sistemas que se auto-curam, tolerantes a falhas e escaláveis a milhões de conexões simultâneas.

Um processo BEAM consome ~2 KB de memória e pode ser criado em microssegundos. Uma máquina comum comporta milhões de processos simultâneos. Cada processo tem seu próprio heap, sem memória compartilhada — eliminando race conditions por design.

Processos — spawn, send, receive

Criação e comunicação de processos
# spawn — cria processo e retorna PID
pid = spawn(fn ->
  IO.puts("Olá do processo #{inspect(self())}")
end)
# PID do processo filho

# self() — PID do processo atual
IO.puts("Principal: #{inspect(self())}")

# Enviar mensagem — operador send
send(pid, {:ola, "mundo"})

# receive — bloqueia até receber mensagem correspondente
receive do
  {:ola, msg} ->
    IO.puts("Recebi: #{msg}")
  {:erro, motivo} ->
    IO.puts("Erro: #{motivo}")
after
  5000 -> IO.puts("Timeout!")  # timeout em ms
end

# Loop de servidor genérico manual
defmodule Contador do
  def iniciar, do: spawn(__MODULE__, :loop, [0])

  def loop(n) do
    receive do
      :incrementar              -> loop(n + 1)
      {:buscar, caller_pid}     -> send(caller_pid, n); loop(n)
      :parar                    -> :ok
    end
  end
end

pid = Contador.iniciar()
send(pid, :incrementar)
send(pid, :incrementar)
send(pid, {:buscar, self()})
receive do n -> IO.inspect(n) end   # 2

GenServer — Servidor Genérico OTP

GenServer completo — callbacks e API pública
defmodule Estoque do
  use GenServer

  # ── API Pública ───────────────────────────────────────────────
  def start_link(opts \\ []) do
    GenServer.start_link(__MODULE__, %{}, name: __MODULE__)
  end

  def adicionar(produto, qtd) do
    GenServer.cast(__MODULE__, {:adicionar, produto, qtd})
    # cast = assíncrono, não aguarda resposta
  end

  def consultar(produto) do
    GenServer.call(__MODULE__, {:consultar, produto})
    # call = síncrono, espera resposta
  end

  def remover(produto, qtd) do
    GenServer.call(__MODULE__, {:remover, produto, qtd})
  end

  # ── Callbacks (lado servidor) ────────────────────────────────
  @impl true
  def init(_args) do
    {:ok, %{}}    # estado inicial = mapa vazio
  end

  @impl true
  def handle_cast({:adicionar, produto, qtd}, estado) do
    novo_estado = Map.update(estado, produto, qtd, &(&1 + qtd))
    {:noreply, novo_estado}
  end

  @impl true
  def handle_call({:consultar, produto}, _from, estado) do
    qtd = Map.get(estado, produto, 0)
    {:reply, qtd, estado}   # {:reply, resposta, novo_estado}
  end

  @impl true
  def handle_call({:remover, produto, qtd}, _from, estado) do
    atual = Map.get(estado, produto, 0)
    cond do
      atual >= qtd ->
        novo = Map.put(estado, produto, atual - qtd)
        {:reply, {:ok, atual - qtd}, novo}
      true ->
        {:reply, {:error, "estoque insuficiente"}, estado}
    end
  end

  @impl true
  def handle_info(:verificar_estoque_critico, estado) do
    # handle_info para mensagens não-call/cast (Process.send_after, etc.)
    estado |> Enum.each(fn {p, q} ->
      if q < 10, do: IO.puts("ALERTA: #{p} com apenas #{q} unidades")
    end)
    {:noreply, estado}
  end
end

# Uso
{:ok, _pid} = Estoque.start_link()
Estoque.adicionar("maçã", 100)
Estoque.adicionar("banana", 50)
Estoque.consultar("maçã")     # 100
Estoque.remover("maçã", 30)   # {:ok, 70}

Supervisor — Tolerância a Falhas

Supervisor tree — "let it crash" philosophy
defmodule MeuApp.Application do
  use Application

  @impl true
  def start(_type, _args) do
    children = [
      Estoque,                           # módulo com start_link/1
      {CacheServer, ttl: 3600},          # com opções
      {Task.Supervisor, name: MeuApp.TaskSupervisor}
    ]

    opts = [
      strategy: :one_for_one,  # reiniciar só o filho que falhou
      # :one_for_all   — reiniciar todos se um falhar
      # :rest_for_one  — reiniciar o que falhou + os iniciados depois
      max_restarts: 3,          # máximo 3 reinícios
      max_seconds: 5            # dentro de 5 segundos
    ]

    Supervisor.start_link(children, opts)
  end
end

# Supervisor dinâmico — adicionar filhos em runtime
defmodule MeuApp.SessionSupervisor do
  use DynamicSupervisor

  def start_link(opts), do: DynamicSupervisor.start_link(__MODULE__, opts, name: __MODULE__)

  @impl true
  def init(_opts), do: DynamicSupervisor.init(strategy: :one_for_one)

  def iniciar_sessao(usuario_id) do
    spec = {SessaoServer, usuario_id}
    DynamicSupervisor.start_child(__MODULE__, spec)
  end
end

# Uma falha em um processo filho NÃO afeta os irmãos
# O Supervisor detecta a queda e reinicia automaticamente
# Isso é o "let it crash" — não defensivo, mas resiliente

Task e Agent — Concorrência Simplificada

Task para paralelismo e Agent para estado simples
# Task — executar trabalho assíncrono
task = Task.async(fn -> operacao_longa() end)
# continua executando código aqui...
resultado = Task.await(task, 10_000)  # timeout 10s

# Múltiplas tasks em paralelo
tarefas = Enum.map(1..10, fn i ->
  Task.async(fn -> processar(i) end)
end)
resultados = Task.await_many(tarefas, 30_000)

# Task.Supervisor para tasks supervisionadas
Task.Supervisor.async_nolink(MeuApp.TaskSupervisor, fn ->
  enviar_email(destinatario, conteudo)
end)

# Agent — wrapper de GenServer para estado simples
{:ok, ag} = Agent.start_link(fn -> %{} end)

Agent.update(ag, fn estado -> Map.put(estado, :chave, "valor") end)
valor = Agent.get(ag, fn estado -> estado[:chave] end)
# "valor"

Agent.get_and_update(ag, fn estado ->
  {estado[:chave], Map.put(estado, :chave, "novo")}
end)

Protocolos e Behaviours

protocol — polimorfismo para qualquer tipo
# Definir protocolo (como interface em OOP)
defprotocol Serializavel do
  @doc "Converte estrutura para string JSON-like"
  def serializar(valor)
end

# Implementar para diferentes tipos
defimpl Serializavel, for: Integer do
  def serializar(n), do: Integer.to_string(n)
end

defimpl Serializavel, for: Map do
  def serializar(mapa) do
    pares = mapa
    |> Enum.map(fn {k, v} -> "\"#{k}\": #{Serializavel.serializar(v)}" end)
    |> Enum.join(", ")
    "{#{pares}}"
  end
end

Serializavel.serializar(42)             # "42"
Serializavel.serializar(%{nome: "Ana"}) # "{\"nome\": \"Ana\"}"

# Behaviour — definir contrato para módulos
defmodule Pagamento do
  @callback processar(Float.t()) :: {:ok, String.t()} | {:error, String.t()}
  @callback estornar(String.t()) :: :ok | {:error, String.t()}
end

defmodule Boleto do
  @behaviour Pagamento

  @impl true
  def processar(valor), do: {:ok, "boleto_#{:rand.uniform(9999)}"}

  @impl true
  def estornar(id), do: :ok
end

Metaprogramação — Macros

defmacro, quote e unquote — código que gera código
defmodule MinhasMacros do

  # unless personalizado
  defmacro menos_que(valor, limite, do: bloco) do
    quote do
      if unquote(valor) < unquote(limite) do
        unquote(bloco)
      end
    end
  end

  # Macro para benchmark
  defmacro medir_tempo(label, do: bloco) do
    quote do
      inicio = System.monotonic_time(:microsecond)
      resultado = unquote(bloco)
      fim = System.monotonic_time(:microsecond)
      IO.puts("#{unquote(label)}: #{fim - inicio} µs")
      resultado
    end
  end
end

require MinhasMacros

MinhasMacros.menos_que estoque, 10 do
  IO.puts("Estoque crítico!")
end

MinhasMacros.medir_tempo "sort de 1M elementos" do
  Enum.sort(Enum.to_list(1..1_000_000))
end

A filosofia "let it crash" do OTP é contraintuitiva mas poderosa: em vez de escrever código defensivo com try/catch em cada ponto de falha possível, você deixa o processo falhar e conta com o Supervisor para reiniciá-lo em estado limpo. Isso resulta em código mais simples, mais legível e sistemas que se auto-curam — provado em produção pelo Erlang/OTP há décadas em telecomunicações com 99.9999999% de uptime (nine nines).
