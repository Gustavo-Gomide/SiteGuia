Fortran — Projetos e Boas Práticas

Código Fortran de qualidade usa módulos para encapsulamento, implicit none em todos os escopos, intent em todos os argumentos, tipos derivados para agrupamento de dados relacionados, e testes unitários via test-drive. O profiling com gprof e a análise de cache com Valgrind/perf completam o ciclo de desenvolvimento de alto desempenho.

A armadilha mais comum em código Fortran é a ordem de iteração de arrays 2D: Fortran armazena por coluna (column-major), então iterar do i; do j; A(i,j) salta na memória. A ordem correta é do j; do i; A(i,j) — o loop interno percorre a coluna contiguamente na memória, resultando em 5–10× mais velocidade por localidade de cache.

Estrutura de Projeto Fortran Moderno

Organização por módulos com encapsulamento
! src/constantes.f90 — parâmetros globais em módulo dedicado
module constantes
    implicit none
    private
    public :: dp, pi, g, c_luz

    integer, parameter :: dp     = selected_real_kind(15, 307)
    real(dp), parameter :: pi    = 3.14159265358979323846_dp
    real(dp), parameter :: g     = 9.80665_dp          ! m/s² (ISO 80000)
    real(dp), parameter :: c_luz = 299792458.0_dp       ! m/s

end module constantes

! src/vetor3d.f90 — tipo derivado com métodos
module vetor3d_mod
    use constantes, only: dp
    implicit none
    private
    public :: Vetor3D, op_norm, op_cross

    type :: Vetor3D
        real(dp) :: x = 0.0_dp
        real(dp) :: y = 0.0_dp
        real(dp) :: z = 0.0_dp
    contains
        procedure :: norma
        procedure :: normalizar
        generic   :: operator(+)  => somar
        generic   :: operator(*)  => escalar
    private
        procedure :: somar
        procedure :: escalar
    end type Vetor3D

contains

    pure real(dp) function norma(self) result(n)
        class(Vetor3D), intent(in) :: self
        n = sqrt(self%x**2 + self%y**2 + self%z**2)
    end function norma

    pure function normalizar(self) result(v)
        class(Vetor3D), intent(in) :: self
        type(Vetor3D) :: v
        real(dp) :: n
        n = self%norma()
        if (n > tiny(n)) then
            v = Vetor3D(self%x/n, self%y/n, self%z/n)
        else
            v = Vetor3D(0.0_dp, 0.0_dp, 0.0_dp)
        end if
    end function normalizar

    pure function somar(a, b) result(c)
        class(Vetor3D), intent(in) :: a
        type(Vetor3D),  intent(in) :: b
        type(Vetor3D) :: c
        c = Vetor3D(a%x + b%x, a%y + b%y, a%z + b%z)
    end function somar

    pure function escalar(v, k) result(c)
        class(Vetor3D), intent(in) :: v
        real(dp),       intent(in) :: k
        type(Vetor3D) :: c
        c = Vetor3D(v%x * k, v%y * k, v%z * k)
    end function escalar

    pure function op_cross(a, b) result(c)
        type(Vetor3D), intent(in) :: a, b
        type(Vetor3D) :: c
        c%x = a%y * b%z - a%z * b%y
        c%y = a%z * b%x - a%x * b%z
        c%z = a%x * b%y - a%y * b%x
    end function op_cross

    pure real(dp) function op_norm(a, b) result(d)
        type(Vetor3D), intent(in) :: a, b
        type(Vetor3D) :: diff
        diff = Vetor3D(a%x - b%x, a%y - b%y, a%z - b%z)
        d = diff%norma()
    end function op_norm

end module vetor3d_mod

Testes Unitários com test-drive

test-drive framework — testes para vetor3d_mod
! test/test_vetor3d.f90
module test_vetor3d
    use testdrive, only: new_unittest, unittest_type, error_type, check
    use vetor3d_mod
    use constantes, only: dp
    implicit none
    private
    public :: collect_tests

contains

    subroutine collect_tests(testsuite)
        type(unittest_type), allocatable, intent(out) :: testsuite(:)
        testsuite = [                                        &
            new_unittest("norma_zero",     test_norma_zero), &
            new_unittest("norma_unitaria", test_norma_unit), &
            new_unittest("soma_vetores",   test_soma)        &
        ]
    end subroutine collect_tests

    subroutine test_norma_zero(error)
        type(error_type), allocatable, intent(out) :: error
        type(Vetor3D) :: v
        v = Vetor3D(0.0_dp, 0.0_dp, 0.0_dp)
        call check(error, abs(v%norma()) < 1.0e-14_dp, "norma zero falhou")
    end subroutine test_norma_zero

    subroutine test_norma_unit(error)
        type(error_type), allocatable, intent(out) :: error
        type(Vetor3D) :: v, vn
        v  = Vetor3D(3.0_dp, 4.0_dp, 0.0_dp)
        vn = v%normalizar()
        call check(error, abs(vn%norma() - 1.0_dp) < 1.0e-14_dp, &
                   "normalização falhou")
    end subroutine test_norma_unit

    subroutine test_soma(error)
        type(error_type), allocatable, intent(out) :: error
        type(Vetor3D) :: a, b, c
        a = Vetor3D(1.0_dp, 0.0_dp, 0.0_dp)
        b = Vetor3D(0.0_dp, 1.0_dp, 0.0_dp)
        c = a + b
        call check(error, abs(c%x - 1.0_dp) < 1.0e-14_dp, "soma x falhou")
        call check(error, abs(c%y - 1.0_dp) < 1.0e-14_dp, "soma y falhou")
    end subroutine test_soma

end module test_vetor3d

Otimização e Profiling

Column-major, alocação contígua e profiling com gprof
! ERRADO — loop externo percorre linhas (não contíguo em Fortran)
subroutine loop_ruim(A, N)
    integer, intent(in) :: N
    real, intent(inout) :: A(N,N)
    integer :: i, j
    do i = 1, N         ! i = linha
        do j = 1, N     ! j = coluna — A(i,j), A(i,j+1)... salta N*4 bytes
            A(i,j) = A(i,j) * 2.0
        end do
    end do
end subroutine loop_ruim

! CORRETO — loop interno percorre coluna (contíguo)
subroutine loop_certo(A, N)
    integer, intent(in) :: N
    real, intent(inout) :: A(N,N)
    integer :: i, j
    do j = 1, N         ! j = coluna
        do i = 1, N     ! i = linha — A(i,j), A(i+1,j)... sequencial
            A(i,j) = A(i,j) * 2.0
        end do
    end do
end subroutine loop_certo

! Alocação contígua para arrays de structs — usar arrays de components

A regra de ouro do Fortran de alto desempenho: meça primeiro, otimize depois. Na maioria dos códigos científicos, 90% do tempo é gasto em 10% do código. Use gprof ou perf para encontrar esses gargalos antes de qualquer otimização manual — e nunca otimize a ordem de loops sem medir o impact real de cache com perf stat --cache-misses.
