Fortran — Avançado

O Fortran moderno oferece OOP com tipos derivados extensíveis, ponteiros de procedimento para funções de ordem superior, DO CONCURRENT para paralelismo automático sem dependências, coarrays SPMD nativos, e interoperabilidade total com C via iso_c_binding — tornando-o uma linguagem completa para HPC moderno.

DO CONCURRENT (Fortran 2008) é uma declaração para o compilador: as iterações não têm dependências e podem rodar em qualquer ordem ou em paralelo. O compilador pode vetorizar com SIMD, paralelizar com threads ou até compilar para GPU — sem que o programador precise especificar como.

OOP — Tipos Derivados e Herança

Tipos derivados com methods e herança (Fortran 2003)
module formas_mod
    implicit none
    private
    public :: Forma, Circulo, Retangulo

    integer, parameter :: dp = selected_real_kind(15, 307)
    real(dp), parameter :: PI = 3.14159265358979323846_dp

    ! Tipo base abstrato
    type, abstract :: Forma
        real(dp) :: cor = 0.0_dp
    contains
        procedure(area_iface), deferred :: area    ! método abstrato
        procedure :: descrever
    end type Forma

    ! Interface para método abstrato
    abstract interface
        real(dp) function area_iface(self)
            import :: Forma, dp
            class(Forma), intent(in) :: self
        end function
    end interface

    ! Tipo derivado concreto — herda de Forma
    type, extends(Forma) :: Circulo
        real(dp) :: raio
    contains
        procedure :: area => area_circulo
    end type Circulo

    type, extends(Forma) :: Retangulo
        real(dp) :: largura, altura
    contains
        procedure :: area => area_retangulo
    end type Retangulo

contains

    real(dp) function area_circulo(self) result(a)
        class(Circulo), intent(in) :: self
        a = PI * self%raio ** 2
    end function area_circulo

    real(dp) function area_retangulo(self) result(a)
        class(Retangulo), intent(in) :: self
        a = self%largura * self%altura
    end function area_retangulo

    subroutine descrever(self)
        class(Forma), intent(in) :: self
        print '(A, F12.4)', "Área = ", self%area()
    end subroutine descrever

end module formas_mod

program oop_demo
    use formas_mod
    implicit none
    type(Circulo)    :: c
    type(Retangulo)  :: r
    class(Forma), allocatable :: f   ! polimorfismo

    c = Circulo(cor=1.0_8, raio=5.0_8)
    r = Retangulo(cor=2.0_8, largura=3.0_8, altura=4.0_8)

    call c%descrever()    ! Área = 78.5398
    call r%descrever()    ! Área = 12.0000

    ! Polimorfismo dinâmico
    allocate(Circulo :: f)
    select type(f)
        type is (Circulo)
            f%raio = 3.0_8
    end select
    call f%descrever()

end program oop_demo

DO CONCURRENT e Paralelismo Implícito

do concurrent com locality specifiers
program do_concurrent_ex
    implicit none
    integer, parameter :: N = 1000000
    real :: a(N), b(N), c(N)
    real :: local_var
    integer :: i

    a = 1.0; b = 2.0

    ! DO CONCURRENT — iterações independentes, compilador vetoriza/paraleliza
    do concurrent (i = 1:N)
        c(i) = sqrt(a(i) ** 2 + b(i) ** 2)
    end do

    ! Com locality specifiers (Fortran 2018)
    do concurrent (i = 1:N) local(local_var) shared(a, b, c)
        local_var = a(i) + b(i)
        c(i) = local_var * local_var
    end do

    ! Múltiplas dimensões
    real :: mat(N, N)
    integer :: j
    do concurrent (i = 1:N, j = 1:N)
        mat(i, j) = real(i) + real(j)
    end do

    print *, sum(c)

end program do_concurrent_ex

OpenMP — Paralelismo com Threads

Diretivas OpenMP para loops e seções paralelas
program openmp_ex
    use omp_lib
    implicit none
    integer, parameter :: N = 10000000
    real :: a(N), b(N), c(N)
    integer :: i
    real :: soma
    double precision :: t_ini, t_fim

    a = 1.0; b = 2.0

    t_ini = omp_get_wtime()

    !$omp parallel do schedule(static) shared(a, b, c)
    do i = 1, N
        c(i) = a(i) * b(i) + sqrt(a(i))
    end do
    !$omp end parallel do

    t_fim = omp_get_wtime()
    print '(A, F8.4, A)', "Tempo paralelo: ", t_fim - t_ini, " s"

    ! Redução paralela
    soma = 0.0
    !$omp parallel do reduction(+:soma)
    do i = 1, N
        soma = soma + c(i)
    end do
    !$omp end parallel do
    print *, "Soma =", soma

    ! Informações OpenMP
    print *, "Threads disponíveis:", omp_get_max_threads()

    ! Compilar: gfortran -O3 -fopenmp openmp_ex.f90 -o openmp_ex
    ! Executar: OMP_NUM_THREADS=8 ./openmp_ex

end program openmp_ex

Interoperabilidade com C (iso_c_binding)

Chamar funções C de Fortran e vice-versa
! Módulo de interface para funções C
module c_interface
    use iso_c_binding
    implicit none

    ! Interface para função C: int strlen(const char *s)
    interface
        function c_strlen(s) bind(C, name="strlen") result(n)
            use iso_c_binding
            type(c_ptr),    value :: s
            integer(c_size_t)    :: n
        end function c_strlen

        ! void *malloc(size_t size)
        function c_malloc(size) bind(C, name="malloc") result(ptr)
            use iso_c_binding
            integer(c_size_t), value :: size
            type(c_ptr)              :: ptr
        end function c_malloc

        ! void free(void *ptr)
        subroutine c_free(ptr) bind(C, name="free")
            use iso_c_binding
            type(c_ptr), value :: ptr
        end subroutine c_free
    end interface

end module c_interface

! Exportar subrotina Fortran para ser chamada em C
subroutine calcular_norma(v, n, resultado) bind(C, name="calcular_norma")
    use iso_c_binding
    implicit none
    integer(c_int), value    :: n
    real(c_double), intent(in)  :: v(n)
    real(c_double), intent(out) :: resultado
    integer :: i

    resultado = 0.0_c_double
    do i = 1, n
        resultado = resultado + v(i) ** 2
    end do
    resultado = sqrt(resultado)
end subroutine calcular_norma
