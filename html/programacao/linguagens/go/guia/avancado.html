Go — Avançado

O nível avançado de Go cobre o modelo de concorrência completo (goroutines, channels bufferizados, select e sync), o pacote context para cancelamento e prazos, generics com constraints (Go 1.18+), reflexão com reflect, profiling com pprof e as garantias do memory model de Go.

O mantra de Go para concorrência é "Do not communicate by sharing memory; share memory by communicating." Em vez de proteger dados compartilhados com mutexes o tempo todo, canais transformam a comunicação entre goroutines na única forma de acesso — o dado pertence a uma goroutine de cada vez, passado pelo canal. Simples de raciocinar, difícil de errar.

Goroutines e Channels

Goroutines, channels buferizados e WaitGroup
package main

import (
    "fmt"
    "sync"
    "time"
)

// Padrão fan-out: distribuir trabalho entre N goroutines
func worker(id int, jobs <-chan int, results chan<- int, wg *sync.WaitGroup) {
    defer wg.Done()
    for j := range jobs {
        // simula trabalho
        time.Sleep(10 * time.Millisecond)
        results <- j * j
    }
}

func main() {
    // Channel buferizado — não bloqueia até encher
    jobs    := make(chan int, 10)
    results := make(chan int, 10)
    var wg sync.WaitGroup

    // Lançar 3 workers
    for w := 1; w <= 3; w++ {
        wg.Add(1)
        go worker(w, jobs, results, &wg)
    }

    // Enviar 9 tarefas
    for j := 1; j <= 9; j++ {
        jobs <- j
    }
    close(jobs)   // fecha: workers param de receber

    // Esperar workers e fechar results
    go func() {
        wg.Wait()
        close(results)
    }()

    // Coletar resultados
    for r := range results {
        fmt.Println(r)
    }

    // select — multiplexar channels (timeout, cancelamento)
    ch1 := make(chan string, 1)
    ch2 := make(chan string, 1)
    go func() { time.Sleep(50*time.Millisecond); ch1 <- "um" }()
    go func() { time.Sleep(30*time.Millisecond); ch2 <- "dois" }()

    for i := 0; i < 2; i++ {
        select {
        case msg := <-ch1:
            fmt.Println("ch1:", msg)
        case msg := <-ch2:
            fmt.Println("ch2:", msg)
        case <-time.After(100 * time.Millisecond):
            fmt.Println("timeout")
        }
    }
}

Context — Cancelamento e Prazos

context.WithCancel, WithTimeout e propagação
package main

import (
    "context"
    "fmt"
    "time"
)

// Função que respeita cancelamento
func buscarDados(ctx context.Context, url string) (string, error) {
    // simula trabalho que pode ser cancelado
    ch := make(chan string, 1)
    go func() {
        time.Sleep(200 * time.Millisecond)
        ch <- "dados de " + url
    }()

    select {
    case resultado := <-ch:
        return resultado, nil
    case <-ctx.Done():
        return "", ctx.Err()   // context.DeadlineExceeded ou Canceled
    }
}

func main() {
    // WithTimeout — cancela após 150ms
    ctx, cancel := context.WithTimeout(context.Background(), 150*time.Millisecond)
    defer cancel()   // sempre chamar cancel para liberar recursos

    dados, err := buscarDados(ctx, "https://api.exemplo.com")
    if err != nil {
        fmt.Println("Erro:", err)   // context deadline exceeded
    } else {
        fmt.Println(dados)
    }

    // WithValue — propagar valores (request ID, user info)
    type ctxKey string
    const reqIDKey ctxKey = "requestID"

    ctx2 := context.WithValue(context.Background(), reqIDKey, "abc-123")
    reqID := ctx2.Value(reqIDKey).(string)
    fmt.Println("Request ID:", reqID)

    // WithCancel — cancelamento manual
    ctx3, cancel3 := context.WithCancel(context.Background())
    go func() {
        time.Sleep(50 * time.Millisecond)
        cancel3()   // cancela após 50ms
    }()

    _, err3 := buscarDados(ctx3, "https://lento.com")
    fmt.Println("ctx3 err:", err3)   // context canceled
}

sync — Mutex, RWMutex e atomic

Proteger estado compartilhado quando channels não bastam
package main

import (
    "fmt"
    "sync"
    "sync/atomic"
)

// Cache thread-safe com RWMutex
type Cache struct {
    mu   sync.RWMutex
    data map[string]string
}

func (c *Cache) Get(key string) (string, bool) {
    c.mu.RLock()           // múltiplos leitores simultâneos
    defer c.mu.RUnlock()
    v, ok := c.data[key]
    return v, ok
}

func (c *Cache) Set(key, value string) {
    c.mu.Lock()            // exclusão mútua para escrita
    defer c.mu.Unlock()
    c.data[key] = value
}

// sync.Once — executar uma vez (singleton, inicialização lazy)
var (
    instancia *Cache
    once      sync.Once
)

func getCache() *Cache {
    once.Do(func() {
        instancia = &Cache{data: make(map[string]string)}
    })
    return instancia
}

// atomic — operações atômicas sem mutex (contadores)
var contador atomic.Int64

func main() {
    c := getCache()
    c.Set("nome", "Go")
    v, _ := c.Get("nome")
    fmt.Println(v)

    // sync.Map — map thread-safe para reads frequentes
    var sm sync.Map
    sm.Store("chave", 42)
    val, _ := sm.Load("chave")
    fmt.Println(val.(int))

    // atomic counter
    var wg sync.WaitGroup
    for i := 0; i < 1000; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            contador.Add(1)
        }()
    }
    wg.Wait()
    fmt.Println("Contador:", contador.Load())   // sempre 1000
}

Generics (Go 1.18+)

Type parameters, constraints e funções genéricas
package main

import (
    "cmp"
    "fmt"
)

// Constraint: qualquer tipo numérico
type Numero interface {
    ~int | ~int8 | ~int16 | ~int32 | ~int64 |
    ~float32 | ~float64
}

// Função genérica
func Soma[T Numero](nums []T) T {
    var total T
    for _, n := range nums {
        total += n
    }
    return total
}

// Map e Filter genéricos
func MapFn[T, U any](slice []T, f func(T) U) []U {
    resultado := make([]U, len(slice))
    for i, v := range slice {
        resultado[i] = f(v)
    }
    return resultado
}

func Filter[T any](slice []T, pred func(T) bool) []T {
    var res []T
    for _, v := range slice {
        if pred(v) {
            res = append(res, v)
        }
    }
    return res
}

// Struct genérica
type Par[A, B any] struct {
    Primeiro A
    Segundo  B
}

// cmp.Ordered — qualquer tipo ordenável
func Min[T cmp.Ordered](a, b T) T {
    if a < b { return a }
    return b
}

func main() {
    fmt.Println(Soma([]int{1, 2, 3, 4, 5}))       // 15
    fmt.Println(Soma([]float64{1.1, 2.2, 3.3}))   // 6.6

    dobrados := MapFn([]int{1,2,3}, func(n int) int { return n * 2 })
    fmt.Println(dobrados)   // [2 4 6]

    pares := Filter([]int{1,2,3,4,5,6}, func(n int) bool { return n%2==0 })
    fmt.Println(pares)      // [2 4 6]

    p := Par[string, int]{Primeiro: "um", Segundo: 1}
    fmt.Println(p)

    fmt.Println(Min(3, 7))        // 3
    fmt.Println(Min("a", "b"))   // a
}

import (

    "runtime/pprof"

// CPU profiling em qualquer programa

Go usa goroutines multiplexadas em threads do OS pelo runtime (scheduler M:N). Um servidor Go pode manter dezenas de milhares de goroutines simultâneas com apenas alguns MBs de stack — cada goroutine começa com 2KB e cresce conforme necessário. Um thread Java ou Python consome 1MB de stack e requer um thread OS. Por isso APIs Go frequentemente lidam com 100.000+ conexões simultâneas em hardware modesto.
