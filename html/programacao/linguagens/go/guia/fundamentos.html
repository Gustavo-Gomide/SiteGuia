Go — Fundamentos

Os fundamentos do Go cobrem o sistema de tipos (tipos primitivos, structs, interfaces e ponteiros), slices e maps como estruturas de dados centrais, funções com múltiplos retornos, controle de fluxo e o padrão idiomático de tratamento de erros com (value, error).

Em Go, tudo o que você declara deve ser usado — uma variável não usada é erro de compilação. Da mesma forma, imports não usados causam erro. Essa regra força o código a ser limpo por construção: sem variáveis fantasma, sem imports acumulados. É frustrante no início e libertador depois.

Tipos e Variáveis

Tipos primitivos, declaração e inferência
package main

import "fmt"

func main() {
    // Declaração explícita
    var nome string = "Go"
    var versao int = 22
    var preco float64 = 3.14
    var ativo bool = true

    // Inferência de tipo — := (short variable declaration)
    linguagem := "Go"       // string
    ano       := 2009        // int
    pi        := 3.14159     // float64

    // Múltipla atribuição
    a, b := 10, 20
    a, b = b, a              // swap idiomático

    // Zero values — tipos têm valor padrão (não há null/undefined)
    var n int      // 0
    var s string   // ""
    var f float64  // 0.0
    var b2 bool    // false

    // Constantes
    const MaxConn = 100
    const Pi = 3.14159265358979

    // iota — contadores em const block
    const (
        Domingo = iota  // 0
        Segunda         // 1
        Terca           // 2
        Quarta          // 3
    )

    fmt.Println(nome, versao, linguagem, ano, pi, a, b, n, s, f, b2, ativo, Pi, MaxConn, Quarta)
}

Funções

Funções, múltiplos retornos, variádicas e closures
package main

import "fmt"

// Função com múltiplos retornos (idioma central de Go)
func dividir(a, b float64) (float64, error) {
    if b == 0 {
        return 0, fmt.Errorf("divisão por zero")
    }
    return a / b, nil
}

// Named return values
func minMax(nums []int) (min, max int) {
    min, max = nums[0], nums[0]
    for _, v := range nums {
        if v < min { min = v }
        if v > max { max = v }
    }
    return   // naked return — retorna min e max
}

// Função variádica
func soma(nums ...int) int {
    total := 0
    for _, n := range nums {
        total += n
    }
    return total
}

// Funções como valores (first-class)
func aplicar(nums []int, f func(int) int) []int {
    resultado := make([]int, len(nums))
    for i, v := range nums {
        resultado[i] = f(v)
    }
    return resultado
}

// Closure
func contador() func() int {
    n := 0
    return func() int {
        n++
        return n
    }
}

func main() {
    res, err := dividir(10, 3)
    if err != nil {
        fmt.Println("Erro:", err)
    } else {
        fmt.Printf("%.4f\n", res)
    }

    fmt.Println(minMax([]int{5, 2, 8, 1, 9}))   // 1 9
    fmt.Println(soma(1, 2, 3, 4, 5))             // 15

    nums := []int{1, 2, 3, 4, 5}
    dobrados := aplicar(nums, func(n int) int { return n * 2 })
    fmt.Println(dobrados)    // [2 4 6 8 10]

    c := contador()
    fmt.Println(c(), c(), c())  // 1 2 3
}

Structs e Interfaces

Definir structs, métodos e interfaces implícitas
package main

import (
    "fmt"
    "math"
)

// Interface — satisfeita implicitamente
type Forma interface {
    Area() float64
    Perimetro() float64
}

// Struct com campos
type Circulo struct {
    Raio float64
}

// Métodos sobre struct — pointer receiver para mutação
func (c Circulo) Area() float64      { return math.Pi * c.Raio * c.Raio }
func (c Circulo) Perimetro() float64  { return 2 * math.Pi * c.Raio }

type Retangulo struct {
    Largura, Altura float64
}

func (r Retangulo) Area() float64      { return r.Largura * r.Altura }
func (r Retangulo) Perimetro() float64 { return 2 * (r.Largura + r.Altura) }

// Função que aceita qualquer Forma
func descrever(f Forma) {
    fmt.Printf("Área: %.2f, Perímetro: %.2f\n", f.Area(), f.Perimetro())
}

// Struct embedding (composição, não herança)
type Animal struct {
    Nome string
}

func (a Animal) Apresentar() string {
    return "Eu sou " + a.Nome
}

type Cachorro struct {
    Animal          // embedding — herda métodos de Animal
    Raca string
}

func main() {
    c := Circulo{Raio: 5}
    r := Retangulo{Largura: 4, Altura: 6}

    descrever(c)    // Circulo satisfaz Forma implicitamente
    descrever(r)

    // Slice de interfaces
    formas := []Forma{c, r}
    for _, f := range formas {
        fmt.Printf("%.2f\n", f.Area())
    }

    // Struct embedding
    d := Cachorro{Animal: Animal{Nome: "Rex"}, Raca: "Labrador"}
    fmt.Println(d.Apresentar())  // d.Animal.Apresentar() promovido
    fmt.Println(d.Nome)          // campo promovido
}

Slices e Maps

Arrays, slices crescentes e maps hash
package main

import "fmt"

func main() {
    // Slice — fatia de array dinâmica
    nums := []int{1, 2, 3, 4, 5}
    nums = append(nums, 6, 7)               // cresce
    fmt.Println(nums[2:5])                  // [3 4 5] — fatiamento [low:high]

    // make — aloca slice com len e cap
    s := make([]int, 3, 10)    // len=3, cap=10
    fmt.Println(len(s), cap(s))

    // copy
    src := []int{1, 2, 3}
    dst := make([]int, len(src))
    copy(dst, src)

    // Iterar com range
    for i, v := range nums {
        fmt.Printf("[%d]=%d ", i, v)
    }
    fmt.Println()

    // Slice de strings
    linguagens := []string{"Go", "Rust", "Python"}
    linguagens = append(linguagens, "Kotlin")

    // 2D slice
    matriz := [][]int{
        {1, 2, 3},
        {4, 5, 6},
        {7, 8, 9},
    }
    fmt.Println(matriz[1][2])  // 6

    // Map
    idades := map[string]int{
        "Alice": 30,
        "Bob":   25,
    }
    idades["Carol"] = 28

    // Lookup com ok-idiom
    idade, ok := idades["Alice"]
    if ok {
        fmt.Printf("Alice tem %d anos\n", idade)
    }

    // Deletar
    delete(idades, "Bob")

    // Iterar mapa (ordem não determinística!)
    for nome, idade := range idades {
        fmt.Printf("%s: %d\n", nome, idade)
    }

    // make map
    contagem := make(map[string]int)
    palavras := []string{"go", "e", "go", "c", "go"}
    for _, p := range palavras {
        contagem[p]++
    }
    fmt.Println(contagem)  // map[c:1 e:1 go:3]
}

Controle de Fluxo

if, for, switch, defer e ponteiros
package main

import "fmt"

func main() {
    // if com statement de inicialização
    if n := 42; n%2 == 0 {
        fmt.Println("par")
    } else {
        fmt.Println("ímpar")
    }

    // for — único laço em Go (3 formas)
    for i := 0; i < 5; i++ {        // C-style
        fmt.Print(i, " ")
    }
    fmt.Println()

    for i := 0; i < 10; i += 2 {   // while-style (só condição)
        fmt.Print(i, " ")
    }
    fmt.Println()

    // switch — sem break explícito (não há fallthrough por padrão)
    dia := 3
    switch dia {
    case 1, 7:
        fmt.Println("fim de semana")
    case 2, 3, 4, 5, 6:
        fmt.Println("dia útil")
    default:
        fmt.Println("inválido")
    }

    // switch sem expressão — substitui if/else if
    x := 75
    switch {
    case x >= 90: fmt.Println("A")
    case x >= 70: fmt.Println("B")
    default:      fmt.Println("C")
    }

    // defer — executa ao retornar da função (LIFO)
    defer fmt.Println("3: limpeza final")
    defer fmt.Println("2: liberando recursos")
    fmt.Println("1: trabalhando...")
    // Output: 1, 2, 3

    // Ponteiros
    v := 42
    p := &v          // endereço de v
    *p = 100         // desreferenciar e modificar
    fmt.Println(v)   // 100

    // Go não tem aritmética de ponteiros — evita bugs
    // new() aloca zero value
    q := new(int)
    *q = 99
    fmt.Println(*q)  // 99
}

Tratamento de Erros

Padrão de erros: (value, error), errors.Is/As, wrapping
package main

import (
    "errors"
    "fmt"
)

// Definir tipo de erro customizado
type ErrValidacao struct {
    Campo   string
    Mensagem string
}

func (e *ErrValidacao) Error() string {
    return fmt.Sprintf("validação: campo '%s' — %s", e.Campo, e.Mensagem)
}

// Sentinel errors
var ErrNaoEncontrado = errors.New("não encontrado")

func buscar(id int) (string, error) {
    db := map[int]string{1: "Alice", 2: "Bob"}
    v, ok := db[id]
    if !ok {
        return "", fmt.Errorf("buscar(id=%d): %w", id, ErrNaoEncontrado)
    }
    return v, nil
}

func validar(nome string) error {
    if len(nome) < 3 {
        return &ErrValidacao{Campo: "nome", Mensagem: "mínimo 3 caracteres"}
    }
    return nil
}

func main() {
    // Verificação padrão
    nome, err := buscar(99)
    if err != nil {
        fmt.Println(err)
    } else {
        fmt.Println(nome)
    }

    // errors.Is — verifica sentinel (atravessa Unwrap)
    _, err = buscar(99)
    if errors.Is(err, ErrNaoEncontrado) {
        fmt.Println("recurso não existe")
    }

    // errors.As — extrai tipo específico
    err2 := validar("Jo")
    var errVal *ErrValidacao
    if errors.As(err2, &errVal) {
        fmt.Printf("Campo com erro: %s\n", errVal.Campo)
    }

    // panic e recover (use raramente)
    defer func() {
        if r := recover(); r != nil {
            fmt.Println("Recuperado de panic:", r)
        }
    }()
    // panic("algo inesperado")  // descomente para testar
}

O padrão (value, error) em Go é explícito por design. Em Java, uma exceção pode emergir de qualquer ponto da pilha invisível para o chamador. Em Go, toda função que pode falhar anuncia isso na assinatura — o compilador força você a lidar com o erro ou descartá-lo (_) conscientemente. Não é belo, mas é auditável: você pode ler qualquer código Go e entender exatamente o que pode falhar.
