Go — Bibliotecas e Frameworks

O ecossistema Go oferece bibliotecas de alta qualidade para qualquer domínio: web (Gin, Echo, Chi, net/http), banco de dados (GORM, sqlx, pgx), gRPC, mensageria (kafka-go, NATS), CLI (Cobra, urfave/cli), logging (zerolog, zap, slog), observabilidade (Prometheus, OpenTelemetry) e testabilidade (testify).

A biblioteca padrão net/http de Go é production-ready por si só. O servidor HTTP integrado usa goroutines automáticas por conexão, suporta TLS nativo, HTTP/2 e timeouts configuráveis. Frameworks como Gin e Echo adicionam roteamento com parâmetros, middleware composable e binding de JSON — mas para muitos projetos, net/http puro é suficiente.

Gin — Web Framework

API REST completa com Gin: rotas, middleware, JSON, validação
// go get github.com/gin-gonic/gin
package main

import (
    "net/http"
    "github.com/gin-gonic/gin"
)

type Produto struct {
    ID    int    `json:"id"`
    Nome  string `json:"nome"  binding:"required,min=2"`
    Preco float64`json:"preco" binding:"required,gt=0"`
}

var produtos = []Produto{
    {1, "Teclado", 250.0},
    {2, "Mouse",   120.0},
}

func main() {
    r := gin.Default()   // Logger + Recovery middleware

    // Middleware customizado
    r.Use(func(c *gin.Context) {
        c.Header("X-App", "MeuServico")
        c.Next()
    })

    // Grupo de rotas com prefixo
    api := r.Group("/api/v1")
    {
        api.GET("/produtos", func(c *gin.Context) {
            c.JSON(http.StatusOK, gin.H{"dados": produtos})
        })

        api.GET("/produtos/:id", func(c *gin.Context) {
            id := c.Param("id")
            for _, p := range produtos {
                if fmt.Sprint(p.ID) == id {
                    c.JSON(http.StatusOK, p)
                    return
                }
            }
            c.JSON(http.StatusNotFound, gin.H{"erro": "não encontrado"})
        })

        api.POST("/produtos", func(c *gin.Context) {
            var novo Produto
            if err := c.ShouldBindJSON(&novo); err != nil {
                c.JSON(http.StatusBadRequest, gin.H{"erro": err.Error()})
                return
            }
            novo.ID = len(produtos) + 1
            produtos = append(produtos, novo)
            c.JSON(http.StatusCreated, novo)
        })

        api.DELETE("/produtos/:id", func(c *gin.Context) {
            c.Status(http.StatusNoContent)
        })
    }

    r.Run(":8080")
}

gRPC — RPC de Alta Performance

Definir proto, gerar código Go, implementar servidor
// produtos.proto
syntax = "proto3";
option go_package = "github.com/usuario/app/pb";

service ProdutoService {
    rpc BuscarProduto (BuscarRequest) returns (Produto);
    rpc ListarProdutos(ListarRequest) returns (stream Produto);
}

message Produto { int32 id=1; string nome=2; float preco=3; }
message BuscarRequest { int32 id=1; }
message ListarRequest { }

// Gerar: protoc --go_out=. --go-grpc_out=. produtos.proto

// Implementação do servidor
type servidor struct {
    pb.UnimplementedProdutoServiceServer
}

func (s *servidor) BuscarProduto(ctx context.Context, req *pb.BuscarRequest) (*pb.Produto, error) {
    for _, p := range produtosDB {
        if p.Id == req.Id {
            return p, nil
        }
    }
    return nil, status.Errorf(codes.NotFound, "produto %d não encontrado", req.Id)
}

func (s *servidor) ListarProdutos(req *pb.ListarRequest, stream pb.ProdutoService_ListarProdutosServer) error {
    for _, p := range produtosDB {
        if err := stream.Send(p); err != nil {
            return err
        }
    }
    return nil
}

// Iniciar servidor gRPC
lis, _ := net.Listen("tcp", ":50051")
s := grpc.NewServer()
pb.RegisterProdutoServiceServer(s, &servidor{})
s.Serve(lis)

Banco de Dados — sqlx e GORM

sqlx para SQL idiomático e GORM para ORM
// sqlx — thin wrapper sobre database/sql
// go get github.com/jmoiron/sqlx
// go get github.com/lib/pq

import (
    "github.com/jmoiron/sqlx"
    _ "github.com/lib/pq"
)

db, _ := sqlx.Connect("postgres",
    "host=localhost port=5432 user=user password=pass dbname=app sslmode=disable")
defer db.Close()

type Usuario struct {
    ID    int    `db:"id"`
    Nome  string `db:"nome"`
    Email string `db:"email"`
}

// Select lista
var usuarios []Usuario
db.Select(&usuarios, "SELECT * FROM usuarios WHERE ativo = $1", true)

// Get um registro
var u Usuario
db.Get(&u, "SELECT * FROM usuarios WHERE id = $1", 1)

// Named queries
db.NamedExec(`INSERT INTO usuarios (nome, email) VALUES (:nome, :email)`, u)

// Transação
tx := db.MustBegin()
tx.NamedExec(`UPDATE usuarios SET nome=:nome WHERE id=:id`, u)
tx.Commit()

// ---- GORM ----
// go get gorm.io/gorm
// go get gorm.io/driver/postgres

import (
    "gorm.io/gorm"
    "gorm.io/driver/postgres"
)

db2, _ := gorm.Open(postgres.Open(dsn), &gorm.Config{})
db2.AutoMigrate(&Usuario{})

// CRUD com GORM
db2.Create(&Usuario{Nome: "Alice", Email: "alice@ex.com"})
db2.First(&u, 1)
db2.Model(&u).Update("Nome", "Alice Silva")
db2.Delete(&u)

// Queries complexas
db2.Where("nome LIKE ?", "%Alice%").Find(&usuarios)
db2.Joins("JOIN pedidos ON pedidos.usuario_id = usuarios.id").Find(&usuarios)

Cobra — CLIs Profissionais

Subcomandos, flags e argumentos com Cobra
// go get github.com/spf13/cobra
// Estrutura usada por kubectl, Hugo, Docker CLI
package main

import (
    "fmt"
    "os"
    "github.com/spf13/cobra"
)

var verbose bool
var output  string

var rootCmd = &cobra.Command{
    Use:   "meutool",
    Short: "Ferramenta de exemplo",
}

var processar = &cobra.Command{
    Use:   "processar [arquivo]",
    Short: "Processa um arquivo",
    Args:  cobra.ExactArgs(1),
    RunE: func(cmd *cobra.Command, args []string) error {
        arquivo := args[0]
        if verbose {
            fmt.Printf("Processando: %s → %s\n", arquivo, output)
        }
        // lógica de processamento...
        return nil
    },
}

func main() {
    rootCmd.PersistentFlags().BoolVarP(&verbose, "verbose", "v", false, "modo verboso")
    processar.Flags().StringVarP(&output, "output", "o", "output.txt", "arquivo de saída")
    rootCmd.AddCommand(processar)
    if err := rootCmd.Execute(); err != nil {
        os.Exit(1)
    }
}

Observabilidade — Prometheus, Zap e OpenTelemetry

Métricas, logging estruturado e tracing
// Prometheus — métricas
// go get github.com/prometheus/client_golang/prometheus

httpReqs := prometheus.NewCounterVec(
    prometheus.CounterOpts{Name: "http_requests_total", Help: "Total HTTP"},
    []string{"method", "status"},
)
prometheus.MustRegister(httpReqs)

latencia := prometheus.NewHistogramVec(
    prometheus.HistogramOpts{
        Name:    "http_latency_seconds",
        Buckets: prometheus.DefBuckets,
    }, []string{"path"},
)
prometheus.MustRegister(latencia)

// Incrementar no handler
httpReqs.With(prometheus.Labels{"method": "GET", "status": "200"}).Inc()

// Expor /metrics
http.Handle("/metrics", promhttp.Handler())

// Zap — logging estruturado de alto desempenho
// go get go.uber.org/zap
logger, _ := zap.NewProduction()
defer logger.Sync()

logger.Info("servidor iniciado",
    zap.String("addr", ":8080"),
    zap.Int("pid", os.Getpid()),
)
logger.Error("erro ao conectar",
    zap.String("host", "db:5432"),
    zap.Error(err),
)

Cobra é a base da maioria das CLIs profissionais em Go — kubectl, Hugo, GitHub CLI, Docker CLI, Helm e Terraform usam Cobra ou padrão similar. A razão é a hierarquia de subcomandos: kubectl get pods, kubectl apply -f, kubectl logs --follow são todos subcomandos com seus próprios flags, argumentos e documentação, sem boilerplate manual. Cobra gera help consistente, completação de shell (bash/zsh/fish) e documentação markdown automaticamente.
