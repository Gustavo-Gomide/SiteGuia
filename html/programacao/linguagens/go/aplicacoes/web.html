Go — Web

Go — Desenvolvimento Web

Go é uma das linguagens mais usadas para backend web de alto desempenho. A biblioteca padrão net/http inclui servidor HTTP/2 completo, e frameworks como Gin e Echo adicionam roteamento avançado, middleware e binding JSON. O modelo goroutine/handler trata cada requisição concorrentemente por padrão.

O benchmark TechEmpower Framework Benchmarks consistentemente coloca servidores Go nas primeiras posições entre linguagens com GC. Um servidor Go típico atinge 100.000–500.000 requisições por segundo em hardware modesto — sem ajustes, sem tunning de thread pool, sem configuração de reactive pipeline. A concorrência é automática.

net/http — Servidor Padrão

Servidor HTTP completo sem frameworks externos
package main

import (
    "encoding/json"
    "fmt"
    "log/slog"
    "net/http"
    "time"
)

// Middleware: logging de requisições
func logMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        inicio := time.Now()
        next.ServeHTTP(w, r)
        slog.Info("request", "method", r.Method, "path", r.URL.Path,
            "dur", time.Since(inicio))
    })
}

// Handler como função
func healthHandler(w http.ResponseWriter, r *http.Request) {
    w.Header().Set("Content-Type", "application/json")
    json.NewEncoder(w).Encode(map[string]string{"status": "ok"})
}

// Handler como struct (para injeção de dependências)
type produtoHandler struct {
    service ProdutoService
}

func (h *produtoHandler) ServeHTTP(w http.ResponseWriter, r *http.Request) {
    switch r.Method {
    case http.MethodGet:
        lista, err := h.service.List(r.Context())
        if err != nil {
            http.Error(w, err.Error(), http.StatusInternalServerError)
            return
        }
        w.Header().Set("Content-Type", "application/json")
        json.NewEncoder(w).Encode(lista)
    case http.MethodPost:
        var p Produto
        if err := json.NewDecoder(r.Body).Decode(&p); err != nil {
            http.Error(w, err.Error(), http.StatusBadRequest)
            return
        }
        // criar produto ...
        w.WriteHeader(http.StatusCreated)
        json.NewEncoder(w).Encode(p)
    }
}

func main() {
    mux := http.NewServeMux()
    mux.HandleFunc("GET /health", healthHandler)   // Go 1.22: método no padrão!
    mux.HandleFunc("GET /api/v1/produtos", func(w http.ResponseWriter, r *http.Request) {
        fmt.Fprintln(w, "lista de produtos")
    })

    srv := &http.Server{
        Addr:         ":8080",
        Handler:      logMiddleware(mux),
        ReadTimeout:  5 * time.Second,
        WriteTimeout: 10 * time.Second,
        IdleTimeout:  120 * time.Second,
    }

    slog.Info("servidor iniciado", "addr", ":8080")
    srv.ListenAndServe()
}

HTML Templates

html/template — server-side rendering seguro
package main

import (
    "html/template"
    "net/http"
)

var tmpl = template.Must(template.ParseGlob("templates/*.html"))

// templates/base.html
// <!DOCTYPE html>
// <html><head><title>{{.Titulo}}</title></head>
// <body>
// {{template "conteudo" .}}
// </body></html>

// templates/lista.html
// {{define "conteudo"}}
// <ul>{{range .Itens}}<li>{{.Nome}} — R$ {{printf "%.2f" .Preco}}</li>{{end}}</ul>
// {{end}}

type ListaViewModel struct {
    Titulo string
    Itens  []Produto
}

func listaHandler(w http.ResponseWriter, r *http.Request) {
    data := ListaViewModel{
        Titulo: "Catálogo de Produtos",
        Itens: []Produto{
            {Nome: "Teclado", Preco: 250},
            {Nome: "Mouse",   Preco: 120},
        },
    }
    if err := tmpl.ExecuteTemplate(w, "base.html", data); err != nil {
        http.Error(w, err.Error(), 500)
    }
}

// Não usar text/template para HTML — não escapa!

WebSockets com gorilla/websocket

Chat em tempo real com WebSockets
// go get github.com/gorilla/websocket
package main

import (
    "net/http"
    "github.com/gorilla/websocket"
)

var upgrader = websocket.Upgrader{
    ReadBufferSize:  1024,
    WriteBufferSize: 1024,
    CheckOrigin: func(r *http.Request) bool {
        return true   // em produção: verificar origin!
    },
}

type Hub struct {
    clients    map[*websocket.Conn]bool
    broadcast  chan []byte
    register   chan *websocket.Conn
    unregister chan *websocket.Conn
}

func (h *Hub) Run() {
    for {
        select {
        case conn := <-h.register:
            h.clients[conn] = true
        case conn := <-h.unregister:
            delete(h.clients, conn)
            conn.Close()
        case msg := <-h.broadcast:
            for conn := range h.clients {
                conn.WriteMessage(websocket.TextMessage, msg)
            }
        }
    }
}

func wsHandler(hub *Hub, w http.ResponseWriter, r *http.Request) {
    conn, err := upgrader.Upgrade(w, r, nil)
    if err != nil { return }
    hub.register <- conn

    for {
        _, msg, err := conn.ReadMessage()
        if err != nil {
            hub.unregister <- conn
            break
        }
        hub.broadcast <- msg
    }
}

Go 1.22 introduziu routing patterns com método HTTP e parâmetros de path direto no http.ServeMux padrão: mux.HandleFunc("GET /users/{id}", handler). Antes era necessário frameworks como Gin apenas para isso. Para muitos projetos simples, a stdlib agora é suficiente — um servidor HTTP de produção sem nenhuma dependência externa, compilando em um binário estático de poucos megabytes.
