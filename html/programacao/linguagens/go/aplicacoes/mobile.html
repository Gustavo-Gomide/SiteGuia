Go — Mobile

Go pode ser executado em dispositivos móveis de duas formas: gomobile — compila Go para bibliotecas nativas (.aar Android / .framework iOS) ou apps completos; e Fyne — framework GUI que compila para Android e iOS com o mesmo codebase desktop. Ambas as abordagens eliminam a necessidade de Java/Kotlin/Swift para lógica Go.

A abordagem mais prática é usar Go como biblioteca de lógica compartilhada: o backend em Go é compilado para .aar (Android) e .framework (iOS) via gomobile bind, e os UIs nativos são escritos em Kotlin/Swift — o que é o melhor de dois mundos para apps complexos.

gomobile bind — Biblioteca Nativa
Expor lógica Go para Android (AAR) e iOS (framework)
// Instalar gomobile
go install golang.org/x/mobile/cmd/gomobile@latest
gomobile init

// Estrutura: apenas tipos exportados são visíveis em Java/Kotlin/Swift
package meulib    // nome do pacote = nome da biblioteca

import (
    "errors"
    "fmt"
)

// Regras gomobile bind:
// - Apenas tipos: bool, int, float64, string, []byte
// - Structs com métodos exportados
// - Interfaces (retornadas por funções)
// - error como segundo valor de retorno

type Calculadora struct {
    Historico []string
}

func NovaCalculadora() *Calculadora {
    return &Calculadora{}
}

func (c *Calculadora) Somar(a, b float64) float64 {
    resultado := a + b
    c.Historico = append(c.Historico, fmt.Sprintf("%.2f + %.2f = %.2f", a, b, resultado))
    return resultado
}

func (c *Calculadora) Dividir(a, b float64) (float64, error) {
    if b == 0 {
        return 0, errors.New("divisão por zero")
    }
    return a / b, nil
}

func (c *Calculadora) UltimoHistorico() string {
    if len(c.Historico) == 0 { return "" }
    return c.Historico[len(c.Historico)-1]
}

// Compilar para Android (.aar)
gomobile bind -target android -o meulib.aar ./meulib/

// Compilar para iOS (.framework)
gomobile bind -target ios -o Meulib.xcframework ./meulib/

// Uso em Kotlin (Android)
// val calc = Meulib.NovaCalculadora()
// val res = calc.somar(10.0, 5.0)       → 15.0
// val div = calc.dividir(10.0, 0.0)     → throws Exception

// Uso em Swift (iOS)
// let calc = MeulibNovaCalculadora()!
// let res = calc.somar(10, b: 5)        → 15.0

gomobile build — App Completo

App Go nativo sem UI framework externo
// go get golang.org/x/mobile/app
// go get golang.org/x/mobile/event/lifecycle
import (
    "golang.org/x/mobile/app"
    "golang.org/x/mobile/event/lifecycle"
    "golang.org/x/mobile/event/paint"
    "golang.org/x/mobile/event/touch"
    "golang.org/x/mobile/gl"
    "golang.org/x/mobile/exp/gl/glutil"
)

func main() {
    app.Main(func(a app.App) {
        var glctx gl.Context

        for e := range a.Events() {
            switch e := a.Filter(e).(type) {

            case lifecycle.Event:
                switch e.Crosses(lifecycle.StageVisible) {
                case lifecycle.CrossOn:
                    glctx, _ = e.DrawContext.(gl.Context)
                case lifecycle.CrossOff:
                    glctx = nil
                }

            case paint.Event:
                if glctx != nil {
                    glctx.ClearColor(0.1, 0.1, 0.2, 1)
                    glctx.Clear(gl.COLOR_BUFFER_BIT)
                    a.Publish()
                }

            case touch.Event:
                fmt.Printf("Touch: (%.0f, %.0f) tipo=%v\n", e.X, e.Y, e.Type)
            }
        }
    })
}

# Buildar APK de debug direto no device conectado
gomobile build -target android -o app.apk .
adb install -r app.apk

# Buildar IPA para iOS (requer macOS + Xcode)
gomobile build -target ios -o app.app .

Fyne Mobile — UI Cross-Platform

O mesmo app Fyne compila para Android e iOS
// fyne package -os android -appID com.exemplo.app
// fyne package -os ios     -appID com.exemplo.app
// Requer: Android Studio / Xcode instalados

package main

import (
    "fyne.io/fyne/v2/app"
    "fyne.io/fyne/v2/container"
    "fyne.io/fyne/v2/widget"
    "fyne.io/fyne/v2/storage"
)

func main() {
    a := app.New()
    w := a.NewWindow("Notas")

    // Armazenamento persistente em mobile (usa storage nativo)
    uri, _ := storage.Child(storage.RootURI(), "notas.txt")

    entrada := widget.NewMultiLineEntry()
    entrada.SetPlaceHolder("Escreva suas notas...")

    // Carregar notas salvas
    if r, err := storage.Reader(uri); err == nil {
        defer r.Close()
        conteudo, _ := io.ReadAll(r)
        entrada.SetText(string(conteudo))
    }

    botaoSalvar := widget.NewButton("Salvar", func() {
        w, _ := storage.Writer(uri)
        defer w.Close()
        w.Write([]byte(entrada.Text))
        dialog.ShowInformation("OK", "Notas salvas!", mainWindow)
    })

    mainWindow = w
    w.SetContent(container.NewBorder(nil, botaoSalvar, nil, nil, entrada))
    w.ShowAndRun()
}

gomobile bind é a estratégia preferida para times que já têm app nativo: escreva a lógica de negócios, criptografia, sincronização e protocolos em Go — testável, portável e de alta performance — e deixe o UI para Kotlin/Swift que têm os melhores componentes nativos de cada plataforma. É a arquitetura usada por apps como o Dropbox e o Keybase em produção.
