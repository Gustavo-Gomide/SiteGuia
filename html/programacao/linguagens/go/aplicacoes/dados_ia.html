Go — Dados e IA

Go é excelente para pipelines de dados de alta performance: leitura de CSV/JSON em streaming, paralelismo com goroutines, e serviços de inferência que servem modelos Python via gRPC. As bibliotecas gonum, Gota e tfgo complementam o ecossistema.

Go é frequentemente usado como camada de serviço entre modelos Python e consumidores: o modelo é treinado em Python/TensorFlow, exportado como SavedModel, e servido via Go com alta concorrência e baixa latência. Combinação prática: Python para ciência de dados, Go para engineering de produção.

CSV e JSON — Processamento em Streaming

Ler CSV linha a linha sem carregar tudo na memória
package main

import (
    "encoding/csv"
    "encoding/json"
    "io"
    "os"
    "strconv"
)

type Venda struct {
    ID       int     `json:"id"`
    Produto  string  `json:"produto"`
    Valor    float64 `json:"valor"`
    Regiao   string  `json:"regiao"`
}

// Leitura streaming de CSV — O(1) memória
func processarCSV(caminho string) error {
    f, _ := os.Open(caminho)
    defer f.Close()

    r := csv.NewReader(f)
    r.Read()  // pular cabeçalho

    totalPorRegiao := make(map[string]float64)

    for {
        reg, err := r.Read()
        if err == io.EOF { break }
        if err != nil { return err }

        valor, _ := strconv.ParseFloat(reg[2], 64)
        totalPorRegiao[reg[3]] += valor
    }

    return json.NewEncoder(os.Stdout).Encode(totalPorRegiao)
}

// JSON Lines streaming (JSONL)
func lerJSONLines(r io.Reader, out chan<- Venda) {
    decoder := json.NewDecoder(r)
    for decoder.More() {
        var v Venda
        if err := decoder.Decode(&v); err == nil {
            out <- v
        }
    }
    close(out)
}

// Pipeline paralelo: ler + processar com goroutines
func pipeline() {
    vendas := make(chan Venda, 100)
    results := make(chan float64, 100)

    go func() {
        lerJSONLines(os.Stdin, vendas)
    }()

    // Worker pool — 4 goroutines processando em paralelo
    for i := 0; i < 4; i++ {
        go func() {
            for v := range vendas {
                results <- calcularTaxa(v)
            }
        }()
    }
}

gonum — Matemática e Estatística

Álgebra linear, estatística e distribuições
// go get gonum.org/v1/gonum
import (
    "gonum.org/v1/gonum/mat"
    "gonum.org/v1/gonum/stat"
    "gonum.org/v1/gonum/stat/distuv"
)

// Matrizes
a := mat.NewDense(2, 2, []float64{1, 2, 3, 4})
b := mat.NewDense(2, 2, []float64{5, 6, 7, 8})

result := mat.NewDense(2, 2, nil)
result.Mul(a, b)    // multiplicação de matrizes
result.Add(a, b)
result.Scale(2.0, a)

// Transposta e determinante
at := a.T()
det := mat.Det(a)

// Estatística descritiva
dados := []float64{2, 4, 4, 4, 5, 5, 7, 9}
pesos := make([]float64, len(dados))
for i := range pesos { pesos[i] = 1 }

media := stat.Mean(dados, pesos)        // 5.0
variância := stat.Variance(dados, pesos)
desvio := stat.StdDev(dados, pesos)
correlação := stat.Correlation(dados, dados, nil)

percentil90 := stat.Quantile(0.9, stat.Empirical, dados, nil)

// Distribuição Normal
norm := distuv.Normal{Mu: 0, Sigma: 1}
prob := norm.CDF(1.96)    // P(X <= 1.96) ≈ 0.975
amostra := norm.Rand()    // gerar amostra aleatória

// Regressão linear simples
x := []float64{1, 2, 3, 4, 5}
y := []float64{2.1, 3.9, 6.0, 8.1, 9.8}
alpha, beta := stat.LinearRegression(x, y, nil, false)
r2 := stat.RSquared(y, x, nil, alpha, beta)
fmt.Printf("y = %.2fx + %.2f  R²=%.4f\n", beta, alpha, r2)

Gota — DataFrames em Go

Manipulação tabular similar ao pandas
// go get github.com/go-gota/gota/dataframe
// go get github.com/go-gota/gota/series
import (
    "github.com/go-gota/gota/dataframe"
    "github.com/go-gota/gota/series"
)

// Carregar CSV
f, _ := os.Open("vendas.csv")
df := dataframe.ReadCSV(f)

// Inspecionar
fmt.Println(df.Describe())
fmt.Println(df.Nrow(), df.Ncol())
fmt.Println(df.Names())

// Filtrar linhas: valor > 100
filtrado := df.Filter(
    dataframe.F{Colname: "valor", Comparator: series.Greater, Comparando: "100"},
)

// Selecionar colunas
sub := df.Select([]string{"produto", "valor", "regiao"})

// Adicionar coluna calculada
desconto := series.New([]float64{0.1, 0.2, 0.15}, series.Float, "desconto")
df = df.Mutate(desconto)

// Group By + Agregação manual
grupos := df.GroupBy("regiao")
ag := grupos.Aggregation(
    []dataframe.AggregationType{dataframe.Aggregation_SUM},
    []string{"valor"},
)
fmt.Println(ag)

// Exportar para JSON
jsonData, _ := df.Select([]string{"produto", "valor"}).MarshalJSON()

Inferência com TensorFlow Go

Carregar SavedModel exportado do Python e executar inferência
// go get github.com/galeone/tfgo
// Requer libtensorflow instalado no sistema
import (
    tf "github.com/galeone/tfgo"
    "github.com/galeone/tfgo/image"
)

// Carregar modelo SavedModel exportado com tf.saved_model.save()
model := tf.LoadModel("./modelo_tf", []string{"serve"}, nil)

// Criar tensor de entrada — batch de imagens 224x224x3
inputData := [1][224][224][3]float32{}  // preencher com pixel values
inputTensor, _ := tf.NewTensor(inputData)

// Inferência
resultados := model.Exec(
    []tf.Output{
        model.Op("StatefulPartitionedCall", 0),
    },
    map[tf.Output]*tf.Tensor{
        model.Op("serving_default_input", 0): inputTensor,
    },
)

// Extrair probabilidades
probs := resultados[0].Value().([][]float32)[0]
classeMax := 0
for i, p := range probs {
    if p > probs[classeMax] { classeMax = i }
}
fmt.Printf("Classe predita: %d (%.2f%%)\n", classeMax, probs[classeMax]*100)

Para pipelines de dados que processam arquivos de GB, a combinação de CSV streaming + channel buffer + worker pool é imbatível: você mantém uso de memória constante enquanto paraleliza o processamento com no máximo N goroutines. O truque é calibrar o tamanho do channel buffer para evitar que produtores bloqueiem enquanto consumidores processam.
