Go — Embedded e IoT

TinyGo compila Go para microcontroladores (Arduino, Raspberry Pi Pico, ESP32) produzindo binários de 2–50 KB. Para dispositivos Linux-based com mais recursos, Go padrão com periph.io oferece acesso completo a GPIO, I²C, SPI e UART. MQTT via paho-mqtt conecta dispositivos ao cloud.

TinyGo usa LLVM como backend e remove o runtime Go completo — sem GC, sem goroutines. O subset de Go disponível é limitado mas cobre a maioria dos casos embarcados: tipos, funções, interfaces, arrays, structs e ponteiros. Para Raspberry Pi OS (Linux ARM), Go padrão roda normalmente com goroutines, GC e todo o ecossistema.

TinyGo — Microcontroladores

Arduino Uno, Raspberry Pi Pico e ESP32
// Instalar TinyGo
// https://tinygo.org/getting-started/
// tinygo flash -target arduino main.go
// tinygo flash -target pico main.go

package main

import (
    "machine"       // hardware abstraction — package especial do TinyGo
    "time"
)

// Blink LED — Arduino / Pico / ESP32
func main() {
    led := machine.LED    // pino LED built-in da placa
    led.Configure(machine.PinConfig{Mode: machine.PinOutput})

    for {
        led.High()
        time.Sleep(500 * time.Millisecond)
        led.Low()
        time.Sleep(500 * time.Millisecond)
    }
}

// Botão com pull-up + interrupção
func botaoInterrupcao() {
    btn := machine.BUTTON    // pino de botão embutido
    btn.Configure(machine.PinConfig{Mode: machine.PinInputPullup})

    btn.SetInterrupt(machine.PinFalling, func(pin machine.Pin) {
        machine.LED.Toggle()
    })
    for { time.Sleep(time.Hour) }    // esperar por interrupções
}

// PWM — controlar intensidade de LED ou servomotor
func pwmFade() {
    pwm := machine.TIM1
    pwm.Configure(machine.PWMConfig{Period: 20_000_000})  // 50 Hz
    ch, _ := pwm.Channel(machine.PA8)

    for {
        for i := uint32(0); i <= 100; i++ {
            pwm.Set(ch, pwm.Top()*i/100)
            time.Sleep(10 * time.Millisecond)
        }
    }
}

periph.io — GPIO, I²C e SPI no Linux

Raspberry Pi OS — sensores reais via I²C
// go get periph.io/x/periph
// go get periph.io/x/conn/v3
import (
    "periph.io/x/conn/v3/gpio"
    "periph.io/x/conn/v3/gpio/gpioreg"
    "periph.io/x/conn/v3/i2c/i2creg"
    "periph.io/x/host/v3"
    "periph.io/x/devices/v3/bmxx80"  // sensor BME280: temp/umidade/pressão
)

func main() {
    // Inicializar drivers de hardware
    if _, err := host.Init(); err != nil { panic(err) }

    // GPIO — LED e botão
    led := gpioreg.ByName("GPIO23")
    led.Out(gpio.High)
    time.Sleep(time.Second)
    led.Out(gpio.Low)

    botao := gpioreg.ByName("GPIO24")
    botao.In(gpio.PullUp, gpio.FallingEdge)
    botao.WaitForEdge(-1)   // bloqueia até o botão ser pressionado

    // I²C — sensor BME280 (endereço 0x76)
    bus, _ := i2creg.Open("")   // abre /dev/i2c-1 automaticamente
    defer bus.Close()

    sensor, _ := bmxx80.NewI2C(bus, 0x76, &bmxx80.DefaultOpts)
    defer sensor.Halt()

    var env devices.Environment
    sensor.Sense(&env)
    fmt.Printf("Temperatura: %v\nUmidade: %v\nPressão: %v\n",
        env.Temperature, env.Humidity, env.Pressure)
}

MQTT — Comunicação IoT em Nuvem

Publicar leituras de sensores e receber comandos
// go get github.com/eclipse/paho.mqtt.golang
import mqtt "github.com/eclipse/paho.mqtt.golang"

opts := mqtt.NewClientOptions().
    AddBroker("tcp://broker.hivemq.com:1883").
    SetClientID("sensor-01").
    SetUsername("user").
    SetPassword("pass").
    SetKeepAlive(30 * time.Second).
    SetAutoReconnect(true)

opts.OnConnectionLost = func(c mqtt.Client, err error) {
    fmt.Println("conexão perdida:", err)
}

client := mqtt.NewClient(opts)
if token := client.Connect(); token.Wait() && token.Error() != nil {
    panic(token.Error())
}
defer client.Disconnect(250)

// Publicar leitura de temperatura (QoS 1)
payload := fmt.Sprintf(`{"sensor":"01","temp":%.1f,"ts":%d}`, 23.5, time.Now().Unix())
client.Publish("sensores/01/temperatura", 1, false, payload)

// Subscribir a comandos (QoS 2)
client.Subscribe("dispositivos/01/cmds", 2, func(_ mqtt.Client, msg mqtt.Message) {
    fmt.Printf("Comando recebido: %s\n", msg.Payload())
    msg.Ack()   // para QoS 2
})

// Loop de leituras periódicas
ticker := time.NewTicker(10 * time.Second)
for range ticker.C {
    temp := lerSensor()
    client.Publish("sensores/01/temperatura", 1, false,
        fmt.Sprintf(`{"temp":%.1f}`, temp))
}

Porta Serial — Comunicação com Arduino

Ler dados seriais e enviar comandos
// go get go.bug.st/serial
import "go.bug.st/serial"

// Listar portas disponíveis
portas, _ := serial.GetPortsList()
for _, p := range portas {
    fmt.Println(p)   // /dev/ttyACM0, COM3, etc.
}

// Abrir porta serial — 9600 baud, 8N1
mode := &serial.Mode{
    BaudRate: 9600,
    Parity:   serial.NoParity,
    DataBits: 8,
    StopBits: serial.OneStopBit,
}
port, err := serial.Open("/dev/ttyACM0", mode)
if err != nil { panic(err) }
defer port.Close()

// Enviar comando
port.Write([]byte("LED:ON\n"))

// Ler resposta linha a linha
buf := make([]byte, 1024)
for {
    n, _ := port.Read(buf)
    linha := string(buf[:n])
    if strings.TrimSpace(linha) != "" {
        fmt.Print(linha)
    }
}
