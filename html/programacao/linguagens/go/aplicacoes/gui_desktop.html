Go — GUI e Desktop

Fyne é o toolkit GUI cross-platform mais popular em Go: compila para Windows, macOS, Linux e mobile com um único codebase. Produz binários nativos sem dependências de runtime externas. Para apps utilitários, systray adiciona ícone na bandeja do sistema.

Fyne usa OpenGL como backend de renderização, portanto requer drivers gráficos. Em servidores headless, use fyne_test ou test.NewApp() para testes. A API segue o padrão widgets declarativos com binding de dados bidirecional — semelhante ao Flutter conceitualmente, mas em Go.

Fyne — Janelas e Widgets

go get fyne.io/fyne/v2 — Widgets fundamentais
package main

import (
    "fyne.io/fyne/v2/app"
    "fyne.io/fyne/v2/container"
    "fyne.io/fyne/v2/widget"
    "fyne.io/fyne/v2/layout"
    "fyne.io/fyne/v2/data/binding"
)

func main() {
    a := app.New()
    w := a.NewWindow("Calculadora IMC")
    w.Resize(fyne.NewSize(400, 300))

    // Widgets básicos
    labelTitulo := widget.NewLabel("Calculadora de IMC")
    labelTitulo.TextStyle = fyne.TextStyle{Bold: true}

    entradaPeso := widget.NewEntry()
    entradaPeso.SetPlaceHolder("Peso (kg)")

    entradaAltura := widget.NewEntry()
    entradaAltura.SetPlaceHolder("Altura (m)")

    resultado := widget.NewLabel("")

    botao := widget.NewButton("Calcular", func() {
        peso, _ := strconv.ParseFloat(entradaPeso.Text, 64)
        altura, _ := strconv.ParseFloat(entradaAltura.Text, 64)
        imc := peso / (altura * altura)
        resultado.SetText(fmt.Sprintf("IMC: %.1f", imc))
    })

    // Layout VBox
    w.SetContent(container.NewVBox(
        labelTitulo,
        widget.NewSeparator(),
        entradaPeso,
        entradaAltura,
        botao,
        resultado,
    ))

    w.ShowAndRun()
}

// Data Binding — atualização automática de UI
func comBinding(w fyne.Window) {
    contador := binding.NewInt()
    contador.Set(0)

    label := widget.NewLabelWithData(binding.IntToString(contador))

    btn := widget.NewButton("+", func() {
        v, _ := contador.Get()
        contador.Set(v + 1)
    })

    w.SetContent(container.NewVBox(label, btn))
}

Fyne — Forms, Menus e Canvas

Formulários, menus nativos e desenho vetorial
import (
    "fyne.io/fyne/v2/canvas"
    "fyne.io/fyne/v2/dialog"
    "image/color"
)

// Form com validação
nomeEntry := widget.NewEntry()
emailEntry := widget.NewEntry()

form := &widget.Form{
    Items: []*widget.FormItem{
        {Text: "Nome", Widget: nomeEntry, HintText: "Nome completo"},
        {Text: "Email", Widget: emailEntry},
    },
    OnSubmit: func() {
        if !strings.Contains(emailEntry.Text, "@") {
            dialog.ShowError(fmt.Errorf("email inválido"), w)
            return
        }
        dialog.ShowInformation("Sucesso", "Formulário enviado!", w)
    },
}

// Menu nativo da aplicação
menuArquivo := fyne.NewMenu("Arquivo",
    fyne.NewMenuItem("Abrir", func() {
        dialog.ShowFileOpen(func(r fyne.URIReadCloser, _ error) {
            if r != nil { fmt.Println("Arquivo:", r.URI()) }
        }, w)
    }),
    fyne.NewMenuItem("Salvar", func() {
        dialog.ShowFileSave(func(w fyne.URIWriteCloser, _ error) {}, w)
    }),
    fyne.NewMenuItemSeparator(),
    fyne.NewMenuItem("Sair", a.Quit),
)
w.SetMainMenu(fyne.NewMainMenu(menuArquivo))

// Canvas — desenho vetorial
rect := canvas.NewRectangle(color.NRGBA{R: 0, G: 120, B: 255, A: 255})
circle := canvas.NewCircle(color.NRGBA{R: 255, G: 60, B: 0, A: 200})
text := canvas.NewText("Go GUI", color.White)
text.TextSize = 24

w.SetContent(container.NewWithoutLayout(rect, circle, text))

Systray — Bandeja do Sistema

Ícone na bandeja + app rodando em background
// go get github.com/getlantern/systray
import "github.com/getlantern/systray"

func main() {
    systray.Run(onReady, onExit)
}

func onReady() {
    // Ícone PNG em bytes (ex: arquivo embutido com //go:embed)
    systray.SetIcon(iconData)
    systray.SetTitle("Monitor")
    systray.SetTooltip("Monitor de sistema em Go")

    mStatus := systray.AddMenuItem("Status: OK", "Exibir status")
    mAbrir := systray.AddMenuItem("Abrir Dashboard", "Abre janela principal")
    systray.AddSeparator()
    mSair := systray.AddMenuItem("Sair", "Encerrar aplicação")

    go func() {
        for {
            select {
            case <-mStatus.ClickedCh:
                dialog.ShowInformation("Status", obterStatus(), nil)
            case <-mAbrir.ClickedCh:
                abrirJanelaFyne()
            case <-mSair.ClickedCh:
                systray.Quit()
            }
        }
    }()

    // Atualizar tooltip periodicamente
    ticker := time.NewTicker(5 * time.Second)
    go func() {
        for range ticker.C {
            uso := obterUSOCPU()
            systray.SetTooltip(fmt.Sprintf("CPU: %.0f%%", uso))
        }
    }()
}

func onExit() {
    fmt.Println("Encerrando...")
}

Fyne compila o mesmo código para Windows, macOS e Linux usando fyne package — gera .exe, .app e binário Linux respectivamente. Para distribuição, use fyne-cross para compilação cruzada a partir de qualquer plataforma. O resultado é um único executável portável sem instalação de runtime, o que é incomum no ecossistema de GUIs cross-platform.
