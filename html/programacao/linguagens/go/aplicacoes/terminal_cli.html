Go — Terminal e CLI

Go é uma das melhores linguagens para ferramentas CLI: binários estáticos sem dependências, cross-compile trivial, e uma das melhores bibliotecas do ecossistema — Cobra para argumentos/subcomandos, Viper para configuração hierárquica, Bubble Tea para TUIs interativas com arquitetura Elm, e lipgloss para estilo ANSI.

O ecossistema Charmbracelet (bubbletea + lipgloss + bubbles + glamour) tornou Go na melhor plataforma para TUIs modernas. Ferramentas como gh, lazygit, k9s e charm são construídas com Go — provando que aplicações de terminal podem ter UX comparável a interfaces gráficas.

Cobra + Viper — CLI Profissional

Subcomandos, flags, configuração e auto-complete
package main

import (
    "github.com/spf13/cobra"
    "github.com/spf13/viper"
    "fmt"
    "os"
)

var rootCmd = &cobra.Command{
    Use:   "mytool",
    Short: "Ferramenta de processamento de dados",
    Long:  `mytool processa arquivos CSV, JSON e bancos de dados.`,
}

var converterCmd = &cobra.Command{
    Use:     "converter [arquivo]",
    Short:   "Converte formato de arquivo",
    Args:    cobra.ExactArgs(1),
    Example: "  mytool converter dados.csv --output dados.json --formato json",
    RunE: func(cmd *cobra.Command, args []string) error {
        entrada := args[0]
        saida, _ := cmd.Flags().GetString("output")
        formato, _ := cmd.Flags().GetString("formato")
        verbose, _ := cmd.Flags().GetBool("verbose")

        if verbose {
            fmt.Printf("Convertendo %s → %s (formato: %s)\n", entrada, saida, formato)
        }
        // ... lógica real
        return nil
    },
}

func init() {
    // Flags persistentes (herdadas por subcomandos)
    rootCmd.PersistentFlags().Bool("verbose", false, "saída detalhada")

    // Flags locais do subcomando
    converterCmd.Flags().StringP("output", "o", "output.json", "arquivo de saída")
    converterCmd.Flags().StringP("formato", "f", "json", "formato: json, csv, yaml")

    rootCmd.AddCommand(converterCmd)

    // Viper — arquivo de configuração + env vars
    viper.SetConfigName("config")
    viper.SetConfigType("yaml")
    viper.AddConfigPath("$HOME/.mytool")
    viper.AddConfigPath(".")
    viper.AutomaticEnv()   // MYTOOL_DATABASE_URL → database.url
    viper.SetEnvPrefix("MYTOOL")

    if err := viper.ReadInConfig(); err == nil {
        fmt.Println("Config:", viper.ConfigFileUsed())
    }

    // Bind flag ao Viper
    viper.BindPFlag("verbose", rootCmd.PersistentFlags().Lookup("verbose"))
}

func main() {
    if err := rootCmd.Execute(); err != nil {
        os.Exit(1)
    }
}

Bubble Tea — TUI Interativa (Arquitetura Elm)

Model + Update + View — lista interativa no terminal
// go get github.com/charmbracelet/bubbletea
// go get github.com/charmbracelet/lipgloss
// go get github.com/charmbracelet/bubbles
import (
    tea "github.com/charmbracelet/bubbletea"
    "github.com/charmbracelet/lipgloss"
    "github.com/charmbracelet/bubbles/textinput"
    "github.com/charmbracelet/bubbles/spinner"
    "fmt"
)

type Model struct {
    itens     []string
    cursor    int
    selecionados map[int]bool
    input     textinput.Model
    loading   bool
    spinner   spinner.Model
}

// Mensagens (Events)
type dadosCarregados []string

func novoModel() Model {
    ti := textinput.New()
    ti.Placeholder = "Buscar..."
    ti.Focus()

    sp := spinner.New()
    sp.Spinner = spinner.Dot

    return Model{
        itens:     []string{"Go", "Rust", "Python", "TypeScript"},
        selecionados: make(map[int]bool),
        input:     ti,
        spinner:   sp,
    }
}

// Init — comandos iniciais
func (m Model) Init() tea.Cmd {
    return tea.Batch(textinput.Blink, m.spinner.Tick)
}

// Update — processam mensagens e retornam novo Model
func (m Model) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
    switch msg := msg.(type) {
    case tea.KeyMsg:
        switch msg.String() {
        case "ctrl+c", "q":
            return m, tea.Quit
        case "up", "k":
            if m.cursor > 0 { m.cursor-- }
        case "down", "j":
            if m.cursor < len(m.itens)-1 { m.cursor++ }
        case " ":
            m.selecionados[m.cursor] = !m.selecionados[m.cursor]
        }
    }
    var cmd tea.Cmd
    m.input, cmd = m.input.Update(msg)
    return m, cmd
}

// View — renderização em string
func (m Model) View() string {
    estiloCursor := lipgloss.NewStyle().Foreground(lipgloss.Color("#00D7FF")).Bold(true)
    estiloSel    := lipgloss.NewStyle().Foreground(lipgloss.Color("#00FF87"))
    estiloNormal := lipgloss.NewStyle().Foreground(lipgloss.Color("#FFFDF5"))

    s := "Selecione linguagens (espaço = selecionar, q = sair)\n\n"
    s += m.input.View() + "\n\n"

    for i, item := range m.itens {
        cursor := "  "
        if m.cursor == i { cursor = estiloCursor.Render("▶ ") }

        check := "○ "
        if m.selecionados[i] { check = estiloSel.Render("● ") }

        line := estiloNormal.Render(item)
        s += fmt.Sprintf("%s%s%s\n", cursor, check, line)
    }
    return s
}

func main() {
    p := tea.NewProgram(novoModel(), tea.WithAltScreen())
    if _, err := p.Run(); err != nil { panic(err) }
}

// Progress bar simples sem dependências
func progressBar(atual, total int, largura int) string {
    pct := float64(atual) / float64(total)
    preenchido := int(pct * float64(largura))
    barra := strings.Repeat("█", preenchido) + strings.Repeat("░", largura-preenchido)
    return fmt.Sprintf("\r[%s] %d/%d (%.0f%%)", barra, atual, total, pct*100)
}

// Usar em loop
for i := 0; i <= 100; i++ {
    fmt.Print(progressBar(i, 100, 30))
    time.Sleep(20 * time.Millisecond)
}
fmt.Println()    // nova linha ao finalizar

// Cores ANSI sem dependências
const (
    Reset  = "\033[0m"
    Verde  = "\033[32m"
    Vermelho = "\033[31m"
    Azul   = "\033[34m"
    Negrito = "\033[1m"
)

func sucesso(msg string) { fmt.Printf("%s%s✓ %s%s\n", Negrito, Verde, msg, Reset) }
func erro(msg string)    { fmt.Fprintf(os.Stderr, "%s✗ %s%s\n", Vermelho, msg, Reset) }
func info(msg string)    { fmt.Printf("%sℹ %s%s\n", Azul, msg, Reset) }

// Spinner sem dependências
frames := []string{"⠋", "⠙", "⠹", "⠸", "⠼", "⠴", "⠦", "⠧", "⠇", "⠏"}
i := 0
ticker := time.NewTicker(80 * time.Millisecond)
done := make(chan bool)
go func() {
    for {
        select {
        case <-ticker.C:
            fmt.Printf("\r%s Processando...", frames[i%len(frames)])
            i++
        case <-done:
            fmt.Println("\r✓ Concluído!   ")
            return
        }
    }
}()

A arquitetura Elm no Bubble Tea (Model imutável → Update(msg) retorna novo Model → View() renderiza string) é revolucionária para TUIs: estado sempre previsível, sem callbacks entrelaçados, sem mutação concorrente. Cada tecla pressionada é uma mensagem tipada; cada efeito assíncrono (chamada HTTP, leitura de arquivo) retorna uma mensagem — tornando o fluxo de dados 100% rastreável e testável.
