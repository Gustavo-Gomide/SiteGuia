Go — Banco de Dados

Go oferece acesso a bancos de dados relacionais via database/sql (stdlib) com drivers específicos, pgx para PostgreSQL nativo de alta performance, sqlx para mapeamento de structs, e GORM para ORM completo. Migrações são gerenciadas com golang-migrate ou goose.

database/sql gerencia um connection pool automático. Cada chamada db.QueryContext() pega uma conexão do pool e devolve ao terminar. O pool se ajusta automaticamente à carga — configure SetMaxOpenConns, SetMaxIdleConns e SetConnMaxLifetime para ambientes de produção. Não há necessidade de gerenciar conexões manualmente.

database/sql — Nível Baixo

Queries, scan de resultados e transações
package main

import (
    "context"
    "database/sql"
    "fmt"
    "time"
    _ "github.com/lib/pq"   // driver PostgreSQL — registra via init()
)

type Produto struct {
    ID        int
    Nome      string
    Preco     float64
    CriadoEm  time.Time
}

func conectar() *sql.DB {
    db, err := sql.Open("postgres",
        "host=localhost port=5432 user=app password=senha dbname=shop sslmode=disable")
    if err != nil {
        panic(err)
    }
    // Configurar connection pool
    db.SetMaxOpenConns(25)
    db.SetMaxIdleConns(5)
    db.SetConnMaxLifetime(5 * time.Minute)
    db.SetConnMaxIdleTime(1 * time.Minute)
    return db
}

// Query — múltiplas linhas
func listarProdutos(ctx context.Context, db *sql.DB) ([]Produto, error) {
    rows, err := db.QueryContext(ctx,
        "SELECT id, nome, preco, criado_em FROM produtos WHERE ativo = $1", true)
    if err != nil {
        return nil, fmt.Errorf("listarProdutos: %w", err)
    }
    defer rows.Close()

    var produtos []Produto
    for rows.Next() {
        var p Produto
        if err := rows.Scan(&p.ID, &p.Nome, &p.Preco, &p.CriadoEm); err != nil {
            return nil, fmt.Errorf("scan: %w", err)
        }
        produtos = append(produtos, p)
    }
    return produtos, rows.Err()
}

// QueryRow — uma linha
func buscarPorID(ctx context.Context, db *sql.DB, id int) (*Produto, error) {
    var p Produto
    err := db.QueryRowContext(ctx,
        "SELECT id, nome, preco, criado_em FROM produtos WHERE id = $1", id).
        Scan(&p.ID, &p.Nome, &p.Preco, &p.CriadoEm)
    if err == sql.ErrNoRows {
        return nil, nil   // não encontrado
    }
    return &p, err
}

// Transação
func transferir(ctx context.Context, db *sql.DB, deID, paraID int, valor float64) error {
    tx, err := db.BeginTx(ctx, &sql.TxOptions{Isolation: sql.LevelSerializable})
    if err != nil { return err }
    defer tx.Rollback()   // no-op se Commit() chamou antes

    _, err = tx.ExecContext(ctx,
        "UPDATE contas SET saldo = saldo - $1 WHERE id = $2", valor, deID)
    if err != nil { return err }

    _, err = tx.ExecContext(ctx,
        "UPDATE contas SET saldo = saldo + $1 WHERE id = $2", valor, paraID)
    if err != nil { return err }

    return tx.Commit()
}

pgx — Driver PostgreSQL Nativo

pgx v5: batch, COPY e tipos PostgreSQL nativos
// go get github.com/jackc/pgx/v5
import (
    "github.com/jackc/pgx/v5"
    "github.com/jackc/pgx/v5/pgxpool"
)

// Pool de conexões pgx
pool, _ := pgxpool.New(ctx, "postgres://user:pass@localhost/shop")
defer pool.Close()

// Row scan sem reflect (tipado)
row := pool.QueryRow(ctx, "SELECT id, nome, preco FROM produtos WHERE id=$1", 1)
var id int; var nome string; var preco float64
row.Scan(&id, &nome, &preco)

// pgx.CollectRows — scan para slice de structs
rows, _ := pool.Query(ctx, "SELECT id, nome, preco FROM produtos")
produtos, _ := pgx.CollectRows(rows, pgx.RowToStructByName[Produto])

// Batch — enviar múltiplos comandos em 1 roundtrip
batch := &pgx.Batch{}
batch.Queue("INSERT INTO produtos (nome, preco) VALUES ($1, $2)", "A", 10.0)
batch.Queue("INSERT INTO produtos (nome, preco) VALUES ($1, $2)", "B", 20.0)
pool.SendBatch(ctx, batch).Close()

// COPY — inserção em massa (10x mais rápido que INSERTs)
_, err := pool.CopyFrom(
    ctx,
    pgx.Identifier{"produtos"},
    []string{"nome", "preco"},
    pgx.CopyFromSlice(len(dados), func(i int) ([]any, error) {
        return []any{dados[i].Nome, dados[i].Preco}, nil
    }),
)

Migrações com golang-migrate

Criar e aplicar migrações SQL versionadas
# Instalar golang-migrate
go install -tags 'postgres' github.com/golang-migrate/migrate/v4/cmd/migrate@latest

# Criar migração
migrate create -ext sql -dir migrations/ -seq create_produtos

# migrations/000001_create_produtos.up.sql
CREATE TABLE produtos (
    id         SERIAL PRIMARY KEY,
    nome       VARCHAR(255) NOT NULL,
    preco      DECIMAL(10,2) NOT NULL CHECK (preco >= 0),
    ativo      BOOLEAN NOT NULL DEFAULT true,
    criado_em  TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_produtos_nome ON produtos(nome);

# migrations/000001_create_produtos.down.sql
DROP TABLE IF EXISTS produtos;

# Aplicar migrações
migrate -path migrations/ -database $DATABASE_URL up
migrate -path migrations/ -database $DATABASE_URL down 1
migrate -path migrations/ -database $DATABASE_URL version

// Usar golang-migrate programaticamente no Go
import "github.com/golang-migrate/migrate/v4"

m, err := migrate.New("file://migrations", databaseURL)
if err != nil { log.Fatal(err) }
if err := m.Up(); err != nil && err != migrate.ErrNoChange {
    log.Fatal(err)
}

Redis com go-redis

Cache, sessões e pub/sub com Redis
// go get github.com/redis/go-redis/v9
import "github.com/redis/go-redis/v9"

rdb := redis.NewClient(&redis.Options{
    Addr:     "localhost:6379",
    Password: "",
    DB:       0,
})
defer rdb.Close()

// SET com TTL
rdb.Set(ctx, "sessao:abc", "userid:42", 24*time.Hour)

// GET
val, err := rdb.Get(ctx, "sessao:abc").Result()
if err == redis.Nil {
    fmt.Println("sessão expirada")
}

// Cache-aside pattern
func buscarComCache(ctx context.Context, id int) (*Produto, error) {
    chave := fmt.Sprintf("produto:%d", id)
    
    cached, err := rdb.Get(ctx, chave).Bytes()
    if err == nil {
        var p Produto
        json.Unmarshal(cached, &p)
        return &p, nil
    }

    p, err := buscarDoBanco(ctx, id)
    if err != nil { return nil, err }

    data, _ := json.Marshal(p)
    rdb.Set(ctx, chave, data, 5*time.Minute)
    return p, nil
}

// Pub/Sub
pubsub := rdb.Subscribe(ctx, "eventos")
ch := pubsub.Channel()
go func() {
    for msg := range ch {
        fmt.Println("evento:", msg.Payload)
    }
}()
rdb.Publish(ctx, "eventos", "produto_criado:42")

O padrão de usar defer tx.Rollback() logo após BeginTx() é idiomático em Go e garante que a transação seja revertida em qualquer caminho de erro — mesmo que você retorne cedo, mesmo que um panic ocorra (com recover). Quando Commit() é chamado com sucesso, o Rollback() posterior é um no-op. Sem necessidade de try/finally, sem handler de exceção. Lógica clara, segura por construção.
