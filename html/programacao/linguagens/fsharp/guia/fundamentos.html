F# — Fundamentos

Os fundamentos do F# cobrem o sistema de tipos (primitivos, listas, tuples, records, DUs), funções como valores de primeira classe com aplicação parcial e currying, pattern matching exaustivo, e o estilo de programação com pipelines — tudo com inferência de tipos que elimina a maior parte das anotações.

Em F#, tudo é uma expressão que retorna um valor — inclusive if/else, match e try/with. Uma função que não retorna nada útil retorna unit (equivalente ao void de C#, mas é um tipo real). Isso permite composição sem casos especiais e elimina funções void que são difíceis de encadear.

Tipos e Bindings

Primitivos, tuples, listas e opções
// Primitivos — anotação de tipo opcional
let inteiro : int         = 42
let float_   : float      = 3.14
let bool_    : bool       = true
let texto    : string     = "F#"
let char_    : char       = 'A'
let big      : int64      = 9_223_372_036_854_775_807L
let decimal_ : decimal    = 1.99m

// Tuple — múltiplos valores sem nome
let par: int * string = (1, "um")
let a, b = par    // desestruturar
let primeiro = fst par
let segundo  = snd par

// Lista imutável (Linked list)
let nums = [1; 2; 3; 4; 5]
let nums2 = 0 :: nums               // cons: [0;1;2;3;4;5]
let combinado = [1;2] @ [3;4]       // concatenar
nums |> List.head                   // 1
nums |> List.tail                   // [2;3;4;5]
nums |> List.length                 // 5

// Array mutável de tamanho fixo
let arr = [| 10; 20; 30 |]
arr.[1]                             // 20 (ou arr[1] F# 6+)
arr.[0] <- 100                      // mutação

// Option — ausência de valor sem null
let alguns : int option = Some 42
let nenhum : int option = None

let valor =
    match alguns with
    | Some n -> n * 2
    | None   -> 0

// Operadores de option
alguns |> Option.defaultValue 0     // 42
nenhum |> Option.defaultValue 0     // 0
alguns |> Option.map (fun n -> n + 1) // Some 43
alguns |> Option.bind (fun n -> if n > 10 then Some n else None)

Records e Discriminated Unions

Records imutáveis e DUs para modelagem de domínio
// Record — produto de campos nomeados
type Pessoa = {
    Nome  : string
    Idade : int
    Email : string
}

let joao = { Nome = "João"; Idade = 30; Email = "jo@ex.com" }
// Acessar campos
joao.Nome                   // "João"
// Cópia com modificação (with expression)
let joaoMaisVelho = { joao with Idade = 31 }

// Record genérico
type Paginado<'T> = {
    Itens      : 'T list
    Total      : int
    Pagina     : int
    TotalPages : int
}

// Discriminated Union — soma de casos
type Forma =
    | Circulo   of raio: float
    | Retangulo of largura: float * altura: float
    | Triangulo of base_: float * altura: float

let area forma =
    match forma with
    | Circulo r       -> System.Math.PI * r * r
    | Retangulo (l,h) -> l * h
    | Triangulo (b,h) -> b * h / 2.0

let c = Circulo 5.0
printfn "Área: %.2f" (area c)   // 78.54

// DU para modelar resultado (Result)
type Resultado<'OK, 'Erro> =
    | Sucesso of 'OK
    | Falha   of 'Erro

let dividir a b =
    if b = 0 then Falha "divisão por zero"
    else Sucesso (a / b)

// DU para estado de pedido — estados inválidos irrepresentáveis
type StatusPedido =
    | PedidoPendente  of criadoEm: System.DateTime
    | PedidoPago      of pago: System.DateTime * valor: decimal
    | PedidoEnviado   of pago: System.DateTime * rastreio: string
    | PedidoCancelado of motivo: string

Funções e Pipeline

HOFs, currying, pipe |> e composição >>
// Função de primeira classe
let dobrar x = x * 2
let fnRef = dobrar                  // funções são valores

// Lambda (função anônima)
let triple = fun x -> x * 3

// Funções de ordem superior
let nums = [1..10]
nums |> List.map (fun x -> x * x)   // [1;4;9;16;25;36;49;64;81;100]
nums |> List.filter (fun x -> x % 2 = 0)  // [2;4;6;8;10]
nums |> List.reduce (+)              // 55
nums |> List.fold (fun acc x -> acc + x) 0  // 55

// Pipeline |> — encadear transformações
let resultado =
    [1..20]
    |> List.filter (fun n -> n % 2 = 0)
    |> List.map    (fun n -> n * n)
    |> List.sum

// Composição >> — combinar funções
let dobrarEAdicionar1 = dobrar >> (fun x -> x + 1)
dobrarEAdicionar1 5  // 11

// Aplicação parcial
let multiplicar a b = a * b
let dobrar2       = multiplicar 2
let triplicar     = multiplicar 3
[1..5] |> List.map dobrar2          // [2;4;6;8;10]

// Recursão com rec
let rec fib n =
    match n with
    | 0 | 1 -> n
    | n     -> fib (n-1) + fib (n-2)

// Tail-recursive com acumulador
let rec fibA n (a, b) =
    if n = 0 then a
    else fibA (n-1) (b, a+b)
let fibonacci n = fibA n (0, 1)

Pattern Matching

Match expression — exaustivo, com guards e destructuring
// Match básico
let descrever n =
    match n with
    | 0         -> "zero"
    | 1         -> "um"
    | n when n < 0 -> "negativo"   // guard com when
    | _         -> "outro"          // wildcard

// Match em lista
let primeiroElemento lista =
    match lista with
    | []     -> None
    | x :: _ -> Some x

// Match em tuple
let classificar_ponto (x, y) =
    match (x, y) with
    | (0, 0) -> "origem"
    | (x, 0) -> sprintf "eixo X, x=%d" x
    | (0, y) -> sprintf "eixo Y, y=%d" y
    | (x, y) -> sprintf "(%d, %d)" x y

// Match em record
let saudar pessoa =
    match pessoa with
    | { Nome = "Admin" } -> "Bem-vindo, administrador!"
    | { Idade = a } when a < 18 -> "Acesso restrito"
    | { Nome = n; Email = e } -> sprintf "Olá %s <%s>" n e

// Active Patterns — customizar como match funciona
let (|Par|Impar|) n = if n % 2 = 0 then Par else Impar

let descreverNum n =
    match n with
    | Par   -> sprintf "%d é par"   n
    | Impar -> sprintf "%d é ímpar" n

Discriminated Unions com pattern matching tornam impossível esquecer um caso: se você adicionar um novo estado ao seu DU, todos os match existentes deixam de compilar até serem atualizados. É o equivalente a um aviso de compilação global que não pode ser ignorado — contraste com herança de classe onde adicionar um novo subtipo pode passar silenciosamente sem atualizar código existente.
