F# — Terminal e CLI

F# é excelente para ferramentas de linha de comando e scripts de automação: a biblioteca Argu usa Discriminated Unions para definir argumentos CLI completamente tipados com help automático; FSI permite scripts standalone sem compilação; o pipeline funcional |> lê arquivos como sequências lazy, processa e salva com código conciso e seguro.

Com Argu, você define seus argumentos CLI como uma DU: cada case é uma flag ou opção. O parser lê argv, valida tipos e retorna um valor F# — não strings. Erro de tipo na CLI vira erro de tipo em F#. Help é gerado automaticamente a partir dos nomes e atributos dos cases.

Argu — Argumentos CLI Tipados

CLI completa com subcomandos, flags e validação
// dotnet add package Argu
open Argu

// DU define os argumentos — cada case é um argumento
type CliArgs =
    | [<AltCommandLine("-i")>] Input  of path: string
    | [<AltCommandLine("-o")>] Output of path: string
    | [<AltCommandLine("-v")>] Verbose
    | [<AltCommandLine("-n")>] Lines  of count: int
    | Force

    interface IArgParserTemplate with
        member s.Usage =
            match s with
            | Input  _ -> "Arquivo de entrada"
            | Output _ -> "Arquivo de saída"
            | Verbose   -> "Modo verboso"
            | Lines  _  -> "Número de linhas a processar"
            | Force     -> "Sobrescrever arquivo existente"

[<EntryPoint>]
let main argv =
    let parser  = ArgumentParser.Create<CliArgs>(programName = "meutool")
    let results = parser.ParseCommandLine(argv)

    // Extrair valores tipados — não strings!
    let entrada  = results.GetResult(Input)
    let saida    = results.GetResult(Output, defaultValue = "output.txt")
    let verbose  = results.Contains Verbose
    let nLinhas  = results.GetResult(Lines, defaultValue = -1)
    let forcar   = results.Contains Force

    if verbose then
        printfn "Entrada: %s → Saída: %s (linhas: %d)" entrada saida nLinhas

    // Lógica...
    0

// Uso:  meutool --input arq.txt --output res.txt -v --lines 100
// Help: meutool --help  (gerado automaticamente)

System.IO — Arquivos com Pipeline Funcional

Ler, transformar e salvar arquivos de forma funcional
open System.IO

// Ler linhas como seq (lazy — não carrega tudo na memória)
let lerLinhas arquivo : string seq =
    File.ReadLines(arquivo)   // lazy

// Processar CSV grande sem DataFrame
let processarCsv entrada saida =
    use writer = new StreamWriter(saida)
    writer.WriteLine("Nome,Valor,Dobro")
    lerLinhas entrada
    |> Seq.skip 1                         // pular header
    |> Seq.map (fun linha -> linha.Split(","))
    |> Seq.filter (fun cols -> cols.Length = 2)
    |> Seq.map (fun cols ->
        let nome  = cols.[0].Trim()
        let valor = cols.[1].Trim() |> float
        nome, valor, valor * 2.0)
    |> Seq.iter (fun (nome, v, d) ->
        writer.WriteLine(sprintf "%s,%.2f,%.2f" nome v d))

// Percorrer diretório recursivamente
let buscarArquivos raiz extensao =
    Directory.EnumerateFiles(raiz, sprintf "*.%s" extensao, SearchOption.AllDirectories)
    |> Seq.toList

// Copiar e renomear em batch
let renomearArquivos dir prefixo =
    Directory.EnumerateFiles(dir)
    |> Seq.iteri (fun i caminho ->
        let ext   = Path.GetExtension(caminho)
        let novo  = Path.Combine(dir, sprintf "%s_%03d%s" prefixo i ext)
        File.Move(caminho, novo))

// Ler JSON de arquivo
open System.Text.Json
let lerJson<'T> caminho =
    File.ReadAllText(caminho)
    |> fun txt -> JsonSerializer.Deserialize<'T>(txt)

// Escrever JSON formatado
let salvarJson<'T> caminho (valor: 'T) =
    let opts = JsonSerializerOptions(WriteIndented = true)
    JsonSerializer.Serialize(valor, opts)
    |> fun json -> File.WriteAllText(caminho, json)

Console Colorido e Formatação

Cores ANSI, tabelas no terminal e progress reporting
open System

// Cores ANSI (funciona no Windows Terminal, macOS, Linux)
let verde   = "\x1b[32m"
let vermelho = "\x1b[31m"
let amarelo  = "\x1b[33m"
let negrito  = "\x1b[1m"
let reset    = "\x1b[0m"

let ok   msg = printfn "%s✓%s %s" verde reset msg
let erro  msg = printfn "%s✗%s %s" vermelho reset msg
let aviso msg = printfn "%s⚠%s %s" amarelo reset msg

ok    "Arquivo processado com sucesso"
erro  "Arquivo não encontrado: dados.csv"
aviso "Nenhum item correspondeu ao filtro"

// Tabela no terminal
let imprimirTabela (headers: string list) (rows: string list list) =
    let widths =
        headers |> List.mapi (fun i h ->
            rows
            |> List.map (fun r -> if i < r.Length then r.[i].Length else 0)
            |> List.max
            |> max h.Length)
    let linha = widths |> List.map (fun w -> String.replicate w "-") |> String.concat "-+-"
    let formatar cols =
        List.mapi (fun i c -> (cols |> List.item i).PadRight(widths.[i])) [0..widths.Length-1]
        |> String.concat " | "
    printfn "| %s |" (formatar headers)
    printfn "+-%-s-+" linha
    rows |> List.iter (fun r -> printfn "| %s |" (formatar r))

// Barra de progresso
let progresso atual total =
    let pct  = int (float atual / float total * 40.0)
    let bar  = String.replicate pct "#" + String.replicate (40-pct) "."
    printf "\r[%s] %d/%d" bar atual total

for i in 0..100 do
    progresso i 100
    System.Threading.Thread.Sleep(20)
printfn "\nConcluído!"

Processos Externos e Shell

Executar comandos externos e capturar saída
open System.Diagnostics

// Helper para executar um comando e capturar stdout+stderr
let executar (cmd: string) (args: string) : int * string =
    let info =
        ProcessStartInfo(
            FileName               = cmd,
            Arguments              = args,
            RedirectStandardOutput = true,
            RedirectStandardError  = true,
            UseShellExecute        = false)
    use proc = new Process(StartInfo = info)
    proc.Start() |> ignore
    let stdout = proc.StandardOutput.ReadToEnd()
    let stderr = proc.StandardError.ReadToEnd()
    proc.WaitForExit()
    proc.ExitCode, stdout + stderr

// Usar
let (code, output) = executar "git" "log --oneline -10"
if code = 0 then
    output.Split('\n')
    |> Array.filter (fun l -> l.Length > 0)
    |> Array.iter (printfn "  %s")
else
    eprintfn "Erro: %s" output

// Listar commits modificados no último dia
let (_, commits) = executar "git" "log --since=1.day --name-only --format=%n"
commits.Split('\n')
|> Array.filter (fun l -> l.EndsWith(".fs"))
|> Array.distinct
|> Array.iter (printfn "Modificado: %s")

Exemplo Completo — Ferramenta de Relatório

report.fsx — ferramenta CLI standalone
#!/usr/bin/env dotnet-script
// report.fsx — executar: dotnet fsi report.fsx --dir . --ext fs

#r "nuget: Argu"
open System.IO
open Argu

type Args =
    | [<AltCommandLine("-d")>] Dir of directory: string
    | [<AltCommandLine("-e")>] Ext of extension: string
    | [<AltCommandLine("-o")>] Output of file: string
    interface IArgParserTemplate with
        member s.Usage =
            match s with
            | Dir _ -> "Diretório raiz"
            | Ext _ -> "Extensão (sem ponto)"
            | Output _ -> "Arquivo de saída"

let parser = ArgumentParser.Create<Args>()
let args   = parser.ParseCommandLine(fsi.CommandLineArgs.[1..])

let raiz = args.GetResult(Dir, defaultValue = ".")
let ext  = args.GetResult(Ext, defaultValue = "fs")
let saida = args.TryGetResult Output

let arquivos =
    Directory.EnumerateFiles(raiz, sprintf "*.%s" ext, SearchOption.AllDirectories)
    |> Seq.toArray

let estatisticas () =
    arquivos |> Array.map (fun arq ->
        let linhas = File.ReadLines(arq) |> Seq.length
        arq, linhas)

let stats = estatisticas()
let total = stats |> Array.sumBy snd
let relatorio =
    stats
    |> Array.sortByDescending snd
    |> Array.map (fun (f, n) -> sprintf "%-60s %5d linhas" f n)
    |> Array.append [| sprintf "=== %d arquivos .%s, %d linhas total ===" arquivos.Length ext total |]

relatorio |> Array.iter (printfn "%s")

match saida with
| Some caminho -> File.WriteAllLines(caminho, relatorio); printfn "Salvo em %s" caminho
| None         -> ()

F# como linguagem de script (.fsx) elimina a dicotomia shell script (frágil, sem tipos) vs aplicativo compilado (verboso, burocrático). Um .fsx tem tipos estáticos, acesso total ao .NET BCL, pacotes NuGet inline com #r, e roda com dotnet fsi arquivo.fsx — sem compilação manual, sem projeto, sem csproj. É o ponto ideal entre a agilidade do bash e a robustez de uma aplicação real.
