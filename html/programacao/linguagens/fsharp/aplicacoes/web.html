F# — Desenvolvimento Web

F# para web combina a solidez do ASP.NET Core com abstrações funcionais de alto nível. Giraffe e Saturn transformam o pipeline do Kestrel em funções compostas; Suave é um servidor HTTP puramente funcional sem dependência do ASP.NET; Fable + Elmish levam F# ao frontend via JavaScript.

No modelo Giraffe/Saturn, uma rota é simplesmente uma função HttpContext -> Task<HttpContext option>. Rotas são compostas com o operador >=> (peixe Kleisli) — cada handler passa o contexto adiante ou curto-circuita retornando None. Todo o roteamento é verificado em tempo de compilação.

Giraffe — ASP.NET Core Funcional

API REST com Giraffe — roteamento, JSON, middleware
// dotnet add package Giraffe
open System
open Microsoft.AspNetCore.Builder
open Microsoft.AspNetCore.Hosting
open Microsoft.Extensions.DependencyInjection
open Giraffe

// Tipos de domínio
type Produto = {
    Id   : int
    Nome : string
    Preco: decimal
}

// Estado simulado
let produtos = [
    { Id = 1; Nome = "Teclado"; Preco = 250m }
    { Id = 2; Nome = "Mouse";   Preco = 120m }
]

// Handlers — HttpHandler = HttpContext -> Task<HttpContext option>
let listarProdutos : HttpHandler =
    fun next ctx ->
        json produtos next ctx

let buscarProduto (id: int) : HttpHandler =
    fun next ctx ->
        match produtos |> List.tryFind (fun p -> p.Id = id) with
        | Some p -> json p next ctx
        | None   -> RequestErrors.notFound (text "Produto não encontrado") next ctx

let criarProduto : HttpHandler =
    fun next ctx ->
        task {
            let! novoProduto = ctx.BindJsonAsync<Produto>()
            // salvar no banco...
            return! Successful.created (json novoProduto) next ctx
        }

// Roteamento com >=> (Kleisli composition)
let webApp =
    choose [
        GET  >=> route  "/produtos"     >=> listarProdutos
        GET  >=> routef "/produtos/%i"  buscarProduto
        POST >=> route  "/produtos"     >=> criarProduto
        RequestErrors.notFound (text "Rota não encontrada")
    ]

// Configuração ASP.NET
let configureApp (app: IApplicationBuilder) =
    app.UseGiraffe(webApp)

let configureServices (services: IServiceCollection) =
    services.AddGiraffe() |> ignore

[<EntryPoint>]
let main _ =
    WebHostBuilder()
        .UseKestrel()
        .Configure(Action<IApplicationBuilder> configureApp)
        .ConfigureServices(configureServices)
        .UseUrls("http://localhost:5000")
        .Build()
        .Run()

Saturn — MVC Funcional sobre Giraffe

Saturn — routers, controllers e pipeline
// dotnet add package Saturn
open Saturn
open Giraffe

type Tarefa = { Id: int; Titulo: string; Feita: bool }

// Controller CRUD
let tarefaController = controller {
    index   (fun ctx -> task { return! json [{ Id=1; Titulo="Estudar F#"; Feita=false }] next ctx })
    show    (fun ctx id -> task {
        return! json { Id = id; Titulo = "Tarefa " + string id; Feita = false } next ctx
    })
    create  (fun ctx -> task {
        let! nova = ctx.BindJsonAsync<Tarefa>()
        return! Successful.created (json nova) next ctx
    })
}

// Router com scope
let apiRouter = router {
    forward "/tarefas" tarefaController
    getf "/ping" (fun _ ctx -> task { return! text "pong" next ctx })
}

let appRouter = router {
    forward "/api" apiRouter
}

// Application — CORS, JSON, pipelines
let app = application {
    url "http://0.0.0.0:5000"
    use_router appRouter
    use_json_serializer (Thoth.Json.Giraffe.ThothSerializer())
    use_cors "CORS" (fun builder ->
        builder.AllowAnyOrigin()
               .AllowAnyMethod()
               .AllowAnyHeader() |> ignore)
    memory_cache
    use_gzip
}

run app

Serialização JSON com F#

System.Text.Json + FSharpLu para DUs e Options
open System.Text.Json
open System.Text.Json.Serialization

// Configurar conversor F# para System.Text.Json
// dotnet add package FSharp.SystemTextJson
open System.Text.Json
open Helpers = FSharp.SystemTextJson

let opts = JsonSerializerOptions()
opts.Converters.Add(JsonFSharpConverter())   // suporta DUs, Options, tuples

type Status = Pendente | EmAndamento | Concluido

type Pedido = {
    Id     : int
    Status : Status
    Tags   : string list
    Notas  : string option
}

let pedido = {
    Id     = 42
    Status = EmAndamento
    Tags   = ["urgente"; "vip"]
    Notas  = Some "Cliente preferencial"
}

let json = JsonSerializer.Serialize(pedido, opts)
// {"Id":42,"Status":"EmAndamento","Tags":["urgente","vip"],"Notas":"Cliente preferencial"}

let pedido2 = JsonSerializer.Deserialize<Pedido>(json, opts)
// DUs e Options corretamente desserializados

// Thoth.Json — codecs manuais mas previsíveis
// dotnet add package Thoth.Json.Net
open Thoth.Json.Net

let encodePedido p =
    Encode.object [
        "id"    , Encode.int    p.Id
        "notas" , Encode.option Encode.string p.Notas
    ]

let decodePedido : Decoder<Pedido> =
    Decode.object (fun get -> {
        Id     = get.Required.Field "id"    Decode.int
        Status = Pendente
        Tags   = []
        Notas  = get.Optional.Field "notas" Decode.string
    })

Suave — HTTP Puro em F#

Servidor HTTP sem ASP.NET
// dotnet add package Suave
open Suave
open Suave.Filters
open Suave.Operators
open Suave.Successful

// WebPart = HttpContext -> Async<HttpContext option>
let app =
    choose [
        GET >=> choose [
            path "/hello" >=> OK "Hello, World!"
            pathScan "/greet/%s" (fun nome -> OK (sprintf "Olá, %s!" nome))
        ]
        POST >=> path "/echo" >=>
            request (fun req ->
                let corpo = System.Text.Encoding.UTF8.GetString(req.rawForm)
                OK (sprintf "Você enviou: %s" corpo))
        RequestErrors.NOT_FOUND "Página não encontrada"
    ]

startWebServer defaultConfig app

O operador >=> (Kleisli Fish) é o coração do Giraffe e Suave. Em vez de registrar middlewares com efeitos colaterais em uma pipeline imperativa, você compõe funções puras que passam contexto adiante ou aborta: autenticar >=> autorizar >=> handler. Se qualquer etapa retornar None, a composição para — exatamente como Option.bind, mas aplicado a requisições HTTP.
