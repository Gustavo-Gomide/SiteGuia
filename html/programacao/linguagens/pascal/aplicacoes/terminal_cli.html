Pascal — Aplicações CLI

Construa ferramentas de linha de comando robustas com Free Pascal: parsing de argumentos, leitura de configuração INI, logs em arquivo e execução de processos externos.

Pascal compila para executáveis nativos pequenos e rápidos — sem runtime externo, sem JVM, sem interpretador. Uma ferramenta CLI em Pascal pode ser distribuída como um único arquivo binário que roda em qualquer máquina com o mesmo SO e arquitetura.

Argumentos da Linha de Comando

ParamCount retorna o número de argumentos passados. ParamStr(0) é o nome do executável. ParamStr(n) retorna o n-ésimo argumento (iniciando em 1).

args.pas — Parsing básico de argumentos
program FerramentaCLI;

uses SysUtils;

procedure ExibirAjuda;
begin
  WriteLn('Uso: ferramenta [opções] arquivo');
  WriteLn('');
  WriteLn('Opções:');
  WriteLn('  -h, --help      Esta ajuda');
  WriteLn('  -v, --verbose   Modo verboso');
  WriteLn('  -o SAIDA        Arquivo de saída');
  WriteLn('  --versao        Exibe versão');
  Halt(0);   { termina o processo com código 0 }
end;

var
  i          : Integer;
  verbose    : Boolean = False;
  arquivoIn  : String = '';
  arquivoOut : String = 'saida.txt';
  arg        : String;

begin
  { Sem argumentos: exibe ajuda }
  if ParamCount = 0 then
    ExibirAjuda;

  { Iterar sobre todos os argumentos }
  i := 1;
  while i <= ParamCount do
  begin
    arg := ParamStr(i);

    if (arg = '-h') or (arg = '--help') then
      ExibirAjuda

    else if (arg = '-v') or (arg = '--verbose') then
      verbose := True

    else if arg = '--versao' then
    begin
      WriteLn('ferramenta v1.0.0');
      Halt(0);
    end

    else if (arg = '-o') and (i < ParamCount) then
    begin
      Inc(i);
      arquivoOut := ParamStr(i);   { argumento seguinte é o valor }
    end

    else if arg[1] <> '-' then
      arquivoIn := arg   { parâmetro posicional }

    else
    begin
      WriteLn(StdErr, 'Erro: opção desconhecida: ', arg);
      Halt(1);   { código de erro 1 = argumento inválido }
    end;

    Inc(i);
  end;

  if arquivoIn = '' then
  begin
    WriteLn(StdErr, 'Erro: arquivo de entrada não especificado');
    Halt(1);
  end;

  if verbose then
    WriteLn('Modo verboso ativo');
  WriteLn('Entrada: ', arquivoIn);
  WriteLn('Saída:   ', arquivoOut);
end.

Arquivos de Configuração INI

A classe TIniFile da unit IniFiles lê e escreve arquivos no formato INI. É a forma mais simples de persistir configuração em Pascal sem dependências externas.

config.ini — Arquivo de configuração de exemplo
[Geral]
versao=1.0
idioma=pt-BR
debug=false

[Banco]
host=localhost
porta=5432
banco=mydb
usuario=admin

[Logs]
nivel=INFO
arquivo=app.log
tamanho_max=10
configuracao.pas — TIniFile: leitura e escrita
unit Configuracao;

{$mode objfpc}{$H+}

interface

uses
  SysUtils, IniFiles;

type
  TConfigBanco = record
    Host    : String;
    Porta   : Integer;
    Banco   : String;
    Usuario : String;
  end;

  TConfig = class
  private
    FIni    : TIniFile;
    FCaminho: String;
  public
    constructor Create(const caminho: String);
    destructor  Destroy; override;

    function  LerString(const secao, chave, padrao: String): String;
    function  LerInt(const secao, chave: String; padrao: Integer): Integer;
    function  LerBool(const secao, chave: String; padrao: Boolean): Boolean;
    procedure GravarString(const secao, chave, valor: String);
    procedure GravarInt(const secao, chave: String; valor: Integer);
    function  LerConfigBanco: TConfigBanco;
    procedure Salvar;
  end;

implementation

constructor TConfig.Create(const caminho: String);
begin
  FCaminho := caminho;
  FIni := TIniFile.Create(caminho);
end;

destructor TConfig.Destroy;
begin
  FIni.Free;
  inherited;
end;

function TConfig.LerString(const secao, chave, padrao: String): String;
begin
  Result := FIni.ReadString(secao, chave, padrao);
end;

function TConfig.LerInt(const secao, chave: String; padrao: Integer): Integer;
begin
  Result := FIni.ReadInteger(secao, chave, padrao);
end;

function TConfig.LerBool(const secao, chave: String; padrao: Boolean): Boolean;
begin
  Result := FIni.ReadBool(secao, chave, padrao);
end;

procedure TConfig.GravarString(const secao, chave, valor: String);
begin
  FIni.WriteString(secao, chave, valor);
end;

procedure TConfig.GravarInt(const secao, chave: String; valor: Integer);
begin
  FIni.WriteInteger(secao, chave, valor);
end;

function TConfig.LerConfigBanco: TConfigBanco;
const
  S = 'Banco';
begin
  Result.Host    := FIni.ReadString(S, 'host',    'localhost');
  Result.Porta   := FIni.ReadInteger(S, 'porta',   5432);
  Result.Banco   := FIni.ReadString(S, 'banco',   'mydb');
  Result.Usuario := FIni.ReadString(S, 'usuario', 'postgres');
end;

procedure TConfig.Salvar;
begin
  FIni.UpdateFile;   { força escrita para disco }
end;

end.

Sistema de Logs em Arquivo

logger.pas — Logger thread-safe com rotação de arquivo
unit Logger;

{$mode objfpc}{$H+}

interface

uses
  SysUtils, Classes, SyncObjs;

type
  TNivelLog = (nlDEBUG, nlINFO, nlWARN, nlERROR, nlFATAL);

  TLogger = class
  private
    FArquivo       : TextFile;
    FNivelMinimo   : TNivelLog;
    FCritical      : TCriticalSection;   { thread-safe }
    FTamanhoMaxKB  : Int64;
    FCaminho       : String;
    FAberto        : Boolean;
    procedure RotacionarSeNecessario;
    function  NivelParaStr(nivel: TNivelLog): String;
  public
    constructor Create(const caminho: String;
                         nivel: TNivelLog = nlINFO;
                         tamanhoMaxKB: Int64 = 10240);
    destructor  Destroy; override;
    procedure Log(nivel: TNivelLog; const msg: String);
    procedure Debug(const msg: String);
    procedure Info(const msg: String);
    procedure Warn(const msg: String);
    procedure Error(const msg: String);
  end;

implementation

function TLogger.NivelParaStr(nivel: TNivelLog): String;
const
  Nomes: array[TNivelLog] of String =
    ('DEBUG', 'INFO ', 'WARN ', 'ERROR', 'FATAL');
begin
  Result := Nomes[nivel];
end;

constructor TLogger.Create(const caminho: String; nivel: TNivelLog; tamanhoMaxKB: Int64);
begin
  FCaminho      := caminho;
  FNivelMinimo  := nivel;
  FTamanhoMaxKB := tamanhoMaxKB;
  FCritical     := TCriticalSection.Create;
  AssignFile(FArquivo, caminho);
  if FileExists(caminho) then
    Append(FArquivo)
  else
    Rewrite(FArquivo);
  FAberto := True;
end;

destructor TLogger.Destroy;
begin
  if FAberto then
    CloseFile(FArquivo);
  FCritical.Free;
  inherited;
end;

procedure TLogger.RotacionarSeNecessario;
var
  backup: String;
begin
  if FileSize(FArquivo) > FTamanhoMaxKB * 1024 then
  begin
    CloseFile(FArquivo);
    backup := ChangeFileExt(FCaminho, '.bak');
    if FileExists(backup) then
      DeleteFile(backup);
    RenameFile(FCaminho, backup);
    AssignFile(FArquivo, FCaminho);
    Rewrite(FArquivo);
  end;
end;

procedure TLogger.Log(nivel: TNivelLog; const msg: String);
var
  linha: String;
begin
  if nivel < FNivelMinimo then Exit;

  linha := Format('%s [%s] %s',
    [FormatDateTime('yyyy-mm-dd hh:nn:ss.zzz', Now),
     NivelParaStr(nivel),
     msg]);

  FCritical.Acquire;
  try
    RotacionarSeNecessario;
    WriteLn(FArquivo, linha);
    Flush(FArquivo);   { força escrita imediata — essencial para logs }
  finally
    FCritical.Release;
  end;

  { Erros também vão para stderr }
  if nivel >= nlERROR then
    WriteLn(StdErr, linha);
end;

procedure TLogger.Debug(const msg: String); begin Log(nlDEBUG, msg); end;
procedure TLogger.Info(const msg: String);  begin Log(nlINFO,  msg); end;
procedure TLogger.Warn(const msg: String);  begin Log(nlWARN,  msg); end;
procedure TLogger.Error(const msg: String); begin Log(nlERROR, msg); end;

end.

Execução de Processos Externos

processos.pas — fpSystem, Process e captura de saída
program Processos;

uses
  SysUtils,
  Process;     { unit de subprocess do FPC }

{ ── Método 1: SysUtils.ExecuteProcess — simples, sem captura de saída ─── }
procedure ExemplosSimples;
var
  codigo: Integer;
begin
  { Executa e retorna código de saída }
  codigo := ExecuteProcess('ls', '-la /tmp');
  WriteLn('Código de saída: ', codigo);

  { Equivalente ao system() do C }
  fpSystem('echo "Olá do subprocesso"');
end;

{ ── Método 2: TProcess — controle total ─────────────────────────────── }
function CapturarSaida(const comando: String; args: array of String): String;
var
  proc    : TProcess;
  lista   : TStringList;
  i       : Integer;
begin
  proc := TProcess.Create(nil);
  lista := TStringList.Create;
  try
    proc.Executable := comando;
    for i := 0 to High(args) do
      proc.Parameters.Add(args[i]);

    proc.Options := [poWaitOnExit, poUsePipes];
    proc.Execute;

    lista.LoadFromStream(proc.Output);
    Result := lista.Text;
  finally
    lista.Free;
    proc.Free;
  end;
end;

{ ── Método 3: subprocess assíncrono com leitura em tempo real ────────── }
procedure ExecutarComProgress(const cmd: String; params: array of String);
var
  proc   : TProcess;
  buffer : array[0..4095] of Byte;
  lidos  : Integer;
  i      : Integer;
begin
  proc := TProcess.Create(nil);
  try
    proc.Executable := cmd;
    for i := 0 to High(params) do
      proc.Parameters.Add(params[i]);

    proc.Options := [poUsePipes];   { sem poWaitOnExit = assíncrono }
    proc.Execute;

    { Ler enquanto o processo roda }
    while proc.Running do
    begin
      lidos := proc.Output.Read(buffer, SizeOf(buffer));
      if lidos > 0 then
        Write(Copy(PChar(@buffer), 1, lidos));
    end;

    { Drenar buffer restante }
    repeat
      lidos := proc.Output.Read(buffer, SizeOf(buffer));
      if lidos > 0 then
        Write(Copy(PChar(@buffer), 1, lidos));
    until lidos = 0;

    WriteLn('Código de saída: ', proc.ExitCode);
  finally
    proc.Free;
  end;
end;

var
  saida: String;

begin
  ExemplosSimples;

  saida := CapturarSaida('uname', ['-a']);
  WriteLn('Sistema: ', Trim(saida));

  WriteLn('--- Listagem ---');
  ExecutarComProgress('ls', ['-la', '/tmp']);
end.

Projeto Completo: Ferramenta de Backup

backup.pas — Ferramenta CLI de backup com config INI e log
program BackupTool;

{$mode objfpc}{$H+}

uses
  SysUtils, IniFiles, Classes, Process;

type
  TBackupConfig = record
    Origem  : String;
    Destino : String;
    Compactar : Boolean;
    LogArquivo : String;
  end;

function CarregarConfig(const iniPath: String): TBackupConfig;
var
  ini: TIniFile;
begin
  ini := TIniFile.Create(iniPath);
  try
    Result.Origem     := ini.ReadString('Backup', 'origem',    '/dados');
    Result.Destino    := ini.ReadString('Backup', 'destino',   '/backup');
    Result.Compactar  := ini.ReadBool('Backup',   'compactar', True);
    Result.LogArquivo := ini.ReadString('Log', 'arquivo', 'backup.log');
  finally
    ini.Free;
  end;
end;

procedure EscreverLog(const logPath, msg: String);
var
  f: TextFile;
begin
  AssignFile(f, logPath);
  if FileExists(logPath) then
    Append(f)
  else
    Rewrite(f);
  try
    WriteLn(f, FormatDateTime('yyyy-mm-dd hh:nn:ss', Now) + ' ' + msg);
    Flush(f);
  finally
    CloseFile(f);
  end;
end;

function ExecutarComando(const cmd: String; args: array of String): Integer;
var
  p: TProcess;
  i: Integer;
begin
  p := TProcess.Create(nil);
  try
    p.Executable := cmd;
    for i := 0 to High(args) do
      p.Parameters.Add(args[i]);
    p.Options := [poWaitOnExit];
    p.Execute;
    Result := p.ExitCode;
  finally
    p.Free;
  end;
end;

var
  cfg     : TBackupConfig;
  destino : String;
  codigo  : Integer;

begin
  if ParamCount < 1 then
  begin
    WriteLn('Uso: backup config.ini');
    Halt(1);
  end;

  if not FileExists(ParamStr(1)) then
  begin
    WriteLn(StdErr, 'Arquivo de configuração não encontrado: ', ParamStr(1));
    Halt(2);
  end;

  cfg := CarregarConfig(ParamStr(1));
  EscreverLog(cfg.LogArquivo, 'Iniciando backup de ' + cfg.Origem);

  destino := IncludeTrailingPathDelimiter(cfg.Destino) +
             FormatDateTime('yyyymmdd_hhnnss', Now);

  if cfg.Compactar then
  begin
    destino := destino + '.tar.gz';
    codigo := ExecutarComando('tar', ['-czf', destino, cfg.Origem]);
  end
  else
  begin
    ForceDirectories(destino);
    codigo := ExecutarComando('cp', ['-r', cfg.Origem, destino]);
  end;

  if codigo = 0 then
  begin
    EscreverLog(cfg.LogArquivo, 'Backup concluído: ' + destino);
    WriteLn('OK: ', destino);
  end
  else
  begin
    EscreverLog(cfg.LogArquivo, 'ERRO: código ' + IntToStr(codigo));
    WriteLn(StdErr, 'Falha! Código: ', codigo);
    Halt(3);
  end;
end.
bash — Compilar e usar a ferramenta
# Compilar com otimização
fpc -O2 -FEbin backup.pas

# Configuração de exemplo: config.ini
cat > config.ini <<EOF
[Backup]
origem=/home/user/documentos
destino=/backup
compactar=true

[Log]
arquivo=backup.log
EOF

# Executar
./backup config.ini

# Saída esperada:
# OK: /backup/20240115_143022.tar.gz

# Ver log
cat backup.log
# 2024-01-15 14:30:21 Iniciando backup de /home/user/documentos
# 2024-01-15 14:30:22 Backup concluído: /backup/20240115_143022.tar.gz
