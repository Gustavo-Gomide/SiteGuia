Pascal — GUI Desktop com Lazarus

Crie aplicações desktop multiplataforma com Lazarus LCL — a alternativa open-source ao Delphi que compila para Linux, Windows e macOS sem alterar o código.

A LCL (Lazarus Component Library) abstrai os widgets nativos de cada plataforma: GTK3/Qt6 no Linux, Win32/Win64 no Windows, Cocoa no macOS. O mesmo arquivo .pas compila em todas — e a aplicação usa os controles nativos de cada SO, respeitando o tema do usuário.

Arquitetura da LCL

A LCL separa a API pública dos widgets (em Pascal) da implementação nativa (widgetsets). Você programa contra a API LCL; o widgetset traduz chamadas para GTK, WinAPI ou Cocoa.

Formulário Básico: TForm e Componentes

Em Lazarus, a maior parte da UI é criada visualmente no designer de formulários. O resultado é serializado em um .lfm. O código Pascal associado fica no .pas. Você só escreve código nos event handlers.

MainForm.lfm — Arquivo de design (gerado pelo Lazarus)
object FormPrincipal: TFormPrincipal
  Caption = 'Cadastro de Pessoas'
  Width = 480
  Height = 400
  Position = poScreenCenter

  object LabelNome: TLabel
    Caption = 'Nome:'
    Left = 12; Top = 16
  end

  object EditNome: TEdit
    Left = 80; Top = 12; Width = 350
    Text = ''
    MaxLength = 100
  end

  object BtnAdicionar: TButton
    Caption = 'Adicionar'
    Left = 12; Top = 50; Width = 100
    OnClick = BtnAdicionarClick
  end

  object ListBoxPessoas: TListBox
    Left = 12; Top = 90; Width = 440; Height = 200
    Sorted = True
  end
end
MainForm.pas — Event handlers e lógica
unit MainForm;

{$mode objfpc}{$H+}

interface

uses
  Classes, SysUtils, Forms, Controls, StdCtrls, Dialogs;

type
  TFormPrincipal = class(TForm)
    LabelNome     : TLabel;
    EditNome      : TEdit;
    BtnAdicionar  : TButton;
    ListBoxPessoas: TListBox;
    BtnRemover    : TButton;
    MemoLog       : TMemo;

    { Event handlers — conectados via .lfm }
    procedure BtnAdicionarClick(Sender: TObject);
    procedure BtnRemoverClick(Sender: TObject);
    procedure ListBoxPessoasDblClick(Sender: TObject);
    procedure FormCreate(Sender: TObject);
    procedure FormDestroy(Sender: TObject);
  private
    procedure Log(const msg: String);
  end;

var
  FormPrincipal: TFormPrincipal;

implementation

{$R *.lfm}   { liga o design do .lfm ao código Pascal }

procedure TFormPrincipal.FormCreate(Sender: TObject);
begin
  Caption := 'Cadastro de Pessoas — Lazarus';
  EditNome.SetFocus;
end;

procedure TFormPrincipal.FormDestroy(Sender: TObject);
begin
  { liberar recursos aqui se necessário }
end;

procedure TFormPrincipal.BtnAdicionarClick(Sender: TObject);
var
  nome: String;
begin
  nome := Trim(EditNome.Text);
  if nome = '' then
  begin
    ShowMessage('Nome não pode ser vazio!');
    EditNome.SetFocus;
    Exit;
  end;

  if ListBoxPessoas.Items.IndexOf(nome) >= 0 then
  begin
    ShowMessage('Pessoa já cadastrada!');
    Exit;
  end;

  ListBoxPessoas.Items.Add(nome);
  Log('Adicionado: ' + nome);
  EditNome.Text := '';
  EditNome.SetFocus;
end;

procedure TFormPrincipal.BtnRemoverClick(Sender: TObject);
var
  idx: Integer;
begin
  idx := ListBoxPessoas.ItemIndex;
  if idx < 0 then
  begin
    ShowMessage('Selecione um item para remover.');
    Exit;
  end;

  if MessageDlg('Confirmar remoção de "' + ListBoxPessoas.Items[idx] + '"?',
                mtConfirmation, [mbYes, mbNo], 0) = mrYes then
  begin
    Log('Removido: ' + ListBoxPessoas.Items[idx]);
    ListBoxPessoas.Items.Delete(idx);
  end;
end;

procedure TFormPrincipal.ListBoxPessoasDblClick(Sender: TObject);
begin
  if ListBoxPessoas.ItemIndex >= 0 then
    ShowMessage('Selecionado: ' + ListBoxPessoas.Items[ListBoxPessoas.ItemIndex]);
end;

procedure TFormPrincipal.Log(const msg: String);
begin
  MemoLog.Lines.Add(FormatDateTime('hh:nn:ss', Now) + ' — ' + msg);
end;

end.

Componentes Avançados

componentes.pas — TStringGrid, TTreeView, TPageControl por código
unit ComponentesAvancados;

{$mode objfpc}{$H+}

interface
uses
  Classes, SysUtils, Forms, Controls, Grids, ComCtrls, StdCtrls;

type
  TFormGrade = class(TForm)
  private
    FGrade  : TStringGrid;
    FArvore : TTreeView;
    FPagina : TPageControl;
    procedure PreencherGrade;
    procedure PreencherArvore;
  public
    constructor Create(AOwner: TComponent); override;
  end;

implementation

constructor TFormGrade.Create(AOwner: TComponent);
var
  tab1, tab2 : TTabSheet;
begin
  inherited Create(AOwner);
  Caption := 'Demo Componentes';
  Width := 700; Height := 500;

  { TPageControl — notebook com abas }
  FPagina := TPageControl.Create(Self);
  FPagina.Parent := Self;
  FPagina.Align := alClient;

  tab1 := FPagina.AddTabSheet;
  tab1.Caption := 'Tabela';

  tab2 := FPagina.AddTabSheet;
  tab2.Caption := 'Árvore';

  { TStringGrid — grade editável }
  FGrade := TStringGrid.Create(Self);
  FGrade.Parent := tab1;
  FGrade.Align := alClient;
  FGrade.RowCount := 6;
  FGrade.ColCount := 4;
  FGrade.FixedRows := 1;   { primeira linha é cabeçalho }
  FGrade.Options := FGrade.Options + [goEditing];  { permite editar }
  PreencherGrade;

  { TTreeView — hierarquia de nós }
  FArvore := TTreeView.Create(Self);
  FArvore.Parent := tab2;
  FArvore.Align := alClient;
  FArvore.ReadOnly := True;
  PreencherArvore;
end;

procedure TFormGrade.PreencherGrade;
type
  TRow = array[0..3] of String;
const
  Dados : array[0..4] of TRow = (
    ('Nome', 'Cargo', 'Salário', 'Ativo'),
    ('Alice',  'Dev Sr',  '12000.00',  'Sim'),
    ('Bob',    'Dev Jr',  '6000.00',   'Sim'),
    ('Carol',  'QA',      '8000.00',   'Não'),
    ('David',  'DevOps',  '10000.00',  'Sim')
  );
var
  row, col: Integer;
begin
  for row := 0 to High(Dados) do
    for col := 0 to 3 do
      FGrade.Cells[col, row] := Dados[row][col];
end;

procedure TFormGrade.PreencherArvore;
var
  raiz, filho: TTreeNode;
begin
  FArvore.Items.Clear;
  raiz := FArvore.Items.Add(nil, 'Linguagens');
    filho := FArvore.Items.AddChild(raiz, 'Pascal');
      FArvore.Items.AddChild(filho, 'Free Pascal');
      FArvore.Items.AddChild(filho, 'Delphi');
    filho := FArvore.Items.AddChild(raiz, 'OCaml');
      FArvore.Items.AddChild(filho, 'Core OCaml');
      FArvore.Items.AddChild(filho, 'Lwt');
  raiz.Expand(True);
end;

end.

Acesso a Banco de Dados — ZeosLib

ZeosLib é uma biblioteca open-source que conecta Free Pascal/Lazarus a PostgreSQL, MySQL, SQLite, Firebird, Oracle e outros. É instalada como pacote Lazarus e expõe componentes TZConnection, TZQuery, TZTable.

BancoDados.pas — ZeosLib com SQLite
unit BancoDados;

{$mode objfpc}{$H+}

interface
uses
  Classes, SysUtils, ZConnection, ZDataset, ZDbcIntfs, DB;

type
  TRepoClientes = class
  private
    FConn  : TZConnection;
    FQuery : TZQuery;
  public
    constructor Create(const dbPath: String);
    destructor  Destroy; override;
    procedure CriarTabela;
    procedure Inserir(const nome, email: String; idade: Integer);
    procedure ListarTodos;
    function  BuscarPorNome(const nome: String): TZQuery;
  end;

implementation

constructor TRepoClientes.Create(const dbPath: String);
begin
  FConn := TZConnection.Create(nil);
  FConn.Protocol := 'sqlite3';
  FConn.Database := dbPath;
  FConn.Connect;

  FQuery := TZQuery.Create(nil);
  FQuery.Connection := FConn;
end;

destructor TRepoClientes.Destroy;
begin
  FQuery.Free;
  FConn.Disconnect;
  FConn.Free;
  inherited;
end;

procedure TRepoClientes.CriarTabela;
begin
  FQuery.SQL.Text :=
    'CREATE TABLE IF NOT EXISTS clientes (' +
    '  id    INTEGER PRIMARY KEY AUTOINCREMENT,' +
    '  nome  TEXT    NOT NULL,' +
    '  email TEXT    UNIQUE NOT NULL,' +
    '  idade INTEGER' +
    ')';
  FQuery.ExecSQL;
end;

procedure TRepoClientes.Inserir(const nome, email: String; idade: Integer);
begin
  FQuery.SQL.Text :=
    'INSERT INTO clientes (nome, email, idade) VALUES (:nome, :email, :idade)';
  { Parâmetros — nunca concatenar strings SQL diretamente! }
  FQuery.ParamByName('nome').AsString  := nome;
  FQuery.ParamByName('email').AsString := email;
  FQuery.ParamByName('idade').AsInteger := idade;
  FQuery.ExecSQL;
end;

procedure TRepoClientes.ListarTodos;
begin
  FQuery.SQL.Text := 'SELECT * FROM clientes ORDER BY nome';
  FQuery.Open;
  while not FQuery.EOF do
  begin
    WriteLn(Format('%3d  %-20s  %-30s  %3d', [
      FQuery.FieldByName('id').AsInteger,
      FQuery.FieldByName('nome').AsString,
      FQuery.FieldByName('email').AsString,
      FQuery.FieldByName('idade').AsInteger
    ]));
    FQuery.Next;
  end;
  FQuery.Close;
end;

function TRepoClientes.BuscarPorNome(const nome: String): TZQuery;
begin
  FQuery.SQL.Text := 'SELECT * FROM clientes WHERE nome LIKE :padrao';
  FQuery.ParamByName('padrao').AsString := '%' + nome + '%';
  FQuery.Open;
  Result := FQuery;
end;

end.

Publicação Multiplataforma

bash — Compilar para múltiplas plataformas com Lazarus CLI
# Compilar projeto Lazarus (.lpi) para a plataforma atual
lazbuild --build-all MeuSistema.lpi

# Compilar para Windows 64-bit (de Linux)
lazbuild --os=win64 --cpu=x86_64 --build-all MeuSistema.lpi

# Compilar para macOS (requer SDK disponível)
lazbuild --os=darwin --cpu=x86_64 --build-all MeuSistema.lpi

# Compilar para ARM Linux (Raspberry Pi 32-bit)
lazbuild --os=linux --cpu=arm --build-all MeuSistema.lpi

# Otimizado para release (sem debug)
lazbuild --build-mode=Release --build-all MeuSistema.lpi

# Verificar dependências em runtime (Linux)
ldd ./MeuSistema

# Strip binary — reduz tamanho
strip ./MeuSistema
