Pascal — Fundamentos

Tipos, variáveis, estruturas de controle, procedures, functions, arrays e records — os blocos estruturais de qualquer programa Pascal.

Pascal foi deliberadamente projetado para que os conceitos fundamentais de programação — declaração de variáveis, controle de fluxo, modularização — fossem expressos de forma clara e não ambígua. Aprender esses fundamentos em Pascal cria uma base sólida para qualquer linguagem imperativa posterior.

Tipos Básicos

Pascal tem tipagem estática forte. Cada variável deve ser declarada com seu tipo na seção var antes do begin. O compilador rejeita atribuições entre tipos incompatíveis.

tipos.pas — Tipos primitivos e operações
program TiposBasicos;

uses SysUtils;

var
  { ── Inteiros ─────────────────────────────────────────────── }
  i8  : ShortInt;    { -128 a 127 (8 bits) }
  i16 : SmallInt;   { -32768 a 32767 (16 bits) }
  i32 : Integer;    { -2^31 a 2^31-1 (32 bits) — mais comum }
  i64 : Int64;      { 64 bits com sinal }
  u8  : Byte;       { 0 a 255 (sem sinal) }
  u32 : Cardinal;   { 0 a 2^32-1 (sem sinal) }
  nat : NativeInt;  { 32 ou 64 bits conforme a plataforma }

  { ── Reais ────────────────────────────────────────────────── }
  r32 : Single;     { 32 bits, ~7 dígitos }
  r64 : Double;     { 64 bits, ~15 dígitos — mais comum }
  r80 : Extended;   { 80 bits, ~18 dígitos (x87 FPU) }
  re  : Real;       { alias para Double no FPC moderno }
  cur : Currency;  { valor monetário (4 casas decimais fixas) }

  { ── Booleano ─────────────────────────────────────────────── }
  ativo  : Boolean;
  b8     : ByteBool;   { 1 byte, compatível com C }
  b32    : LongBool;   { 4 bytes, compatível com Win32 }

  { ── Char e String ────────────────────────────────────────── }
  letra  : Char;
  texto  : String;        { AnsiString dinâmica (FPC) }
  curto  : String[50];    { ShortString — máx 50 chars }
  wide   : WideString;    { Unicode UTF-16 }

begin
  { Atribuição e operações com inteiros }
  i32 := 100;
  i32 := i32 * 2 + 5;         { 205 }
  i32 := i32 div 3;           { divisão inteira: 68 }
  i32 := i32 mod 10;          { resto: 8 }
  WriteLn('i32 = ', i32);

  { Operações com reais }
  r64 := 3.14159265;
  r64 := Sqrt(2.0);           { raiz quadrada }
  r64 := Power(2.0, 10.0);    { 2^10 = 1024.0 }
  r64 := Abs(-3.7);           { valor absoluto }
  WriteLn('r64 = ', r64:10:4);

  { Conversões explícitas }
  i32 := Round(3.7);      { arredonda para 4 }
  i32 := Trunc(3.9);      { trunca para 3 }
  r64 := i32;              { inteiro para real: implícito }
  i32 := Ord('A');        { char para código ASCII: 65 }
  letra := Chr(65);        { código para char: 'A' }

  { Booleano }
  ativo := True;
  ativo := (5 > 3) and not (2 = 3);  { True }
  WriteLn('ativo = ', ativo);

  { String }
  texto := 'Olá, ';
  texto := texto + 'Pascal!';          { concatenação com + }
  WriteLn('Texto: ', texto);
  WriteLn('Tamanho: ', Length(texto));   { 13 }
  WriteLn('Maiúsculo: ', UpperCase(texto));
  WriteLn('Pos de "Pascal": ', Pos('Pascal', texto));  { 7 }
end.

Estruturas de Controle

controle.pas — if-then-else, case-of, loops
program ControleFluxo;

var
  n, i, soma : Integer;
  c : Char;
  continuar : Boolean;

begin
  { ── if-then-else ──────────────────────────────────────── }
  n := 7;
  if n > 0 then
    WriteLn('positivo')
  else if n < 0 then
    WriteLn('negativo')
  else
    WriteLn('zero');

  { begin/end para múltiplas instruções no ramo }
  if n mod 2 = 0 then
  begin
    WriteLn('par');
    WriteLn('n/2 = ', n div 2);
  end
  else
    WriteLn('ímpar');

  { ── case-of — switch tipado ──────────────────────────── }
  c := 'A';
  case c of
    'A'..'Z' : WriteLn('Letra maiúscula');
    'a'..'z' : WriteLn('Letra minúscula');
    '0'..'9' : WriteLn('Dígito');
    ' ', ',', '.' : WriteLn('Pontuação ou espaço');
  else
    WriteLn('Outro caractere');
  end;

  { case com inteiro }
  case n of
    1       : WriteLn('um');
    2, 3    : WriteLn('dois ou três');
    4..10   : WriteLn('quatro a dez');
  else
    WriteLn('outro');
  end;

  { ── for-to-do ────────────────────────────────────────── }
  soma := 0;
  for i := 1 to 100 do
    soma := soma + i;
  WriteLn('Soma 1..100 = ', soma);   { 5050 }

  { for-downto: decrementar }
  for i := 10 downto 1 do
    Write(i, ' ');
  WriteLn;

  { ── while-do ─────────────────────────────────────────── }
  i := 1;
  soma := 0;
  while i <= 100 do
  begin
    soma := soma + i;
    Inc(i);   { equivale a i := i + 1 }
  end;

  { ── repeat-until — executa ao menos uma vez ──────────── }
  i := 0;
  repeat
    Inc(i);
    WriteLn('Iteração: ', i);
  until i >= 3;   { para quando true (oposto do while) }

  { break e continue (extensão Free Pascal) }
  for i := 1 to 10 do
  begin
    if i = 5 then Break;      { sai do loop }
    if i mod 2 = 0 then Continue; { próxima iteração }
    Write(i, ' ');
  end;
end.

Procedures e Functions

Pascal distingue procedure (sem valor de retorno, equivalente ao void de C) de function (retorna um valor). O retorno da função é atribuído ao nome dela ou à variável especial Result.

subprogramas.pas — Procedures, functions e parâmetros
program Subprogramas;

{ ── Procedure: sem valor de retorno ──────────────────────────────── }
procedure Saudar(nome: String);
begin
  WriteLn('Olá, ', nome, '!');
end;

{ ── Function: retorna valor ──────────────────────────────────────── }
function Quadrado(x: Integer): Integer;
begin
  Result := x * x;   { ou: Quadrado := x * x; }
end;

{ ── Parâmetros var: passagem por referência ──────────────────────── }
procedure Trocar(var a, b: Integer);
var
  temp: Integer;
begin
  temp := a;
  a := b;
  b := temp;
end;

{ ── Parâmetros const: por valor, somente leitura ─────────────────── }
function TamanhoString(const s: String): Integer;
begin
  Result := Length(s);
end;

{ ── Parâmetros out: saída explícita ──────────────────────────────── }
procedure DivResto(a, b: Integer; out quoc, resto: Integer);
begin
  quoc  := a div b;
  resto := a mod b;
end;

{ ── Recursão ─────────────────────────────────────────────────────── }
function Fatorial(n: Integer): Int64;
begin
  if n <= 1 then
    Result := 1
  else
    Result := n * Fatorial(n - 1);
end;

{ ── Funções com múltiplas saídas via var ──────────────────────────── }
procedure MinMax(const arr: array of Integer;
                   var minVal, maxVal: Integer);
var
  i: Integer;
begin
  minVal := arr[0];
  maxVal := arr[0];
  for i := 1 to High(arr) do
  begin
    if arr[i] < minVal then minVal := arr[i];
    if arr[i] > maxVal then maxVal := arr[i];
  end;
end;

var
  x, y, q, r : Integer;
  dados : array of Integer;
  minV, maxV : Integer;

begin
  Saudar('Pascal');
  WriteLn('5² = ', Quadrado(5));
  WriteLn('10! = ', Fatorial(10));

  x := 10; y := 20;
  Trocar(x, y);
  WriteLn('x=', x, ' y=', y);   { x=20 y=10 }

  DivResto(17, 5, q, r);
  WriteLn('17 div 5: quoc=', q, ' resto=', r);

  dados := [3, 1, 4, 1, 5, 9, 2, 6];
  MinMax(dados, minV, maxV);
  WriteLn('Min=', minV, ' Max=', maxV);
end.

Arrays e Strings

arrays.pas — Arrays estáticos, dinâmicos e strings
program Arrays;

type
  TIntArray   = array[1..10]  of Integer;    { 1-indexed }
  TMatriz3x3  = array[0..2, 0..2] of Double;  { bidimensional }
  TCharArray  = array['A'..'Z'] of Integer;   { índice char }

var
  fixo   : TIntArray;
  mat    : TMatriz3x3;
  contas : TCharArray;
  dyn    : array of Integer;   { array dinâmico (heap) }
  i, j, n : Integer;

begin
  { Array estático: inicialização }
  for i := 1 to 10 do
    fixo[i] := i * i;             { quadrados: 1, 4, 9, ... }
  WriteLn('fixo[5] = ', fixo[5]);  { 25 }
  WriteLn('Low = ',  Low(fixo));   { 1 }
  WriteLn('High = ', High(fixo));  { 10 }

  { Matriz bidimensional }
  for i := 0 to 2 do
    for j := 0 to 2 do
      mat[i][j] := if i = j then 1.0 else 0.0;  { matriz identidade }
  { Nota: condicional inline não existe em Pascal padrão;
    use if-then-else separado }

  { Array dinâmico }
  n := 5;
  SetLength(dyn, n);              { aloca memória }
  for i := 0 to High(dyn) do   { índice 0..n-1 }
    dyn[i] := i * 10;
  WriteLn('Length(dyn) = ', Length(dyn));  { 5 }

  SetLength(dyn, 10);   { redimensionar (preserva dados) }

  { Inicialização literal (FPC 3.0+) }
  dyn := [10, 20, 30, 40, 50];

  { Funções de array úteis (SysUtils/Math) }
  WriteLn(MinIntValue(dyn));    { menor valor }
  WriteLn(MaxIntValue(dyn));    { maior valor }
  WriteLn(SumInt(dyn));         { soma de todos }
end.

Records — Estruturas de Dados

O tipo record agrupa campos de tipos diferentes sob um único nome. É o equivalente a struct em C. Em Object Pascal, records podem ter métodos e propriedades (extensão do FPC/Delphi).

records.pas — Records simples e aninhados
program Records;

uses SysUtils;

type
  { Record básico }
  TData = record
    Dia  : 1..31;     { subtipo com range }
    Mes  : 1..12;
    Ano  : Word;
  end;

  TEndereco = record
    Rua    : String[100];
    Numero : Integer;
    Cidade : String[50];
    CEP    : String[9];
  end;

  TPessoa = record
    Nome       : String;
    CPF        : String[14];
    DataNasc   : TData;       { record aninhado }
    Endereco   : TEndereco;
    Ativo      : Boolean;
  end;

  TListaPessoas = array of TPessoa;

function CriarPessoa(const nomePessoa, cpf: String; dia, mes, ano: Integer): TPessoa;
begin
  Result.Nome := nomePessoa;
  Result.CPF  := cpf;
  Result.DataNasc.Dia := dia;
  Result.DataNasc.Mes := mes;
  Result.DataNasc.Ano := ano;
  Result.Ativo := True;
end;

procedure ExibirPessoa(const p: TPessoa);
begin
  WriteLn('Nome: ', p.Nome);
  WriteLn('CPF:  ', p.CPF);
  WriteLn('Nasc: ', p.DataNasc.Dia, '/', p.DataNasc.Mes, '/', p.DataNasc.Ano);
  WriteLn('Ativo: ', p.Ativo);
end;

var
  alice, bob : TPessoa;
  pessoas    : TListaPessoas;
  i          : Integer;

begin
  alice := CriarPessoa('Alice Silva', '123.456.789-00', 15, 6, 1990);
  ExibirPessoa(alice);

  { with — acessa campos sem repetir o nome da variável }
  with bob do
  begin
    Nome            := 'Bob Souza';
    CPF             := '987.654.321-00';
    DataNasc.Dia    := 20;
    DataNasc.Mes    := 12;
    DataNasc.Ano    := 1985;
    Endereco.Cidade := 'São Paulo';
    Ativo           := True;
  end;

  pessoas := [alice, bob];
  WriteLn('--- Lista de Pessoas ---');
  for i := 0 to High(pessoas) do
  begin
    WriteLn('Pessoa ', i + 1, ':');
    ExibirPessoa(pessoas[i]);
  end;
end.
