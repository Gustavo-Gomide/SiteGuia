Pascal — Avançado

Object Pascal moderno: OOP completo com herança, polimorfismo e interfaces, gerenciamento manual de memória, generics, exceções e manipulação de arquivos binários.

Object Pascal (o dialeto usado pelo Delphi e Free Pascal) é uma das implementações mais completas de OOP disponíveis. Classes, interfaces, delegação, generics e RTTI coexistem com o acesso de baixo nível que torna Pascal adequado para sistemas embarcados e software crítico.

Ponteiros e Gerenciamento Manual de Memória

Pascal suporta ponteiros explícitos (^Tipo). New aloca no heap, Dispose libera. O símbolo ^ serve tanto para declarar tipo ponteiro quanto para desreferenciar.

ponteiros.pas — Alocação manual e listas encadeadas
program Ponteiros;

type
  { Ponteiro para inteiro }
  PInt = ^Integer;

  { Nó de lista encadeada }
  PNo = ^TNo;
  TNo = record
    Valor   : Integer;
    Proximo : PNo;
  end;

procedure InserirFrente(var lista: PNo; valor: Integer);
var
  novo: PNo;
begin
  New(novo);         { aloca TNo no heap }
  novo^.Valor   := valor;
  novo^.Proximo := lista;
  lista := novo;   { novo nó é a nova cabeça }
end;

procedure ImprimirLista(lista: PNo);
var
  atual: PNo;
begin
  atual := lista;
  while atual <> nil do
  begin
    Write(atual^.Valor, ' ');
    atual := atual^.Proximo;
  end;
  WriteLn;
end;

procedure LiberarLista(var lista: PNo);
var
  atual, proximo: PNo;
begin
  atual := lista;
  while atual <> nil do
  begin
    proximo := atual^.Proximo;
    Dispose(atual);   { libera memória do nó }
    atual := proximo;
  end;
  lista := nil;
end;

var
  lista    : PNo;
  pInteiro : PInt;

begin
  { Uso básico de ponteiro }
  New(pInteiro);
  pInteiro^ := 42;    { ^ desreferencia }
  WriteLn('Valor: ', pInteiro^);
  Dispose(pInteiro);
  pInteiro := nil;   { boa prática: nil após dispose }

  { Lista encadeada }
  lista := nil;
  InserirFrente(lista, 10);
  InserirFrente(lista, 20);
  InserirFrente(lista, 30);
  ImprimirLista(lista);    { 30 20 10 }
  LiberarLista(lista);
end.

Object Pascal — OOP Completo

Object Pascal estende Pascal com class. Diferente do record, uma class é alocada no heap por padrão. O construtor (constructor Create) inicializa, o destrutor (destructor Destroy) libera recursos. O compilador não gerencia a vida do objeto — você é responsável por chamar .Free.

oop_hierarquia.pas — Classes, herança e polimorfismo
program OOP;

uses SysUtils;

type
  { ── Classe base abstrata ─────────────────────────────────── }
  TAnimal = class
  private
    FNome  : String;
    FIdade : Integer;
  protected
    function GetDescricao: String; virtual; abstract;
  public
    constructor Create(const nome: String; idade: Integer);
    destructor Destroy; override;

    procedure Falar; virtual; abstract;
    procedure ExibirInfo;

    property Nome  : String  read FNome;
    property Idade : Integer read FIdade;
  end;

  { ── Subclasse Cachorro ──────────────────────────────────── }
  TCachorro = class(TAnimal)
  private
    FRaca : String;
  protected
    function GetDescricao: String; override;
  public
    constructor Create(const nome, raca: String; idade: Integer);
    procedure Falar; override;
    property Raca: String read FRaca;
  end;

  { ── Subclasse Gato ─────────────────────────────────────── }
  TGato = class(TAnimal)
  protected
    function GetDescricao: String; override;
  public
    procedure Falar; override;
  end;

{ ── Implementações ────────────────────────────────────────── }
constructor TAnimal.Create(const nome: String; idade: Integer);
begin
  inherited Create;   { chama TObject.Create }
  FNome  := nome;
  FIdade := idade;
end;

destructor TAnimal.Destroy;
begin
  WriteLn('Destruindo ', FNome);
  inherited Destroy;   { sempre chamar inherited }
end;

procedure TAnimal.ExibirInfo;
begin
  WriteLn(GetDescricao, ' | Idade: ', FIdade, ' anos');
end;

constructor TCachorro.Create(const nome, raca: String; idade: Integer);
begin
  inherited Create(nome, idade);
  FRaca := raca;
end;

function TCachorro.GetDescricao: String;
begin
  Result := Format('Cachorro: %s (%s)', [FNome, FRaca]);
end;

procedure TCachorro.Falar;
begin
  WriteLn(FNome, ': Au au!');
end;

function TGato.GetDescricao: String;
begin
  Result := 'Gato: ' + FNome;
end;

procedure TGato.Falar;
begin
  WriteLn(FNome, ': Miau!');
end;

var
  animais : array of TAnimal;
  i       : Integer;

begin
  { Polimorfismo — array de TAnimal com objetos de subclasses }
  SetLength(animais, 3);
  animais[0] := TCachorro.Create('Rex', 'Labrador', 3);
  animais[1] := TGato.Create('Mimi', 2);
  animais[2] := TCachorro.Create('Bolt', 'Dálmata', 5);

  for i := 0 to High(animais) do
  begin
    animais[i].ExibirInfo;
    animais[i].Falar;
    animais[i].Free;   { liberar memória }
  end;
end.

Interfaces

Interfaces em Object Pascal são contratos puros — sem implementação, sem estado. Uma classe pode implementar múltiplas interfaces (Pascal não tem herança múltipla de classe). Use as para conversão e is para checagem de tipo.

interfaces.pas — Contratos e implementações múltiplas
program Interfaces;

uses SysUtils;

type
  { Interfaces — inicia com I por convenção }
  ISerializable = interface
    function ToJSON: String;
    procedure FromJSON(const json: String);
  end;

  IComparavel = interface
    function ComparaCom(outro: IComparavel): Integer;
  end;

  { Classe implementa AMBAS as interfaces }
  TProduto = class(TObject, ISerializable, IComparavel)
  private
    FNome  : String;
    FPreco : Double;
  public
    constructor Create(const nome: String; preco: Double);
    { ISerializable }
    function  ToJSON: String;
    procedure FromJSON(const json: String);
    { IComparavel }
    function ComparaCom(outro: IComparavel): Integer;
    property Nome  : String  read FNome;
    property Preco : Double  read FPreco;
  end;

constructor TProduto.Create(const nome: String; preco: Double);
begin
  FNome  := nome;
  FPreco := preco;
end;

function TProduto.ToJSON: String;
begin
  Result := Format('{"nome":"%s","preco":%.2f}', [FNome, FPreco]);
end;

procedure TProduto.FromJSON(const json: String);
begin
  { implementação simplificada }
  WriteLn('Deserializando: ', json);
end;

function TProduto.ComparaCom(outro: IComparavel): Integer;
var
  outroProd: TProduto;
begin
  outroProd := outro as TProduto;   { cast de interface para classe }
  if FPreco < outroProd.Preco then Result := -1
  else if FPreco > outroProd.Preco then Result := 1
  else Result := 0;
end;

procedure SerializarQualquer(obj: ISerializable);
begin
  WriteLn('JSON: ', obj.ToJSON);
end;

var
  p1, p2 : TProduto;

begin
  p1 := TProduto.Create('Teclado', 299.90);
  p2 := TProduto.Create('Mouse', 149.90);

  SerializarQualquer(p1);
  SerializarQualquer(p2);

  if p1.ComparaCom(p2) > 0 then
    WriteLn(p1.Nome, ' é mais caro que ', p2.Nome);

  { verify: is operador verifica se objeto implementa interface }
  if p1 is TProduto then
    WriteLn('p1 é um TProduto');

  p1.Free;
  p2.Free;
end.

Generics — Tipos Paramétricos (FPC 2.2+)

generics.pas — Classes e procedures genéricas
program Generics;

uses SysUtils, Generics.Collections;

type
  { Pilha genérica }
  TPilha<T> = class
  private
    FDados : array of T;
    FTopo  : Integer;
  public
    constructor Create(capacidade: Integer = 16);
    procedure Empilhar(const valor: T);
    function  Desempilhar: T;
    function  Topo: T;
    function  EstaVazia: Boolean;
    property  Tamanho: Integer read FTopo;
  end;

constructor TPilha<T>.Create(capacidade: Integer);
begin
  SetLength(FDados, capacidade);
  FTopo := 0;
end;

procedure TPilha<T>.Empilhar(const valor: T);
begin
  if FTopo >= Length(FDados) then
    SetLength(FDados, Length(FDados) * 2);
  FDados[FTopo] := valor;
  Inc(FTopo);
end;

function TPilha<T>.Desempilhar: T;
begin
  if EstaVazia then
    raise Exception.Create('Pilha vazia!');
  Dec(FTopo);
  Result := FDados[FTopo];
end;

function  TPilha<T>.Topo: T; begin Result := FDados[FTopo - 1]; end;
function  TPilha<T>.EstaVazia: Boolean; begin Result := FTopo = 0; end;

{ Usando TDictionary de Generics.Collections (biblioteca padrão FPC) }
var
  pilhaInt  : TPilha<Integer>;
  pilhaStr  : TPilha<String>;
  dici      : TDictionary<String, Integer>;
  lista     : TList<Double>;

begin
  pilhaInt := TPilha<Integer>.Create;
  pilhaInt.Empilhar(10);
  pilhaInt.Empilhar(20);
  pilhaInt.Empilhar(30);
  WriteLn('Topo: ', pilhaInt.Desempilhar);   { 30 }

  dici := TDictionary<String, Integer>.Create;
  dici.Add('alpha', 1);
  dici.Add('beta',  2);
  WriteLn('alpha = ', dici['alpha']);

  pilhaInt.Free;
  dici.Free;
end.

Exceções — try-except-finally

excecoes.pas — Tratamento e criação de exceções
program Excecoes;

uses SysUtils;

type
  { Exceção customizada }
  EValidacao = class(Exception)
  private
    FCampo : String;
  public
    constructor Create(const campo, mensagem: String);
    property Campo: String read FCampo;
  end;

constructor EValidacao.Create(const campo, mensagem: String);
begin
  inherited Create(mensagem);
  FCampo := campo;
end;

procedure ValidarIdade(idade: Integer);
begin
  if idade < 0 then
    raise EValidacao.Create('idade', Format('Idade %d inválida', [idade]));
  if idade > 150 then
    raise EValidacao.Create('idade', 'Idade superior a 150 não é válida');
end;

function AbrirArquivo(const caminho: String): TextFile;
var
  f : TextFile;
begin
  AssignFile(f, caminho);
  Reset(f);    { raise EInOutError se arquivo não existe }
  Result := f;
end;

procedure ProcessarComRecursos;
var
  obj : TObject;
begin
  obj := TObject.Create;
  try
    { código que pode lançar exceção }
    ValidarIdade(-5);
    WriteLn('Nunca chega aqui');
  except
    on e: EValidacao do
      WriteLn('Erro no campo ', e.Campo, ': ', e.Message);
    on e: EInOutError do
      WriteLn('Erro de I/O: ', e.Message);
    on e: Exception do
      WriteLn('Erro genérico: ', e.Message);
  end;

  try
    { finally SEMPRE executa — mesmo com exceção }
    ValidarIdade(200);
  finally
    obj.Free;   { liberado mesmo se ValidarIdade lançar }
    WriteLn('Recurso liberado');
  end;
end;

begin
  ProcessarComRecursos;
end.

Streams e Arquivos Binários

arquivos.pas — TFileStream e serialização binária
program ArquivosBinarios;

uses SysUtils, Classes;

type
  TRegistro = packed record   { packed: sem padding }
    ID    : Integer;
    Nome  : array[0..49] of AnsiChar;   { tamanho fixo }
    Valor : Double;
  end;

procedure EscreverRegistros(const caminho: String);
var
  fs  : TFileStream;
  reg : TRegistro;
  i   : Integer;
begin
  fs := TFileStream.Create(caminho, fmCreate);
  try
    for i := 1 to 5 do
    begin
      FillChar(reg, SizeOf(reg), 0);     { zerar memória }
      reg.ID    := i;
      reg.Valor := i * 1.5;
      StrPCopy(reg.Nome, Format('Item %d', [i]));
      fs.WriteBuffer(reg, SizeOf(reg));  { escreve exatamente SizeOf bytes }
    end;
  finally
    fs.Free;
  end;
end;

procedure LerRegistros(const caminho: String);
var
  fs  : TFileStream;
  reg : TRegistro;
begin
  fs := TFileStream.Create(caminho, fmOpenRead);
  try
    while fs.Position < fs.Size do
    begin
      fs.ReadBuffer(reg, SizeOf(reg));
      WriteLn(Format('ID=%d Nome=%-10s Valor=%.1f',
        [reg.ID, reg.Nome, reg.Valor]));
    end;
  finally
    fs.Free;
  end;
end;

begin
  EscreverRegistros('dados.bin');
  LerRegistros('dados.bin');
end.
