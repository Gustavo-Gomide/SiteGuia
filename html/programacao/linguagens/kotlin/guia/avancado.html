Kotlin — Tópicos Avançados

Nesta etapa, Kotlin deixa de ser apenas sintaxe amigável e vira ferramenta de arquitetura: concorrência estruturada com coroutines, modelagem de domínios fechados com sealed classes e APIs expressivas com extension functions e DSLs.

Coroutines e Concorrência Estruturada

suspend, launch, async e cancellation
import kotlinx.coroutines.*

suspend fun buscarUsuario(id: Int): String {
		delay(200)
		return "usuario-$id"
}

fun main() = runBlocking {
		val job = launch {
				repeat(5) { i ->
						println("tick $i")
						delay(100)
				}
		}

		val perfil = async { buscarUsuario(42) }
		println("Perfil: ${perfil.await()}")

		delay(250)
		job.cancelAndJoin()
		println("cancelado com seguranca")
}
Flow para streams assíncronos
import kotlinx.coroutines.*
import kotlinx.coroutines.flow.*

fun eventos(): Flow<Int> = flow {
		for (i in 1..5) {
				emit(i)
				delay(100)
		}
}

fun main() = runBlocking {
		eventos()
				.map { it * 10 }
				.filter { it >= 20 }
				.collect { println(it) }
}

Sealed Classes e Modelagem de Estado

Hierarquias fechadas e when exaustivo
sealed interface Resultado<out T> {
		data class Sucesso<T>(val dado: T) : Resultado<T>
		data class Erro(val mensagem: String) : Resultado<Nothing>
		object Carregando : Resultado<Nothing>
}

fun render(r: Resultado<String>): String = when (r) {
		is Resultado.Sucesso -> "OK: ${r.dado}"
		is Resultado.Erro -> "ERRO: ${r.mensagem}"
		Resultado.Carregando -> "Aguarde..."
}

// when não precisa de else quando a sealed está completa.

Generics, Inline e Reified

APIs genéricas mais seguras e eficientes
inline fun <reified T> Any?.asOrNull(): T? = this as? T

fun main() {
		val x: Any = "abc"
		val s: String? = x.asOrNull<String>()
		val n: Int? = x.asOrNull<Int>()

		println(s)
		println(n)
}

// reified permite acessar o tipo em runtime dentro de inline.

inline fun medir(nome: String, bloco: () -> Unit) {
		val ini = System.nanoTime()
		bloco()
		val fim = System.nanoTime()
		println("$nome levou ${(fim - ini) / 1_000_000.0} ms")
}

Extension Functions e DSL Builders

Criando APIs expressivas em Kotlin
data class Query(
		var tabela: String = "",
		var filtro: String = "",
)

class QueryBuilder {
		private val q = Query()
		fun from(t: String) { q.tabela = t }
		fun where(f: String) { q.filtro = f }
		fun build() = q
}

fun query(block: QueryBuilder.() -> Unit): Query {
		val b = QueryBuilder()
		b.block()
		return b.build()
}

fun String.slug() = lowercase().replace(" ", "-")

fun main() {
		val q = query {
				from("usuarios")
				where("ativo = true")
		}

		println(q)
		println("Kotlin Avancado".slug())
}
