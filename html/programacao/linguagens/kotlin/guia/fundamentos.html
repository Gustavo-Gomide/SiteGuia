Kotlin — Fundamentos

Fundamentos de Kotlin

Os pilares de Kotlin incluem tipagem forte com inferência, null safety, funções de primeira classe, orientação a objetos moderna e APIs funcionais sobre coleções.

Variáveis e Tipos

val, var, inferência e tipos primitivos
fun main() {
		val nome: String = "Kotlin"      // imutável
		var contador: Int = 0             // mutável

		val idade = 30                    // inferido como Int
		val altura = 1.75                 // Double
		val ativo = true                  // Boolean

		contador += 1
		println("$nome: contador=$contador")
}

Funções e Controle de Fluxo

fun, if, when, loops
fun soma(a: Int, b: Int): Int {
		return a + b
}

fun dobro(x: Int) = x * 2             // sintaxe curta

fun classificar(nota: Int): String = when {
		nota >= 90 -> "A"
		nota >= 80 -> "B"
		nota >= 70 -> "C"
		else -> "D"
}

fun main() {
		for (i in 1..5) println(i)

		var x = 3
		while (x > 0) {
				println("x=$x")
				x--
		}

		println(classificar(87))
}

Null Safety

Tipos anuláveis, safe call e Elvis operator
fun tamanhoSeguro(nome: String?): Int {
		return nome?.length ?: 0
}

fun main() {
		val a: String = "Ana"
		val b: String? = null

		println(a.length)                 // sempre seguro
		println(b?.length)                // Int?
		println(tamanhoSeguro(b))

		val texto = b ?: "valor padrao"  // Elvis operator
		println(texto)
}

Classes, Data Classes e Herança

Modelagem orientada a objetos em Kotlin
open class Pessoa(val nome: String) {
		open fun apresentar() = "Ola, eu sou $nome"
}

class Aluno(nome: String, val curso: String) : Pessoa(nome) {
		override fun apresentar() = "Sou $nome e estudo $curso"
}

data class Produto(
		val id: Int,
		val nome: String,
		val preco: Double,
)

fun main() {
		val aluno = Aluno("Carlos", "Computacao")
		println(aluno.apresentar())

		val p1 = Produto(1, "Teclado", 199.9)
		val p2 = p1.copy(preco = 179.9)
		println(p1)
		println(p2)
}

Coleções e Lambdas

map, filter, reduce e imutabilidade prática
fun main() {
		val numeros = listOf(1, 2, 3, 4, 5, 6)

		val pares = numeros.filter { it % 2 == 0 }
		val quadrados = numeros.map { it * it }
		val soma = numeros.reduce { acc, n -> acc + n }

		println(pares)
		println(quadrados)
		println(soma)

		val nomes = listOf("ana", "bruno", "carla")
		val nomesFormatados = nomes
				.map { it.replaceFirstChar(Char::uppercase) }
				.sorted()

		println(nomesFormatados)
}
