C# — Fundamentos

Fundamentos do C#

Os pilares do C#: sistema de tipos rico com value types e reference types, OOP com classes e interfaces, delegates e eventos, generics type-safe, colecoes e o poderoso LINQ. Dominar esses fundamentos e obrigatorio para trabalhar com qualquer parte do ecossistema .NET.

C# tem um sistema de tipos em dois mundos: value types (struct, int, bool) vivem na stack e semantica por valor; reference types (class, interface, record) vivem no heap e semantica por referencia. Compreender essa distincao e fundamental para performance e comportamento correto.

Sistema de Tipos

C# — tipos primitivos, inferencia, nullable, pattern matching
// Tipos primitivos (value types — alias do System.*)
int    i  = 42;          // System.Int32 (32-bit signed)
long   l  = 100_000L;    // separador _ para legibilidade
double d  = 3.14;
float  f  = 1.5f;
decimal m = 19.99m;     // monetario — sem erro float
bool   b  = true;
char   c  = 'A';
string s  = "texto";    // reference type mas imutavel

// Inferencia com var (tipo resolvido em compile-time)
var x = 10;              // int
var lista = new List<string>();

// Tipos nulaveis (value types)
int? opcao = null;
int  valor = opcao ?? -1;        // null-coalescing
opcao ??= 42;                    // atribuir se null
if (opcao.HasValue) { var v = opcao.Value; }

// string: verbatim, interpolado e combinado
string caminho = @"C:\Users\João\docs";  // verbatim @ ignora escapes
string msg     = $"Total: {10 * 2:N0}";    // interpolado com formatacao
string raw     = """
  Texto com "aspas" e \backslash
  sem precisar escapar
  """;  // raw string literal (C# 11+)

// Type checking e casting
object obj = "ola";
if (obj is string texto)        // is-pattern (C# 7+)
  Console.WriteLine(texto.ToUpper());

var comprimento = obj switch {
  string str when str.Length > 5 => "longa",
  string str                    => "curta",
  null                          => "nulo",
  _                             => "outro tipo"
};

Colecoes e Generics

C# — List, Dictionary, HashSet, Queue, Stack e LINQ
using System.Collections.Generic;
using System.Linq;

// List<T> — array dinâmico (O(1) amortizado append)
var nomes = new List<string> { "Ana", "Bia", "Carlos" };
nomes.Add("Diego");
nomes.RemoveAt(0);
int idx = nomes.IndexOf("Bia");

// Collection expressions (C# 12+)
List<int> nums = [1, 2, 3, 4, 5];  // sintaxe compacta
int[] arr  = [..nums, 6, 7];      // spread operator ..

// Dictionary<TKey, TValue>
var pop = new Dictionary<string, int> {
  ["Brasil"] = 215_000_000,
  ["EUA"]    = 330_000_000
};
pop["India"] = 1_400_000_000;
if (pop.TryGetValue("Brasil", out var n)) Console.WriteLine(n);

// HashSet — unicidade O(1) lookup
var tags = new HashSet<string> { "cs", "dotnet", "api" };
tags.Add("cs");           // ignorado (ja existe)
bool tem = tags.Contains("api");

// ====== LINQ ======
var pessoas = new[] {
  new { Nome = "Ana",    Idade = 30, Cidade = "SP" },
  new { Nome = "Carlos", Idade = 25, Cidade = "RJ" },
  new { Nome = "Bia",    Idade = 35, Cidade = "SP" },
};

// Sintaxe de metodos (fluent)
var adultosSP = pessoas
  .Where(p => p.Cidade == "SP" && p.Idade > 20)
  .OrderBy(p => p.Nome)
  .Select(p => new { p.Nome, Desc = $"{p.Cidade}/{p.Idade}" })
  .ToList();

// Sintaxe de query (SQL-like)
var queryResult =
  from p in pessoas
  where p.Cidade == "SP"
  orderby p.Idade descending
  select p.Nome;

// Aggregacoes
int    total  = pessoas.Sum(p => p.Idade);
double media  = pessoas.Average(p => p.Idade);
var    grupoC = pessoas
  .GroupBy(p => p.Cidade)
  .ToDictionary(g => g.Key, g => g.Count());

OOP — Classes, Structs, Interfaces e Records

C# — classes com heranca, interfaces, records e pattern matching
// Interface com metodos default (C# 8+)
interface IForma
{
  double Area();
  double Perimetro();
  string Descricao() => $"Area={Area():F2}, Perimetro={Perimetro():F2}";
}

// Classe base abstrata
abstract class Forma : IForma
{
  public string Cor { get; init; } = "branco";  // init-only property
  public abstract double Area();
  public abstract double Perimetro();
  public override string ToString() => $"{GetType().Name}(cor={Cor})";
}

// Subclasse com primary constructor (C# 12+)
class Circulo(double raio, string cor = "azul") : Forma
{
  public double Raio { get; } = raio;
  public override string Cor { get; init; } = cor;
  public override double Area()      => Math.PI * Raio * Raio;
  public override double Perimetro() => 2 * Math.PI * Raio;
}

class Retangulo(double largura, double altura) : Forma
{
  public double Largura { get; } = largura;
  public double Altura  { get; } = altura;
  public override double Area()      => Largura * Altura;
  public override double Perimetro() => 2 * (Largura + Altura);
}

// Record — value semantics, imutavel, Equals/GetHashCode/ToString automaticos
record Ponto(double X, double Y)
{
  public double DistanciaOrigem() => Math.Sqrt(X * X + Y * Y);
}

// Pattern matching com switch expression
static string Classificar(IForma forma) => forma switch {
  Circulo { Raio: > 10 } => "circulo grande",
  Circulo c               => $"circulo r={c.Raio:F1}",
  Retangulo r when r.Largura == r.Altura => "quadrado",
  Retangulo _             => "retangulo",
  _                        => "forma desconhecida"
};

// Uso — com-expression em record (cria copia com campo alterado)
Ponto p1 = new(3, 4);
Ponto p2 = p1 with { X = 0 };      // nova instancia { X=0, Y=4 }
bool iguais = p1 == new Ponto(3, 4); // true — structural equality

Delegates, Eventos e Lambdas

C# — delegate, Action, Func, EventHandler e expressoes lambda
// Delegate — tipo que representa referencias a metodos
delegate double Operacao(double a, double b);

Operacao soma   = (double a, double b) => a + b;
Operacao mult   = (a, b) => a * b;       // tipos inferidos
double   result = soma(3, 4);            // 7.0

// Func<T, TResult> e Action<T> — delegates genericos nativos
Func<int, int, int> pow     = (int b, int e) => (int)Math.Pow(b, e);
Func<string, bool>    vazio  = str => string.IsNullOrEmpty(str);
Action<string>         log   = msg => Console.WriteLine($"[LOG] {msg}");
Predicate<int>         par   = n => n % 2 == 0;

// Multicast: delegates combinados com +=
Action pipeline = () => Console.WriteLine("Passo 1");
pipeline        += () => Console.WriteLine("Passo 2");
pipeline        += () => Console.WriteLine("Passo 3");
pipeline();   // executa todos os passos em sequencia

// Eventos — encapsulam delegate com restrict add/remove
class Botao
{
  public event EventHandler<EventArgs>? Clicado;
  public virtual void OnClicado() => Clicado?.Invoke(this, EventArgs.Empty);
}

class EventArgs : System.EventArgs { public string Mensagem { get; init; } = ""; }

var botao = new Botao();
botao.Clicado += (sender, e) => log("Botao clicado!");
botao.OnClicado();  // dispara os handlers

// LINQ com lambdas compostas
var numeros = Enumerable.Range(1, 100);
Func<int, bool> ePrimoCandidato = n => n > 1 && Enumerable.Range(2, (int)Math.Sqrt(n) - 1)
                                       .All(d => n % d != 0);
var primos = numeros.Where(ePrimoCandidato).ToList();

Controle de Fluxo e Iteradores

C# — switch, for, foreach, yield return e exceções
// switch expression com pattern matching (C# 8+)
static string DescDia(DayOfWeek dia) => dia switch {
  DayOfWeek.Saturday or DayOfWeek.Sunday => "final de semana",
  DayOfWeek.Monday                         => "segunda, dificil",
  DayOfWeek.Friday                         => "sexta, quase la",
  _                                         => "dia util"
};

// for, foreach, while, do-while
for (int i = 0; i < 10; i++) { if (i == 5) continue; }

int[] dados = { 1, 2, 3 };
foreach (var item in dados)
  Console.Write($"{item} ");

// Iterador com yield — gera sequencia lazy sem alocar tudo
static IEnumerable<int> Fibonacci(int max) {
  (int a, int b) = (0, 1);
  while (a <= max) {
    yield return a;
    (a, b) = (b, a + b);
  }
}
foreach (var n in Fibonacci(100))
  Console.Write($"{n} ");

// Excecoes — try/catch/when/finally
try {
  int resultado = dividir(10, 0);
}
catch (DivideByZeroException ex) when (ex.Message.Contains("zero")) {
  Console.WriteLine("Divisao por zero!");
}
catch (Exception ex) {
  Console.WriteLine($"Erro inesperado: {ex.Message}");
  throw;  // re-throw preservando stack trace
}
finally {
  Console.WriteLine("Bloco finally sempre executado");
}

// Excecao customizada
class ValidacaoException(string campo, string mensagem)
    : Exception($"Campo '{campo}': {mensagem}")
{
  public string Campo { get; } = campo;
}

// using e IDisposable — liberacao deterministica de recursos
using var stream = File.OpenRead("dados.txt");  // Dispose() no fim do escopo
using var reader = new StreamReader(stream);
string? linha;
while ((linha = reader.ReadLine()) != null)
  Console.WriteLine(linha);
