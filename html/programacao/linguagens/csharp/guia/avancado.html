async / await e Task

C# — async/await, Task, ValueTask, IAsyncEnumerable e CancellationToken
using System.Threading;
using System.Threading.Tasks;
using System.Net.Http;

// Task — operacao assincrona I/O-bound
static async Task<string> BuscarDadosAsync(string url, CancellationToken ct)
{
  using var http = new HttpClient();
  var resposta = await http.GetAsync(url, ct);
  resposta.EnsureSuccessStatusCode();
  return await resposta.Content.ReadAsStringAsync(ct);
}

// Aguardar multiplas tasks em paralelo
static async Task BuscarParaleloAsync()
{
  using var cts = new CancellationTokenSource(TimeSpan.FromSeconds(10));
  Task<string>[] tarefas = [
    BuscarDadosAsync("https://api.exemplo.com/users",  cts.Token),
    BuscarDadosAsync("https://api.exemplo.com/orders", cts.Token),
  ];
  string[] resultados = await Task.WhenAll(tarefas);
  Console.WriteLine($"Recebidos: {string.Join(", ", resultados.Select(r => r.Length))}");
}

// ValueTask — evita alocar Task quando resultado ja disponivel (hot path)
static async ValueTask<int> LerDoCache(int id)
{
  if (_cache.TryGetValue(id, out var val)) return val;  // sincrono, sem Task
  return await LerDoBancoDeDadosAsync(id);
}

// IAsyncEnumerable — stream assincrono de dados (backpressure)
static async IAsyncEnumerable<LogEntry> StreamLogsAsync(
    Stream stream,
    [EnumeratorCancellation] CancellationToken ct = default)
{
  using var reader = new StreamReader(stream);
  while (!reader.EndOfStream)
  {
    ct.ThrowIfCancellationRequested();
    var linha = await reader.ReadLineAsync(ct);
    if (linha is not null)
      yield return LogEntry.Parse(linha);
  }
}

// Consumir IAsyncEnumerable
await foreach (var entry in StreamLogsAsync(stream).WithCancellation(ct))
  Console.WriteLine(entry);

// ConfigureAwait — evitar capturar SynchronizationContext em bibliotecas
static async Task<byte[]> LerArquivoAsync(string path)
{
  return await File.ReadAllBytesAsync(path).ConfigureAwait(false);
}

Span<T>, Memory<T> e Zero-Copy

C# — Span<T>, ReadOnlySpan, MemoryMarshal e processamento de buffer sem alocacao
using System.Runtime.InteropServices;

// Span — view tipo fatia (slice) sem copiar memoria
int[] array = { 1, 2, 3, 4, 5, 6 };
Span<int> fatia = array.AsSpan(2, 3);   // { 3, 4, 5 } — sem alocacao
fatia[0] = 99;                         // modifica o array original

// Processar string sem alocar substrings
static int ContarPalavras(ReadOnlySpan<char> texto)
{
  int count = 0;
  bool emPalavra = false;
  foreach (char c in texto) {
    if (char.IsWhiteSpace(c)) { emPalavra = false; }
    else if (!emPalavra)      { emPalavra = true; count++; }
  }
  return count;
}
int palavras = ContarPalavras("ola mundo como vai");  // 4, sem alocs

// Parsing numerico sem alocar string
ReadOnlySpan<char> numStr = "3.14159".AsSpan();
double pi = double.Parse(numStr);

// MemoryMarshal — interpretar bytes como structs sem copiar
struct PacoteUDP { public int Seq; public short Tipo; public short Len; }

byte[] buffer = new byte[1024];
ref PacoteUDP header = ref MemoryMarshal.AsRef<PacoteUDP>(buffer);
Console.WriteLine($"Seq={header.Seq} Tipo={header.Tipo}");

// stackalloc — alocar na stack (sem GC pressure)
Span<byte> chave = stackalloc byte[32];  // seguro — Span impede escape
System.Security.Cryptography.RandomNumberGenerator.Fill(chave);

// ArrayPool — reutilizar arrays grandes sem pressionar GC
var pool = System.Buffers.ArrayPool<byte>.Shared;
byte[] alugado = pool.Rent(4096);
try {
  // usar alugado...
} finally {
  pool.Return(alugado, clearArray: true);
}

Generics Avancados e Constraints

C# — constraints, variancia, static abstract members e generic math
using System.Numerics;

// Constraints — restringir T para garantir operacoes seguras
static T Max<T>(T a, T b) where T : IComparable<T>
  => a.CompareTo(b) >= 0 ? a : b;

static T CriarESalvar<T>() where T : new(), IDisposable
{
  var obj = new T();     // new() permite construtor default
  // ...
  return obj;
}

// Generic math (C# 11+) — static abstract members em interfaces
static T Somar<T>(IEnumerable<T> items) where T : IAdditionOperators<T, T, T>, new()
{
  T acc = new() { }; // ou T.Zero se T : IAdditiveIdentity
  foreach (var item in items) acc += item;
  return acc;
}

// Variancia — covariance e contravariance
IEnumerable<string> strings = ["a", "b"];
IEnumerable<object> objects = strings;  // IEnumerable<out T> e covariante

Action<object> logObj = o => Console.WriteLine(o);
Action<string> logStr = logObj;         // Action<in T> e contravariante

// Interface com static abstract — tipo de retorno calculado
interface IParseable<TSelf> where TSelf : IParseable<TSelf>
{
  static abstract TSelf Parse(string input);
}

record struct Temperatura(double Celsius) : IParseable<Temperatura>
{
  public static Temperatura Parse(string s) => new(double.Parse(s));
  public double Fahrenheit => Celsius * 9 / 5 + 32;
}

static T ParseGenerico<T>(string s) where T : IParseable<T> => T.Parse(s);
Temperatura t = ParseGenerico<Temperatura>("100");

Source Generators e Reflection

C# — atributos, Reflection em runtime e Source Generator em compile-time
using System.Reflection;

// Atributo customizado
[AttributeUsage(AttributeTargets.Property)]
class ValidarAttribute(int min, int max) : Attribute
{
  public int Min { get; } = min;
  public int Max { get; } = max;
}

class Produto
{
  [Validar(1, 100)]
  public int Quantidade { get; set; }
  [Validar(0, 10000)]
  public decimal Preco { get; set; } = 0m;
}

// Validacao via Reflection
static IEnumerable<string> Validar<T>(T obj)
{
  foreach (var prop in typeof(T).GetProperties())
  {
    var attr = prop.GetCustomAttribute<ValidarAttribute>();
    if (attr == null) continue;
    var valor = (double)Convert.ChangeType(prop.GetValue(obj)!, typeof(double));
    if (valor < attr.Min || valor > attr.Max)
      yield return $"{prop.Name} fora do intervalo [{attr.Min}, {attr.Max}]: {valor}";
  }
}

// Expression trees — inspecionar codigo como dado
using System.Linq.Expressions;

Expression<Func<int, bool>> expr = x => x > 5 && x < 20;
var body = (BinaryExpression)expr.Body;
Console.WriteLine($"Operador: {body.NodeType}");  // AndAlso
var compilada = expr.Compile();
bool ok = compilada(10);  // true

P/Invoke e Interop com Codigo Nativo

C# — P/Invoke, unsafe, fixed e LibraryImport (C# 11+)
using System.Runtime.InteropServices;

// P/Invoke classico — chamar DLL nativa
static class Libc
{
  [DllImport("libc", EntryPoint = "getpid")]
  public static extern int GetPid();

  [DllImport("libc", CharSet = CharSet.Ansi)]
  public static unsafe extern int write(int fd, byte* buf, nint count);
}

// LibraryImport — API moderna com source gen, sem boxing (C# 11+)
static partial class Native
{
  [LibraryImport("libcalc.so", EntryPoint = "add_doubles")]
  public static partial double Somar(double a, double b);

  [LibraryImport("libcalc.so")]
  // MarshalAs para controlar marshaling de string
  [return: MarshalAs(UnmanagedType.LPStr)]
  public static partial string Versao();
}

// unsafe — ponteiros e aritmética de ponteiro
unsafe static void ProcessarBuffer(byte* ptr, int len)
{
  for (int i = 0; i < len; i++)
    *(ptr + i) ^= 0xFF;  // inverter bits
}

// fixed — fixar objeto gerenciado para ponteiro nao se mover durante GC
byte[] dados = { 1, 2, 3 };
unsafe {
  fixed (byte* p = dados) {
    ProcessarBuffer(p, dados.Length);
  }
}

// Struct com layout explícito para interop com C (ex: cabecalho de arquivo)
[StructLayout(LayoutKind.Sequential, Pack = 1)]
struct CabecalhoBMP
{
  public ushort Tipo;     // 0x4D42 = "BM"
  public uint   Tamanho;
  public uint   Reservado;
  public uint   Offset;
}
