Bibliotecas e Frameworks C#

O ecossistema NuGet conta com mais de 300 mil pacotes. As bibliotecas mais usadas incluem ASP.NET Core para web, Entity Framework Core e Dapper para banco de dados, MediatR para CQRS, FluentValidation para validacao, Serilog para logging estruturado, SignalR para WebSockets e Polly para resiliencia.

ASP.NET Core — Web e API

C# + ASP.NET Core — Minimal API, controllers, middleware e DI
using Microsoft.AspNetCore.Builder;
using Microsoft.Extensions.DependencyInjection;

var builder = WebApplication.CreateBuilder(args);

// Registrar servicos no container de DI
builder.Services.AddSingleton<ICache, MemoryCache>();
builder.Services.AddScoped<IProdutoRepo, ProdutoRepository>();
builder.Services.AddTransient<IEmailService, SmtpEmailService>();
builder.Services.AddDbContext<AppDbContext>(opt =>
  opt.UseNpgsql(builder.Configuration.GetConnectionString("Postgres")));
builder.Services.AddControllers();
builder.Services.AddEndpointsApiExplorer();
builder.Services.AddSwaggerGen();

var app = builder.Build();

// Middleware pipeline (ordem importa)
app.UseHttpsRedirection();
app.UseAuthentication();
app.UseAuthorization();
app.UseSwagger(); app.UseSwaggerUI();

// Minimal API — endpoints sem controller
app.MapGet("/produtos/{id:int}", async (int id, IProdutoRepo repo) =>
{
  var produto = await repo.FindAsync(id);
  return produto is null
    ? Results.NotFound(new { Mensagem = "Produto nao encontrado" })
    : Results.Ok(produto);
}).WithName("GetProduto").WithTags("Produtos");

app.MapPost("/produtos", async (CriarProdutoDto dto, IProdutoRepo repo) =>
{
  var id = await repo.CriarAsync(dto);
  return Results.CreatedAtRoute("GetProduto", new { id }, dto);
});

app.MapControllers();
app.Run();

// Controller tradicional com atributos
[ApiController, Route("api/[controller]")]
class PedidosController(IPedidoService svc) : ControllerBase
{
  [HttpGet("{id:guid}")]
  [ProducesResponseType<PedidoDto>(200)]
  [ProducesResponseType(404)]
  public async Task<IActionResult> Get(Guid id) {
    var pedido = await svc.BuscarAsync(id);
    return pedido == null ? NotFound() : Ok(pedido);
  }
}

Entity Framework Core e Dapper

C# + EF Core — migrations, queries LINQ e Dapper para queries otimizadas
using Microsoft.EntityFrameworkCore;
using Dapper;

// Entidades
class Produto
{
  public int     Id      { get; set; }
  public string  Nome    { get; set; } = "";
  public decimal Preco   { get; set; }
  public int     CatId  { get; set; }
  public Categoria? Categoria { get; set; }
}

// DbContext com configuracao Fluent API
class AppDbContext : DbContext
{
  public AppDbContext(DbContextOptions<AppDbContext> opts) : base(opts) {}
  public DbSet<Produto> Produtos => Set<Produto>();

  protected override void OnModelCreating(ModelBuilder mb)
  {
    mb.Entity<Produto>(p => {
      p.HasKey(x => x.Id);
      p.Property(x => x.Nome).HasMaxLength(200).IsRequired();
      p.Property(x => x.Preco).HasPrecision(18, 2);
      p.HasOne(x => x.Categoria)
       .WithMany(c => c.Produtos)
       .HasForeignKey(x => x.CatId)
       .OnDelete(DeleteBehavior.Restrict);
      p.HasIndex(x => x.Nome).IsUnique();
    });
  }
}

// Repositorio com EF Core
class ProdutoRepository(AppDbContext db)
{
  public async Task<List<Produto>> BuscarBaratosAsync(decimal limite) =>
    await db.Produtos
      .Include(p => p.Categoria)
      .Where(p => p.Preco < limite)
      .OrderBy(p => p.Preco)
      .AsNoTracking()    // read-only: sem track de mudancas
      .ToListAsync();

  public async Task SalvarAsync(Produto p) {
    db.Produtos.Add(p);
    await db.SaveChangesAsync();
  }
}

// Dapper — micro-ORM para queries SQL diretas
class ProdutoDapperRepo(IDbConnection conn)
{
  // Parametrizado — previne SQL injection
  public async Task<IEnumerable<Produto>> BuscarPorCategoriaAsync(int catId) =>
    await conn.QueryAsync<Produto>(
      "SELECT p.*, c.Nome AS CategoriaNome FROM Produtos p JOIN Categorias c ON c.Id = p.CatId WHERE p.CatId = @CatId",
      new { CatId = catId });

  public async Task<int> InserirAsync(Produto p) =>
    await conn.ExecuteScalarAsync<int>(
      "INSERT INTO Produtos(Nome, Preco, CatId) VALUES(@Nome, @Preco, @CatId); SELECT LAST_INSERT_ID();",
      p);
}

MediatR, FluentValidation e Polly

C# + MediatR — CQRS com Commands, Queries e Handlers
using MediatR;
using FluentValidation;

// === MediatR — CQRS pattern ===
// Command
record CriarPedidoCommand(Guid ClienteId, List<ItemDto> Itens) : IRequest<Guid>;

// Handler
class CriarPedidoHandler(AppDbContext db, IEventBus bus)
    : IRequestHandler<CriarPedidoCommand, Guid>
{
  public async Task<Guid> Handle(CriarPedidoCommand req, CancellationToken ct)
  {
    var pedido = new Pedido { ClienteId = req.ClienteId };
    foreach (var item in req.Itens)
      pedido.AdicionarItem(item.ProdutoId, item.Qtd);
    db.Pedidos.Add(pedido);
    await db.SaveChangesAsync(ct);
    await bus.PublicarAsync(new PedidoCriadoEvent(pedido.Id), ct);
    return pedido.Id;
  }
}

// Pipeline Behavior — cross-cutting (log, validacao, retry)
class ValidationBehavior<TReq, TResp>(IEnumerable<IValidator<TReq>> validators)
    : IPipelineBehavior<TReq, TResp> where TReq : notnull
{
  public async Task<TResp> Handle(TReq req, RequestHandlerDelegate<TResp> next, CancellationToken ct)
  {
    var erros = validators
      .SelectMany(v => v.Validate(req).Errors)
      .Where(e => e != null)
      .ToList();
    if (erros.Count > 0)
      throw new ValidationException(erros);
    return await next();
  }
}

// FluentValidation
class CriarPedidoValidator : AbstractValidator<CriarPedidoCommand>
{
  public CriarPedidoValidator() {
    RuleFor(x => x.ClienteId).NotEmpty();
    RuleFor(x => x.Itens).NotEmpty().WithMessage("Pedido precisa de ao menos 1 item");
    RuleForEach(x => x.Itens)
      .ChildRules(item => {
        item.RuleFor(x => x.Qtd).GreaterThan(0);
        item.RuleFor(x => x.ProdutoId).NotEmpty();
      });
  }
}

// Polly — resiliencia: retry, circuit breaker, timeout
using Polly;
using Polly.Retry;

var retryPolicy = new ResiliencePipelineBuilder<string>()
  .AddRetry(new RetryStrategyOptions<string> {
    MaxRetryAttempts   = 3,
    Delay              = TimeSpan.FromSeconds(1),
    BackoffType        = DelayBackoffType.Exponential,
    ShouldHandle       = new PredicateBuilder<string>().Handle<HttpRequestException>(),
    OnRetry            = args => { Console.WriteLine($"Retry {args.AttemptNumber}"); return ValueTask.CompletedTask; }
  })
  .AddTimeout(TimeSpan.FromSeconds(5))
  .Build();
var resp = await retryPolicy.ExecuteAsync(ct => http.GetStringAsync("https://api.exemplo.com", ct));

SignalR — WebSockets e Tempo Real

C# + SignalR — hub de notificacoes e cliente JavaScript
// Hub — servidor
using Microsoft.AspNetCore.SignalR;

class NotificacoesHub : Hub
{
  public async Task EnviarMensagem(string grupo, string msg) =>
    await Clients.Group(grupo).SendAsync("ReceberMensagem", Context.User?.Identity?.Name, msg);

  public async Task EntrarGrupo(string grupo) =>
    await Groups.AddToGroupAsync(Context.ConnectionId, grupo);

  public override async Task OnConnectedAsync() {
    await Clients.Caller.SendAsync("Welcome", $"Conectado: {Context.ConnectionId}");
    await base.OnConnectedAsync();
  }
}

// Registrar no Program.cs
builder.Services.AddSignalR();
app.MapHub<NotificacoesHub>("/hubs/notificacoes");

// Enviar do servidor para clientes via IHubContext (fora do Hub)
class PedidoService(IHubContext<NotificacoesHub> hub)
{
  public async Task ProcessarPedidoAsync(Pedido p, CancellationToken ct) {
    // ... processar ...
    await hub.Clients.Group($"cliente-{p.ClienteId}")
      .SendAsync("PedidoAtualizado", new { p.Id, Status = "Processado" }, ct);
  }
}
