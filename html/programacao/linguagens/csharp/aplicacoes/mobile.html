C# para Mobile

.NET MAUI (Multi-platform App UI) e a plataforma oficial Microsoft para desenvolvimento mobile com C# — um unico codebase compila para iOS, Android, macOS e Windows com acesso a APIs nativas de cada plataforma. Usa XAML para UI declarativa, CommunityToolkit.Mvvm para MVVM, e Essentials para permissoes, camera, GPS, armazenamento seguro e notificacoes.

Em .NET MAUI, o codigo C# e compilado para nativo via AOT (Ahead-of-Time) no iOS e via JIT no Android. O acesso a APIs nativas especificas de plataforma e feito via #if ANDROID / #elif IOS ou via abstraccao de interface + implementacao por plataforma.

// MauiProgram.cs — ponto de entrada e DI
using CommunityToolkit.Maui;

public static class MauiProgram
{
  public static MauiApp CreateMauiApp()
  {
    var builder = MauiApp.CreateBuilder();
    builder
      .UseMauiApp<App>()
      .UseMauiCommunityToolkit()
      .ConfigureFonts(fonts => {
        fonts.AddFont("Inter-Regular.ttf", "InterRegular");
        fonts.AddFont("Inter-Bold.ttf", "InterBold");
      });

    // Services
    builder.Services
      .AddSingleton<IContatosService, ContatosService>()
      .AddSingleton<ILocalDb, SqliteDb>()
      .AddTransient<ContatosViewModel>()
      .AddTransient<ContatosPage>()
      .AddTransient<DetalheContatoPage>();

    return builder.Build();
  }
}

C# + .NET MAUI Essentials — GPS, camera, secure storage e notificacoes
using Microsoft.Maui.ApplicationModel;
using Microsoft.Maui.Devices.Sensors;
using Microsoft.Maui.Media;
using Microsoft.Maui.Storage;

// 1. GPS — solicitar permissao e obter localizacao
async Task<Location?> ObterLocalizacaoAsync()
{
  var status = await Permissions.RequestAsync<Permissions.LocationWhenInUse>();
  if (status != PermissionStatus.Granted)
  {
    await Shell.Current.DisplayAlert("Permissao", "GPS nao autorizado.", "OK");
    return null;
  }
  var req = new GeolocationRequest(GeolocationAccuracy.Medium, TimeSpan.FromSeconds(10));
  return await Geolocation.GetLocationAsync(req);
}

// 2. Camera / galeria — capturar foto
async Task<FileResult?> CapturarFotoAsync()
{
  if (!MediaPicker.Default.IsCaptureSupported) return null;
  return await MediaPicker.Default.CapturePhotoAsync(new MediaPickerOptions {
    Title = "Tire uma foto do local"
  });
}

// 3. Secure Storage — salvar tokens com seguranca (Keychain iOS / Keystore Android)
await SecureStorage.Default.SetAsync("access_token", token);
var tkn = await SecureStorage.Default.GetAsync("access_token");

// 4. Preferences — configuracoes simples (nao sigilosas)
Preferences.Default.Set("tema", "escuro");
var tema = Preferences.Default.Get("tema", "claro");

// 5. Connectivity — verificar conexao de rede
if (Connectivity.Current.NetworkAccess != NetworkAccess.Internet)
  await Shell.Current.DisplayAlert("Offline", "Sem conexao com a internet.", "OK");

SQLite — Banco de Dados Local

C# + SQLite-net-pcl — CRUD local persistido no dispositivo
using SQLite;

[Table("contatos")]
public class Contato
{
  [PrimaryKey, AutoIncrement]
  public int    Id     { get; set; }
  [MaxLength(200), NotNull]
  public string Nome   { get; set; } = "";
  public string Email  { get; set; } = "";
  [Indexed]
  public string Telefone { get; set; } = "";
}

class SqliteDb : ILocalDb
{
  private readonly SQLiteAsyncConnection _conn;

  public SqliteDb()
  {
    var dbPath = Path.Combine(FileSystem.AppDataDirectory, "app.db3");
    _conn = new SQLiteAsyncConnection(dbPath,
      SQLiteOpenFlags.ReadWrite | SQLiteOpenFlags.Create | SQLiteOpenFlags.FullMutex);
  }

  public async Task InicializarAsync() =>
    await _conn.CreateTableAsync<Contato>();

  public async Task<List<Contato>> ListarAsync() =>
    await _conn.Table<Contato>().ToListAsync();

  public async Task<List<Contato>> BuscarAsync(string termo) =>
    await _conn.Table<Contato>()
               .Where(c => c.Nome.Contains(termo))
               .ToListAsync();

  public async Task<int> SalvarAsync(Contato c) =>
    c.Id != 0 ? await _conn.UpdateAsync(c) : await _conn.InsertAsync(c);

  public async Task<int> RemoverAsync(Contato c) =>
    await _conn.DeleteAsync(c);
}

Codigo Especifico por Plataforma

C# + .NET MAUI — interface abstrata com implementacoes por plataforma
// Interface compartilhada — Platforms/
public interface IVibracao
{
  void Vibrar(int duracaoMs);
}

// Platforms/Android/VibracaoAndroid.cs
public class VibracaoAndroid : IVibracao
{
  public void Vibrar(int ms)
  {
    var vibrator = Android.App.Application.Context
      .GetSystemService(Android.Content.Context.VibratorService) as Android.OS.Vibrator;
    if (Android.OS.Build.VERSION.SdkInt >= Android.OS.BuildVersionCodes.O)
      vibrator?.Vibrate(Android.OS.VibrationEffect
        .CreateOneShot(ms, Android.OS.VibrationEffect.DefaultAmplitude));
    else
      vibrator?.Vibrate(ms);
  }
}

// Platforms/iOS/VibracaoIos.cs
public class VibracaoIos : IVibracao
{
  public void Vibrar(int _)
    => AudioToolbox.SystemSound.Vibrate.PlayAlertSound();
}

// Registro condicional em MauiProgram.cs
// Abordagem alternativa com #if
builder.Services.AddSingleton<IVibracao>(
#if ANDROID
  new VibracaoAndroid()
#elif IOS
  new VibracaoIos()
#else
  new VibracaoNoop()
#endif
);
