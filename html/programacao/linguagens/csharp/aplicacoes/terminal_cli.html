C# — Terminal e CLI

C# para Terminal e CLI

C# e .NET permitem construir ferramentas de linha de comando (CLI) de alta qualidade com dois paradigmas complementares: System.CommandLine (library oficial Microsoft) para parsing robusto de argumentos, subcomandos e --help automatico; e Spectre.Console para renderizacao rica no terminal com tabelas, progress bars, paineis, live display, prompt interativo e markup colorido.

CLIs em .NET compilados com dotnet publish -r linux-x64 --self-contained -p:PublishSingleFile=true produzem um unico executavel nativo sem dependencia do .NET runtime instalado na maquina alvo — ideal para distribuicao de ferramentas devops.

System.CommandLine — Parsing e Subcomandos

C# + System.CommandLine — CLI com subcomandos, opcoes tipadas e validacao
using System.CommandLine;

// Opcoes globais
var verboseOpt = new Option<bool>(
  "--verbose", "-v"],
  description: "Exibe saida detalhada");

// Subcomando: deploy
var ambienteArg = new Argument<string>("ambiente", "prod, staging ou dev");
ambienteArg.AddCompletions("prod", "staging", "dev");

var tagOpt = new Option<string>("--tag", "Image tag (ex: v1.3.0)") { IsRequired = true };
tagOpt.AddValidator(r => {
  if (!r.GetValueForOption(tagOpt)!.StartsWith("v"))
    r.ErrorMessage = "Tag deve comecar com 'v' (ex: v1.0.0)";
});

var deployCmd = new Command("deploy", "Faz deploy de uma imagem Docker")
  { ambienteArg, tagOpt };
deployCmd.SetHandler(async (ambiente, tag, verbose, ctx) =>
{
  await ExecutarDeployAsync(ambiente, tag, verbose);
  ctx.ExitCode = 0;
}, ambienteArg, tagOpt, verboseOpt);

// Subcomando: status
var statusCmd = new Command("status", "Lista deployments ativos");
statusCmd.SetHandler(async (verbose) => await MostrarStatusAsync(verbose), verboseOpt);

// Root command
var root = new RootCommand("Deploy CLI — gerencia deployments de containers")
  { verboseOpt, deployCmd, statusCmd };

return await root.InvokeAsync(args);
C# + System.CommandLine — leitura de stdin e piping Unix-style
// Ferramenta de processamento de texto via pipe
// Uso: cat arquivo.csv | minha-ferramenta transform --upper

var upperOpt   = new Option<bool>("--upper", "Converte para maiusculas");
var arquivoOpt = new Option<FileInfo?>("--input", "Arquivo de entrada (default: stdin)");

var cmd = new RootCommand("Processa texto linha a linha") { upperOpt, arquivoOpt };
cmd.SetHandler(async (bool upper, FileInfo? arquivo) =>
{
  TextReader reader = arquivo is not null
    ? new StreamReader(arquivo.FullName)
    : Console.In;

  string? linha;
  while ((linha = await reader.ReadLineAsync()) is not null)
  {
    var saida = upper ? linha.ToUpperInvariant() : linha;
    await Console.Out.WriteLineAsync(saida);
  }

  if (arquivo is not null) reader.Dispose();
}, upperOpt, arquivoOpt);

Spectre.Console — UI Terminal Rica

C# + Spectre.Console — tabelas, progress bars, prompts e Live Display
using Spectre.Console;

// Markup colorido
AnsiConsole.MarkupLine("[bold green]Iniciando deploy...[/]");
AnsiConsole.MarkupLine("[yellow]AVISO:[/] ambiente [bold]{ambiente}[/] selecionado.");

// Tabela formatada automaticamente
var tabela = new Table()
  .AddColumn(new TableColumn("[bold]Servico[/]").LeftAligned())
  .AddColumn(new TableColumn("[bold]Status[/]").Centered())
  .AddColumn("Versao")
  .AddRow("api-gateway",    "[green]Running[/]",   "v2.1.0")
  .AddRow("auth-service",  "[green]Running[/]",   "v1.4.2")
  .AddRow("worker",        "[red]Stopped[/]",   "v1.0.1")
  .Border(TableBorder.Rounded);
AnsiConsole.Write(tabela);

// Prompt interativo com selecao multipla
var servicos = AnsiConsole.Prompt(
  new MultiSelectionPrompt<string>()
    .Title("[bold]Quais servicos reiniciar?[/]")
    .PageSize(10)
    .AddChoices("api-gateway", "auth-service", "worker", "scheduler"));

// Progress bar com multiplas tarefas
await AnsiConsole.Progress()
  .AutoRefresh(true)
  .AutoClear(false)
  .Columns(new TaskDescriptionColumn(), new ProgressBarColumn(), new PercentageColumn(), new ElapsedTimeColumn())
  .StartAsync(async ctx => {
    var t1 = ctx.AddTask("[green]Build imagem Docker[/]", maxValue: 100);
    var t2 = ctx.AddTask("[blue]Push para registry[/]", maxValue: 100);
    while (!ctx.IsFinished)
    {
      t1.Increment(3);
      await Task.Delay(50);
      if (t1.Value >= 50) t2.Increment(2);
    }
  });

// Live Display — atualiza em tempo real sem apagar tela
await AnsiConsole.Live(new Table().AddColumn("Log"))
  .StartAsync(async ctx => {
    for (int i = 0; i < 20; i++)
    {
      ctx.UpdateTarget(new Table().AddColumn("Log").AddRow($"[grey]{DateTime.Now:HH:mm:ss}[/] linha {i}"));
      await Task.Delay(200);
    }
  });
C# + Spectre.Console — FigletText, Status spinner e ExceptionCapture
// FigletText — arte ASCII
AnsiConsole.Write(new FigletText("Deploy CLI").Color(Color.Blue));

// Status spinner — tarefa longa com animacao
await AnsiConsole.Status()
  .Spinner(Spinner.Known.Dots12)
  .SpinnerStyle(Style.Parse("cyan"))
  .StartAsync("Conectando ao cluster Kubernetes...", async ctx => {
    await Task.Delay(2000);  // simula conexao
    ctx.Status = "Aplicando manifesto...";
    await Task.Delay(1500);
    ctx.Status = "Aguardando pods ficarem prontos...";
    await Task.Delay(3000);
  });

// Captura de excecao formatada
try
{
  await ExecutarOperacaoRiscoAsync();
}
catch (Exception ex)
{
  AnsiConsole.WriteException(ex, ExceptionFormats.ShortenEverything);
  Environment.Exit(1);
}

// Confirmar acao destrutiva
if (!AnsiConsole.Confirm("[red bold]Tem certeza que deseja DELETAR os dados de producao?[/]", false))
{
  AnsiConsole.MarkupLine("[grey]Operacao cancelada.[/]");
  return;
}

Publicar como Dotnet Global Tool

XML — .csproj para dotnet tool e publicacao self-contained
<!-- MinhaTool.csproj -->
<Project Sdk="Microsoft.NET.Sdk">
  <PropertyGroup>
    <OutputType>Exe</OutputType>
    <TargetFramework>net9.0</TargetFramework>
    <Nullable>enable</Nullable>
    <ImplicitUsings>enable</ImplicitUsings>
    <!-- Metadados dotnet tool -->
    <PackAsTool>true</PackAsTool>
    <ToolCommandName>minha-tool</ToolCommandName>
    <PackageId>MinhaEmpresa.MinhaTool</PackageId>
    <Version>1.0.0</Version>
    <Description>CLI para gerenciar deployments</Description>
    <Authors>Gustavo</Authors>
    <!-- AOT + trimming para executavel menor -->
    <PublishAot>true</PublishAot>
    <TrimmerRootAssembly Include="MinhaTool" />
  </PropertyGroup>
  <ItemGroup>
    <PackageReference Include="System.CommandLine" Version="2.0.0-beta4.*" />
    <PackageReference Include="Spectre.Console" Version="0.*" />
  </ItemGroup>
</Project>

// --- Publicar ---
// Instalar localmente (para dev)
// dotnet pack -c Release -o ./nupkg
// dotnet tool install --global --add-source ./nupkg MinhaEmpresa.MinhaTool
// minha-tool deploy prod --tag v2.0.0

// Publicar no NuGet.org
// dotnet nuget push ./nupkg/*.nupkg --api-key $NUGET_KEY --source https://api.nuget.org/v3/index.json

// Publicar como executavel self-contained (sem .NET instalado)
// dotnet publish -r linux-x64 --self-contained -p:PublishSingleFile=true -p:StripSymbols=true
// dotnet publish -r win-x64  --self-contained -p:PublishSingleFile=true
