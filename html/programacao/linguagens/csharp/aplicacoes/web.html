C# para Web

ASP.NET Core e o framework web de maior performance no ecossistema moderno — consistentemente no top 5 do benchmark TechEmpower. Suporta REST, gRPC, WebSockets (SignalR), Blazor (C# no browser via WASM), autenticacao JWT/OAuth2, OpenAPI/Swagger e deploy nativo em Docker e Kubernetes.

ASP.NET Core Minimal API (introducida no .NET 6) permite montar uma API REST completa em poucos arquivos sem boilerplate de controllers, mantendo toda a potencia de DI, middleware e Swagger. Para arquiteturas maiores, controllers e MVC oferecem mais estrutura e convencoes.

Minimal API — REST com DTOs e Validacao

C# + ASP.NET Core — API REST completa com CRUD, FluentValidation e Swagger
// Program.cs — API completa com grupos e filtros
using FluentValidation;

var builder = WebApplication.CreateBuilder(args);
builder.Services.AddEndpointsApiExplorer();
builder.Services.AddSwaggerGen();
builder.Services.AddValidatorsFromAssemblyContaining<Program>();
builder.Services.AddSingleton<ProdutoDb>();

var app = builder.Build();
if (app.Environment.IsDevelopment()) { app.UseSwagger(); app.UseSwaggerUI(); }
app.UseHttpsRedirection();

// Grupo de rotas — prefixo e metadados compartilhados
var api = app.MapGroup("/api/v1/produtos")
             .WithTags("Produtos")
             .WithOpenApi();

api.MapGet("/", (ProdutoDb db) => db.Todos())
   .WithName("ListarProdutos");

api.MapGet("/{id:int}", (int id, ProdutoDb db) =>
  db.Buscar(id) is { } p ? Results.Ok(p) : Results.NotFound());

api.MapPost("/", (CriarProdutoDto dto, IValidator<CriarProdutoDto> v, ProdutoDb db) => {
  var erros = v.Validate(dto);
  if (!erros.IsValid) return Results.ValidationProblem(erros.ToDictionary());
  var p = db.Criar(dto);
  return Results.CreatedAtRoute("ListarProdutos", null, p);
});

api.MapPut("/{id:int}", (int id, AtualizarProdutoDto dto, ProdutoDb db) =>
  db.Atualizar(id, dto) ? Results.NoContent() : Results.NotFound());

api.MapDelete("/{id:int}", (int id, ProdutoDb db) =>
  db.Remover(id) ? Results.NoContent() : Results.NotFound());

app.Run();

// DTOs e validacao
record CriarProdutoDto(string Nome, decimal Preco, int Estoque);
record AtualizarProdutoDto(string? Nome, decimal? Preco);

class CriarValidator : AbstractValidator<CriarProdutoDto>
{
  public CriarValidator() {
    RuleFor(x => x.Nome).NotEmpty().MaximumLength(200);
    RuleFor(x => x.Preco).GreaterThan(0);
    RuleFor(x => x.Estoque).GreaterThanOrEqualTo(0);
  }
}

Autenticacao JWT e Autorizacao

C# + ASP.NET Core — emissao e validacao de JWT Bearer, roles e policies
using System.IdentityModel.Tokens.Jwt;
using System.Security.Claims;
using Microsoft.IdentityModel.Tokens;
using Microsoft.AspNetCore.Authentication.JwtBearer;

// Configurar autenticacao JWT
builder.Services.AddAuthentication(JwtBearerDefaults.AuthenticationScheme)
  .AddJwtBearer(opt => {
    opt.TokenValidationParameters = new TokenValidationParameters {
      ValidateIssuer           = true,
      ValidateAudience         = true,
      ValidateLifetime         = true,
      ValidateIssuerSigningKey = true,
      ValidIssuer    = builder.Configuration["Jwt:Issuer"],
      ValidAudience  = builder.Configuration["Jwt:Audience"],
      IssuerSigningKey = new SymmetricSecurityKey(
        Encoding.UTF8.GetBytes(builder.Configuration["Jwt:Key"]!))
    };
  });

builder.Services.AddAuthorizationBuilder()
  .AddPolicy("Admin", p => p.RequireRole("admin"))
  .AddPolicy("PremiumOrAdmin", p => p.RequireAssertion(ctx =>
    ctx.User.IsInRole("admin") || ctx.User.IsInRole("premium")));

// Endpoint de login — gera token JWT
app.MapPost("/auth/login", (LoginDto dto) =>
{
  // Validar credenciais (nunca exponha senha em texto puro)
  if (dto.Email != "admin@ex.com" || dto.Senha != "senhaHash")
    return Results.Unauthorized();

  var claims = new[] {
    new Claim(ClaimTypes.Sub,   "user-guid-aqui"),
    new Claim(ClaimTypes.Email, dto.Email),
    new Claim(ClaimTypes.Role,  "admin")
  };
  var key    = new SymmetricSecurityKey(Encoding.UTF8.GetBytes("chave-super-secreta-32chars!!"));
  var creds  = new SigningCredentials(key, SecurityAlgorithms.HmacSha256);
  var token  = new JwtSecurityToken(
    issuer:   "MinhaApi", audience: "MinhaApiClients",
    claims:   claims, expires: DateTime.UtcNow.AddHours(1),
    signingCredentials: creds);

  return Results.Ok(new { Token = new JwtSecurityTokenHandler().WriteToken(token) });
});

// Endpoint protegido por role
app.MapGet("/admin/dashboard", (ClaimsPrincipal user) =>
  new { Mensagem = $"Bem-vindo, {user.FindFirst(ClaimTypes.Email)?.Value}" })
   .RequireAuthorization("Admin");
Seguranca JWT: A chave simetrica deve ter ao menos 256 bits (32 caracteres). Nunca armazene a chave em codigo — use dotnet user-secrets para desenvolvimento e variaveis de ambiente (ou Key Vault) em producao. Use HTTPS para que tokens nao sejam interceptados. Nunca armazene senhas em texto puro — use BCrypt/Argon2 via ASP.NET Core Identity.

gRPC com Protobuf

C# + gRPC — definicao .proto, servico e cliente streaming
// produtos.proto — definicao do contrato
// syntax = "proto3";
// package produtos;
// service ProdutoService {
//   rpc Buscar (BuscarRequest) returns (ProdutoReply);
//   rpc ListarStream (FiltroRequest) returns (stream ProdutoReply);
// }
// message BuscarRequest { int32 id = 1; }
// message ProdutoReply { int32 id = 1; string nome = 2; double preco = 3; }
// message FiltroRequest { double preco_maximo = 1; }

// Implementacao do servico gRPC
using Grpc.Core;
using Produtos;

class ProdutoGrpcService : ProdutoService.ProdutoServiceBase
{
  private readonly IProdutoRepository _repo;
  public ProdutoGrpcService(IProdutoRepository repo) => _repo = repo;

  public override async Task<ProdutoReply> Buscar(BuscarRequest req, ServerCallContext ctx)
  {
    var p = await _repo.FindAsync(req.Id, ctx.CancellationToken)
          ?? throw new RpcException(new Status(StatusCode.NotFound, "Produto nao encontrado"));
    return new ProdutoReply { Id = p.Id, Nome = p.Nome, Preco = (double)p.Preco };
  }

  public override async Task ListarStream(
    FiltroRequest req, IServerStreamWriter<ProdutoReply> stream, ServerCallContext ctx)
  {
    await foreach (var p in _repo.BuscarBaratosAsync((decimal)req.PrecoMaximo, ctx.CancellationToken))
    {
      await stream.WriteAsync(new ProdutoReply { Id = p.Id, Nome = p.Nome });
    }
  }
}

// Registrar no Program.cs
builder.Services.AddGrpc();
app.MapGrpcService<ProdutoGrpcService>();

// Cliente gRPC tipado
var channel = GrpcChannel.ForAddress("https://localhost:5001");
var client  = new ProdutoService.ProdutoServiceClient(channel);
var reply   = await client.BuscarAsync(new BuscarRequest { Id = 42 });
Console.WriteLine(reply.Nome);

Middleware Customizado

C# — middleware para logging, rate limiting e tratamento de excecoes
// Middleware de request logging
class RequestLoggingMiddleware(RequestDelegate next, ILogger<RequestLoggingMiddleware> logger)
{
  public async Task InvokeAsync(HttpContext ctx)
  {
    var sw = Stopwatch.StartNew();
    try   { await next(ctx); }
    finally
    {
      sw.Stop();
      logger.LogInformation(
        "{Method} {Path} {Status} {ElapsedMs}ms",
        ctx.Request.Method, ctx.Request.Path, ctx.Response.StatusCode, sw.ElapsedMilliseconds);
    }
  }
}

// Middleware de tratamento global de excecoes
class ExceptionMiddleware(RequestDelegate next, ILogger<ExceptionMiddleware> logger)
{
  public async Task InvokeAsync(HttpContext ctx)
  {
    try { await next(ctx); }
    catch (ValidationException ex)
    {
      ctx.Response.StatusCode = 400;
      await ctx.Response.WriteAsJsonAsync(new { Erros = ex.Errors.Select(e => e.ErrorMessage) });
    }
    catch (Exception ex)
    {
      logger.LogError(ex, "Erro nao tratado");
      ctx.Response.StatusCode = 500;
      await ctx.Response.WriteAsJsonAsync(new { Mensagem = "Erro interno" });
    }
  }
}

// Registrar middleware
app.UseMiddleware<ExceptionMiddleware>();  // deve ser o primeiro
app.UseMiddleware<RequestLoggingMiddleware>();

// Rate limiting nativo (.NET 7+)
builder.Services.AddRateLimiter(opt => {
  opt.AddFixedWindowLimiter("api", o => {
    o.PermitLimit        = 100;
    o.Window             = TimeSpan.FromMinutes(1);
    o.QueueLimit         = 10;
    o.QueueProcessingOrder = QueueProcessingOrder.OldestFirst;
  });
});
app.UseRateLimiter();
api.RequireRateLimiting("api");
