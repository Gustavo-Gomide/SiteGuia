C# para GUI e Desktop

O ecossistema .NET oferece varias opcoes para GUI nativa: WPF (Windows Presentation Foundation) para aplicacoes Windows ricas com XAML e MVVM; WinForms para RAD Windows simples e moderno; .NET MAUI para apps multi-plataforma verdadeiros (iOS, Android, macOS, Windows) com um unico codebase C#; e Avalonia UI para UI cross-platform com XAML (incluindo Linux).

WPF com o padrao MVVM (Model-View-ViewModel) continua sendo o padrao da industria para aplicacoes Windows enterprise com dados complexos. .NET MAUI e Avalonia UI sao a escolha moderna quando portabilidade entre plataformas e requisito.

WPF — MVVM com INotifyPropertyChanged e Commands

C# + WPF — ViewModel com ObservableProperty, RelayCommand e data binding
using System.Collections.ObjectModel;
using System.ComponentModel;
using System.Windows.Input;

// Base ViewModel — INotifyPropertyChanged
abstract class ViewModelBase : INotifyPropertyChanged
{
  public event PropertyChangedEventHandler? PropertyChanged;
  protected void SetProperty<T>(ref T campo, T valor, [CallerMemberName] string? prop = null)
  {
    if (EqualityComparer<T>.Default.Equals(campo, valor)) return;
    campo = valor;
    PropertyChanged?.Invoke(this, new PropertyChangedEventArgs(prop));
  }
}

// RelayCommand — converte lambda em ICommand
class RelayCommand(Action execute, Func<bool>? canExecute = null) : ICommand
{
  public event EventHandler? CanExecuteChanged;
  public bool   CanExecute(object? p) => canExecute?.Invoke() ?? true;
  public void   Execute(object? p) => execute();
  public void   RaiseCanExecuteChanged() =>
    CanExecuteChanged?.Invoke(this, EventArgs.Empty);
}

// ViewModel de lista de tarefas
class TarefasViewModel : ViewModelBase
{
  private string _novaTarefa = "";
  public string NovaTarefa
  {
    get => _novaTarefa;
    set { SetProperty(ref _novaTarefa, value); AdicionarCommand.RaiseCanExecuteChanged(); }
  }

  public ObservableCollection<Tarefa> Tarefas { get; } = [];

  public RelayCommand AdicionarCommand { get; }
  public RelayCommand<Tarefa> RemoverCommand { get; }

  public TarefasViewModel()
  {
    AdicionarCommand = new(Adicionar, () => !string.IsNullOrWhiteSpace(NovaTarefa));
    RemoverCommand   = new(Tarefa? t => { if (t is not null) Tarefas.Remove(t); });
  }

  private void Adicionar()
  {
    Tarefas.Add(new Tarefa(NovaTarefa));
    NovaTarefa = "";
  }
}

record Tarefa(string Titulo) : ViewModelBase
{
  private bool _concluida;
  public bool Concluida { get => _concluida; set => SetProperty(ref _concluida, value); }
}

.NET MAUI — Multi-Plataforma com C#

C# + .NET MAUI — ContentPage, Shell navigation, MVVM com CommunityToolkit
using CommunityToolkit.Mvvm.ComponentModel;
using CommunityToolkit.Mvvm.Input;

// ViewModel com source generator — elimina boilerplate
[ObservableObject]
partial class ProdutosViewModel
{
  [ObservableProperty]
  [NotifyCanExecuteChangedFor(nameof(BuscarCommand))]
  private string _filtro = "";

  [ObservableProperty]
  private bool _carregando;

  public ObservableCollection<Produto> Produtos { get; } = [];

  private readonly IProdutoService _service;
  public ProdutosViewModel(IProdutoService service) => _service = service;

  [RelayCommand(CanExecute = nameof(PodeBuscar))]
  async Task BuscarAsync()
  {
    Carregando = true;
    try
    {
      var lista = await _service.BuscarAsync(Filtro);
      Produtos.Clear();
      foreach (var p in lista) Produtos.Add(p);
    }
    finally { Carregando = false; }
  }

  bool PodeBuscar() => !string.IsNullOrWhiteSpace(Filtro);

  [RelayCommand]
  async Task AbrirDetalheAsync(Produto produto) =>
    await Shell.Current.GoToAsync($"{nameof(DetalhePage)}?id={produto.Id}");
}

// Registro de DI e rotas no MauiProgram.cs
var builder = MauiApp.CreateBuilder();
builder.UseMauiApp<App>()
  .ConfigureFonts(f => f.AddFont("Inter-Regular.ttf", "InterRegular"));
builder.Services
  .AddSingleton<IProdutoService, ProdutoService>()
  .AddTransient<ProdutosViewModel>()
  .AddTransient<ProdutosPage>()
  .AddTransient<DetalhePage>();

Routing.RegisterRoute(nameof(DetalhePage), typeof(DetalhePage));
await builder.Build().RunAsync();

Avalonia UI — Cross-Platform (Windows/Linux/macOS)

C# + Avalonia 11 — janela XAML, ViewModel e reactive bindings
using Avalonia;
using Avalonia.Controls;
using Avalonia.ReactiveUI;
using ReactiveUI;
using System.Reactive;

// ViewModel com ReactiveUI
class ContadorViewModel : ReactiveObject
{
  private int _contador;
  public int Contador
  {
    get => _contador;
    set => this.RaiseAndSetIfChanged(ref _contador, value);
  }

  public ReactiveCommand<Unit, Unit> IncrementarCommand { get; }
  public ReactiveCommand<Unit, Unit> ResetarCommand    { get; }

  public ContadorViewModel()
  {
    var podeResetar = this.WhenAnyValue(x => x.Contador, c => c > 0);
    IncrementarCommand = ReactiveCommand.Create(() => Contador++);
    ResetarCommand     = ReactiveCommand.Create(() => Contador = 0, podeResetar);
  }
}

// Window code-behind
public class MainWindow : ReactiveWindow<ContadorViewModel>
{
  public MainWindow()
  {
    InitializeComponent();
    ViewModel = new ContadorViewModel();
  }
}

// Program.cs — Avalonia bootstrap
AppBuilder.Configure<App>()
  .UsePlatformDetect()
  .UseReactiveUI()
  .StartWithClassicDesktopLifetime(args);
