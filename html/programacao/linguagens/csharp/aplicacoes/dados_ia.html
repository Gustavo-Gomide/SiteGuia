C# para IA e Dados

C# e .NET oferecem um ecossistema robusto para machine learning e inteligenca artificial: ML.NET para treinamento on-premise de modelos de classificacao, regressao e clustering; Semantic Kernel para orquestrar agentes com LLMs como GPT-4 e Mistral; ONNX Runtime para inferencia ultra-rapida de modelos exportados de PyTorch/scikit-learn; e Microsoft.Data.Analysis (DataFrame) para manipulacao tabular estilo Pandas.

O diferencial do .NET em IA e a integracao nativa com producao: voce treina ou carrega um modelo e ja expoe como endpoint ASP.NET Core com todo o ecossistema de observabilidade, autenticacao e escalabilidade do Azure — sem precisar de um microservico Python separado.

ML.NET — Machine Learning Nativo em C#

C# + ML.NET — classificacao binaria de sentimento com pipeline completo
using Microsoft.ML;
using Microsoft.ML.Data;

// Esquemas de entrada/saida
class ReviewInput
{
  [LoadColumn(0)] public string Texto  { get; set; } = "";
  [LoadColumn(1), ColumnName("Label")] public bool Positivo { get; set; }
}
class ReviewOutput
{
  [ColumnName("PredictedLabel")] public bool   Positivo { get; set; }
  public float  Score       { get; set; }
  public float  Probability { get; set; }
}

var ctx = new MLContext(seed: 42);

// Carregar dados de CSV
var dados     = ctx.Data.LoadFromTextFile<ReviewInput>("reviews.tsv", separatorChar: '\t', hasHeader: true);
var splits    = ctx.Data.TrainTestSplit(dados, testFraction: 0.2);

// Pipeline: tokenizacao → TF-IDF → classificador SdcaLogisticRegression
var pipeline  = ctx.Transforms.Text
  .FeaturizeText("Features", outputColumnName: "Features",
    inputColumnName: nameof(ReviewInput.Texto),
    options: new TextFeaturizingEstimator.Options {
      WordFeatureExtractor = new WordBagEstimator.Options { NgramLength = 2, UseAllLengths = true }
    })
  .Append(ctx.BinaryClassification.Trainers
    .SdcaLogisticRegression(labelColumnName: "Label", featureColumnName: "Features"));

// Treinar
var modelo = pipeline.Fit(splits.TrainSet);

// Avaliar metricas
var predicoes = modelo.Transform(splits.TestSet);
var metricas  = ctx.BinaryClassification.Evaluate(predicoes, "Label");
Console.WriteLine($"Accuracy={metricas.Accuracy:P2}  AUC={metricas.AreaUnderRocCurve:P2}  F1={metricas.F1Score:P2}");

// Salvar e carregar modelo
ctx.Model.Save(modelo, dados.Schema, "sentimento.zip");
var modeloLoaded = ctx.Model.Load("sentimento.zip", out _);

// PredictionEngine thread-safe via pool
var engPool = ctx.Model.CreatePredictionEnginePool<ReviewInput, ReviewOutput>(
  serviceProvider: null,
  modelLoader: ctx.Model.Load("sentimento.zip", out _));

// Predicao em runtime
var resultado = new PredictionEngine<ReviewInput, ReviewOutput>(ctx, modeloLoaded, true)
  .Predict(new ReviewInput { Texto = "Produto incrivel, superou expectativas!" });
Console.WriteLine($"Positivo={resultado.Positivo}  Prob={resultado.Probability:P1}");
C# + ML.NET — regressao de preco com features numericas e validacao cruzada
class ImovelInput
{
  public float Area        { get; set; }
  public float Quartos     { get; set; }
  public float DistanciaMetro { get; set; }
  [ColumnName("Label")]
  public float Preco       { get; set; }
}

var featureCols = new[] { nameof(ImovelInput.Area), nameof(ImovelInput.Quartos), nameof(ImovelInput.DistanciaMetro) };

var pipeline = ctx.Transforms
  .Concatenate("Features", featureCols)
  .Append(ctx.Transforms.NormalizeMinMax("Features"))
  .Append(ctx.Regression.Trainers
    .FastForest(numberOfTrees: 200, minimumExampleCountPerLeaf: 5));

// Validacao cruzada k-fold
var cv = ctx.Regression.CrossValidate(dados, pipeline, numberOfFolds: 5);
Console.WriteLine($"R² medio = {cv.Average(r => r.Metrics.RSquared):F4}");
Console.WriteLine($"RMSE medio = {cv.Average(r => r.Metrics.RootMeanSquaredError):F2}");

Semantic Kernel — Agentes e LLMs

C# + Semantic Kernel — agente com plugins, memoria e RAG
using Microsoft.SemanticKernel;
using Microsoft.SemanticKernel.Connectors.OpenAI;

// Configurar kernel com OpenAI
var kernel = Kernel.CreateBuilder()
  .AddOpenAIChatCompletion("gpt-4o-mini", apiKey: Environment.GetEnvironmentVariable("OPENAI_KEY")!)
  .Build();

// Plugin nativo — funcoes C# chamadas pelo LLM via function calling
class ClimaPlugin
{
  [KernelFunction, Description("Retorna temperatura atual de uma cidade")]
  public async Task<string> GetTemperaturaAsync(
    [Description("Nome da cidade")] string cidade) =>
    $"A temperatura em {cidade} e 22°C com sol";   // exemplo simplificado
}

class ProdutosPlugin(IProdutoRepository repo)
{
  [KernelFunction, Description("Busca produto pelo nome")]
  public async Task<string> BuscarProdutoAsync(string nome)
  {
    var p = await repo.BuscarPorNomeAsync(nome);
    return p is null ? "Produto nao encontrado" : $"{p.Nome}: R$ {p.Preco:N2} — {p.Estoque} em estoque";
  }
}

kernel.Plugins.AddFromType<ClimaPlugin>();
kernel.Plugins.AddFromObject(new ProdutosPlugin(repo), "Produtos");

// Chat loop com historico e function calling automatico
var config   = new OpenAIPromptExecutionSettings {
  ToolCallBehavior = ToolCallBehavior.AutoInvokeKernelFunctions
};
var chat     = kernel.GetRequiredService<IChatCompletionService>();
var historico = new ChatHistory("Voce e um assistente de loja. Responda em portugues.");
historico.AddUserMessage("Tem notebook disponivel? Qual o preco?");

var resposta = await chat.GetChatMessageContentAsync(historico, config, kernel);
Console.WriteLine(resposta.Content);

// Prompt template com variaveis
var fn = kernel.CreateFunctionFromPrompt("""
  Resuma em {{$idioma}} o seguinte texto em ate {{$palavras}} palavras:
  {{$texto}}
""", executionSettings: new OpenAIPromptExecutionSettings { MaxTokens = 500 });

var resumo = await kernel.InvokeAsync(fn, new KernelArguments {
  ["idioma"] = "portugues",
  ["palavras"] = "50",
  ["texto"] = textoLongo
});
Console.WriteLine(resumo);

ONNX Runtime — Inferencia de Modelos PyTorch/scikit-learn

C# + Microsoft.ML.OnnxRuntime — inferencia com tensor tipado e batch
using Microsoft.ML.OnnxRuntime;
using Microsoft.ML.OnnxRuntime.Tensors;

// Carregar modelo ONNX exportado do PyTorch ou scikit-learn
using var session = new InferenceSession("modelo.onnx",
  SessionOptions.MakeSessionOptionWithCudaProvider());  // GPU opcional

// Inspecionar entradas esperadas
foreach (var node in session.InputMetadata)
  Console.WriteLine($"Input: {node.Key}  Dims: [{string.Join(",", node.Value.Dimensions)}]");

// Preprocessar imagem 28x28 (MNIST)
float[] pixels = CarregarImagem("digito.png");  // normalizado [0,1]
var tensor = new DenseTensor<float>(pixels, new[] { 1, 1, 28, 28 });

// Inferencia
var inputs  = new[] { NamedOnnxValue.CreateFromTensor("input", tensor) };
using var results  = session.Run(inputs);
var output  = results.First().AsTensor<float>();
int predicao = Enumerable.Range(0, 10).MaxBy(i => output[0, i]);
Console.WriteLine($"Digito reconhecido: {predicao}");

// Batch inference — processar varios inputs de uma vez
float[] batchPixels = CarregarImagensBatch(caminhos); // [N x C x H x W]
var batchTensor = new DenseTensor<float>(batchPixels, new[] { caminhos.Length, 1, 28, 28 });
using var batchResults = session.Run(new[] { NamedOnnxValue.CreateFromTensor("input", batchTensor) });
var batchOutput = batchResults.First().AsTensor<float>();
for (int i = 0; i < caminhos.Length; i++)
{
  int pred = Enumerable.Range(0, 10).MaxBy(j => batchOutput[i, j]);
  Console.WriteLine($"[{i}] = {pred}");
}

Microsoft.Data.Analysis — DataFrame em C#

C# + Microsoft.Data.Analysis — analise tabular, filtros e estatisticas
using Microsoft.Data.Analysis;

// Criar DataFrame a partir de CSV
var df = DataFrame.LoadCsv("vendas.csv");

Console.WriteLine($"Linhas: {df.Rows.Count}  Colunas: {df.Columns.Count}");
Console.WriteLine(df.Description());  // min, max, media, desvio padrao, etc.

// Filtrar e selecionar colunas
var filtro  = (PrimitiveDataFrameColumn<float>) df["Valor"];
var mascara = filtro.ElementwiseGreaterThan(1000f);
var grandes = df[mascara];

// Adicionar coluna calculada
var colValor = (PrimitiveDataFrameColumn<float>) df["Valor"];
var colQtd   = (PrimitiveDataFrameColumn<int>) df["Quantidade"];
df.Columns.Add(new SingleDataFrameColumn("Total", colValor * colQtd.Apply(x => (float)x)));

// GroupBy e agregacao
var porCategoria = df.GroupBy("Categoria").Sum("Total");
Console.WriteLine(porCategoria);

// Ordenar e exportar
var ordenado = df.OrderByDescending("Total");
DataFrame.SaveCsv(ordenado, "vendas_ordenado.csv");
