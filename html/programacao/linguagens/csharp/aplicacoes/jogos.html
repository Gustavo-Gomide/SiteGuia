C# — Jogos

C# para Desenvolvimento de Jogos

C# e a linguagem principal do Unity — o engine de jogos mais utilizado no mundo — e uma das opcoes de scripting no Godot 4. Alem disso, MonoGame permite construir jogos 2D multiplataforma com controle total sobre o pipeline de renderizacao. C# no contexto de jogos exige atencao especial a gerencia de memoria (evitar alocacoes em runtime) e ao modelo de execucao de frame-by-frame.

No Unity, cada MonoBehaviour roda em uma thread principal. Operacoes custosas como carregamento de assets ou chamadas de rede devem usar Coroutine (yield/await) ou o Job System com Burst Compiler para aproveitar threads sem bloquear o frame.

Unity — MonoBehaviour, Física e Input

C# + Unity — movimento de personagem com Rigidbody, Input System e animacoes
using UnityEngine;
using UnityEngine.InputSystem;

public class PlayerController : MonoBehaviour
{
  [Header("Movimento")]
  [SerializeField] private float _velocidade  = 5f;
  [SerializeField] private float _forcaPulo   = 8f;

  private Rigidbody2D _rb;
  private Animator    _anim;
  private Vector2     _input;
  private bool        _noChao;

  void Awake()
  {
    _rb   = GetComponent<Rigidbody2D>();
    _anim = GetComponent<Animator>();
  }

  // Chamado pelo Input System Action Map
  public void OnMove(InputValue value)  => _input = value.Get<Vector2>();
  public void OnJump(InputValue value)
  {
    if (value.isPressed && _noChao)
      _rb.AddForce(Vector2.up * _forcaPulo, ForceMode2D.Impulse);
  }

  void FixedUpdate()
  {
    = new Vector2(_input.x * _velocidade, _rb.velocity.y);
    _anim.SetFloat("Speed", Mathf.Abs(_input.x));
    if (_input.x != 0)
      transform.localScale = new Vector3(Mathf.Sign(_input.x), 1, 1);
  }

  void OnCollisionEnter2D(Collision2D col)
  {
    if (col.gameObject.CompareTag("Ground")) _noChao = true;
  }
  void OnCollisionExit2D(Collision2D col)
  {
    if (col.gameObject.CompareTag("Ground")) _noChao = false;
  }
}

Unity — ScriptableObjects, Events e Object Pooling

C# + Unity — ScriptableObject como data container e evento global desacoplado
using UnityEngine;
using System.Collections.Generic;

// ScriptableObject — configuracao de inimigo editavel no Inspector
[CreateAssetMenu(menuName = "Game/InimigoConfig")]
public class InimigoConfig : ScriptableObject
{
  public string NomeInimigo;
  public float  Vida;
  public float  Velocidade;
  public float  DanoAtaque;
  public Sprite  Sprite;
}

// Evento global via ScriptableObject — sem dependencia de singleton
[CreateAssetMenu(menuName = "Game/EventoJogo")]
public class EventoJogo : ScriptableObject
{
  private readonly List<IEventoListener> _listeners = new();
  public void Disparar()  => _listeners.ForEach(l => l.AoReceberEvento());
  public void Registrar(IEventoListener l)   => _listeners.Add(l);
  public void Cancelar(IEventoListener l)   => _listeners.Remove(l);
}
public interface IEventoListener { void AoReceberEvento(); }

// Object Pool nativo (Unity 2021+)
using UnityEngine.Pool;

public class ProjetilPool : MonoBehaviour
{
  [SerializeField] private Projetil _prefab;

  private ObjectPool<Projetil> _pool;

  void Awake()
  {
    _pool = new ObjectPool<Projetil>(
      createFunc:   () => Instantiate(_prefab),
      actionOnGet:  p  => p.gameObject.SetActive(true),
      actionOnRelease: p => p.gameObject.SetActive(false),
      actionOnDestroy: p => Destroy(p.gameObject),
      defaultCapacity: 20, maxSize: 100);
  }

  public Projetil Obter() => _pool.Get();
  public void Retornar(Projetil p) => _pool.Release(p);
}

// Coroutine — carregamento progressivo com progresso
public class GerenciadorCena : MonoBehaviour
{
  public IEnumerator CarregarCenaAsync(string nome, System.Action<float> onProgresso)
  {
    var op = SceneManager.LoadSceneAsync(nome);
    op.allowSceneActivation = false;
    while (op.progress < 0.9f)
    {
      onProgresso(op.progress / 0.9f);
      yield return null;
    }
    onProgresso(1f);
    op.allowSceneActivation = true;
  }
}

MonoGame — Jogo 2D do Zero

C# + MonoGame — game loop, sprite, colisao e gerenciador de estados
using Microsoft.Xna.Framework;
using Microsoft.Xna.Framework.Graphics;
using Microsoft.Xna.Framework.Input;

public class Jogo : Game
{
  private GraphicsDeviceManager _graphics;
  private SpriteBatch            _batch;
  private Texture2D              _texJogador, _texInimigo;
  private Rectangle              _jogador = new(100, 100, 64, 64);
  private Rectangle              _inimigo  = new(400, 200, 64, 64);

  public Jogo()
  {
    _graphics = new GraphicsDeviceManager(this);
    Content.RootDirectory = "Content";
  }

  protected override void LoadContent()
  {
    _batch    = new SpriteBatch(GraphicsDevice);
    _texJogador = Content.Load<Texture2D>("jogador");
    _texInimigo  = Content.Load<Texture2D>("inimigo");
  }

  protected override void Update(GameTime gt)
  {
    var ks  = Keyboard.GetState();
    var dt  = (float) gt.ElapsedGameTime.TotalSeconds;
    var vel = (int) (200 * dt);

    if (ks.IsKeyDown(Keys.Left))  _jogador.X -= vel;
    if (ks.IsKeyDown(Keys.Right)) _jogador.X += vel;
    if (ks.IsKeyDown(Keys.Up))    _jogador.Y -= vel;
    if (ks.IsKeyDown(Keys.Down))  _jogador.Y += vel;

    // Colisao AABB
    if (_jogador.Intersects(_inimigo))
      Console.WriteLine("Colisao detectada!");

    base.Update(gt);
  }

  protected override void Draw(GameTime gt)
  {
    GraphicsDevice.Clear(Color.CornflowerBlue);
    _batch.Begin(SpriteSortMode.Deferred, BlendState.AlphaBlend,
                  SamplerState.PointClamp);
    _batch.Draw(_texJogador, _jogador, Color.White);
    _batch.Draw(_texInimigo,  _inimigo,  Color.White);
    _batch.End();
  }
}

Boas Praticas em Jogos C#

Boas Praticas

  • Cache de componentes no Awake() com GetComponent<T>
  • Usar ObjectPool<T> para projeteis, particulas e efeitos
  • Usar struct para dados de estado (posicao, velocidade) — zero alocacao
  • Preferir Physics.SphereCastNonAlloc sobre SphereCastAll
  • Usar profiler (ProfilerMarker) para medir hotspots
  • ScriptableObjects para configs editaveis sem acoplar cenas

Anti-Padroes

  • Chamar GetComponent<T> no Update() — muito lento
  • Usar Find("nome") ou FindObjectOfType<T> em runtime
  • Alocar strings, listas ou arrays no Update() — trigger de GC
  • Usar Debug.Log() em producao — custo de GC alocacao de string
  • yield return new WaitForSeconds() — use instancia cached
  • Singleton Game Manager com estado global bagunçado
