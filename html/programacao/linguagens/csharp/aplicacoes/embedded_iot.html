C# para Embedded e IoT

C# pode rodar diretamente em microcontroladores via .NET nanoFramework (ESP32, STM32, NXP — firmware CLR minimalista sem SO), em dispositivos Linux embarcados como Raspberry Pi via .NET IoT Libraries com acesso nativo a GPIO, I2C, SPI e UART, e em gateways industriais conectados ao Azure IoT Hub via MQTT ou AMQP.

.NET nanoFramework implementa um subconjunto do CLR em C++ que cabe em microcontroladores com 256 KB de Flash. Voce escreve, compila e depura no Visual Studio com o mesmo C# que usa em backend — sem aprender C/C++ embarcado para prototipagem.

.NET nanoFramework — C# em Microcontroladores

C# + nanoFramework — GPIO, PWM, ADC e comunicacao I2C no ESP32
using nanoFramework.Hardware.Esp32;
using System.Device.Gpio;
using System.Device.I2c;
using System.Device.Pwm;

class Program
{
  static void Main()
  {
    // GPIO output — pisca LED na porta 2
    using var gpio = new GpioController();
    gpio.OpenPin(2, PinMode.Output);
    while (true)
    {
      gpio.Write(2, PinValue.High);
      Thread.Sleep(500);
      gpio.Write(2, PinValue.Low);
      Thread.Sleep(500);
    }
  }
}

// PWM — controle de servo motor
using var pwm = new PwmChannel(0, 0, frequency: 50);  // 50 Hz para servo
pwm.Start();
for (double dc = 0.05; dc <= 0.10; dc += 0.01)    // 5% a 10% duty cycle = 0° a 180°
{
  pwm.DutyCycle = dc;
  Thread.Sleep(200);
}
pwm.Stop();

// I2C — leitura de sensor de temperatura BME280
const int BME280_ADDR = 0x76;
var i2cConf = new I2cConnectionSettings(1, BME280_ADDR);
using var i2c = I2cDevice.Create(i2cConf);

// Verificar chip_id (registro 0xD0)
SpanByte cmd     = new byte[] { 0xD0 };
SpanByte resposta = new byte[1];
i2c.WriteRead(cmd, resposta);
Debug.WriteLine($"BME280 chip_id = 0x{resposta[0]:X2}"); // deve ser 0x60
C# + nanoFramework — Wi-Fi, HTTPS e JSON no ESP32
using nanoFramework.Networking;
using nanoFramework.Json;

// Conectar ao Wi-Fi
var success = WifiNetworkHelper.ConnectDhcp("MinhaRede", "senha123",
  requiresDateTime: true, token: new CancellationTokenSource(60_000).Token);

if (!success) { Debug.WriteLine("Wi-Fi falhou"); return; }

// POST JSON para API (TLS 1.2)
var payload = JsonConvert.SerializeObject(new { temperatura = 25.3f, umidade = 60 });
using var req = (HttpWebRequest) WebRequest.Create("https://api.exemplo.com/dados");
req.Method      = "POST";
req.ContentType = "application/json";
req.GetRequestStream().Write(Encoding.UTF8.GetBytes(payload));

using var resp = req.GetResponse();
Debug.WriteLine("Dados enviados com sucesso");

.NET IoT Libraries — Raspberry Pi e Linux Embarcado

C# + System.Device.Gpio / Iot.Device.Bindings — sensores, displays e servos
using System.Device.Gpio;
using System.Device.I2c;
using Iot.Device.Dht;             // DHT11/DHT22
using Iot.Device.Ssd1306;        // display OLED 128x64
using Iot.Device.Hcsr04;         // sensor ultrassonico HC-SR04
using UnitsNet;

// Leitura de temperatura e umidade com DHT22 (GPIO 4)
using var gpio = new GpioController(PinNumberingScheme.Board);
using var dht  = new Dht22(4, PinNumberingScheme.Board, gpio);

if (dht.TryReadHumidity(out var hum) && dht.TryReadTemperature(out var temp))
  Console.WriteLine($"Temp={temp.DegreesCelsius:F1}°C  Umidade={hum.Percent:F1}%");

// Display OLED via I2C
using var i2c  = I2cDevice.Create(new I2cConnectionSettings(busId: 1, deviceAddress: 0x3C));
using var oled = new Ssd1306(i2c, 128, 64);
var img      = new BitmapImage(PixelFormat.Format1bppIndexed, 128, 64);
using var canvas = BitmapImage.CreateGraphics(img);
canvas.DrawString($"T={temp.DegreesCelsius:F1}C", new PointF(0, 0));
oled.DrawBitmap(img);
oled.Display();

// Sensor ultrassonico
using var sonar = new Hcsr04(triggerPin: 23, echoPin: 24);
if (sonar.TryGetDistance(out Length dist))
  Console.WriteLine($"Distancia: {dist.Centimeters:F1} cm");

// GPIO interrupt — botao com debounce
gpio.OpenPin(17, PinMode.InputPullUp);
gpio.RegisterCallbackForPinValueChangedEvent(17,
  PinEventTypes.Falling,
  (_, e) => Console.WriteLine($"Botao pressionado no GPIO {e.PinNumber}"));
Console.ReadLine();

MQTT com MQTTnet

C# + MQTTnet — publicar telemetria e receber comandos com QoS e TLS
using MQTTnet;
using MQTTnet.Client;
using MQTTnet.Protocol;

var factory = new MqttFactory();
using var client  = factory.CreateMqttClient();

var options = new MqttClientOptionsBuilder()
  .WithTcpServer("broker.hivemq.com", 8883)
  .WithClientId("dispositivo-001")
  .WithCredentials("usuario", "senha")
  .WithTlsOptions(o => o.UseTls())
  .WithCleanSession()
  .Build();

// Handler de mensagens recebidas
client.ApplicationMessageReceivedAsync += async e => {
  var topico  = e.ApplicationMessage.Topic;
  var payload = e.ApplicationMessage.ConvertPayloadToString();
  Console.WriteLine($"[{topico}] {payload}");
  await Task.CompletedTask;
};

// Reconexao automatica
client.DisconnectedAsync += async e => {
  await Task.Delay(TimeSpan.FromSeconds(5));
  try { await client.ConnectAsync(options); }
  catch (Exception ex) { Console.WriteLine($"Reconexao falhou: {ex.Message}"); }
};

await client.ConnectAsync(options, CancellationToken.None);

// Subscrever topico de comandos
var subOpts = factory.CreateSubscribeOptionsBuilder()
  .WithTopicFilter(f => f.WithTopic("dispositivo/001/cmd").WithAtLeastOnceQoS())
  .Build();
await client.SubscribeAsync(subOpts);

// Publicar telemetria a cada 5 segundos
var cts = new CancellationTokenSource();
await Task.Run(async () => {
  while (!cts.Token.IsCancellationRequested)
  {
    var payload = JsonSerializer.Serialize(new {
      ts          = DateTimeOffset.UtcNow.ToUnixTimeSeconds(),
      temperatura = 22.5f + Random.Shared.NextSingle() * 2,
      umidade     = 60 + Random.Shared.Next(-5, 5)
    });
    var msg = new MqttApplicationMessageBuilder()
      .WithTopic("dispositivo/001/telemetria")
      .WithPayload(payload)
      .WithQualityOfServiceLevel(MqttQualityOfServiceLevel.AtLeastOnce)
      .WithRetainFlag()
      .Build();
    await client.PublishAsync(msg);
    await Task.Delay(5000, cts.Token);
  }
}, cts.Token);

Azure IoT Hub — Dispositivo Gerenciado

C# + Azure.IoT.Device — telemetria, Device Twin e metodos diretos
using Microsoft.Azure.Devices.Client;
using Microsoft.Azure.Devices.Shared;

const string connStr = "HostName=meu-hub.azure-devices.net;DeviceId=dev001;SharedAccessKey=...";
using var device = DeviceClient.CreateFromConnectionString(connStr, TransportType.Mqtt);
await device.OpenAsync();

// Enviar telemetria
var json = JsonSerializer.Serialize(new { temperatura = 23.1, bateria = 87 });
var msg  = new Message(Encoding.UTF8.GetBytes(json)) {
  ContentType     = "application/json",
  ContentEncoding = "utf-8"
};
msg.Properties["tipo"] = "temperatura";
await device.SendEventAsync(msg);

// Device Twin — reportar propriedades
var reported = new TwinCollection();
reported["firmware"] = "1.3.0";
reported["ultimaConexao"] = DateTimeOffset.UtcNow;
await device.UpdateReportedPropertiesAsync(reported);

// Device Twin — receber desired properties
await device.SetDesiredPropertyUpdateCallbackAsync(async (twin, _) => {
  if (twin.Contains("intervaloLeitura"))
    Console.WriteLine($"Novo intervalo: {twin["intervaloLeitura"]} s");
  await Task.CompletedTask;
}, null);

// Metodo direto — reiniciar dispositivo remotamente
await device.SetMethodHandlerAsync("reiniciar", async (req, _) => {
  Console.WriteLine("Reiniciando o dispositivo...");
  _ = Task.Run(async () => { await Task.Delay(2000); Environment.Exit(0); });
  return new MethodResponse(Encoding.UTF8.GetBytes("{\"status\":\"ok\"}"), 200);
}, null);
