C# — Banco de Dados

C# com Banco de Dados

O ecossistema .NET oferece acesso a banco de dados em todos os niveis de abstracao: EF Core como ORM completo com Code-First, Dapper para queries SQL diretas com mapeamento minimalista, MongoDB Driver para documentos flexiveis e StackExchange.Redis para cache e pub/sub. Cada ferramenta tem seu lugar e conhecer todas e essencial para projetos profissionais.

EF Core 9 com .NET 9 traz melhorias significativas: queries compostas sem traducao de N+1, split queries, tabelas temporais (SQL Server), compiled queries para zero overhead de traducao e suporte a JSON columns no PostgreSQL via Npgsql.

// Entidades do dominio
public class Categoria
{
  public int    Id        { get; set; }
  public required string Nome     { get; set; }
  public ICollection<Produto> Produtos { get; } = [];
}

public class Produto
{
  public int     Id           { get; set; }
  public required string Nome { get; set; }
  public decimal Preco        { get; set; }
  public int     Estoque      { get; set; }
  public int     CategoriaId  { get; set; }
  public Categoria? Categoria  { get; set; }
  // Concorrencia otimista — EF Core rastreia
  [Timestamp]
  public byte[] RowVersion   { get; set; } = [];
}

// DbContext com configuracoes via Fluent API
class LojaDb(DbContextOptions<LojaDb> opts) : DbContext(opts)
{
  public DbSet<Produto>   Produtos   => Set<Produto>();
  public DbSet<Categoria> Categorias => Set<Categoria>();

  protected override void OnModelCreating(ModelBuilder mb)
  {
    mb.Entity<Produto>(e => {
      e.HasKey(p => p.Id);
      e.Property(p => p.Nome).HasMaxLength(200).IsRequired();
      e.Property(p => p.Preco).HasPrecision(18, 2);
      e.HasIndex(p => p.CategoriaId);
      e.HasOne(p => p.Categoria)
       .WithMany(c => c.Produtos)
       .HasForeignKey(p => p.CategoriaId)
       .OnDelete(DeleteBehavior.Restrict);
      // Filtro global — produtos com estoque positivo
      e.HasQueryFilter(p => p.Estoque > 0);
    });
    mb.Entity<Categoria>(e => {
      e.HasKey(c => c.Id);
      e.Property(c => c.Nome).HasMaxLength(100);
    });
  }
}

EF Core — Queries Avancadas

C# + EF Core — LINQ avancado, split query, compiled query e raw SQL
// Paginacao eficiente com Keyset Pagination (evita OFFSET lento)
async Task<List<Produto>> BuscarPaginaAsync(LojaDb db, int ultimoId, int page = 20)
  => await db.Produtos
      .Where(p => p.Id > ultimoId)
      .OrderBy(p => p.Id)
      .Take(page)
      .AsNoTracking()   // sem rastreamento = mais rapido para reads
      .ToListAsync();

// Eager loading — evita N+1
var cats = await db.Categorias
  .Include(c => c.Produtos.Where(p => p.Preco < 1000))  // filtered include
  .AsSplitQuery()           // evita produto cartesiano, emite 2 SELECTs
  .AsNoTrackingWithIdentityResolution()
  .ToListAsync();

// Projections — busca apenas o necessario
var resumos = await db.Produtos
  .Select(p => new { p.Id, p.Nome, p.Preco })
  .ToListAsync();

// Compiled Query — reutiliza plano de consulta
private static readonly Func<LojaDb, int, Task<Produto?>> BuscarPorIdQuery =
  EF.CompileAsyncQuery((LojaDb db, int id) =>
    db.Produtos.FirstOrDefault(p => p.Id == id));

var produto = await BuscarPorIdQuery(db, 42);

// Raw SQL parametrizado — sem risco de SQL injection
var lista = await db.Produtos
  .FromSqlRaw("SELECT * FROM loja.Produtos WHERE Preco BETWEEN {0} AND {1}", 100, 500)
  .AsNoTracking()
  .ToListAsync();

// Interpolated SQL — EF Core parametriza automaticamente
decimal min = 100, max = 500;
var lista2 = await db.Produtos
  .FromSql($"SELECT * FROM loja.Produtos WHERE Preco BETWEEN {min} AND {max}")
  .ToListAsync();

// Concorrencia otimista — trata DbUpdateConcurrencyException
try
{
  produtoExistente.Preco = 4000;
  await db.SaveChangesAsync();
}
catch (DbUpdateConcurrencyException ex)
{
  var entry   = ex.Entries.Single();
  var dbVals  = await entry.GetDatabaseValuesAsync();
  entry.OriginalValues.SetValues(dbVals!);
  await db.SaveChangesAsync();  // retry com valores atualizados
}

Dapper — Micro-ORM SQL Direto

C# + Dapper — queries parametrizadas, multi-map, Table-Valued Parameters e transactions
using Dapper;
using System.Data;
using Npgsql;

class ProdutoDapperRepo(NpgsqlDataSource ds)
{
  // Query simples — retorna IEnumerable<Produto>
  public async Task<IEnumerable<Produto>> ListarAsync(decimal precoMax)
  {
    await using var conn = await ds.OpenConnectionAsync();
    return await conn.QueryAsync<Produto>(
      "SELECT id, nome, preco, estoque, categoria_id FROM loja.produtos WHERE preco <= @PrecoMax",
      new { PrecoMax = precoMax });
  }

  // Multi-map — JOIN em um unico roundtrip
  public async Task<IEnumerable<Produto>> ListarComCategoriaAsync()
  {
    const string sql = """
      SELECT p.*, c.id, c.nome
      FROM loja.produtos p
      JOIN loja.categorias c ON c.id = p.categoria_id
    """;
    await using var conn = await ds.OpenConnectionAsync();
    return await conn.QueryAsync<Produto, Categoria, Produto>(sql,
      (prod, cat) => { prod.Categoria = cat; return prod; },
      splitOn: "id");
  }

  // Transacao — inserir pedido + itens atomicamente
  public async Task CriarPedidoAsync(Pedido pedido, IEnumerable<ItemPedido> itens)
  {
    await using var conn = await ds.OpenConnectionAsync();
    await using var tx   = await conn.BeginTransactionAsync();
    try
    {
      var id = await conn.ExecuteScalarAsync<int>(
        "INSERT INTO pedidos (cliente_id, total) VALUES (@ClienteId, @Total) RETURNING id",
        pedido, tx);
      foreach (var item in itens)
        item.PedidoId = id;
      await conn.ExecuteAsync(
        "INSERT INTO itens_pedido (pedido_id, produto_id, quantidade) VALUES (@PedidoId, @ProdutoId, @Quantidade)",
        itens, tx);
      await tx.CommitAsync();
    }
    catch { await tx.RollbackAsync(); throw; }
  }

  // QueryMultiple — varios result sets em uma unica query
  public async Task<(IEnumerable<Produto> Produtos, int Total)> BuscarPaginaAsync(int pag)
  {
    const string sql = """
      SELECT * FROM loja.produtos ORDER BY id LIMIT 20 OFFSET @Offset;
      SELECT COUNT(*) FROM loja.produtos;
    """;
    await using var conn = await ds.OpenConnectionAsync();
    await using var multi = await conn.QueryMultipleAsync(sql, new { Offset = (pag - 1) * 20 });
    var produtos = await multi.ReadAsync<Produto>();
    var total    = await multi.ReadSingleAsync<int>();
    return (produtos, total);
  }
}

MongoDB — Driver Oficial C#

C# + MongoDB.Driver — CRUD, indices, aggregation e transactions
using MongoDB.Driver;
using MongoDB.Bson;

// Documento — BsonId mapeia para _id
record LogEvento(
  [BsonId, BsonRepresentation(BsonType.ObjectId)] string? Id,
  string Evento, string Usuario, DateTime CriadoAt, BsonDocument Dados);

class LogRepository(IMongoDatabase db)
{
  private readonly IMongoCollection<LogEvento> _col = db.GetCollection<LogEvento>("logs");

  // Criar indices na inicializacao
  public async Task CriarIndicesAsync()
  {
    var idx = Builders<LogEvento>.IndexKeys;
    await _col.Indexes.CreateManyAsync([
      new CreateIndexModel<LogEvento>(idx.Ascending(l => l.Usuario)),
      new CreateIndexModel<LogEvento>(idx.Descending(l => l.CriadoAt),
        new CreateIndexOptions { ExpireAfter = TimeSpan.FromDays(90) })  // TTL index
    ]);
  }

  public async Task InserirAsync(LogEvento log) => await _col.InsertOneAsync(log);

  // Filtro tipado via Builders
  public async Task<List<LogEvento>> BuscarPorUsuarioAsync(string usuario)
  {
    var filtro = Builders<LogEvento>.Filter;
    return await _col
      .Find(filtro.Eq(l => l.Usuario, usuario) & filtro.Gte(l => l.CriadoAt, DateTime.UtcNow.AddDays(-7)))
      .SortByDescending(l => l.CriadoAt)
      .Limit(100)
      .ToListAsync();
  }

  // Aggregation pipeline
  public async Task<List<BsonDocument>> EventosPorDiaAsync()
    => await _col.Aggregate()
        .Group(new BsonDocument {
          { "_id", new BsonDocument("$dateToString",
              new BsonDocument {{ "format", "%Y-%m-%d"}, { "date", "$criadoAt" }}) },
          { "total", new BsonDocument("$sum", 1) }
        })
        .ToListAsync();
}

Redis — Cache e Pub/Sub com StackExchange.Redis

C# + StackExchange.Redis — cache com IDistributedCache, pub/sub e Lua scripts atomicos
using Microsoft.Extensions.Caching.StackExchangeRedis;
using StackExchange.Redis;

// Registrar
builder.Services.AddStackExchangeRedisCache(o => o.Configuration = "localhost:6379");
builder.Services.AddSingleton<IConnectionMultiplexer>(
  ConnectionMultiplexer.Connect("localhost:6379"));

// Cache-aside pattern com IDistributedCache
class ProdutoCacheService(IDistributedCache cache, IProdutoRepository repo)
{
  public async Task<Produto?> GetOrSetAsync(int id, CancellationToken ct = default)
  {
    var key   = $"produto:{id}";
    var bytes = await cache.GetAsync(key, ct);
    if (bytes is { Length: > 0 })
      return JsonSerializer.Deserialize<Produto>(bytes);

    var produto = await repo.FindAsync(id, ct);
    if (produto is not null)
      await cache.SetAsync(key,
        JsonSerializer.SerializeToUtf8Bytes(produto),
        new DistributedCacheEntryOptions {
          AbsoluteExpirationRelativeToNow = TimeSpan.FromMinutes(10) }, ct);

    return produto;
  }

  public async Task InvalidarAsync(int id) =>
    await cache.RemoveAsync($"produto:{id}");
}

// Pub/Sub para invalidacao de cache distribuida
class CacheInvalidacaoPublisher(IConnectionMultiplexer mux)
{
  public async Task PublicarAsync(int produtoId)
  {
    var pub = mux.GetSubscriber();
    await pub.PublishAsync(RedisChannel.Literal("cache:invalidar"), produtoId.ToString());
  }
}

// Script Lua atomico — incremento com limite (rate limiting manual)
class RateLimiterRedis(IConnectionMultiplexer mux)
{
  private static readonly LuaScript _script = LuaScript.Prepare("""
    local cur = redis.call('incr', @key)
    if cur == 1 then redis.call('expire', @key, @ttl) end
    return cur
  """);

  public async Task<bool> PermitidoAsync(string ip, int limite = 100)
  {
    var db  = mux.GetDatabase();
    var res = (int) await db.ScriptEvaluateAsync(_script, new { key = $"rl:{ip}", ttl = 60 });
    return res <= limite;
  }
}
