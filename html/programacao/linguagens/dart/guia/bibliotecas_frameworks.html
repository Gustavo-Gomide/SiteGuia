Dart — Bibliotecas e Frameworks

Bibliotecas e Frameworks Dart

O ecossistema Dart e centrado no Flutter para UI cross-platform, mas conta tambem com uma suite de bibliotecas especializadas: Shelf para servidores HTTP minimalistas, Riverpod para gerenciamento de estado reativo, Drift para banco de dados SQLite type-safe, Dio para requisicoes HTTP com interceptors, RxDart para extensoes reativas e Bloc para arquitetura baseada em eventos.

Todas as bibliotecas Dart sao distribuidas via pub.dev — o repositorio oficial de pacotes. O arquivo pubspec.yaml gerencia dependencias e o comando dart pub get instala os pacotes listados.

HTTP: http e Dio

Dart — package:http para requisicoes simples
import 'dart:convert';
import 'package:http/http.dart' as http;

class ApiClient {
  final http.Client _client;
  final String _baseUrl;

  ApiClient(this._baseUrl, [http.Client? client])
    : _client = client ?? http.Client();

  Future<T> get<T>(String path, T Function(Map<String, dynamic>) parse) async {
    final uri = Uri.parse('$_baseUrl$path');
    final resp = await _client.get(uri, headers: {'Accept': 'application/json'});

    if (resp.statusCode >= 400) {
      throw HttpException('HTTP ${resp.statusCode}: ${resp.body}', uri: uri);
    }
    return parse(jsonDecode(resp.body) as Map<String, dynamic>);
  }

  Future<Map<String, dynamic>> post(String path, Map<String, dynamic> body) async {
    final resp = await _client.post(
      Uri.parse('$_baseUrl$path'),
      headers: {'Content-Type': 'application/json'},
      body: jsonEncode(body),
    );
    if (resp.statusCode >= 400) throw HttpException('${resp.statusCode}');
    return jsonDecode(resp.body) as Map<String, dynamic>;
  }

  void dispose() => _client.close();
}
Dart + Dio — interceptors, retry, upload com FormData e cancelamento
import 'package:dio/dio.dart';

final dio = Dio(BaseOptions(
  baseUrl:        'https://api.exemplo.com/v1',
  connectTimeout: const Duration(seconds: 10),
  receiveTimeout: const Duration(seconds: 30),
  headers: {'Accept': 'application/json'},
));

// Interceptor de autenticacao
dio.interceptors.add(InterceptorsWrapper(
  onRequest: (opts, handler) {
    opts.headers['Authorization'] = 'Bearer ${tokenStorage.get()}';
    handler.next(opts);
  },
  onError: (DioException e, handler) async {
    if (e.response?.statusCode == 401) {
      await renovarToken();
      final re = await dio.request(e.requestOptions.path,
        options: Options(method: e.requestOptions.method));
      handler.resolve(re);
    } else {
      handler.next(e);
    }
  },
));

// Requisicao simples com tipo generico
Future<List<Produto>> buscarProdutos() async {
  final res = await dio.get<List>('/produtos');
  return res.data!.map((j) => Produto.fromJson(j as Map<String, dynamic>)).toList();
}

// Upload multipart
Future<void> enviarFoto(String caminho) async {
  final form = FormData.fromMap({
    'foto': await MultipartFile.fromFile(caminho, filename: 'foto.jpg'),
    'descricao': 'Imagem de perfil',
  });
  await dio.post('/upload', data: form,
    onSendProgress: (sent, total) => print('${(sent/total*100).toInt()}%'));
}

// Cancelamento
final cancelToken = CancelToken();
try {
  final res = await dio.get('/lento', cancelToken: cancelToken);
} on DioException catch (e) {
  if (CancelToken.isCancel(e)) print('Cancelado!');
}
cancelToken.cancel('usuario saiu da tela');

Shelf — Servidor HTTP

Dart + Shelf — API REST com roteamento, middleware e JSON
import 'dart:convert';
import 'dart:io';
import 'package:shelf/shelf.dart';
import 'package:shelf/shelf_io.dart' as io;
import 'package:shelf_router/shelf_router.dart';

// Middleware de logging
Middleware logger() => (handler) => (request) async {
  final sw = Stopwatch()..start();
  final resp = await handler(request);
  print('${request.method} ${request.url} ${resp.statusCode} ${sw.elapsedMilliseconds}ms');
  return resp;
};

// Helper de resposta JSON
Response jsonResponse(dynamic data, {int status = 200}) => Response(
  status,
  body: jsonEncode(data),
  headers: {'Content-Type': 'application/json'},
);

// Service
class ProdutosService {
  final _dados = <String, Map<String, dynamic>>{
    '1': {'id': '1', 'nome': 'Mouse', 'preco': 89.9},
    '2': {'id': '2', 'nome': 'Teclado', 'preco': 249.0},
  };
  List<Map<String, dynamic>> listar()           => _dados.values.toList();
  Map<String, dynamic>?  buscar(String id)    => _dados[id];
}

// Handler com Router
Handler criarApp(ProdutosService svc) {
  final router = Router();

  router.get('/produtos', (req) => jsonResponse(svc.listar()));

  router.get('/produtos/<id>', (req, String id) {
    final p = svc.buscar(id);
    return p != null
      ? jsonResponse(p)
      : Response.notFound('{"erro":"nao encontrado"}');
  });

  router.post('/produtos', (req) async {
    final body = jsonDecode(await req.readAsString()) as Map<String, dynamic>;
    return jsonResponse(body, status: 201);
  });

  return Pipeline()
    .addMiddleware(logRequests())
    .addMiddleware(logger())
    .addHandler(router.call);
}

Future<void> main() async {
  final app    = criarApp(ProdutosService());
  final server = await io.serve(app, InternetAddress.anyIPv4, 8080);
  print('Servidor em http://localhost:${server.port}');
}

Dart + Riverpod — providers, AsyncNotifier e Ref
import 'package:flutter_riverpod/flutter_riverpod.dart';
import 'package:riverpod_annotation/riverpod_annotation.dart';
part 'providers.g.dart';

// Provider simples (valor imutavel)
final baseUrlProvider = Provider((ref) => 'https://api.exemplo.com');

// FutureProvider — async automatico
final produtosProvider = FutureProvider<List<Produto>>((ref) async {
  final client = ref.watch(apiClientProvider);
  return client.buscarProdutos();
});

// Anotacao @riverpod — gera produtosProvider automaticamente
@riverpod
Future<List<Produto>> produtos(ProdutosRef ref) async {
  final client = ref.watch(apiClientProvider);
  ref.keepAlive(); // nao descarta mesmo sem listener
  return client.buscarProdutos();
}

// StateNotifier para estado mutavel complexo
class CarrinhoNotifier extends Notifier<List<ItemCarrinho>> {
  @override
  List<ItemCarrinho> build() => [];

  void adicionar(Produto p) {
    final idx = state.indexWhere((i) => i.produto.id == p.id);
    if (idx >= 0) {
      state = [
        for (final (i, item) in state.indexed)
          if (i == idx) item.copyWith(qtd: item.qtd + 1) else item,
      ];
    } else {
      state = [...state, ItemCarrinho(produto: p, qtd: 1)];
    }
  }

  void remover(String produtoId) =>
    state = state.where((i) => i.produto.id != produtoId).toList();

  double get total =>
    state.fold(0, (acc, i) => acc + i.produto.preco * i.qtd);
}

final carrinhoProvider = NotifierProvider<CarrinhoNotifier, List<ItemCarrinho>>(
  CarrinhoNotifier.new,
);

// No widget
class ProdutosPage extends ConsumerWidget {
  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final state = ref.watch(produtosProvider);
    return state.when(
      loading: () => const CircularProgressIndicator(),
      error:   (e, st) => Text('Erro: $e'),
      data:    (lista) => ListView.builder(
        itemCount: lista.length,
        itemBuilder: (ctx, i) => ListTile(title: Text(lista[i].nome)),
      ),
    );
  }
}

Bloc — Arquitetura Baseada em Eventos

Dart + flutter_bloc — BLoC com events, states e Cubit
import 'package:flutter_bloc/flutter_bloc.dart';

// --- Cubit (BLoC simplificado sem eventos explícitos) ---
class ContadorCubit extends Cubit<int> {
  ContadorCubit() : super(0);
  void incrementar() => emit(state + 1);
  void decrementar() => emit(state - 1);
  void resetar()     => emit(0);
}

// --- BLoC completo com eventos ---
// Eventos
sealed class AuthEvent {}
class LoginRequested extends AuthEvent {
  final String email, senha;
  LoginRequested(this.email, this.senha);
}
class LogoutRequested extends AuthEvent {}

// Estados
sealed class AuthState {}
class AuthInitial      extends AuthState {}
class AuthLoading      extends AuthState {}
class AuthAuthenticated extends AuthState { final Usuario user; AuthAuthenticated(this.user); }
class AuthError         extends AuthState { final String msg;   AuthError(this.msg);  }

// BLoC
class AuthBloc extends Bloc<AuthEvent, AuthState> {
  final AuthService _authService;

  AuthBloc(this._authService) : super(AuthInitial()) {
    on<LoginRequested>(_onLogin);
    on<LogoutRequested>(_onLogout);
  }

  Future<void> _onLogin(LoginRequested event, Emitter<AuthState> emit) async {
    emit(AuthLoading());
    try {
      final user = await _authService.login(event.email, event.senha);
      emit(AuthAuthenticated(user));
    } catch (e) {
      emit(AuthError(e.toString()));
    }
  }

  Future<void> _onLogout(LogoutRequested _, Emitter<AuthState> emit) async {
    await _authService.logout();
    emit(AuthInitial());
  }
}

// No widget BlocBuilder
BlocBuilder<AuthBloc, AuthState>(
  builder: (ctx, state) => switch (state) {
    AuthInitial()             => const LoginScreen(),
    AuthLoading()             => const CircularProgressIndicator(),
    AuthAuthenticated(:var user) => HomeScreen(user: user),
    AuthError(:var msg)       => ErrorScreen(msg: msg),
  },
);
