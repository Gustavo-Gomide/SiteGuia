Dart — Avancado

Dart Avancado

Dart e construido sobre um event loop single-thread com async/await e Streams para concorrencia nao-bloqueante, e Isolates para paralelismo real com memoria isolada. Alem disso, extension methods permitem adicionar metodos a tipos existentes, code generation via build_runner elimina boilerplate, e dart:ffi garante interoperabilidade com bibliotecas C nativas com performance de chamada de funcao direta.

O modelo de concorrencia de Dart e baseado em message passing entre Isolates — sem estado compartilhado e sem race conditions. Dentro de um Isolate, tudo e single-thread, mas a VM pode executar multiplos Isolates em threads do sistema operacional distintas.

Async/Await e Futures

Dart — Future, async/await, tratamento de erros e combinators
import 'dart:convert';
import 'package:http/http.dart' as http;

// Funcao assincrona — retorna Future
Future<List<Produto>> buscarProdutos(String categoria) async {
  final uri = Uri.https('api.exemplo.com', '/produtos', {'cat': categoria});
  final resp = await http.get(uri);

  if (resp.statusCode != 200) {
    throw HttpException('Erro HTTP ${resp.statusCode}');
  }
  final json = jsonDecode(resp.body) as List;
  return json.map((j) => Produto.fromJson(j as Map<String, dynamic>)).toList();
}

// try/catch em codigo async
Future<void> main() async {
  try {
    final produtos = await buscarProdutos('eletronicos');
    print('${produtos.length} produtos encontrados');
  } on HttpException catch (e) {
    print('Falha de rede: $e');
  }

  // Future.wait — paralelo
  final [eletronicos, roupas, alimentos] = await Future.wait([
    buscarProdutos('eletronicos'),
    buscarProdutos('roupas'),
    buscarProdutos('alimentos'),
  ]);

  // Future.any — primeira que completar
  final rapidinha = await Future.any([
    buscarDoPrimeiroServidor(),
    buscarDoSegundoServidor(),
  ]);

  // Timeout
  final resultado = await buscarProdutos('gaming')
    .timeout(const Duration(seconds: 5),
     onTimeout: () => <Produto>[]);

  // Future.delayed — simula delay
  await Future.delayed(const Duration(milliseconds: 500));

  // Transformar Future com .then/.catchError (estilo callback)
  buscarProdutos('livros')
    .then((p) => print('Livros: ${p.length}'))
    .catchError((e) => print('Erro: $e'));
}

Streams

Dart — Stream single-subscription, broadcast, StreamController e async*
import 'dart:async';

// async* — generator de stream
Stream<int> contadorPeriodico() async* {
  var i = 0;
  while (true) {
    yield i++;
    await Future.delayed(const Duration(seconds: 1));
  }
}

// Consumir stream com await for
Future<void> lerContador() async {
  await for (final n in contadorPeriodico().take(5)) {
    print('tick: $n');
  }
}

// StreamController — controle manual
final StreamController<String> ctrl = StreamController.broadcast();

ctrl.stream
  .where((s) => s.startsWith('A'))
  .map((s) => s.toUpperCase())
  .listen((s) => print('Filtrado: $s'));

ctrl.add('Alice');       // Filtrado: ALICE
ctrl.add('Bob');         // ignorado
ctrl.add('Ana');         // Filtrado: ANA
ctrl.addError(Exception('falha'));
await ctrl.close();

// Transformar com StreamTransformer
final transformer = StreamTransformer<int, String>.fromHandlers(
  handleData: (data, sink) => sink.add('valor=$data'),
);
final strStream = Stream.fromIterable([1, 2, 3]).transform(transformer);

// StreamSubscription — controle de ciclo de vida
late StreamSubscription<int> sub;
sub = contadorPeriodico().listen(
  (n) { if (n == 3) sub.cancel(); },
  onError: (e) => print('Erro: $e'),
  onDone:  ()  => print('Stream encerrada'),
  cancelOnError: false,
);

Isolates — Paralelismo Real

Dart — Isolate.run, Isolate.spawn e comunicacao via SendPort
import 'dart:isolate';

// Isolate.run — modo mais simples (Dart 2.19+)
Future<void> exemploSimples() async {
  // Executa funcao top-level em isolate separado
  final resultado = await Isolate.run(() => _calcularHash('dados grandes'));
  print('Hash: $resultado');
}

String _calcularHash(String dados) {
  // CPU-intensivo — nao bloqueia o isolate principal
  var hash = 0;
  for (final c in dados.codeUnits) hash = (hash * 31 + c).toUnsigned(32);
  return hash.toRadixString(16).padLeft(8, '0');
}

// Isolate.spawn — comunicacao bidirecional via SendPort
Future<void> exemploSpawn() async {
  final rp = ReceivePort();
  await Isolate.spawn(_workerEntry, rp.sendPort);

  final workerSendPort = await rp.first as SendPort;

  // Enviar mensagem e aguardar resposta
  final responsePort = ReceivePort();
  workerSendPort.send(['calcular', 1000000, responsePort.sendPort]);
  final resultado = await responsePort.first;
  print('Resultado: $resultado');
  rp.close();
}

// Entry point do isolate worker (deve ser top-level ou static)
void _workerEntry(SendPort mainSendPort) {
  final workerRP = ReceivePort();
  mainSendPort.send(workerRP.sendPort); // se apresenta
  workerRP.listen((msg) {
    final [cmd, n, SendPort reply] = msg as List;
    if (cmd == 'calcular') {
      var soma = 0;
      for (var i = 0; i < (n as int); i++) soma += i;
      reply.send(soma);
    }
  });
}

Extension Methods e Callable Classes

Dart — extension methods, extension types e callable classes
// Extension em tipo existente
extension StringUtils on String {
  String get capitalizado => isEmpty ? this :
    this[0].toUpperCase() + substring(1);

  bool get eEmail => RegExp(r'^[^\s@]+@[^\s@]+\.[^\s@]+$').hasMatch(this);

  String truncar(int max, [String sufixo = '...']) =>
    length <= max ? this : substring(0, max - sufixo.length) + sufixo;
}

print('hello'.capitalizado);       // Hello
print('user@mail.com'.eEmail);     // true
print('texto longo aqui'.truncar(10)); // texto lon...

// Extension em nullable
extension NullableInt on int? {
  int get ouZero => this ?? 0;
}
int? x;
print(x.ouZero); // 0

// Extension type — wrapper zero-cost (Dart 3.3+)
extension type Dinheiro(int centavos) implements int {
  Dinheiro operator +(Dinheiro outro) => Dinheiro(centavos + outro.centavos);
  String get formatado => 'R\$ ${(centavos / 100).toStringAsFixed(2)}';
}

final preco = Dinheiro(1999);
print(preco.formatado); // R$ 19.99

// Callable class — instancia chamada como funcao
class Multiplicador {
  final int fator;
  const Multiplicador(this.fator);
  int call(int valor) => valor * fator;
}

final dobrar = Multiplicador(2);
print(dobrar(5));  // 10
final lista = [1, 2, 3];
print(lista.map(dobrar).toList()); // [2, 4, 6]

Code Generation: json_serializable e freezed

Dart — json_serializable para fromJson/toJson sem boilerplate
// pubspec.yaml (dependencias)
// dependencies:
//   json_annotation: ^4.9.0
// dev_dependencies:
//   build_runner: ^2.4.0
//   json_serializable: ^6.8.0

import 'package:json_annotation/json_annotation.dart';
part 'produto.g.dart'; // arquivo gerado

@JsonSerializable()
class Produto {
  final String  id;
  final String  nome;
  final double  preco;
  @JsonKey(name: 'em_estoque')
  final bool    emEstoque;
  final List<String> tags;

  const Produto({
    required this.id,
    required this.nome,
    required this.preco,
    required this.emEstoque,
    this.tags = const [],
  });

  // Metodos gerados pelo build_runner
  factory Produto.fromJson(Map<String, dynamic> json) => _$ProdutoFromJson(json);
  Map<String, dynamic> toJson() => _$ProdutoToJson(this);
}

// Gerar codigo:
// dart run build_runner build --delete-conflicting-outputs

// Uso
final json = {'id': 'p1', 'nome': 'Mouse', 'preco': 89.9, 'em_estoque': true};
final p = Produto.fromJson(json);
print(p.nome);       // Mouse
print(p.toJson()); // {id: p1, nome: Mouse, preco: 89.9, em_estoque: true}
Dart — freezed: classes imutaveis com copyWith, union types e pattern matching
// pubspec.yaml
// dependencies:   freezed_annotation: ^2.4.0, json_annotation: ^4.9.0
// dev_deps:       freezed: ^2.5.0, json_serializable: ^6.8.0, build_runner

import 'package:freezed_annotation/freezed_annotation.dart';
part 'pedido.freezed.dart';
part 'pedido.g.dart';

@freezed
class Pedido with _$Pedido {
  const factory Pedido({
    required String     id,
    required String     cliente,
    required List<ItemPedido> itens,
    @Default(StatusPedido.pendente) StatusPedido status,
  }) = _Pedido;

  factory Pedido.fromJson(Map<String, dynamic> j) => _$PedidoFromJson(j);
}

// Union type com freezed
@freezed
sealed class EstadoPagamento with _$EstadoPagamento {
  const factory EstadoPagamento.aguardando() = Aguardando;
  const factory EstadoPagamento.aprovado(String transacaoId) = Aprovado;
  const factory EstadoPagamento.recusado(String motivo, int codigo) = Recusado;
}

String descreverPagamento(EstadoPagamento estado) => estado.map(
  aguardando: (_)  => 'Aguardando confirmacao...',
  aprovado:   (a)  => 'Aprovado! TX=${a.transacaoId}',
  recusado:   (r)  => 'Recusado [${r.codigo}]: ${r.motivo}',
);

// copyWith automatico
final p1 = Pedido(id: '001', cliente: 'Alice', itens: []);
final p2 = p1.copyWith(status: StatusPedido.finalizado);

Dart FFI — Interoperabilidade com C

Dart — dart:ffi para chamar biblioteca C nativa
import 'dart:ffi';
import 'dart:io';
import 'package:ffi/ffi.dart'; // utilitarios como Utf8/malloc

// Assinatura C: int adicionar(int a, int b)
typedef AdicionarC    = Int32 Function(Int32 a, Int32 b);
typedef AdicionarDart = int   Function(int  a, int  b);

void usarBibliotecaC() {
  final lib = DynamicLibrary.open(
    Platform.isWindows ? 'mylib.dll' : 'libmylib.so'
  );

  final AdicionarDart adicionar = lib
    .lookupFunction<AdicionarC, AdicionarDart>('adicionar');

  print(adicionar(10, 20)); // 30

  // Trabalhar com strings C
  typedef GreetC    = Pointer<Utf8> Function(Pointer<Utf8>);
  typedef GreetDart = Pointer<Utf8> Function(Pointer<Utf8>);
  final greet = lib.lookupFunction<GreetC, GreetDart>('greet');

  final nomeCStr = 'Alice'.toNativeUtf8();
  try {
    final respStr = greet(nomeCStr).toDartString();
    print(respStr); // Hello, Alice!
  } finally {
    malloc.free(nomeCStr);
  }

  // Struct C
  final ponto = calloc<PontoC>();
  ponto.ref.x = 3.0;
  ponto.ref.y = 4.0;
  final dist = lib.lookupFunction<DistanciaC, DistanciaDart>('distancia')(ponto);
  print(dist); // 5.0
  calloc.free(ponto);
}

// Struct Dart espelhando struct C
class PontoC extends Struct {
  @Double() external double x;
  @Double() external double y;
}
