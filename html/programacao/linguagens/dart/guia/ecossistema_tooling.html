Dart — Ecossistema e Tooling

Ecossistema e Tooling Dart

Dart possui um ecossistema de ferramentas integradas e coesas: o dart CLI unifica compilacao, analise, formatacao, gerenciamento de pacotes e execucao de scripts. O gerenciador pub instala dependencias do pub.dev, o maior repositorio de pacotes Dart/Flutter. A ferramenta build_runner automatiza geracao de codigo, e o pacote lints define regras de qualidade do Google.

O Dart SDK e completamente open-source e versionado semanticamente. A atualizacao do SDK e do Flutter e feita com um unico comando: flutter upgrade (que inclui o Dart SDK atualizado).

dart CLI — Comandos Essenciais

Shell — dart CLI completo
## CRIAR PROJETOS
dart create meu_app                         # console
dart create -t package  minha_lib           # pacote publico
dart create -t web-simple meu_site          # web
dart create -t server-shelf meu_servidor    # servidor Shelf

## EXECUTAR
dart run                                    # bin/*.dart
dart run bin/server.dart                    # arquivo especifico
dart run meu_app:ferramenta                 # executavel definido em pubspec

## COMPILAR
dart compile exe  bin/main.dart -o app      # executavel nativo
dart compile js   bin/main.dart -o app.js   # JavaScript otimizado
dart compile wasm bin/main.dart -o app.wasm # WebAssembly (experimental)
dart compile aot-snapshot bin/main.dart     # snapshot AOT
dart compile jit-snapshot  bin/main.dart    # snapshot JIT (mais rapido na 1a exec)
dart compile kernel        bin/main.dart    # bytecode portavel

## ANALISE E QUALIDADE
dart analyze                                # analise estatica completa
dart analyze --fatal-infos                  # trata infos como erros (CI)
dart fix --apply                            # aplica todas as correcoes sugeridas
dart fix --dry-run                          # mostra sem aplicar
dart format .                               # formata todos os .dart
dart format --set-exit-if-changed .         # falha se houver mudancas (CI)

## TESTES
dart test                                   # todos os testes em test/
dart test test/produto_test.dart            # arquivo especifico
dart test --reporter expanded               # saida detalhada
dart test --coverage coverage/              # gera dados de cobertura LCOV

## PACOTES
dart pub get                                # instalar dependencias
dart pub upgrade                            # atualizar para ultimas versoes
dart pub upgrade --major-versions           # atualizar major version
dart pub deps                               # arvore de dependencias
dart pub outdated                           # pacotes desatualizados
dart pub publish                            # publicar no pub.dev
dart pub publish --dry-run                  # verificar antes de publicar
dart pub cache clean                        # limpar cache de pacotes

## BUILD RUNNER (code generation)
dart run build_runner build                       # gerar uma vez
dart run build_runner build --delete-conflicting-outputs
dart run build_runner watch                       # modo watch
dart run build_runner clean                       # limpar arquivos gerados

Analise Estatica e Linting

YAML — analysis_options.yaml para projetos rigorosos
include: package:lints/recommended.yaml   # regras Google oficiais

analyzer:
  language:
    strict-casts: true         # proibe casts implicitos de dynamic
    strict-inference: true     # exige tipos inferidos sem ambiguidade
    strict-raw-types: true     # proibe generics sem tipo (List sem <T>)
  errors:
    invalid_annotation_target: ignore   # silencia warning de annotations
    missing_required_param: error
    dead_code: warning
  exclude:
    - '**/*.g.dart'            # arquivos gerados
    - '**/*.freezed.dart'
    - lib/generated/**

linter:
  rules:
    # Boas praticas obrigatorias
    - prefer_const_constructors
    - prefer_final_fields
    - avoid_print                      # use logger em producao
    - always_declare_return_types
    - prefer_single_quotes
    - sort_pub_dependencies
    - unawaited_futures                # toda Future deve ser awaited
    - cancel_subscriptions             # StreamSubscription deve ser cancelada
    - close_sinks                      # StreamController deve ser fechado
    # Estilo
    - lines_longer_than_80_chars: false  # flexibilizar comprimento de linha

Testes com package:test e Mockito

Dart — testes unitarios, mocks com Mockito e cobertura
import 'package:test/test.dart';
import 'package:mockito/annotations.dart';
import 'package:mockito/mockito.dart';
import 'produto_service.mocks.dart';

@GenerategMocks([ApiClient])
void main() {
  late MockApiClient mockClient;
  late ProdutoService service;

  setUp(() {
    mockClient = MockApiClient();
    service    = ProdutoService(mockClient);
  });

  tearDown(() => verifyNoMoreInteractions(mockClient));

  group('ProdutoService.buscar', () {
    test('retorna produto quando encontrado', () async {
      final produtoMock = Produto(id: '1', nome: 'Mouse', preco: 89.9, emEstoque: true);
      when(mockClient.get('/produtos/1', any))
        .thenAnswer((_) async => produtoMock);

      final resultado = await service.buscar('1');

      expect(resultado.nome, equals('Mouse'));
      expect(resultado.preco, equals(89.9));
      verify(mockClient.get('/produtos/1', any)).called(1);
    });

    test('lanca excecao quando nao encontrado', () async {
      when(mockClient.get('/produtos/99', any))
        .thenThrow(HttpException('HTTP 404'));

      expect(
        () => service.buscar('99'),
        throwsA(isA<HttpException>()),
      );
    });
  });

  group('ProdutoService.listar', () {
    test('retorna lista vazia quando API retorna []', () async {
      when(mockClient.get('/produtos', any)).thenAnswer((_) async => <Produto>[]);
      final lista = await service.listar();
      expect(lista, isEmpty);
    });
  });

  // Teste de stream
  test('stream emite eventos em ordem', () async {
    expect(
      service.atualizacoes(),
      emitsInOrder(['adicionado', 'atualizado', emitsDone]),
    );
  });
}

CI/CD com GitHub Actions

YAML — GitHub Actions para Dart
name: Dart CI

on:
  push:    { branches: [main] }
  pull_request: {}

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: dart-lang/setup-dart@v1
        with: { sdk: stable }

      - name: Instalar dependencias
        run: dart pub get

      - name: Verificar formatacao
        run: dart format --set-exit-if-changed .

      - name: Analise estatica
        run: dart analyze --fatal-infos

      - name: Testes com cobertura
        run: dart test --coverage coverage/

      - name: Gerar relatorio LCOV
        run: dart run coverage:format_coverage --lcov --in=coverage --out=coverage/lcov.info

      - uses: coverallsapp/github-action@v2
        with:
          github-token: ${{ secrets.GITHUB_TOKEN }}

  publish:
    needs: test
    if: github.ref == 'refs/heads/main' && startsWith(github.ref, 'refs/tags/v')
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: dart-lang/setup-dart@v1
      - run: dart pub get
      - run: dart pub publish --force
        env:
          PUB_CREDENTIALS: ${{ secrets.PUB_DEV_TOKEN }}
