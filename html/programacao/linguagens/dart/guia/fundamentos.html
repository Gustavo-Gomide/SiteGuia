Dart — Fundamentos

Fundamentos do Dart

Dart combina a familiaridade do C/Java com funcionalidades modernas: null safety sound, inferencia de tipos, closures de primeira classe, mixins para reutilizacao horizontal de codigo, generics covariantes e um sistema de colecoes rico. Esta secao cobre desde os tipos basicos ate OOP completa com heranca, mixins e interfaces — tudo que voce precisa para escrever Dart idiomatico de qualidade.

Em Dart tudo e objeto, incluindo numeros, funcoes e null. O tipo raiz e Object? — o sufixo ? indica que uma variavel pode conter null. Sem o ?, o compilador garante que a variavel nunca sera null em runtime.

Sistema de Tipos e Variaveis

Dart — Tipos primitivos, inferencia e null safety
// Declaracao explicita
int    contagem = 42;
double preco    = 19.99;
String nome     = 'Alice';
bool   ativo    = true;

// Inferencia de tipo
var    total    = 100;          // int
var    pi       = 3.14;         // double
final  url      = 'https://dart.dev'; // tipo inferido, nao reatribuivel
const  PI       = 3.14159265;  // compile-time constant

// late — inicializacao adiada, segura
late String descricao;
descricao = calcularDescricao(); // deve ser chamado antes de usar

// Nullable
String? apelido;               // null por default
apelido = 'Al';

// Operadores null
int len  = apelido!.length;          // force-unwrap (throw se null)
int? len2 = apelido?.length;        // null propagation
String eff = apelido ?? 'anonimo';  // null coalescing
apelido ??= 'sem apelido';          // assign-if-null

// Type test e cast
Object obj = 'hello';
if (obj is String) {
  print(obj.toUpperCase()); // promovido automaticamente a String
}
if (obj is! int) print('nao e int');
String s = obj as String;       // cast explicito

// dynamic — sem verificacao estatica (evitar!)
dynamic qualquer = 42;
qualquer = 'agora e string'; // ok em runtime, perigoso

final nums = [3, 1, 4, 1, 5];

Funcoes e Closures

Dart — Funcoes: parametros nomeados, opcionais, arrow, closures
// Parametros posicionais obrigatorios
int somar(int a, int b) => a + b;

// Parametros nomeados (obrigatorios com required)
String saudacao({required String nome, String prefixo = 'Ola'}) =>
  '$prefixo, $nome!';
print(saudacao(nome: 'Alice'));                  // Ola, Alice!
print(saudacao(nome: 'Bob', prefixo: 'Oi'));   // Oi, Bob!

// Parametros opcionais posicionais
String formatar(String texto, [int largura = 80, bool alinha = false]) =>
  texto.padLeft(alinha ? largura : 0);

// Funcoes como valores (first-class)
int aplicar(int Function(int) op, int valor) => op(valor);
print(aplicar((x) => x * x, 5)); // 25

// Typedef para alias de funcao
typedef Predicado<T> = bool Function(T value);
Predicado<int> ehPar = (n) => n % 2 == 0;

// Closures — capturam variaveis do escopo
List<int Function()> criarContadores() {
  return List.generate(3, (i) {
    var cnt = 0;
    return () => ++cnt; // cada closure tem seu proprio cnt
  });
}
final ctrs = criarContadores();
print(ctrs[0]()); // 1
print(ctrs[0]()); // 2
print(ctrs[1]()); // 1 — independente

// Funcao generica
T primeiro<T>(List<T> lista) => lista.first;
print(primeiro([10, 20]));           // 10
print(primeiro(['a', 'b']));        // a

OOP: Classes, Mixins, Interfaces e Heranca

Dart — Classe completa com construtor, factory e operador
class Ponto {
  // Campos finais — imutaveis apos construcao
  final double x;
  final double y;

  // Construtor com sintaxe inicializadora (this.x)
  const Ponto(this.x, this.y);

  // Construtor nomeado
  const Ponto.origem() : x = 0, y = 0;

  // Factory constructor — pode retornar instancia cacheada ou subtipo
  factory Ponto.fromMap(Map<String, double> m) =>
    Ponto(m['x']!, m['y']!);

  // Getter calculado
  double get distancia => _magnitude(x, y);

  // Operator overloading
  Ponto operator +(Ponto outro) => Ponto(x + outro.x, y + outro.y);
  bool  operator ==(Object other) =>
    other is Ponto && x == other.x && y == other.y;

  @override
  int get hashCode => Object.hash(x, y);

  @override
  String toString() => 'Ponto($x, $y)';
}

double _magnitude(double x, double y) =>
  double.parse(((x * x + y * y).sqrt()).toStringAsFixed(4));

// Uso
const p1 = Ponto(3, 4);
const p2 = Ponto(1, 1);
print(p1 + p2);          // Ponto(4.0, 5.0)
print(p1.distancia);     // 5.0
Dart — Heranca, abstract, interface e mixins
// Classe abstrata
abstract class Forma {
  double area();
  String get descricao;
  void exibir() => print('$descricao → area = ${area().toStringAsFixed(2)}');
}

// Heranca unica
class Circulo extends Forma {
  final double raio;
  Circulo(this.raio);

  @override
  double area() => 3.14159 * raio * raio;

  @override
  String get descricao => 'Circulo(raio=$raio)';
}

// Interface implicita: toda classe define uma interface
abstract interface class Serializable {
  Map<String, dynamic> toJson();
}

// Mixin — comportamento horizontal
mixin Logavel {
  String get _tag => runtimeType.toString();
  void log(String msg) => print('[$_tag] $msg');
}

// Mixin com restricao de tipo
mixin Validavel on Forma {
  bool eValido() => area() > 0;
}

// Combinar heranca + interface + mixin
class Retangulo extends Forma
    with Logavel, Validavel
    implements Serializable {
  final double largura, altura;
  Retangulo(this.largura, this.altura);

  @override
  double area() => largura * altura;

  @override
  String get descricao => 'Retangulo(${largura}x$altura)';

  @override
  Map<String, dynamic> toJson() =>
    {'tipo': 'retangulo', 'largura': largura, 'altura': altura};
}

final r = Retangulo(5, 3);
r.exibir();       // Retangulo(5.0x3.0) → area = 15.00
r.log('criado');  // [Retangulo] criado
print(r.eValido()); // true
Dart 3 — Records e Pattern Matching
// Record — tupla tipada imutavel
(String, int) par = ('Alice', 30);
print(par.$1); // Alice
print(par.$2); // 30

// Record com campos nomeados
typedef Ponto2D = ({double x, double y});
Ponto2D p = (x: 3.0, y: 4.0);
print('${p.x}, ${p.y}');

// Destructuring de records
var (nome, idade) = ('Bob', 25);
print('$nome tem $idade anos');

// Pattern matching em switch
Object valor = 42;
switch (valor) {
  case int n when n < 0:
    print('negativo: $n');
  case int n:
    print('positivo ou zero: $n');
  case String s:
    print('string: $s');
  case _:
    print('outro');
}

// Sealed class com exhaustivo switch
sealed class Resultado<T> {}
class Sucesso<T> extends Resultado<T> { final T valor; Sucesso(this.valor); }
class Falha<T>   extends Resultado<T> { final String erro; Falha(this.erro); }

String descrever<T>(Resultado<T> r) => switch (r) {
  Sucesso(:final valor) => 'OK: $valor',
  Falha(:final erro)    => 'Erro: $erro',
}; // compilador verifica exaustividade

Excecoes e Tratamento de Erros

Dart — throw, try/on/catch/finally e excecoes customizadas
// Excecao customizada
class ValidacaoException implements Exception {
  final String campo;
  final String mensagem;
  const ValidacaoException(this.campo, this.mensagem);
  @override
  String toString() => 'ValidacaoException[$campo]: $mensagem';
}

void validarIdade(int idade) {
  if (idade < 0)  throw ValidacaoException('idade', 'deve ser >= 0');
  if (idade > 150) throw RangeError.range(idade, 0, 150, 'idade');
}

try {
  validarIdade(-5);
} on ValidacaoException catch (e) {
  print('Validacao: $e');
} on RangeError catch (e, stackTrace) {
  print('Range: $e');
  print(stackTrace);
} catch (e) {
  print('Desconhecido: $e');
} finally {
  print('Sempre executado');
}

// Rethrow
void processar() {
  try {
    operacaoRisco();
  } catch (e) {
    _registrarErro(e);
    rethrow; // preserva stack trace original
  }
}

Generics

Dart — Classes e funcoes genericas com reificacao em runtime
// Classe generica
class Pilha<T> {
  final List<T> _dados = [];

  void push(T valor) => _dados.add(valor);
  T    pop()         => _dados.removeLast();
  T    get topo      => _dados.last;
  bool get vazia     => _dados.isEmpty;
  int  get tamanho   => _dados.length;
}

final pilhaInt = Pilha<int>();
pilhaInt.push(10);
pilhaInt.push(20);
print(pilhaInt.pop()); // 20

// Dart reifica generics — disponivel em runtime!
print(pilhaInt is Pilha<int>);  // true
print(pilhaInt.runtimeType);       // Pilha<int>

// Restricao de tipo
class Repositorio<T extends Serializable> {
  final Map<String, T> _store = {};
  void   salvar(String id, T item) => _store[id] = item;
  T?     buscar(String id)         => _store[id];
  List<T> listar()                  => _store.values.toList();
}

// Funcao generica
T? primeiroPor<T>(List<T> lista, bool Function(T) teste) {
  for (final item in lista) {
    if (teste(item)) return item;
  }
  return null;
}
print(primeiroPor([1, 3, 4, 6], (n) => n.isEven)); // 4
