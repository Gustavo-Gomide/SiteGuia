Dart — Embedded e IoT

Dart — Embedded e IoT

Dart pode ser usado em sistemas embarcados Linux (Raspberry Pi, BeagleBone, Toradex) via dart_periphery para controlar GPIO, I2C, SPI e UART. Para IoT, mqtt_client implementa MQTT 3/5 com wss e QoS. Flutter Embedded permite criar paineis HMI touch-screen diretamente no flutter-pi sem X11/Wayland, rodando direto no framebuffer Linux.

dart_periphery usa dart:ffi para chamar a biblioteca C c-periphery diretamente — sem processos filhos, sem latencia de subprocess, acesso direto aos drivers Linux do kernel via /dev/ e /sys/.

GPIO — Saidas e Entradas Digitais

Dart + dart_periphery — GPIO output (LED blink) e input com interrupt
import 'package:dart_periphery/dart_periphery.dart';

void main() {
  // GPIO 17 como saida (LED)
  final led = GPIO.advanced(17, GPIOdirection.gpioDirOut);

  try {
    for (var i = 0; i < 10; i++) {
      led.write(true);
      sleep(const Duration(milliseconds: 500));
      led.write(false);
      sleep(const Duration(milliseconds: 500));
    }
  } finally {
    led.dispose();
  }
}

// GPIO como entrada com deteccao de borda (interrupt)
void botaoInterrupt() {
  final btn = GPIO.advanced(18, GPIOdirection.gpioDirIn,
    edge: GPIOedge.gpioEdgeBoth);

  print('Aguardando borda...');

  while (true) {
    final evento = btn.readEvent(timeout: 10000);  // 10s timeout
    if (evento != null) {
      print('Borda ${evento.edge.name} em ${evento.timestamp}ns');
      // Debounce simples
      sleep(const Duration(milliseconds: 50));
    }
  }
}

I2C e SPI — Sensores e Perifericos

Dart + dart_periphery — I2C: sensor BME280 (temperatura, umidade, pressao)
import 'package:dart_periphery/dart_periphery.dart';

class BME280 {
  static const _addr    = 0x76;
  static const _regTemp = 0xFA;
  static const _regCtrl  = 0xF4;

  final I2C _i2c;

  BME280(this._i2c) {
    // ctrl_meas: osrs_t=001 osrs_p=001 mode=11 (normal)
    _i2c.writeByte(_addr, _regCtrl, 0x27);
    sleep(const Duration(milliseconds: 100));
  }

  double get temperatura {
    final raw = _i2c.readBytesReg(_addr, _regTemp, 3);
    final adc = ((raw[0] << 12) | (raw[1] << 4) | (raw[2] >> 4));
    return _compensarTemperatura(adc);
  }

  double _compensarTemperatura(int adc) {
    // Compensacao conforme datasheet Bosch
    final digT1 = _i2c.readWordReg(_addr, 0x88);
    final digT2 = _i2c.readWordReg(_addr, 0x8A);
    final digT3 = _i2c.readWordReg(_addr, 0x8C);
    var var1 = ((adc / 16384.0) - (digT1 / 1024.0)) * digT2;
    var var2 = (adc / 131072.0 - digT1 / 8388608.0);
    var2     = var2 * var2 * digT3;
    return (var1 + var2) / 5120.0;
  }

  void fechar() => _i2c.dispose();
}

void main() {
  final i2c = I2C(1);  // /dev/i2c-1
  final sensor = BME280(i2c);

  try {
    while (true) {
      print('Temperatura: ${sensor.temperatura.toStringAsFixed(2)}°C');
      sleep(const Duration(seconds: 2));
    }
  } finally {
    sensor.fechar();
    i2c.dispose();
  }
}
Dart + dart_periphery — SPI: display OLED SSD1306
import 'package:dart_periphery/dart_periphery.dart';
import 'dart:typed_data';

class SSD1306 {
  final SPI   _spi;
  final GPIO  _dc;   // Data/Command
  final GPIO  _rst;  // Reset
  static const _w = 128, _h = 64;
  final _buffer = Uint8List(_w * _h ~/ 8);

  SSD1306(this._spi, this._dc, this._rst) {
    _inicializar();
  }

  void _cmd(int cmd) {
    _dc.write(false);
    _spi.transfer([cmd]);
  }

  void _inicializar() {
    _rst.write(false); sleep(const Duration(milliseconds: 10));
    _rst.write(true);
    for (final c in [0xAE, 0xD5, 0x80, 0xA8, 0x3F, 0xD3, 0x00, 0xAF]) {
      _cmd(c);
    }
  }

  void setPixel(int x, int y, bool ligado) {
    if (x < 0 || x >= _w || y < 0 || y >= _h) return;
    final idx = x + (y ~/ 8) * _w;
    if (ligado) {
      _buffer[idx] |= (1 << (y & 7));
    } else {
      _buffer[idx] &= ~(1 << (y & 7));
    }
  }

  void exibir() {
    _dc.write(true);
    _spi.transfer(_buffer);
  }
}

MQTT — Internet das Coisas

Dart + mqtt_client — Publicar e subscrever topicos com QoS
import 'package:mqtt_client/mqtt_client.dart';
import 'package:mqtt_client/mqtt_server_client.dart';
import 'dart:convert';
import 'dart:io';

class MqttIoT {
  late MqttServerClient _client;
  final String _host, _clientId;

  MqttIoT({required String host, required String clientId})
    : _host = host, _clientId = clientId;

  Future<void> conectar({
    String? usuario, String? senha, bool tls = false}) async {
    _client = MqttServerClient.withPort(_host, _clientId, tls ? 8883 : 1883);
    _client.logging(on: false);
    _client.keepAlivePeriod = 30;
    _client.autoReconnect  = true;
    _client.onAutoReconnect = () => print('Reconectando...');

    if (tls) {
      _client.secure = true;
      _client.securityContext = SecurityContext()
        ..setTrustedCertificates('ca.crt')
        ..useCertificateChain('client.crt')
        ..usePrivateKey('client.key');
    }

    final connMsg = MqttConnectMessage()
      .withClientIdentifier(_clientId)
      .startClean()
      .withWillTopic('dispositivos/$_clientId/status')
      .withWillMessage('offline')
      .withWillQos(MqttQos.atLeastOnce);

    if (usuario != null) connMsg.authenticateAs(usuario, senha);

    _client.connectionMessage = connMsg;

    await _client.connect(usuario, senha);
    if (_client.connectionStatus!.state != MqttConnectionState.connected) {
      throw Exception('Falha: ${_client.connectionStatus!.returnCode}');
    }
    print('Conectado ao broker MQTT!');
  }

  void publicar(String topico, Map<String, dynamic> payload,
    {MqttQos qos = MqttQos.atMostOnce}) {
    final builder = MqttClientPayloadBuilder()
      ..addString(jsonEncode(payload));
    _client.publishMessage(topico, qos, builder.payload!, retain: false);
  }

  void subscrever(String topico, void Function(String, Map<String, dynamic>) handler) {
    _client.subscribe(topico, MqttQos.atLeastOnce);
    _client.updates!.listen((events) {
      for (final ev in events) {
        final msg = ev.payload as MqttPublishMessage;
        final raw = MqttPublishPayload.bytesToStringAsString(msg.payload.message);
        try {
          handler(ev.topic, jsonDecode(raw) as Map<String, dynamic>);
        } catch (_) { /* payload nao e JSON */ }
      }
    });
  }

  void desconectar() => _client.disconnect();
}

// Uso: sensor envia telemetria a cada 5s
Future<void> main() async {
  final mqtt = MqttIoT(host: 'broker.local', clientId: 'sensor-01');
  await mqtt.conectar(usuario: 'iot', senha: 's3cr3t', tls: true);

  mqtt.subscrever('atuadores/sensor-01/cmd', (topico, payload) {
    print('Comando recebido: $payload');
  });

  final i2c    = I2C(1);
  final sensor = BME280(i2c);

  while (true) {
    mqtt.publicar('sensores/sensor-01/telemetria', {
      'temperatura': sensor.temperatura,
      'timestamp': DateTime.now().toIso8601String(),
    }, qos: MqttQos.atLeastOnce);
    sleep(const Duration(seconds: 5));
  }
}

UART — Comunicacao Serial

Dart + dart_periphery — UART: ler GPS NMEA e modulo Modbus RTU
import 'package:dart_periphery/dart_periphery.dart';
import 'dart:typed_data';

class GpsReader {
  final Serial _serial;
  final _buffer = StringBuffer();

  GpsReader({String porta = '/dev/ttyAMA0', int baud = 9600})
    : _serial = Serial(porta, Baudrate.b9600);

  // Le sentencas NMEA ate encontrar GPGGA (posicao)
  Future<Map<String, dynamic>?> lerPosicao({Duration timeout = const Duration(seconds: 5)}) async {
    final limite = DateTime.now().add(timeout);
    while (DateTime.now().isBefore(limite)) {
      if (_serial.inputWaiting() > 0) {
        final bytes = _serial.read(64, timeout: 100);
        _buffer.write(String.fromCharCodes(bytes));
      }
      final linha = _extrairLinha();
      if (linha != null && linha.startsWith(r'$GPGGA')) {
        return _parseGpgga(linha);
      }
      await Future.delayed(const Duration(milliseconds: 50));
    }
    return null;
  }

  String? _extrairLinha() {
    final str = _buffer.toString();
    final nl  = str.indexOf('\n');
    if (nl == -1) return null;
    final linha = str.substring(0, nl).trim();
    _buffer.clear();
    _buffer.write(str.substring(nl + 1));
    return linha;
  }

  Map<String, dynamic>? _parseGpgga(String linha) {
    final campos = linha.split(',');
    if (campos.length < 10 || campos[6] == '0') return null;
    return {
      'lat':  double.tryParse(campos[2]),
      'lon':  double.tryParse(campos[4]),
      'alt':  double.tryParse(campos[9]),
      'sats': int.tryParse(campos[7]),
    };
  }

  void fechar() => _serial.dispose();
}

Flutter Embedded — flutter-pi no Raspberry Pi

Shell — Instalar e rodar Flutter app no Raspberry Pi sem X11
# Instalar flutter-pi no Raspberry Pi (ARM64)
sudo apt install cmake libgles2-mesa-dev libdrm-dev libgbm-dev \
  libsystemd-dev libinput-dev libudev-dev libxkbcommon-dev

git clone --recursive https://github.com/ardera/flutter-pi.git
cd flutter-pi ; mkdir build ; cd build
cmake .. ; make -j$(nproc)
sudo make install

# No host (maquina de desenvolvimento): build do app
flutter build linux --release  # ou bundle para ARM
flutter build bundle            # gera flutter_assets/

# Copiar assets para o Pi
rsync -avz build/flutter_assets/ pi@raspberrypi.local:~/app/

# Rodar no Pi diretamente no framebuffer
flutter-pi --release ~/app

# Configurar como servico systemd
cat > /etc/systemd/system/flutter-app.service <<EOF
[Unit]
Description=Flutter IoT Dashboard
After=network.target

[Service]
ExecStart=/usr/local/bin/flutter-pi --release /home/pi/app
Restart=always
User=pi
Environment=DISPLAY=:0

[Install]
WantedBy=multi-user.target
EOF

systemctl enable flutter-app
systemctl start  flutter-app

Plataformas Suportadas

Raspberry Pi 3/4/5 (ARM32 e ARM64), BeagleBone, Toradex Colibri/Apalis, qualquer SBC Linux com driver DRM/KMS e OpenGL ES 2.0+

Display Suportado

HDMI, DSI (ribbon), SPI (via DRM), touch-screen capacitivo via libinput — totalmente sem X11 ou Wayland, renderizacao direta no framebuffer

Performance

60fps em Raspberry Pi 4 com Impeller (OpenGL ES), uso de CPU/GPU nativo do SoC, ideal para dashboards industriais, paineis de controle HMI e kiosks
