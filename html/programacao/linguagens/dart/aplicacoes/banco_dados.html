Dart — Banco de Dados

O ecossistema Dart/Flutter oferece solucoes para cada camada de persistencia: Drift para SQLite com codigo type-safe gerado em tempo de compilacao, Hive e Isar para NoSQL embarcado de alta performance, e acesso a bancos relacionais remotos (PostgreSQL/MySQL) atraves de servidores Shelf + pacotes server-side como postgres e mysql_client.

Para apps Flutter mobile/desktop, Drift + SQLite e a escolha padrao: queries SQL seguras em tempo de compilacao, migrations versionadas, reactivity via Stream e suporte a WAL (Write-Ahead Logging) para concorrencia. Para dados de usuario pouco estruturados, Hive e mais simples e rapido.

Drift — SQLite Type-Safe

Dart + Drift — Definicao de tabelas e banco
// pubspec.yaml
// dependencies: drift: ^2.x, sqlite3_flutter_libs:, path_provider:, path:
// dev_dependencies: drift_dev:, build_runner:

import 'package:drift/drift.dart';
import 'package:drift/native.dart';
import 'package:path_provider/path_provider.dart';
import 'package:path/path.dart' as p;
import 'dart:io';
part 'banco.g.dart';

// Tabela de categorias
class Categorias extends Table {
  IntColumn    get id       => integer().autoIncrement()();
  TextColumn   get nome     => text().withLength(min: 1, max: 80)();
  DateTimeColumnget criado  => dateTime().withDefault(Constant(DateTime.now()))();
}

// Tabela de tarefas
class Tarefas extends Table {
  IntColumn    get id         => integer().autoIncrement()();
  TextColumn   get titulo     => text().withLength(min: 1, max: 200)();
  TextColumn   get descricao  => text().nullable()();
  BoolColumn   get concluida  => boolean().withDefault(const Constant(false))();
  DateTimeColumnget prazo     => dateTime().nullable()();
  IntColumn    get categoriaId => integer().references(Categorias, #id)();
}

@DriftDatabase(tables: [Categorias, Tarefas])
class AppDatabase extends _$AppDatabase {
  AppDatabase(QueryExecutor e) : super(e);

  @override
  int get schemaVersion => 2;

  @override
  MigrationStrategy get migration => MigrationStrategy(
    onCreate: (m) => m.createAll(),
    onUpgrade: (m, from, to) async {
      if (from < 2) {
        await m.addColumn(tarefas, tarefas.prazo);
      }
    },
  );
}

// Fabrica do banco (singleton)
AppDatabase criarBanco() {
  late NativeDatabase native;
  final executor = LazyDatabase(() async {
    final dir  = await getApplicationDocumentsDirectory();
    final file = File(p.join(dir.path, 'app.db'));
    return NativeDatabase.createInBackground(file);
  });
  return AppDatabase(executor);
}
Dart + Drift — DAO com queries, inserts, watch e joins
part of 'banco.dart';

@DriftAccessor(tables: [Tarefas, Categorias])
class TarefasDao extends DatabaseAccessor<AppDatabase> with _$TarefasDaoMixin {
  TarefasDao(super.db);

  // SELECT com filtro — retorna List
  Future<List<Tarefa>> pendentes() =>
    (select(tarefas)..where((t) => t.concluida.equals(false))).get();

  // Watch — retorna Stream reativo
  Stream<List<Tarefa>> watchAll() =>
    (select(tarefas)..orderBy([(t) => OrderingTerm.desc(t.id)])).watch();

  // INSERT com retorno do id
  Future<int> inserir(TarefasCompanion entry) =>
    into(tarefas).insert(entry);

  // UPDATE parcial
  Future<void> marcarConcluida(int id) =>
    (update(tarefas)..where((t) => t.id.equals(id)))
      .write(const TarefasCompanion(concluida: Value(true)));

  // DELETE
  Future<int> deletar(int id) =>
    (delete(tarefas)..where((t) => t.id.equals(id))).go();

  // JOIN com categoria
  Future<List<TypedResult>> comCategoria() =>
    (select(tarefas)
      .join([innerJoin(categorias, categorias.id.equalsExp(tarefas.categoriaId))])
    ).get();

  // Transacao
  Future<void> moverCategoria(int tarefaId, int novaCategoria) =>
    transaction(() async {
      await marcarConcluida(tarefaId);
      await (update(tarefas)..where((t) => t.id.equals(tarefaId)))
        .write(TarefasCompanion(categoriaId: Value(novaCategoria)));
    });
}

Dart + Hive — boxes, TypeAdapters e criptografia
import 'package:hive_flutter/hive_flutter.dart';
part 'configuracao.g.dart';

// Modelo TypeAdapter (gerado via build_runner)
@HiveType(typeId: 0)
class Configuracao extends HiveObject {
  @HiveField(0) late String tema;
  @HiveField(1) late bool   notificacoes;
  @HiveField(2) late String idioma;
  @HiveField(3) int?   fonteTamanho;
}

Future<void> inicializarHive() async {
  await Hive.initFlutter();
  Hive.registerAdapter(ConfiguracaoAdapter());
}

class ConfiguracaoRepo {
  static const _boxName = 'configuracoes';
  late Box<Configuracao> _box;

  Future<void> abrir() async {
    _box = await Hive.openBox<Configuracao>(_boxName);
  }

  Configuracao ler() => _box.get('config') ?? Configuracao()
    ..tema = 'dark'
    ..notificacoes = true
    ..idioma = 'pt_BR';

  Future<void> salvar(Configuracao cfg) => _box.put('config', cfg);

  // Listenable para reagir a mudancas
  ValueListenable<Box<Configuracao>> listenable() => _box.listenable();
}

// Box criptografado com AES-256
Future<Box> abrirBoxSegura(String nome) async {
  final keySt = await FlutterSecureStorage().read(key: 'hive_key_$nome');
  List<int> chave;
  if (keySt == null) {
    chave = Hive.generateSecureKey();
    await FlutterSecureStorage().write(key: 'hive_key_$nome', value: String.fromCharCodes(chave));
  } else {
    chave = keySt.codeUnits;
  }
  return Hive.openBox(nome, encryptionCipher: HiveAesCipher(chave));
}

Isar — NoSQL de Alta Performance

Dart + Isar — Collections, indices e queries reativas
import 'package:isar/isar.dart';
part 'nota.g.dart';

@collection
class Nota {
  Id id = Isar.autoIncrement;

  @Index(type: IndexType.value)
  late String titulo;

  late String conteudo;

  @Index()
  DateTime? modificado;

  @Enumerated(EnumType.name)
  Prioridade prioridade = Prioridade.normal;
}

enum Prioridade { baixa, normal, alta }

class NotaRepository {
  late Isar _isar;

  Future<void> abrir() async {
    _isar = await Isar.open([NotaSchema],
      name: 'notas_db', directory: await _dirPath());
  }

  Future<void> salvar(Nota nota) =>
    _isar.writeTxn(() => _isar.notas.put(nota));

  Future<List<Nota>> buscarPorTitulo(String query) =>
    _isar.notas
      .filter().tituloContains(query, caseSensitive: false)
      .sortByModificadoDesc()
      .findAll();

  // Query reativa — Stream que emite nova lista ao mudar
  Stream<List<Nota>> watchAltas() =>
    _isar.notas
      .filter().prioridadeEqualTo(Prioridade.alta)
      .sortByModificadoDesc()
      .watch(fireImmediately: true);

  // Transacao com multiplas operacoes
  Future<void> batchImportar(List<Nota> notas) =>
    _isar.writeTxn(() => _isar.notas.putAll(notas));
}

Dart + postgres — Pool de conexoes e queries parametrizadas
import 'package:postgres/postgres.dart';

// Pool de conexoes (recomendado em producao)
final pool = Pool<Connection>(
  await Connection.open(
    Endpoint(
      host:     Platform.environment['DB_HOST']!,
      port:     int.parse(Platform.environment['DB_PORT'] ?? '5432'),
      database: Platform.environment['DB_NAME']!,
      username: Platform.environment['DB_USER']!,
      password: Platform.environment['DB_PASS']!,
    ),
    settings: const ConnectionSettings(sslMode: SslMode.require),
  ),
  size: 10,
);

class ProdutoRepositoryImpl {
  final Connection _db;
  ProdutoRepositoryImpl(this._db);

  // Query parametrizada — previne SQL Injection
  Future<List<Map<String, dynamic>>> listar({int pagina = 1}) async {
    final result = await _db.execute(
      Sql.named(r'SELECT id, nome, preco FROM produtos ORDER BY id LIMIT @limite OFFSET @offset'),
      parameters: {'limite': 20, 'offset': (pagina - 1) * 20},
    );
    return result.map((row) => row.toColumnMap()).toList();
  }

  Future<void> inserir(String nome, double preco, String categoriaId) async {
    await _db.execute(
      Sql.named(r'INSERT INTO produtos (nome, preco, categoria_id) VALUES (@nome, @preco, @cat)'),
      parameters: {'nome': nome, 'preco': preco, 'cat': categoriaId},
    );
  }

  // Transacao explicita
  Future<void> transferirEstoque(int de, int para, int qtd) async {
    await _db.runTx((tx) async {
      await tx.execute(
        Sql.named(r'UPDATE produtos SET estoque = estoque - @q WHERE id = @id'),
        parameters: {'q': qtd, 'id': de},
      );
      await tx.execute(
        Sql.named(r'UPDATE produtos SET estoque = estoque + @q WHERE id = @id'),
        parameters: {'q': qtd, 'id': para},
      );
    });
  }
}
