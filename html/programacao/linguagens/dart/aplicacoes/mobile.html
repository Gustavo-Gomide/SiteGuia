Dart — Desenvolvimento Mobile

Dart para Mobile com Flutter

Flutter e o framework mobile da Google que usa Dart para criar aplicativos nativos de alta performance para iOS, Android, Web e Desktop a partir de um unico codebase. O engine Flutter renderiza cada pixel diretamente via Skia/Impeller sem pontes nativas para widgets, garantindo 60/120fps em qualquer plataforma.

Flutter usa um modelo reactive/declarativo: a UI e descrita como arvore de widgets imutaveis regenerada a cada mudanca de estado. Widgets sao baratos de criar — tanto StatelessWidget quanto StatefulWidget sao compilados AOT para codigo de maquina nativo.

// Widget com estado — contador de likes
class BotaoLike extends StatefulWidget {
  const BotaoLike({super.key, required this.inicial});
  final int inicial;
  @override
  State<BotaoLike> createState() => _BotaoLikeState();
}

class _BotaoLikeState extends State<BotaoLike> {
  late int _likes;
  bool _liked = false;

  @override
  void initState() {
    super.initState();
    _likes = widget.inicial;
  }

  void _toggle() => setState(() {
    _liked = !_liked;
    _likes += _liked ? 1 : -1;
  });

  @override
  Widget build(BuildContext ctx) => IconButton(
    icon: Icon(_liked ? Icons.favorite : Icons.favorite_border,
              color: _liked ? Colors.red : null),
    onPressed: _toggle,
    tooltip: '$_likes curtidas',
  );
}

Navegacao com GoRouter

Dart + Flutter — GoRouter com rotas aninhadas e guards
import 'package:go_router/go_router.dart';

// Guard de autenticacao
String? _redirectAuth(BuildContext ctx, GoRouterState state) {
  final logado = AuthService.instance.isLoggedIn;
  final rotasPublicas = ['/login', '/registro'];
  if (!logado && !rotasPublicas.contains(state.matchedLocation)) return '/login';
  if (logado  &&  rotasPublicas.contains(state.matchedLocation)) return '/';
  return null;
}

final router = GoRouter(
  initialLocation: '/',
  redirect: _redirectAuth,
  routes: [
    ShellRoute(
      builder: (ctx, state, child) => ScaffoldWithBottomNav(child: child),
      routes: [
        GoRoute(path: '/',          builder: (_, __) => const HomePage()),
        GoRoute(path: '/buscar',   builder: (_, __) => const BuscarPage()),
        GoRoute(
          path: '/produto/:id',
          builder: (ctx, state) => ProdutoPage(id: state.pathParameters['id']!),
          routes: [
            GoRoute(
              path: 'avaliacao',     // /produto/:id/avaliacao
              builder: (ctx, state) => AvaliacaoPage(id: state.pathParameters['id']!),
            ),
          ],
        ),
      ],
    ),
    GoRoute(path: '/login',    builder: (_, __) => const LoginPage()),
    GoRoute(path: '/registro', builder: (_, __) => const RegistroPage()),
  ],
);

// Navegar programaticamente
context.go('/produto/42');                  // substitui historico
context.push('/produto/42/avaliacao');     // empilha
context.pop();                              // volta
context.pushReplacement('/');             // substitui topo

Dart + Flutter + Riverpod — Provider, AsyncNotifier e ConsumerWidget
import 'package:riverpod_annotation/riverpod_annotation.dart';
import 'package:flutter_riverpod/flutter_riverpod.dart';
part 'produto_provider.g.dart';

// Provider simples para tema
final temaProvider = StateProvider<ThemeMode>((_) => ThemeMode.dark);

// AsyncNotifier para lista de produtos com CRUD
@riverpod
class ProdutosNotifier extends _$ProdutosNotifier {
  @override
  Future<List<Produto>> build() => ProdutoRepository.instance.listar();

  Future<void> criar(String nome, double preco) async {
    state = const AsyncLoading();
    state = await AsyncValue.guard(() async {
      await ProdutoRepository.instance.criar(nome, preco);
      return ProdutoRepository.instance.listar();
    });
  }

  Future<void> deletar(String id) async {
    await ProdutoRepository.instance.deletar(id);
    ref.invalidateSelf();  // re-fetch
  }
}

// ConsumerWidget — escuta providers
class ProdutosPage extends ConsumerWidget {
  const ProdutosPage({super.key});

  @override
  Widget build(BuildContext ctx, WidgetRef ref) {
    final async = ref.watch(produtosNotifierProvider);

    return Scaffold(
      appBar: AppBar(title: const Text('Produtos')),
      body: async.when(
        loading: () => const Center(child: CircularProgressIndicator()),
        error: (e, _) => ErrorWidget(e),
        data: (lista) => ListView.builder(
          itemCount: lista.length,
          itemBuilder: (_, i) => ListTile(
            title: Text(lista[i].nome),
            subtitle: Text('R\$ ${lista[i].preco.toStringAsFixed(2)}'),
            trailing: IconButton(
              icon: const Icon(Icons.delete),
              onPressed: () => ref.read(produtosNotifierProvider.notifier).deletar(lista[i].id),
            ),
          ),
        ),
      ),
      floatingActionButton: FloatingActionButton.extended(
        onPressed: () => _mostrarFormulario(ctx, ref),
        icon: const Icon(Icons.add),
        label: const Text('Novo'),
      ),
    );
  }
}

Platform Channels — Codigo Nativo

Dart + Flutter — MethodChannel para acessar APIs nativas
import 'package:flutter/services.dart';

class NativeInfo {
  static const _channel = MethodChannel('com.exemplo.app/info');

  static Future<String> getDeviceId() async {
    try {
      return await _channel.invokeMethod<String>('getDeviceId') ?? 'desconhecido';
    } on PlatformException catch (e) {
      throw Exception('Erro nativo: ${e.message}');
    }
  }

  static Future<Map<String, dynamic>> getBatteryInfo() async {
    final result = await _channel.invokeMapMethod<String, dynamic>('getBatteryInfo');
    return result ?? {};
  }
}

// EventChannel para dados continuos (ex: acelerometro)
class SensorService {
  static const _channel = EventChannel('com.exemplo.app/acelerometro');

  static Stream<Map<String, double>> escutar() =>
    _channel
      .receiveBroadcastStream()
      .cast<Map>()
      .map((m) => m.cast<String, double>());
}

// Uso em widget
class AcelerometroWidget extends StatelessWidget {
  const AcelerometroWidget({super.key});

  @override
  Widget build(BuildContext _) => StreamBuilder(
    stream: SensorService.escutar(),
    builder: (ctx, snap) {
      if (!snap.hasData) return const CircularProgressIndicator();
      final data = snap.data!;
      return Text('x:${data["x"]!.toStringAsFixed(2)} '
                  'y:${data["y"]!.toStringAsFixed(2)} '
                  'z:${data["z"]!.toStringAsFixed(2)}');
    },
  );
}
Plugins vs Platform Channels: para a maioria das funcionalidades nativas (camera, geolocalizacao, notificacoes, biometria), use pacotes pub.dev consolidados como camera, geolocator, local_notifications e local_auth. Escreva platform channels proprios apenas quando nao houver plugin disponivel.

Dart + Flutter — flutter_secure_storage + biometric auth
import 'package:flutter_secure_storage/flutter_secure_storage.dart';
import 'package:local_auth/local_auth.dart';

class AuthStorage {
  static const _storage = FlutterSecureStorage(
    aOptions: AndroidOptions(encryptedSharedPreferences: true),
    iOptions: IOSOptions(accessibility: KeychainAccessibility.first_unlock),
  );

  static Future<void> salvarToken(String token) =>
    _storage.write(key: 'jwt_token', value: token);

  static Future<String?> lerToken() =>
    _storage.read(key: 'jwt_token');

  static Future<void> limpar() => _storage.deleteAll();
}

class BiometricService {
  final _auth = LocalAuthentication();

  Future<bool> autenticar() async {
    final disponivel = await _auth.canCheckBiometrics();
    if (!disponivel) return false;

    return _auth.authenticate(
      localizedReason: 'Use sua impressao digital para acessar',
      options: const AuthenticationOptions(
        biometricOnly: true,
        stickyAuth:    true,
      ),
    );
  }
}

flutter build web --release --base-href /meuapp/

# Verificar tamanho do app
flutter build apk --analyze-size
flutter build aab --analyze-size

Boas Praticas Mobile Flutter

  • Use const em widgets sem parametros dinamicos — elimina rebuilds desnecessarios
  • Prefira ListView.builder sobre ListView com filhos fixos para listas longas
  • Isole regras de negocio em repositorios — nao coloque logica em Widgets
  • Use flutter_secure_storage para tokens — nunca shared_preferences
  • Implemente RepaintBoundary em animacoes isoladas para evitar invalidar a arvore toda
  • Use flutter analyze e dart fix --apply antes de cada release

Anti-Padroes Flutter

  • setState desnecessario no nivel raiz do Scaffold — reconstroi a pagina toda
  • Carregar imagens sem cache (Image.network sem cacheWidth/cached_network_image)
  • Executar IO intenso na thread da UI sem usar Isolate.run ou compute()
  • Usar dynamic em vez de tipos — perde verificacao em tempo de compilacao
  • Armazenar contexto alem do ciclo de vida do Widget (GlobalKey desnecessario)
