Dart — Terminal e CLI

Dart — Terminal e CLI

Dart compila para binarios nativos autocontidos via dart compile exe, sem depender da VM em producao. O ecossistema possui args para parsing de flags, dart_console para controle full de terminal ANSI (cores, cursor, raw mode), dcli para scripting estilo shell e dart:io para subprocessos, redirecionamento de stdin/stdout e manipulacao do sistema de arquivos.

Ferramentas CLI em Dart sao publicadas como executaveis globais via dart pub global activate. O proprio tooling Dart (dart, pub, dartfmt, dart2js) e escrito em Dart, demonstrando a maturidade do ecossistema para ferramentas de linha de comando robustas.

dart:io — stdin, stdout, stderr e argumentos

Dart — dart:io basico: leitura, escrita e stdin interativo
import 'dart:io';

void main(List<String> args) {
  // Saida padrao e erro
  stdout.writeln('Olá do stderr');
  stderr.writeln('Erro: algo falhou');

  // Leitura sincrona de linha
  stdout.write('Qual seu nome? ');
  final nome = stdin.readLineSync();
  print('Olá, $nome!');

  // Verificar se ha dados no stdin (modo pipe)
  if (stdin.hasTerminal) {
    print('Rodando em terminal interativo');
  } else {
    // Leitura de pipe: cat arquivo.txt | meutool
    final linhas = stdin.readLineSync();
    print('Recebido do pipe: $linhas');
  }

  // Ler TODOS os bytes do stdin de forma sincrona
  final bytes = stdin.readAsBytesSync();
  print('${bytes.length} bytes lidos');
}

// Leitura assincrona de stdin — melhor para longos fluxos
Future<void> lerStdinAsync() async {
  await for (final linha in stdin.transform(SystemEncoding().decoder).transform(const LineSplitter())) {
    print('Linha: $linha');
  }
}
Dart — Subprocessos com dart:io Process
import 'dart:io';

// Executar processo e capturar saida
Future<void> runProcess() async {
  // Simples: aguarda conclusao e coleta tudo
  final result = await Process.run('git', ['log', '--oneline', '-5']);
  print(result.stdout);
  if (result.exitCode != 0) {
    stderr.writeln(result.stderr);
    exit(1);
  }

  // Streaming: processa saida ao vivo
  final process = await Process.start('dart', ['pub', 'get'],
    workingDirectory: './projeto',
  );
  process.stdout.transform(const Utf8Decoder()).listen(stdout.write);
  process.stderr.transform(const Utf8Decoder()).listen(stderr.write);
  final code = await process.exitCode;
  if (code != 0) exit(code);
}

// Redirecionar stdin do processo pai para filho
Future<void> runInterativo() async {
  final proc = await Process.start(
    'python3', ['-'],
    mode: ProcessStartMode.inheritStdio,
  );
  await proc.exitCode;  // stdin/stdout/stderr herdados do pai
}

Package args — Parsing de Flags e Subcomandos

Dart + args — ArgParser completo com subcomandos
import 'package:args/args.dart';
import 'package:args/command_runner.dart';
import 'dart:io';

// ---- Abordagem 1: ArgParser direto ----
void parseManual(List<String> args) {
  final parser = ArgParser()
    ..addFlag('verbose',   abbr: 'v', help: 'Saida detalhada',         defaultsTo: false)
    ..addFlag('dry-run',   abbr: 'n', help: 'Nao executar de verdade',  negatable: false)
    ..addOption('output',  abbr: 'o', help: 'Arquivo de saida',         defaultsTo: 'out.txt')
    ..addOption('format',             help: 'Formato de saida',
        allowed: ['json', 'csv', 'xml'], defaultsTo: 'json')
    ..addMultiOption('target', abbr: 't', help: 'Alvos (pode repetir)');

  ArgResults results;
  try {
    results = parser.parse(args);
  } on FormatException catch (e) {
    stderr.writeln('Erro: ${e.message}');
    stderr.writeln(parser.usage);
    exit(64); // sysexits.h EX_USAGE
  }

  final verbose = results['verbose']  as bool;
  final output  = results['output']   as String;
  final targets = results['target']   as List<String>;
  final rest    = results.rest;  // argumentos posicionais apos flags

  if (verbose) print('Output: $output  Targets: $targets  Extra: $rest');
}

// ---- Abordagem 2: CommandRunner — subcomandos estilo git/dart ----
void main(List<String> args) => GerenciadorCLI().run(args);

class GerenciadorCLI extends CommandRunner<void> {
  GerenciadorCLI() : super('meutool', 'Ferramenta CLI em Dart') {
    addCommand(BuildCommand());
    addCommand(ServeCommand());
    argParser.addFlag('quiet', abbr: 'q');
  }
}

class BuildCommand extends Command<void> {
  @override String get name        => 'build';
  @override String get description => 'Compila o projeto';
  @override String get usage        =>
      'meutool build [--release] [--target <platform>]';

  BuildCommand() {
    argParser
      ..addFlag('release', defaultsTo: false)
      ..addOption('target', allowed: ['android', 'ios', 'web']);
  }

  @override
  Future<void> run() async {
    final release = argResults!['release'] as bool;
    final target  = argResults!['target']  as String?;
    print('Compilando${release ? " (release)" : ""}${target != null ? " para $target" : ""}...');
  }
}

class ServeCommand extends Command<void> {
  @override String get name        => 'serve';
  @override String get description => 'Inicia servidor de desenvolvimento';

  ServeCommand() {
    argParser.addOption('port', abbr: 'p', defaultsTo: '8080');
  }

  @override
  Future<void> run() async {
    final port = int.parse(argResults!['port'] as String);
    print('Servidor em http://localhost:$port');
  }
}

dart_console — ANSI Colors, Cursor e Raw Mode

Dart + dart_console — Cores ANSI, posicionamento de cursor e input raw
import 'package:dart_console/dart_console.dart';

void demonstrarConsole() {
  final console = Console();

  // Limpar e obter tamanho
  console.clearScreen();
  print('Terminal: ${console.windowWidth}x${console.windowHeight}');

  // Cores e estilos
  console
    ..setForegroundColor(ConsoleColor.brightGreen)
    ..write('OK  ')
    ..resetColorAttributes()
    ..setForegroundColor(ConsoleColor.red)
    ..write('FALHA  ')
    ..resetColorAttributes()
    ..setTextStyle(bold: true, underline: true)
    ..writeLine('Titulo em negrito')
    ..resetColorAttributes();

  // Mover cursor manualmente — util para progress bars
  console.cursorPosition = const Coordinate(5, 0);  // linha 5, coluna 0
  console.write('Linha 5');

  // Ocultar/exibir cursor
  console.hideCursor();
  // ... animacao ...
  console.showCursor();
}

// Progress bar manual para operacoes longas
Future<void> progressBar(int total, Future<void> Function(int) task) async {
  final console = Console();
  console.hideCursor();
  for (var i = 0; i < total; i++) {
    await task(i);
    final pct    = ((i + 1) / total * 100).round();
    final filled = (pct / 5).round();
    final bar    = '#' * filled + '-' * (20 - filled);
    console
      ..eraseLine()
      ..write('\r[$bar] $pct%  (${i + 1}/$total)');
  }
  console.writeLine('');
  console.showCursor();
}

// Input interativo com seta para cima/seta para baixo (menu)
int menuInterativo(List<String> opcoes) {
  final console = Console();
  var selecionado = 0;
  console.hideCursor();
  while (true) {
    console.clearScreen();
    for (var i = 0; i < opcoes.length; i++) {
      if (i == selecionado) {
        console
          ..setForegroundColor(ConsoleColor.yellow)
          ..writeLine('  > ${opcoes[i]}')
          ..resetColorAttributes();
      } else {
        console.writeLine('    ${opcoes[i]}');
      }
    }
    final key = console.readKey();
    if (key.controlChar == ControlCharacter.arrowUp)   selecionado = (selecionado - 1) % opcoes.length;
    if (key.controlChar == ControlCharacter.arrowDown) selecionado = (selecionado + 1) % opcoes.length;
    if (key.controlChar == ControlCharacter.enter) break;
    if (key.char == 'q') { selecionado = -1; break; }
  }
  console.showCursor();
  return selecionado;
}

DCli — Scripting Estilo Shell em Dart

Dart + dcli — Shell scripting fluente
import 'package:dcli/dcli.dart';
import 'dart:io';

void main() {
  // Executar comandos — estilo backtick bash
  final branch = 'git rev-parse --abbrev-ref HEAD'.firstLine!;
  print('Branch atual: $branch');

  final arquivos = 'find . -name "*.dart" -type f'.toList();
  print('${arquivos.length} arquivos Dart');

  // Run e verificar exit code
  'dart format .'.run();  // lanca excecao se != 0

  // Prompt interativo
  final nome   = ask('Qual seu nome?', defaultValue: 'user');
  final senha  = ask('Senha:', hidden: true);
  final confirmar = confirm('Continuar?');

  if (confirmar) {
    print('Bem-vindo, $nome!');
  }

  // Manipulacao de arquivos
  String home = env['HOME'] ?? env['USERPROFILE']!;
  final config = join(home, '.myapp', 'config.json');
  createDir(dirname(config), recursive: true);
  config.write('{"nome": "$nome"}');
  print('Config salvo em: $config');

  // Copiar/mover/deletar
  copy('src.txt', 'dest.txt');
  move('old.txt', 'new.txt');
  delete('lixo.tmp');

  // Glob / find
  find('*.dart', workingDirectory: 'lib')
    .forEach((f) => print(f));

  // Cores rapidas
  print(red('Erro critico!'));
  print(green('Sucesso!'));
  print(yellow('Aviso:') + ' algo estranho aconteceu');
  print(blue('Info'));
  print(bold('Negrito'));
}

dart:io — Sistema de Arquivos Completo

Dart — File, Directory, FileSystemEntity, watch
import 'dart:io';
import 'dart:convert';

Future<void> demoArquivos() async {
  // Leitura completa
  final conteudo = await File('config.json').readAsString();
  final json     = jsonDecode(conteudo);

  // Escrita atomica (via temp + rename)
  final target = File('dados.json');
  final temp   = File('dados.json.tmp');
  await temp.writeAsString(jsonEncode(json), flush: true);
  await temp.rename(target.path);

  // Streaming para arquivos grandes
  final sink = File('log.txt').openWrite(mode: FileMode.append);
  try {
    sink.writeln('[${DateTime.now()}] Evento');
  } finally {
    await sink.flush();
    await sink.close();
  }

  // Ler linha a linha (memoria eficiente)
  final stream = File('grande.csv')
    .openRead()
    .transform(const Utf8Decoder())
    .transform(const LineSplitter());
  await for (final linha in stream) {
    // processar sem carregar tudo na RAM
  }

  // Listar diretorio recursivamente
  await for (final entity in Directory('src').list(recursive: true)) {
    if (entity is File && entity.path.endsWith('.dart')) {
      print(entity.path);
    }
  }

  // Watch — executar ao mudar arquivo (hot reload manual)
  Directory('lib').watch(recursive: true).listen((event) {
    print('Mudou: ${event.path} (${event.type.name})');
  });
}

Compilacao para Binario Nativo

Shell — dart compile exe, pub global e distribuicao
# Compilar para executavel nativo (AOT, sem VM)
dart compile exe bin/meutool.dart -o meutool
dart compile exe bin/meutool.dart -o meutool.exe  # Windows
dart compile exe bin/meutool.dart --define=VERSION=1.0.0 -o meutool

# Verificar tamanho resultante (~5 MB base)
ls -lh meutool

# Compilar para JS (para Node.js ou browser)
dart compile js bin/meutool.dart -o meutool.js
dart compile js bin/meutool.dart -O4 -o meutool.min.js  # optimizations

# Compilar arquivo de snapshot kernel (JIT mais rapido, nao nativo)
dart compile kernel bin/meutool.dart -o meutool.dill
dart run meutool.dill  # precisa da VM

# Publicar como global para usuarios finais
dart pub global activate --source path .        # da pasta local
dart pub global activate meutool                # do pub.dev
dart pub global activate --source git https://github.com/user/meutool

Boas Praticas e Anti-Padroes

Usar exit(0) so apos liberar recursos (fechar sinks, processes)

Usar ProcessSignal.sigint.watch() para Ctrl+C gracioso

Checar stdin.hasTerminal antes de pedir input interativo

Retornar exit codes semanticos (sysexits.h: 64 uso, 65 dados, 70 soft, 74 io)

Prefixar pub global activate com lock de versao em scripts CI

Logar para stderr, dados uteis para stdout (facilita | pipe)

Usar print() para stdout em ferramentas que devem suportar pipe

Usar ANSI colors sem checar stdout.hasTerminal (polui pipes)

Compilar com dart run em producao (JIT — performance e startup pior)

Usar Process.runSync para processos longos (bloqueia event loop)

Esquecer de fechar IOSink — dados podem nao ser flushed

Hardcodar paths com / ou \\ — usar path.join()
