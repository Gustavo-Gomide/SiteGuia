Dart — Desenvolvimento Web

Dart para Desenvolvimento Web

Dart para web tem dois papeis complementares: Flutter Web renderiza interfaces ricas cross-platform (Canvas 2D, CanvasKit/WASM), e Shelf constroe servidores HTTP, APIs REST e middlewares no lado do servidor. Dart tambem pode ser compilado diretamente para JavaScript via dart compile js com tree-shaking agressivo, ou para WebAssembly (WASM) via Flutter Web para maximo desempenho no browser.

Flutter Web 3.x usa CanvasKit por default — renderizacao via WebAssembly + WebGL com fidelidade pixel-perfect ao mobile. Para SEO e carregamento rapido, o modo HTML renderer gera DOM nativo, mais leve para conteudo estatico.

import 'dart:convert';
import 'dart:io';
import 'package:shelf/shelf.dart';
import 'package:shelf/shelf_io.dart' as io;
import 'package:shelf_router/shelf_router.dart';
import 'package:dart_jsonwebtoken/dart_jsonwebtoken.dart';

// Middleware JWT
Middleware jwtAuth(String segredo) => (handler) => (req) async {
  final authHeader = req.headers['authorization'];
  if (authHeader == null || !authHeader.startsWith('Bearer ')) {
    return Response.unauthorized('{"erro":"Token ausente"}');
  }
  try {
    final jwt = JWT.verify(authHeader.substring(7), SecretKey(segredo));
    return handler(req.change(context: {'usuario': jwt.payload}));
  } on JWTException catch (e) {
    return Response.forbidden('{"erro":"${e.message}"}');
  }
};

// Middleware CORS
Middleware cors({String allowedOrigin = '*'}) => (handler) => (req) async {
  final headers = {
    'Access-Control-Allow-Origin':  allowedOrigin,
    'Access-Control-Allow-Methods': 'GET, POST, PUT, DELETE, OPTIONS',
    'Access-Control-Allow-Headers': 'Authorization, Content-Type',
  };
  if (req.method == 'OPTIONS') return Response.ok('', headers: headers);
  final resp = await handler(req);
  return resp.change(headers: {...resp.headers, ...headers});
};

// Controller de produtos
class ProdutosController {
  final ProdutoRepository _repo;
  ProdutosController(this._repo);

  Router get router {
    final r = Router();
    r.get('/produtos', _listar);
    r.get('/produtos/<id>', _buscar);
    r.post('/produtos', _criar);
    r.put('/produtos/<id>', _atualizar);
    r.delete('/produtos/<id>', _deletar);
    return r;
  }

  Future<Response> _listar(Request req) async {
    final pagina = int.tryParse(req.url.queryParameters['page'] ?? '1') ?? 1;
    final lista  = await _repo.listar(pagina: pagina, limite: 20);
    return _json({'dados': lista.map((p) => p.toJson()).toList(), 'pagina': pagina});
  }

  Future<Response> _buscar(Request req, String id) async {
    final resultado = await _repo.buscarPorId(id);
    return switch (resultado) {
      Success(:final value) => _json(value.toJson()),
      Failure(:final error) => Response.notFound(_errorBody(error)),
    };
  }

  Future<Response> _criar(Request req) async {
    final body = jsonDecode(await req.readAsString());
    if (body['nome'] is! String) {
      return Response(422, body: _errorBody('campo nome obrigatorio'));
    }
    final novo = await _repo.criar(
      nome:  body['nome'],
      preco: (body['preco'] as num).toDouble(),
    );
    return _json(novo.toJson(), status: 201);
  }

  Response _json(dynamic data, {int status = 200}) => Response(
    status, body: jsonEncode(data),
    headers: {'Content-Type': 'application/json'},
  );
  String _errorBody(String msg) => jsonEncode({'erro': msg});
}

Future<void> main() async {
  final ctrl   = ProdutosController(ProdutoRepositoryImpl());
  final handler = Pipeline()
    .addMiddleware(logRequests())
    .addMiddleware(cors())
    .addMiddleware(jwtAuth(Platform.environment['JWT_SECRET']!))
    .addHandler(ctrl.router.call);

  final server = await io.serve(handler, InternetAddress.anyIPv4, 8080);
  print('API em http://localhost:${server.port}');
}
Dart + Shelf — WebSockets com shelf_web_socket
import 'package:shelf_web_socket/shelf_web_socket.dart';
import 'package:web_socket_channel/web_socket_channel.dart';

final _clientes = <WebSocketChannel>{};

Handler wsHandler = webSocketHandler((WebSocketChannel channel) {
  _clientes.add(channel);
  print('Cliente conectado. Total: ${_clientes.length}');

  channel.stream.listen(
    (msg) {
      // Broadcast para todos os outros clientes
      for (final cl in _clientes) {
        if (cl != channel) cl.sink.add(msg);
      }
    },
    onDone: () {
      _clientes.remove(channel);
      print('Cliente desconectado. Total: ${_clientes.length}');
    },
  );
});

Flutter Web — Configuracao e Build

Shell — Flutter Web: build e renderizadores
# Habilitar plataforma web
flutter config --enable-web
flutter create --platforms=web meu_app

# Rodar no Chrome (hot reload)
flutter run -d chrome
flutter run -d chrome --web-renderer html       # DOM nativo — menor bundle
flutter run -d chrome --web-renderer canvaskit  # WebGL — fidelidade total

# Build de producao
flutter build web
flutter build web --web-renderer canvaskit      # default em producao
flutter build web --web-renderer html --release
flutter build web --wasm                        # WebAssembly (Dart 3.3+)

# Servir localmente
cd build/web
python -m http.server 8080

  routes: [
    GoRoute(path: '/',           builder: (_, __) => const HomePage()),
    GoRoute(path: '/produtos',  builder: (_, __) => const ProdutosPage()),
    GoRoute(
      path: '/produtos/:id',
      builder: (ctx, state) => ProdutoDetalhe(id: state.pathParameters['id']!),

    GoRoute(path: '/404', builder: (_, __) => const NotFoundPage()),
  ],
  errorBuilder: (_, __) => const NotFoundPage(),
);

void main() {
  WidgetsFlutterBinding.ensureInitialized();
  runApp(MaterialApp.router(routerConfig: _router));
}

// Responsive layout com LayoutBuilder
class ResponsiveLayout extends StatelessWidget {
  const ResponsiveLayout({super.key, required this.mobile, required this.desktop});
  final Widget mobile, desktop;

  @override
  Widget build(BuildContext ctx) =>
    LayoutBuilder(builder: (_, constraints) =>
      constraints.maxWidth < 768 ? mobile : desktop);
}
