Dart — GUI Desktop

Dart — GUI Desktop com Flutter

Flutter Desktop permite criar aplicativos nativos para Windows, macOS e Linux com a mesma codebase Dart. O engine Impeller renderiza via DirectX 11 (Windows), Metal (macOS) e Vulkan (Linux) — sem Electron, sem WebView, widgets nativos desenhados pelo engine com pixel-perfect fidelidade.

Flutter Desktop e estavel desde Flutter 3.0 (2022). O plugin window_manager controla título, tamanho, posicao, bordas e focus da janela nativa. flutter_acrylic adiciona efeitos visuais do OS (Acrylic no Windows 11, NSVisualEffectView no macOS).

Configuracao e Setup

Shell — Habilitar plataformas desktop e build de release
# Habilitar plataformas
flutter config --enable-windows-desktop
flutter config --enable-macos-desktop
flutter config --enable-linux-desktop

# Criar projeto com suporte desktop
flutter create --platforms=windows,macos,linux meu_app_desktop

# Rodar em debug
flutter run -d windows
flutter run -d macos
flutter run -d linux

# Build de release
flutter build windows --release   # gera build/windows/runner/Release/
flutter build macos   --release   # gera build/macos/Build/Products/Release/*.app
flutter build linux   --release   # gera build/linux/x64/release/bundle/

# Verificar dispositivos disponiveis
flutter devices

window_manager — Controle da Janela

Dart + window_manager — Configuracao de janela, frameless e always on top
import 'package:flutter/material.dart';
import 'package:window_manager/window_manager.dart';

Future<void> main() async {
  WidgetsFlutterBinding.ensureInitialized();

  await windowManager.ensureInitialized();

  final opcoes = WindowOptions(
    size:            const Size(1280, 720),
    minimumSize:     const Size(800, 600),
    center:          true,
    title:           'Meu App Desktop',
    titleBarStyle:   TitleBarStyle.normal,
    backgroundColor: Colors.transparent,
    skipTaskbar:     false,
  );

  await windowManager.waitUntilReadyToShow(opcoes, () async {
    await windowManager.show();
    await windowManager.focus();
  });

  runApp(const App());
}

// Mixin para interceptar fechamento da janela
class AppState extends State<App> with WindowListener {
  @override
  void initState() {
    super.initState();
    windowManager.addListener(this);
    windowManager.setPreventClose(true);  // interceptar Alt+F4
  }

  @override
  void onWindowClose() async {
    final confirmar = await showDialog<bool>(
      context: context,
      builder: (_) => AlertDialog(
        title:   const Text('Sair?'),
        content: const Text('Deseja fechar o aplicativo?'),
        actions: [
          TextButton(onPressed: () => Navigator.pop(context, false), child: const Text('Cancelar')),
          TextButton(onPressed: () => Navigator.pop(context, true),  child: const Text('Sair')),
        ],
      ),
    ) ?? false;
    if (confirmar) await windowManager.destroy();
  }

  @override
  void dispose() {
    windowManager.removeListener(this);
    super.dispose();
  }
}

Menus Nativos e Atalhos de Teclado

Dart + Flutter — MenuBar nativo, PlatformMenuBar e SingleActivator
import 'package:flutter/material.dart';
import 'package:flutter/services.dart';

class MainWindow extends StatelessWidget {
  const MainWindow({super.key});

  @override
  Widget build(BuildContext _) => PlatformMenuBar(
    menus: [
      PlatformMenu(label: 'Arquivo', menus: [
        PlatformMenuItem(
          label:       'Novo',
          shortcut:    const SingleActivator(LogicalKeyboardKey.keyN, meta: true),
          onSelected: () => print('Novo arquivo'),
        ),
        PlatformMenuItem(
          label:       'Abrir...',
          shortcut:    const SingleActivator(LogicalKeyboardKey.keyO, meta: true),
          onSelected: () => _abrirArquivo(),
        ),
        const PlatformMenuDivider(),
        PlatformMenuItem(
          label:       'Sair',
          shortcut:    const SingleActivator(LogicalKeyboardKey.keyQ, meta: true),
          onSelected: () => SystemNavigator.pop(),
        ),
      ]),
      PlatformMenu(label: 'Editar', menus: [
        PlatformMenuItemGroup(members: [
          PlatformMenuItem(
            label:    'Copiar',
            shortcut: const SingleActivator(LogicalKeyboardKey.keyC, meta: true),
            onSelected: () {},
          ),
          PlatformMenuItem(
            label:    'Colar',
            shortcut: const SingleActivator(LogicalKeyboardKey.keyV, meta: true),
            onSelected: () {},
          ),
        ]),
      ]),
    ],
    child: const AppBody(),
  );

  void _abrirArquivo() { /* usar file_picker */ }
}

// Atalhos globais com Shortcuts + Actions (funciona em qualquer OS)
class AppBody extends StatelessWidget {
  const AppBody({super.key});
  @override
  Widget build(BuildContext ctx) => Shortcuts(
    shortcuts: {
      LogicalKeySet(LogicalKeyboardKey.control, LogicalKeyboardKey.keyS):
        const SalvarIntent(),
      LogicalKeySet(LogicalKeyboardKey.f5):
        const AtualizarIntent(),
    },
    child: Actions(
      actions: {
        SalvarIntent:    CallbackAction<SalvarIntent>(onInvoke: (_) => _salvar(ctx)),
        AtualizarIntent: CallbackAction<AtualizarIntent>(onInvoke: (_) => _atualizar(ctx)),
      },
      child: Focus(autofocus: true, child: const ConteudoPrincipal()),
    ),
  );
  void _salvar(BuildContext _) => print('Salvar Ctrl+S');
  void _atualizar(BuildContext _) => print('Atualizar F5');
}

class SalvarIntent   extends Intent { const SalvarIntent(); }
class AtualizarIntent extends Intent { const AtualizarIntent(); }

System Tray e Notificacoes Nativas

Dart + tray_manager + local_notifier — Icone na bandeja e notificacoes
import 'package:tray_manager/tray_manager.dart';
import 'package:local_notifier/local_notifier.dart';
import 'package:window_manager/window_manager.dart';

class TrayService with TrayListener {
  Future<void> inicializar() async {
    await trayManager.setIcon('assets/tray_icon.ico');  // .ico (Win), .png (Mac/Lin)
    await trayManager.setToolTip('Meu App');

    final menu = Menu(items: [
      MenuItem(key: 'show',    label: 'Mostrar janela'),
      MenuItem.separator(),
      MenuItem(key: 'quit',    label: 'Sair', role: 'quit'),
    ]);
    await trayManager.setContextMenu(menu);
    trayManager.addListener(this);
  }

  @override
  void onTrayIconMouseDown() {
    windowManager.show();
    windowManager.focus();
  }

  @override
  void onTrayMenuItemClick(MenuItem item) async {
    switch (item.key) {
      case 'show': windowManager.show(); windowManager.focus();
      case 'quit': await windowManager.destroy();
    }
  }
}

// Notificacao nativa do OS
Future<void> notificar(String titulo, String corpo) async {
  await localNotifier.setup(appName: 'Meu App');
  final notif = LocalNotification(title: titulo, body: corpo);
  await notif.show();
}

Empacotamento e Distribuicao

Shell — Empacotar para Windows (MSIX + Inno Setup) e macOS (DMG)
# Windows — MSIX (Microsoft Store / sideload)
# pubspec.yaml: msix: ^3.x
flutter pub add --dev msix

# msix config em pubspec.yaml:
# msix_config:
#   display_name: Meu App
#   publisher_display_name: Minha Empresa
#   identity_name: com.minha-empresa.meu-app
#   msix_version: 1.0.0.0
#   logo_path: assets/logo.png
flutter pub run msix:create

# Windows — Inno Setup tradicional
flutter build windows --release
# criar script .iss referenciando build/windows/runner/Release/
iscc setup_script.iss

# macOS — DMG com create-dmg
flutter build macos --release
npm install -g create-dmg
create-dmg \
  --volname "Meu App" \
  --window-pos 200 120 \
  --window-size 800 400 \
  --icon-size 100 \
  --app-drop-link 600 185 \
  MeuApp-1.0.0.dmg \
  "build/macos/Build/Products/Release/MeuApp.app"

# Linux — AppImage
flutter build linux --release
# usar appimagetool com desktop entry + icon
ARCH=x86_64 appimagetool MeuApp.AppDir MeuApp-x86_64.AppImage
