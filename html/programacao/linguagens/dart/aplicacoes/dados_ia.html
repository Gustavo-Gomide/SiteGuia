Dart — Dados e Inteligencia Artificial

Dart/Flutter permite integrar IA diretamente em apps mobile, desktop e web sem depender de servidores externos. tflite_flutter executa modelos TensorFlow Lite localmente na CPU/GPU/NPU do dispositivo, Google ML Kit oferece APIs de ML prontas (OCR, traducao, deteccao de objetos) e a Gemini API via google_generative_ai conecta ao LLM multimodal da Google com respostas em streaming.

A inferencia on-device com TFLite nao envia dados ao servidor, preserva privacidade, funciona offline e elimina latencia de rede — ideal para apps de saude, financeiro e producao industrial.

TensorFlow Lite com tflite_flutter

Dart + tflite_flutter — Classificacao de imagens on-device
import 'package:tflite_flutter/tflite_flutter.dart';
import 'package:image/image.dart' as img;
import 'dart:typed_data';

class ClassificadorImagens {
  late Interpreter _interpreter;
  late List<String> _labels;

  static const _inputSize = 224;

  Future<void> carregar() async {
    // Opcoes de delegate: GPU acelerada no Android/iOS
    final options = InterpreterOptions();
    if (Platform.isAndroid) {
      options.addDelegate(GpuDelegateV2());
    } else if (Platform.isIOS) {
      options.addDelegate(GpuDelegate());
    }
    _interpreter = await Interpreter.fromAsset(
      'assets/mobilenet_v2.tflite',
      options: options,
    );
    final labelsData = await rootBundle.loadString('assets/labels.txt');
    _labels = labelsData.split('\n').where((l) => l.isNotEmpty).toList();
  }

  List<MapEntry<String, double>> classificar(img.Image imagem) {
    // Pre-processamento: redimensionar + normalizar [-1, 1]
    final redim = img.copyResize(imagem, width: _inputSize, height: _inputSize);
    final input = List.generate(1, (_) =>
      List.generate(_inputSize, (y) =>
        List.generate(_inputSize, (x) {
          final px = redim.getPixel(x, y);
          return [
            (px.r / 127.5) - 1.0,
            (px.g / 127.5) - 1.0,
            (px.b / 127.5) - 1.0,
          ];
        })));

    final output = [List<double>.filled(_labels.length, 0)];
    _interpreter.run(input, output);

    // Top-5 resultados
    final scores = output[0];
    return List.generate(_labels.length, (i) =>
      MapEntry(_labels[i], scores[i]))
      ..sort((a, b) => b.value.compareTo(a.value))
      ..sublist(0, 5);
  }

  void fechar() => _interpreter.close();
}

Dart + google_mlkit — OCR, deteccao de faces e traducao
import 'package:google_mlkit_text_recognition/google_mlkit_text_recognition.dart';
import 'package:google_mlkit_face_detection/google_mlkit_face_detection.dart';
import 'package:google_mlkit_translation/google_mlkit_translation.dart';
import 'package:camera/camera.dart';

// OCR — reconhecimento de texto em imagens
class OcrService {
  final _reconhecedor = TextRecognizer(script: TextRecognitionScript.latin);

  Future<String> extrairTexto(String caminhoImagem) async {
    final input  = InputImage.fromFilePath(caminhoImagem);
    final result = await _reconhecedor.processImage(input);
    return result.blocks
      .map((b) => b.lines.map((l) => l.text).join(' '))
      .join('\n');
  }

  void fechar() => _reconhecedor.close();
}

// Deteccao de faces com landmarks
class FaceDetector {
  final _detector = FaceDetector(
    options: FaceDetectorOptions(
      enableLandmarks:  true,
      enableClassification: true,  // open mouth, smiling
      performanceMode:  FaceDetectorMode.accurate,
    ),
  );

  Future<List<Face>> detectar(CameraImage camImg, InputImageRotation rot) async {
    final planes = camImg.planes;
    final bytes  = Uint8List.fromList(planes.fold<List<int>>([], (b, p) => b..addAll(p.bytes)));
    final input  = InputImage.fromBytes(bytes: bytes, metadata: InputImageMetadata(
      size:     Size(camImg.width.toDouble(), camImg.height.toDouble()),
      rotation: rot,
      format:   InputImageFormat.yuv_420_888,
      bytesPerRow: planes[0].bytesPerRow,
    ));
    return _detector.processImage(input);
  }
}

// Traducao on-device (modelo baixado no primeiro uso)
class TradutorService {
  late OnDeviceTranslator _tradutor;

  Future<void> inicializar() async {
    final gerenciador = OnDeviceTranslatorModelManager();
    if (!await gerenciador.isModelDownloaded('pt')) {
      await gerenciador.downloadModel('pt');
    }
    _tradutor = OnDeviceTranslator(
      sourceLanguage: TranslateLanguage.english,
      targetLanguage: TranslateLanguage.portuguese,
    );
  }

  Future<String> traduzir(String texto) => _tradutor.translateText(texto);
  void fechar() => _tradutor.close();
}

Dart + google_generative_ai — Chat com streaming e visao
import 'package:google_generative_ai/google_generative_ai.dart';

class GeminiService {
  late GenerativeModel _model;
  late ChatSession     _session;

  void inicializar() {
    _model = GenerativeModel(
      model:  'gemini-1.5-flash',
      apiKey: const String.fromEnvironment('GEMINI_KEY'),
      generationConfig: GenerationConfig(
        temperature:    0.7,
        maxOutputTokens: 2048,
        responseMimeType: 'text/plain',
      ),
      safetySettings: [
        SafetySetting(HarmCategory.harassment,      HarmBlockThreshold.mediumAndAbove),
        SafetySetting(HarmCategory.hateSpeech,        HarmBlockThreshold.mediumAndAbove),
        SafetySetting(HarmCategory.sexuallyExplicit,   HarmBlockThreshold.mediumAndAbove),
        SafetySetting(HarmCategory.dangerousContent,   HarmBlockThreshold.mediumAndAbove),
      ],
      systemInstruction: Content.system('Voce e um assistente tecnico especializado em Dart/Flutter.'),
    );
    _session = _model.startChat();
  }

  // Resposta em streaming
  Stream<String> perguntarStream(String mensagem) async* {
    final stream = _session.sendMessageStream(Content.text(mensagem));
    await for (final chunk in stream) {
      final texto = chunk.text;
      if (texto != null) yield texto;
    }
  }

  // Analise de imagem (multimodal)
  Future<String> analisarImagem(Uint8List bytes, String pergunta) async {
    final content = Content.multi([
      DataPart('image/jpeg', bytes),
      TextPart(pergunta),
    ]);
    final resp = await _model.generateContent([content]);
    return resp.text ?? 'Sem resposta';
  }
}

// Widget com streaming de resposta
class ChatWidget extends StatefulWidget {
  const ChatWidget({super.key, required this.gemini});
  final GeminiService gemini;
  @override
  State<ChatWidget> createState() => _ChatWidgetState();
}

class _ChatWidgetState extends State<ChatWidget> {
  final _controller = TextEditingController();
  final _resposta   = ValueNotifier<String>('');

  Future<void> _enviar() async {
    _resposta.value = '';
    widget.gemini.perguntarStream(_controller.text).listen((chunk) {
      _resposta.value += chunk;
    });
    _controller.clear();
  }

  @override
  Widget build(BuildContext ctx) => Column(children: [
    ValueListenableBuilder<String>(
      valueListenable: _resposta,
      builder: (_, text, __) => Text(text),
    ),
    Row(children: [
      Expanded(child: TextField(controller: _controller)),
      IconButton(icon: const Icon(Icons.send), onPressed: _enviar),
    ]),
  ]);
}

Processamento de Dados com dart:math

Dart — Estatistica descritiva, normalizacao e regressao linear
import 'dart:math';

extension Estatistica on List<double> {
  double get media  => reduce((a, b) => a + b) / length;

  double get variancia {
    final m = media;
    return map((x) => pow(x - m, 2).toDouble()).reduce((a, b) => a + b) / length;
  }

  double get desvioPadrao => sqrt(variancia);

  double get mediana {
    final s = [...this]..sort();
    return length.isOdd
      ? s[length ~/ 2]
      : (s[length ~/ 2 - 1] + s[length ~/ 2]) / 2;
  }

  // Normalizacao Min-Max
  List<double> normalizar() {
    final mn = reduce(min);
    final mx = reduce(max);
    final alcance = mx - mn;
    return map((x) => alcance == 0 ? 0.0 : (x - mn) / alcance).toList();
  }

  // Z-Score
  List<double> zScore() {
    final m = media, s = desvioPadrao;
    return s == 0 ? List.filled(length, 0.0) : map((x) => (x - m) / s).toList();
  }
}

// Regressao linear simples (minimos quadrados)
({double a, double b, double r2}) regressaoLinear(
    List<double> x, List<double> y) {
  final n   = x.length;
  final mx  = x.media, my = y.media;
  final num = List.generate(n, (i) => (x[i] - mx) * (y[i] - my)).reduce((a,b) => a+b);
  final den = List.generate(n, (i) => pow(x[i] - mx, 2).toDouble()).reduce((a,b) => a+b);
  final b   = num / den;
  final a   = my - b * mx;

  // Coeficiente R2
  final ssRes = List.generate(n, (i) => pow(y[i] - (a + b * x[i]), 2).toDouble()).reduce((p,q) => p+q);
  final ssTot = List.generate(n, (i) => pow(y[i] - my, 2).toDouble()).reduce((p,q) => p+q);
  return (a: a, b: b, r2: 1 - ssRes / ssTot);
}

void main() {
  final x = [1, 2, 3, 4, 5].map((v) => v.toDouble()).toList();
  final y = [2.1, 3.9, 6.0, 8.2, 9.8];
  final reg = regressaoLinear(x, y);
  print('y = ${reg.b.toStringAsFixed(3)}x + ${reg.a.toStringAsFixed(3)} | R² = ${reg.r2.toStringAsFixed(4)}');
}
