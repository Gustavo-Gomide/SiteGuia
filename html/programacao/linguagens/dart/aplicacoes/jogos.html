Dart — Jogos com Flame Engine

Flame e o game engine 2D open-source para Flutter/Dart. Usa o canvas Flutter como renderer, roda em iOS, Android, Web e Desktop a partir de um unico codebase. Segue o padrao Component/System, com mixins composicionais para colisao, input, animacao, camera e audio — semelhante ao ECS mas mais Dart-idiomatico.

Flame 1.x+ usa o conceito de FlameGame (raiz) + arvore de Components. Cada componente pode ter filhos, ciclo de vida (onLoad, update, render, onRemove) e mixins como HasCollisionDetection, KeyboardHandler, TapCallbacks e DragCallbacks.

Estrutura Basica — FlameGame e Component

Dart + Flame — FlameGame, PositionComponent e ciclo de vida
import 'package:flame/flame.dart';
import 'package:flame/game.dart';
import 'package:flame/components.dart';
import 'package:flame/input.dart';
import 'package:flutter/material.dart';
import 'dart:ui';

void main() {
  WidgetsFlutterBinding.ensureInitialized();
  Flame.device.fullScreen();
  Flame.device.setLandscape();
  runApp(GameWidget(game: MeuJogo()));
}

class MeuJogo extends FlameGame with HasCollisionDetection, KeyboardEvents {
  late Player  _player;
  late TextComponent _pontos;
  int pontos = 0;

  @override
  Future<void> onLoad() async {
    await super.onLoad();

    // Fundo
    await add(ParallaxComponent.load([
      ParallaxImageData('bg_far.png'),
      ParallaxImageData('bg_near.png'),
    ], baseVelocity: Vector2(20, 0)));

    // Jogador
    _player = Player()
      ..position = size / 2
      ..anchor = Anchor.center;
    await add(_player);

    // HUD
    _pontos = TextComponent(
      text: 'Pontos: 0',
      textRenderer: TextPaint(style: const TextStyle(color: Colors.white, fontSize: 18)),
      position: Vector2(10, 10),
    );
    camera.viewport.add(_pontos);  // HUD nao se move com camera
  }

  void adicionarPonto() {
    pontos++;
    _pontos.text = 'Pontos: $pontos';
  }
}

// Componente do jogador
class Player extends SpriteAnimationComponent
    with HasGameRef<MeuJogo>, CollisionCallbacks, KeyboardHandler {
  static const _velocidade = 200.0;
  final _movimento = Vector2.zero();

  Player() : super(size: Vector2.all(64));

  @override
  Future<void> onLoad() async {
    animation = await gameRef.loadSpriteAnimation(
      'player.png',
      SpriteAnimationData.sequenced(amount: 8, stepTime: 0.1, textureSize: Vector2.all(64)),
    );
    await add(RectangleHitbox());
  }

  @override
  void update(double dt) {
    super.update(dt);
    position.add(_movimento * (_velocidade * dt));
    // Clampar dentro da tela
    position.clamp(
      Vector2.zero()     + (size / 2),
      gameRef.size - (size / 2),
    );
  }

  @override
  bool onKeyEvent(RawKeyEvent event, Set<LogicalKeyboardKey> keysPressed) {
    _movimento.setZero();
    if (keysPressed.contains(LogicalKeyboardKey.arrowLeft))  _movimento.x = -1;
    if (keysPressed.contains(LogicalKeyboardKey.arrowRight)) _movimento.x =  1;
    if (keysPressed.contains(LogicalKeyboardKey.arrowUp))    _movimento.y = -1;
    if (keysPressed.contains(LogicalKeyboardKey.arrowDown))  _movimento.y =  1;
    if (_movimento != Vector2.zero()) _movimento.normalize();
    return true;
  }

  @override
  void onCollisionStart(Set<Vector2> pts, PositionComponent other) {
    super.onCollisionStart(pts, other);
    if (other is Moeda) {
      gameRef.adicionarPonto();
      other.removeFromParent();
    }
  }
}

Dart + Flame — Particulas, FlameAudio e CameraComponent
import 'package:flame/particles.dart';
import 'package:flame_audio/flame_audio.dart';

// Sistema de particulas — explosao ao morrer
void explodirParticulas(FlameGame game, Vector2 posicao) {
  game.add(ParticleSystemComponent(
    particle: Particle.generate(
      count:    30,
      lifespan: 0.8,
      generator: (i) => AcceleratedParticle(
        acceleration: (Vector2.random() - Vector2.all(0.5)) * 400,
        speed:        Vector2.random() * 200,
        child: CircleParticle(
          radius: 4,
          paint: Paint()..color = Colors.orange,
        ),
      ),
    ),
    position: posicao,
  ));
}

// CameraComponent — seguir jogador com limites de mundo
class MeuJogoComCamera extends FlameGame {
  @override
  Future<void> onLoad() async {
    final mundo  = World();
    final camera = CameraComponent(world: mundo);

    // Definir limites do mundo (no-scroll alem do mapa)
    camera.setBounds(Rectangle.fromLTWH(0, 0, 3200, 1600));

    final player = Player()..position = Vector2(160, 600);
    mundo.add(player);

    // Seguir o jogador com lerp suave
    camera.viewfinder.follow(player, maxSpeed: 300, horizontalOnly: false);

    await addAll([mundo, camera]);
  }
}

Tiled Maps — Mapas em tile grid

Dart + flame_tiled — Carregar TMX, layers e objetos de spawn
import 'package:flame_tiled/flame_tiled.dart';

class MapaFase1 extends Component with HasGameRef {
  late RenderableTiledMap _mapa;

  @override
  Future<void> onLoad() async {
    _mapa = await TiledComponent.load('fase1.tmx', Vector2.all(16));
    await add(TiledComponent(_mapa));

    // Layer de colisao — gerar hitboxes dos tiles
    final colisoes = _mapa.tileMap.getLayer<TileLayer>('colisao');
    if (colisoes != null) {
      await add(TiledCollisionComponents.fromLayer(colisoes, Vector2.all(16)));
    }

    // Layer de objetos — spawn de inimigos e power-ups
    final objetos = _mapa.tileMap.getLayer<ObjectGroup>('spawn');
    for (final obj in objetos!.objects) {
      final pos = Vector2(obj.x.toDouble(), obj.y.toDouble());
      switch (obj.type) {
        case 'inimigo': await add(Inimigo()..position = pos);
        case 'moeda':   await add(Moeda()..position = pos);
        case 'player':  gameRef.player.position = pos;
      }
    }
  }
}

// Inimigo com FSM simples (patrulhar / atacar)
class Inimigo extends SpriteAnimationGroupComponent<EstadoInimigo>
    with HasGameRef, CollisionCallbacks {
  static const _velPatrulha = 60.0, _velAtaque = 150.0;
  var _estadoAtual = EstadoInimigo.patrulhando;
  var _direcao = 1.0;

  @override
  void update(double dt) {
    super.update(dt);
    final dist = (position - gameRef.player.position).length;
    if (dist < 120) {
      _estadoAtual = EstadoInimigo.atacando;
      current = _estadoAtual;
    } else {
      _estadoAtual = EstadoInimigo.patrulhando;
      current = _estadoAtual;
    }
    final vel = _estadoAtual == EstadoInimigo.atacando ? _velAtaque : _velPatrulha;
    position.x += _direcao * vel * dt;
    if (position.x < 0 || position.x > 3200) _direcao *= -1;
  }
}

enum EstadoInimigo { patrulhando, atacando }

import 'package:flame/components.dart';

// RouterComponent — gerencia telas do jogo
class JogoCompleto extends FlameGame {
  late final RouterComponent router;

  @override
  Future<void> onLoad() async {
    router = RouterComponent(
      routes: {
        'menu':     Route(MenuScreen.new),
        'jogo':     Route(GameScreen.new),
        'gameOver': Route(GameOverScreen.new),
        'pause':    OverlayRoute(
          (ctx, game) => PauseOverlayWidget(game: game),
        ),
      },
      initialRoute: 'menu',
    );
    await add(router);
  }
}

// Save/Load com shared_preferences
class SaveService {
  static Future<void> salvar(int pontuacao, int fase) async {
    final prefs = await SharedPreferences.getInstance();
    await prefs.setInt('pontuacao', pontuacao);
    await prefs.setInt('fase', fase);
  }

  static Future<({int pontuacao, int fase})> carregar() async {
    final prefs = await SharedPreferences.getInstance();
    return (
      pontuacao: prefs.getInt('pontuacao') ?? 0,
      fase:      prefs.getInt('fase')      ?? 1,
    );
  }
}

Fisicas com Forge2D

Dart + flame_forge2d — Corpos rigidos, articulacoes e colisao com fisica
import 'package:flame_forge2d/flame_forge2d.dart';

// Jogo com fisica real (Box2D)
class JogoFisica extends Forge2DGame {
  JogoFisica() : super(gravity: Vector2(0, 9.8));

  @override
  Future<void> onLoad() async {
    await add(Chao());
    await add(Caixa(posicao: Vector2(5, 2)));
    await add(Bola(posicao:  Vector2(3, 0)));
  }
}

class Caixa extends BodyComponent {
  final Vector2 posicao;
  Caixa({required this.posicao});

  @override
  Body createBody() {
    final def = BodyDef()
      ..type     = BodyType.dynamic
      ..position = posicao;
    final shape = PolygonShape()..setAsBoxXY(0.5, 0.5);
    final fix   = FixtureDef(shape)
      ..restitution = 0.3
      ..friction    = 0.5
      ..density     = 1.0;
    return world.createBody(def)..createFixture(fix);
  }
}

class Bola extends BodyComponent {
  final Vector2 posicao;
  Bola({required this.posicao});

  @override
  Body createBody() {
    final def = BodyDef()
      ..type     = BodyType.dynamic
      ..position = posicao;
    final circle = CircleShape()..radius = 0.3;
    final fix    = FixtureDef(circle)
      ..restitution = 0.8  // quicante
      ..density     = 0.5;
    return world.createBody(def)..createFixture(fix);
  }
}
