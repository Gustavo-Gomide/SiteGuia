R é uma das plataformas mais completas para ciência de dados e machine learning estatisticamente rigoroso. O tidymodels unificou o fluxo de modelagem; torch para R trouxe redes neurais sem depender de Python; e ferramentas de interpretabilidade como DALEX e vip respondem à pergunta mais importante: por que o modelo fez essa previsão?

tidymodels — Framework Completo de ML

tidymodels é um meta-pacote que fornece uma interface unificada, consistente e tidy para todo o workflow de machine learning em R. Seus componentes formam uma pipeline modular: preparação de dados, especificação de modelo, ajuste de hiperparâmetros e avaliação.

PacoteResponsabilidade

tidymodels-completo.R — classificação binária
library("tidymodels")

# Dataset: prever se tumor é maligno (1) ou benigno (0)
dados <- read_csv("cancer.csv") |>
  mutate(diagnostico = factor(diagnostico, levels = c("B", "M")))

# 1. Split estratificado
set.seed(42)
splits <- initial_split(dados, prop = 0.75, strata = diagnostico)
treino  <- training(splits)
teste   <- testing(splits)
cv_folds <- vfold_cv(treino, v = 10, strata = diagnostico)

# 2. Receita de pré-processamento
receita <- recipe(diagnostico ~ ., data = treino) |>
  step_impute_median(all_numeric_predictors()) |>
  step_nzv(all_predictors()) |>       # remove features com variância ~zero
  step_normalize(all_numeric_predictors()) |>  # z-score
  step_corr(all_numeric_predictors(), threshold = 0.9)  # remove correlações altas

# 3. Especificação do modelo
modelo_xgb <- boost_tree(
  trees        = tune(),
  tree_depth   = tune(),
  learn_rate   = tune(),
  loss_reduction = tune()
) |>
  set_engine("xgboost") |>
  set_mode("classification")

# 4. Workflow
wf <- workflow() |> add_recipe(receita) |> add_model(modelo_xgb)

# 5. Busca de hiperparâmetros Bayesiana
grid_resultado <- tune_bayes(
  wf, resamples = cv_folds,
  metrics = metric_set(roc_auc, accuracy, f_meas),
  initial = 10, iter = 20
)

# 6. Finalização e avaliação no teste
melhor_params <- select_best(grid_resultado, metric = "roc_auc")
wf_final       <- finalize_workflow(wf, melhor_params)
resultado_teste <- last_fit(wf_final, splits)

collect_metrics(resultado_teste)
# # A tibble: 2 x 4
#   .metric  .estimator .estimate .config
#   accuracy binary        0.977  Preprocessor1_Model1
#   roc_auc  binary        0.996  Preprocessor1_Model1

# Curva ROC
resultado_teste |>
  collect_predictions() |>
  roc_curve(diagnostico, .pred_M) |>
  autoplot()

Deep Learning: Keras e torch para R

R tem dois caminhos para deep learning: keras (via reticulate, usa TensorFlow/Python por baixo) e torch (binding nativo de LibTorch, sem dependência de Python). O pacote torch é mais integrado ao ecossistema R.

torch-rede-neural.R — regressão com torch
library("torch")

# Dados de exemplo
x <- torch_tensor(matrix(rnorm(1000), ncol = 5))
y <- torch_tensor(rnorm(200))

# Definir rede com nn_module
RedeRegressao <- nn_module(
  "RedeRegressao",
  initialize = function() {
    self$l1 <- nn_linear(5, 64)
    self$l2 <- nn_linear(64, 32)
    self$l3 <- nn_linear(32, 1)
    self$dropout <- nn_dropout(p = 0.2)
  },
  forward = function(x) {
    x |> self$l1() |> nnf_relu() |> self$dropout() |>
         self$l2() |> nnf_relu() |>
         self$l3()
  }
)

rede <- RedeRegressao()
otimizador <- optim_adam(rede$parameters, lr = 0.001)
criterio   <- nn_mse_loss()

# Loop de treinamento
for (epoca in 1:100) {
  rede$train()
  prev  <- rede(x)$squeeze()
  perda <- criterio(prev, y)
  otimizador$zero_grad()
  perda$backward()
  otimizador$step()
  if (epoca %% 10 == 0)
    message(sprintf("Época %d | Loss: %.4f", epoca, perda$item()))
}

# Salvar e carregar modelo
torch_save(rede, "modelo.pt")
rede_carregada <- torch_load("modelo.pt")

XGBoost e LightGBM

xgboost-lightgbm.R
library("xgboost")
library("lightgbm")

# ---- XGBoost ----
dtreino <- xgb.DMatrix(data = as.matrix(treino[,-1]), label = treino[[1]])
dteste  <- xgb.DMatrix(data = as.matrix(teste[,-1]))

params <- list(
  objective   = "binary:logistic",
  eval_metric = "auc",
  eta         = 0.05,
  max_depth   = 6,
  subsample   = 0.8,
  colsample_bytree = 0.8
)

xgb_model <- xgb.train(
  params  = params,
  data    = dtreino,
  nrounds = 500,
  watchlist = list(treino = dtreino, teste = dteste),
  early_stopping_rounds = 20,
  verbose = 0
)

previsoes_xgb <- predict(xgb_model, dteste)

# ---- LightGBM ---- (mais rápido, melhor em categorias altas)
dtreino_lgb <- lgb.Dataset(
  data  = as.matrix(treino[,-1]),
  label = treino[[1]]
)

lgb_model <- lgb.train(
  params = list(objective = "binary", metric = "auc", learning_rate = 0.05),
  data   = dtreino_lgb,
  nrounds = 500
)

previsoes_lgb <- predict(lgb_model, as.matrix(teste[,-1]))

Interpretabilidade: DALEX, vip e pdp

Interpretabilidade é crítica para modelos em produção: regulamentações (LGPD, GDPR, setor financeiro) exigem explicações. As ferramentas abaixo funcionam com qualquer modelo — R, Python, ou black box — pois interagem apenas via predições.

interpretabilidade.R — DALEX, vip, pdp
library("DALEX")
library("vip")
library("pdp")

# Criar explainer DALEX — wrapper agnóstico de modelo
explainer <- explain(
  model      = wf_final |> extract_fit_parsnip(),
  data       = teste |> select(-diagnostico),
  y          = as.numeric(teste$diagnostico == "M"),
  label      = "XGBoost Cancer",
  predict_function = function(m, x) predict(m, new_data=x, type="prob")$.pred_M
)

# Importância de variáveis (permutation importance)
imp <- model_parts(explainer, type = "variable_importance")
plot(imp)

# Perfis PDP — efeito marginal de cada variável
pdp_profiles <- model_profile(explainer,
  variables = c("radius_mean", "texture_mean"))
plot(pdp_profiles)

# SHAP — explicação por observação
shap <- local_attributions(explainer, new_observation = teste[1, -1])
plot(shap)

# vip — importância nativa do modelo (rápido)
wf_final |>
  extract_fit_parsnip() |>
  vip(num_features = 15)   # gráfico de importância das 15 top features

AutoML com H2O

h2o-automl.R
library("h2o")
h2o.init(nthreads = -1, max_mem_size = "8G")

# Converter para H2O Frame
treino_h2o <- as.h2o(treino)
teste_h2o  <- as.h2o(teste)

# AutoML — testa múltiplos algoritmos automaticamente
aml <- h2o.automl(
  x               = setdiff(names(treino), "diagnostico"),
  y               = "diagnostico",
  training_frame  = treino_h2o,
  leaderboard_frame = teste_h2o,
  max_models      = 20,
  seed            = 42,
  max_runtime_secs = 300,
  include_algos   = c("GLM", "GBM", "XGBoost", "DRF", "StackedEnsemble")
)

# Leaderboard — ranking dos melhores modelos
print(aml@leaderboard, n = 10)

# Melhor modelo
melhor_modelo <- aml@leader
h2o.performance(melhor_modelo, newdata = teste_h2o)

# Explicabilidade automática
h2o.explain(aml, teste_h2o)   # relatório HTML completo

h2o.shutdown(prompt = FALSE)
