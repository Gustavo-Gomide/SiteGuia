R para Web

R para Desenvolvimento Web

Shiny para apps reativos, Plumber para APIs REST, httr2 como cliente HTTP e múltiplas opções de deploy em nuvem.

R não é uma linguagem de web de propósito geral, mas dentro do seu domínio de análise de dados é excepcional para construir dashboards, APIs analíticas e aplicações científicas interativas. O framework Shiny permite que estatísticos e cientistas de dados publiquem análises como aplicações web sem escrever uma linha de JavaScript.

Shiny — Fundamentos

Uma aplicação Shiny tem dois componentes principais: a UI (interface declarativa descrita em R) e o server (lógica reativa e computação). O framework gerencia automaticamente as dependências reativas — quando um input muda, apenas os outputs que dependem dele são recalculados.

app-fundamentos.R — anatomia de um app Shiny
library("shiny")

# ============ UI — define o layout e controles ============
ui <- fluidPage(
  titlePanel("Análise de Regressão Interativa"),

  sidebarLayout(
    sidebarPanel(
      selectInput("dataset", "Dataset:",
                  choices = c("mtcars", "iris", "airquality")),
      uiOutput("sel_x"),   # UI dinâmica — preenchida pelo server
      uiOutput("sel_y"),
      checkboxInput("linha_regressao", "Exibir regressão linear", value = TRUE),
      downloadButton("baixar_plot", "Baixar Gráfico")
    ),
    mainPanel(
      plotOutput("grafico", height = "400px"),
      verbatimTextOutput("sumario")
    )
  )
)

# ============ Server — lógica reativa ============
server <- function(input, output, session) {

  # reactive() — valor que recalcula quando inputs mudam
  dados <- reactive({
    get(input$dataset)   # carrega o dataset pelo nome
  })

  # UI dinâmica que depende do dataset escolhido
  output$sel_x <- renderUI({
    selectInput("x", "Eixo X:", choices = names(dados()))
  })
  output$sel_y <- renderUI({
    selectInput("y", "Eixo Y:", choices = names(dados()))
  })

  # Gráfico reativo
  output$grafico <- renderPlot({
    req(input$x, input$y)   # req() cancela silenciosamente se NULL
    p <- plot(dados()[[input$x]], dados()[[input$y]],
              xlab = input$x, ylab = input$y, pch = 19, col = "steelblue")
    if (input$linha_regressao) abline(lm(dados()[[input$y]] ~ dados()[[input$x]]),
                                       col = "red", lwd = 2)
  })

  # Sumário da regressão
  output$sumario <- renderPrint({
    req(input$x, input$y)
    if (input$linha_regressao)
      summary(lm(dados()[[input$y]] ~ dados()[[input$x]]))
  })

  # Download de arquivo
  output$baixar_plot <- downloadHandler(
    filename = function() paste0("plot-", Sys.Date(), ".png"),
    content  = function(file) {
      png(file, width = 800, height = 600)
      plot(dados()[[input$x]], dados()[[input$y]])
      dev.off()
    }
  )
}

shinyApp(ui, server)

Shiny — Módulos e Reatividade Avançada

Módulos Shiny encapsulam UI + server em unidades reutilizáveis com namespacing automático de IDs. São essenciais para apps grandes — sem módulos, IDs conflitam e o código torna-se unmaintainable.

modulos.R — padrão de módulo Shiny
# ---- Definição do módulo ----
# O prefixo NS() garante IDs únicos no namespace do módulo

graficoUI <- function(id) {
  ns <- NS(id)   # namespacing function
  tagList(
    selectInput(ns("variavel"), "Variável:", choices = names(mtcars)),
    plotOutput(ns("plot"))
  )
}

graficoServer <- function(id, dados) {
  moduleServer(id, function(input, output, session) {
    output$plot <- renderPlot({
      hist(dados()[[input$variavel]], col = "steelblue",
           main = input$variavel)
    })
  })
}

# ---- App usando o módulo duas vezes ----
ui <- fluidPage(
  fluidRow(
    column(6, graficoUI("plot1")),  # namespace: plot1-variavel, plot1-plot
    column(6, graficoUI("plot2"))   # namespace: plot2-variavel, plot2-plot
  )
)

server <- function(input, output, session) {
  dados_reativos <- reactive(mtcars)
  graficoServer("plot1", dados_reativos)
  graficoServer("plot2", dados_reativos)
}

shinyApp(ui, server)

Plumber — API REST em R

Plumber transforma funções R em endpoints HTTP através de decoradores (comentários especiais). É ideal para expor modelos de ML treinados em R como serviços consumíveis por outras aplicações.

plumber.R — API analítica completa
# plumber.R — salvar este arquivo e executar: plumber::pr_run(plumber::pr("plumber.R"))
library("plumber")

#* @apiTitle API de Análise Preditiva
#* @apiDescription Endpoints para previsão e análise estatística

#* Retorna estatísticas descritivas de um vetor numérico
#* @param dados String JSON de array numérico
#* @post /descritivas
function(dados) {
  x <- jsonlite::fromJSON(dados)
  list(
    media   = mean(x),
    mediana = median(x),
    desvio  = sd(x),
    n       = length(x)
  )
}

#* Previsão com modelo de regressão pré-treinado
#* @param peso Peso em kg
#* @param altura Altura em metros
#* @get /imc
function(peso, altura) {
  peso   <- as.numeric(peso)
  altura <- as.numeric(altura)
  imc    <- peso / altura^2
  list(imc = round(imc, 2),
       categoria = ifelse(imc < 25, "Normal", "Sobrepeso+"))
}

#* Serialização de gráfico como PNG
#* @serializer png
#* @get /grafico
function() {
  plot(mtcars$wt, mtcars$mpg,
       xlab = "Peso", ylab = "Consumo",
       pch = 19, col = "steelblue")
}

httr2 — Cliente HTTP e jsonlite

cliente-http.R
library("httr2")
library("jsonlite")

# GET com autenticação Bearer
resp <- request("https://api.exemplo.com/dados") |>
  req_headers(Accept = "application/json") |>
  req_auth_bearer_token(Sys.getenv("API_TOKEN")) |>
  req_perform()

dados <- resp |> resp_body_json(simplifyVector = TRUE)

# POST com corpo JSON
payload <- list(nome = "Ana", nota = 9.2)
resp_post <- request("https://api.exemplo.com/alunos") |>
  req_body_json(payload) |>
  req_perform()

# Tratamento de erros HTTP
resp_status(resp_post)                 # 201 Created
resp_post |> resp_check_status()        # lança erro se 4xx ou 5xx

# Rate limiting com retry automático
resp_limitado <- request("https://api.exemplo.com/lento") |>
  req_retry(max_tries = 3, backoff = \(i) 2^i) |>
  req_throttle(rate = 10 / 60) |>   # 10 requisições por minuto
  req_perform()

# jsonlite — serialização e desserialização
json_str <- jsonlite::toJSON(list(a = 1, b = c(2, 3)), auto_unbox = TRUE)
obj      <- jsonlite::fromJSON(json_str)

Dockerfile — app Shiny em container
FROM rocker/shiny:4.4.1

# Instalar dependências do sistema
RUN apt-get update && apt-get install -y libcurl4-openssl-dev libssl-dev libxml2-dev

# Instalar pacotes R
RUN Rscript -e "install.packages(c('dplyr','ggplot2','plotly'), repos='https://cran.rstudio.com/')"

# Copiar app
COPY app/ /srv/shiny-server/meuapp/

# Permissões
RUN chown -R shiny:shiny /srv/shiny-server/

EXPOSE 3838
CMD ["/usr/bin/shiny-server"]
