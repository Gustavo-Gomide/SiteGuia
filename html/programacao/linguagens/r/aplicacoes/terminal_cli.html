R para Terminal e CLI

Rscript, optparse e argparse para argumentos de linha de comando, cli para output colorido e barras de progresso, processx para subprocessos e fs para o filesystem.

R é excelente para automação via linha de comando. Com Rscript, scripts R executam como qualquer programa Unix — com shebang, argumentos, saídas padronizadas e integração com make, cron e pipelines de CI/CD. Bioinformáticos usam R em pipelines Snakemake há décadas; agora qualquer equipe de dados pode fazer o mesmo.

Rscript — Fundamentos de Scripts Executáveis

Rscript é o executável para rodar scripts R não-interativos. É parte da instalação padrão de R em todos os sistemas operacionais.

Terminal — modos de execução com Rscript
# Executar script
Rscript meu_script.R

# Executar com argumentos
Rscript analise.R --input dados.csv --output relatorio.html --k 5

# Executar expressão inline
Rscript -e "cat(sqrt(144))"
# 12

# Pipe Unix — R consumindo stdin
cat dados.csv | Rscript -e "
  con <- file('stdin')
  df  <- read.csv(con)
  cat('Linhas:', nrow(df), '\n')
"

# Verificar se pacote existe antes de usar
Rscript -e "if (!requireNamespace('ggplot2')) quit(status=1)"

# Capturar código de saída
Rscript script.R && echo "OK" || echo "ERRO"
script-executavel.R — shebang para Unix/macOS
#!/usr/bin/env Rscript
# Tornar executável: chmod +x script-executavel.R
# Executar: ./script-executavel.R

library("tidyverse")

message("Script iniciado em: ", Sys.time())

# Argumentos passados após --args
args <- commandArgs(trailingOnly = TRUE)
cat("Argumentos recebidos:", args, "\n")

# Exemplo: ./script.R dados.csv 0.05
arquivo <- ifelse(length(args) >= 1, args[1], "dados.csv")
alpha   <- ifelse(length(args) >= 2, as.numeric(args[2]), 0.05)

cat("Arquivo:", arquivo, "| Alpha:", alpha, "\n")

# Código de saída explícito
quit(status = 0)

optparse — Argumentos Estruturados

commandArgs(trailingOnly=TRUE) é primitivo. O pacote optparse (inspirado no optparse do Python) fornece uma API completa para definir opções, valores padrão, tipos e geração automática de mensagem de ajuda (--help).

analise-cli.R — interface optparse completa
#!/usr/bin/env Rscript
library("optparse")

# Definir opções
lista_opcoes <- list(
  make_option(c("-i", "--input"),
              type    = "character",
              help    = "Arquivo CSV de entrada [obrigatório]",
              metavar = "FILE"),
  make_option(c("-o", "--output"),
              type    = "character",
              default = "output.html",
              help    = "Arquivo de saída [padrão: %default]",
              metavar = "FILE"),
  make_option(c("-k", "--clusters"),
              type    = "integer",
              default = 3L,
              help    = "Número de clusters [padrão: %default]"),
  make_option(c("-v", "--verbose"),
              action  = "store_true",
              default = FALSE,
              help    = "Exibir saída detalhada")
)

parser <- OptionParser(
  option_list  = lista_opcoes,
  description  = "Análise de clustering em dados CSV",
  usage        = "usage: %prog [options]"
)
opts <- parse_args(parser)

# Validação — input é obrigatório
if (is.null(opts$input)) {
  print_help(parser)
  quit(status = 2)
}
if (!file.exists(opts$input)) {
  stop(sprintf("Arquivo não encontrado: %s", opts$input))
}

if (opts$verbose) message("Lendo: ", opts$input)

# $ Rscript analise-cli.R --help
# Usage: analise-cli.R [options]
# Análise de clustering em dados CSV
# Options:
#   -i FILE, --input=FILE   Arquivo CSV de entrada [obrigatório]
#   -o FILE, --output=FILE  Arquivo de saída [padrão: output.html]
#   -k INTEGER, --clusters=INTEGER  Número de clusters [padrão: 3]
#   -v, --verbose           Exibir saída detalhada
#   -h, --help              Exibir esta mensagem

cli — Output Profissional no Terminal

O pacote cli (do universo tidyverse) fornece uma API rica para output de terminal: mensagens coloridas por tipo, barras de progresso, spinners, formatação de listas e diffs — tudo com detecção automática de suporte a cores (TTY vs pipe).

cli-output.R — repertório completo
library("cli")

# ---- Mensagens tipadas ----
cli_h1("Pipeline de Análise")                    # === Pipeline de Análise ===
cli_h2("Etapa 1: Carregamento")                  # -- Etapa 1: Carregamento --
cli_alert_info("Lendo arquivo: {.file dados.csv}")   # ℹ Lendo arquivo: dados.csv
cli_alert_success("1.000 registros carregados")     # ✓ 1.000 registros carregados
cli_alert_warning("12 valores NA encontrados")       # ! 12 valores NA encontrados
cli_alert_danger("Arquivo de saída não gravável")   # ✖ Arquivo de saída não gravável

# ---- Listas e formatação ----
cli_ul(c("Limpeza de dados", "Normalização", "Modelagem"))
cli_text("Rodando com {.val {Sys.getenv('USER')}} — núcleos: {.strong {parallel::detectCores()}}")

# ---- Barra de progresso ----
processar_arquivos <- function(arquivos) {
  cli_progress_bar("Processando arquivos", total = length(arquivos))
  for (f in arquivos) {
    Sys.sleep(0.1)   # simulação de trabalho
    cli_progress_update(set = which(arquivos == f))
  }
  cli_progress_done()
}

# ---- Spinner para operações indetermináveis ----
cli_progress_step("Conectando ao banco de dados...")
Sys.sleep(1)
cli_progress_step("Executando query...")
Sys.sleep(0.5)
cli_alert_success("Query concluída em 1.5s")

# ---- Erros com traceback formatado ----
tryCatch({
  stop("Falha na conexão")
}, error = function(e) {
  cli_abort(c(
    "Não foi possível conectar ao banco de dados",
    "x" = conditionMessage(e),
    "i" = "Verifique a variável de ambiente {.envvar PG_HOST}"
  ))
})

processx e fs — Subprocessos e Filesystem

processx-fs.R
library("processx")
library("fs")

# ---- processx — executar subprocesso ----
# Síncrono — esperar a conclusão
resultado <- run("python3", c("preprocessar.py", "--input", "raw.csv"),
                 echo = FALSE, error_on_status = TRUE)
cat(resultado$stdout)

# Assíncrono — iniciar e continuar
proc <- process$new("python3", c("longo_processo.py"),
                   stdout = "|", stderr = "|")
while (proc$is_alive()) {
  saida <- proc$read_output_lines()
  if (length(saida) > 0) cat(saida, sep = "\n")
  Sys.sleep(0.5)
}
proc$get_exit_status()   # 0 = sucesso

# ---- fs — filesystem tidy ----
# Listar arquivos CSV recursivamente
csvs <- fs::dir_ls("./dados", recurse = TRUE, glob = "*.csv")

# Criar estrutura de pastas
fs::dir_create(c("output/plots", "output/tables", "output/models"))

# Copiar renomeando com timestamp
fs::file_copy(csvs, fs::path("backup", paste0(Sys.Date(), "_", fs::path_file(csvs))))

# Informações de arquivo
info <- fs::file_info(csvs)
info |> dplyr::arrange(desc(modification_time)) |> head(5)

# Caminhos seguros e portáveis
caminho <- fs::path("projeto", "dados", "2024-Q1.csv")
# "projeto/dados/2024-Q1.csv" (/ no Unix, \ no Windows automaticamente)

Pipelines com Make e Snakemake

Em projetos de análise reproduzível e bioinformática, R scripts são orquestrados por Make (ou Snakemake especificamente para o workflow de bioinformática). Cada etapa é uma regra com dependências explícitas — o orquestrador reexecuta apenas o necessário quando arquivos mudam.

Makefile — pipeline de análise completa
# Makefile — pipeline de análise de dados
# Executar: make all
# Limpar: make clean

RSCRIPT = Rscript --vanilla

# Alvo padrão
all: output/relatorio_final.html

# Etapa 1: limpeza de dados
output/dados_limpos.rds: dados/raw.csv R/01_limpar.R
	@echo "Limpando dados..."
	$(RSCRIPT) R/01_limpar.R --input $< --output $@

# Etapa 2: features
output/features.rds: output/dados_limpos.rds R/02_features.R
	@echo "Criando features..."
	$(RSCRIPT) R/02_features.R --input $< --output $@

# Etapa 3: modelo
output/modelo.rds: output/features.rds R/03_modelo.R
	@echo "Treinando modelo..."
	$(RSCRIPT) R/03_modelo.R --input $< --output $@

# Etapa 4: relatório
output/relatorio_final.html: output/modelo.rds relatorio.qmd
	@echo "Gerando relatório..."
	quarto render relatorio.qmd --output $@

clean:
	rm -f output/*.rds output/*.html

.PHONY: all clean
