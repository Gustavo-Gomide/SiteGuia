R — Avançado

R Avançado

Ambientes e escopo lexical, sistemas de objetos, metaprogramação com rlang, integração C++ via Rcpp e computação paralela.

Dominar R avançado significa entender a máquina interna da linguagem: como ambientes chain é usada para escopo lexical, como NSE (non-standard evaluation) permite DSLs expressivas como dplyr, como S3/S4/R6 implementam orientação a objetos e como Rcpp permite código C++ compilado a partir do R para gargalos de performance.

Ambientes e Escopo Lexical

Um ambiente (environment) em R é um dicionário de nomes para valores, mais um ponteiro para um ambiente pai. O escopo lexical significa que as funções buscam variáveis no ambiente de sua definição, não de onde são chamadas.

ambientes.R
# Ambiente global e seus pais
environmentName(globalenv())    # "R_GlobalEnv"
environmentName(baseenv())      # "base"
environmentName(emptyenv())     # "R_EmptyEnv" — topo da cadeia

# Criar e inspecionar ambientes
e1 <- new.env(parent = emptyenv())
e1$x <- 42
ls(e1)                          # "x"
get("x", envir = e1)            # 42
exists("x", envir = e1)         # TRUE

# Escopo lexical em ação
x <- 1
f <- function() {
  x <- 2
  function() x   # retorna closure que captura x = 2
}
g <- f()
g()  # 2 — lê x do ambiente de f, não do global

# <<- — atribuição no ambiente pai (permite mutação de estado)
contador <- function() {
  n <- 0
  list(
    incrementar = function() { n <<- n + 1 },
    valor       = function() n
  )
}
c1 <- contador()
c1$incrementar(); c1$incrementar()
c1$valor()   # 2

Sistemas de Objetos: S3, S4, R5 e R6

R tem quatro sistemas de orientação a objetos principais. S3 é informal e prevalente na base; S4 é formal com validação de tipos; R5 (Reference Classes) e R6 introduzem semântica de referência com mutação in-place.

S3 — Sistema Informal (o mais usado)

s3.R
# S3: classe é apenas um atributo "class" no objeto
novo_animal <- function(nome, som) {
  obj <- list(nome = nome, som = som)
  class(obj) <- "Animal"
  obj
}

# Método genérico S3
falar <- function(x, ...) UseMethod("falar")

# Despacho por classe
falar.Animal <- function(x, ...) {
  cat(x$nome, "diz:", x$som, "\n")
}
falar.default <- function(x, ...) cat("Objeto sem voz\n")

gato <- novo_animal("Mimi", "Miau")
falar(gato)   # Mimi diz: Miau

# print.Animal — método print customizado
print.Animal <- function(x, ...) {
  cat("<Animal>", x$nome, "\n")
}
print(gato)   # <Animal> Mimi

R6 — Orientação a Objetos por Referência

r6.R — requer library(R6)
# library(R6)
Conta <- R6::R6Class("Conta",
  public = list(
    saldo = 0,
    titular = NULL,
    initialize = function(titular, saldo_inicial = 0) {
      self$titular <- titular
      self$saldo   <- saldo_inicial
    },
    depositar = function(valor) {
      self$saldo <- self$saldo + valor
      invisible(self)   # retorna self para encadeamento
    },
    sacar = function(valor) {
      if (valor > self$saldo) stop("Saldo insuficiente")
      self$saldo <- self$saldo - valor
      invisible(self)
    },
    print = function(...) {
      cat("Conta de", self$titular, "| Saldo: R$", self$saldo, "\n")
    }
  )
)

conta <- Conta$new("Ana", saldo_inicial = 100)
conta$depositar(50)$sacar(30)   # encadeamento de métodos
conta    # Conta de Ana | Saldo: R$ 120

# Herança em R6
ContaPoupanca <- R6::R6Class("ContaPoupanca",
  inherit = Conta,
  public = list(
    taxa_juros = 0.005,
    render = function() {
      self$saldo <- self$saldo * (1 + self$taxa_juros)
    }
  )
)

Metaprogramação e NSE com rlang

Non-standard evaluation (NSE) é a capacidade de R de capturar expressões sem avaliá-las, permitindo DSLs como dplyr::filter(dados, coluna > 10) onde coluna não precisa existir no escopo global. O pacote rlang fornece ferramentas modernas e seguras para NSE.

metaprogramacao.R — rlang, quasiquotation
# library(rlang)

# enquo() captura uma expressão sem avaliá-la
mostrar_expr <- function(x) {
  expr <- rlang::enquo(x)
  cat("Expressão:", rlang::expr_text(expr), "\n")
  rlang::eval_tidy(expr)
}
mostrar_expr(1 + 2 * 3)
# Expressão: 1 + 2 * 3
# [1] 7

# quo() / expr() — capturar expressão no site da definição
e <- rlang::expr(x + y)
rlang::eval_tidy(e, list(x = 3, y = 4))   # 7

# !! (bang-bang) — unquoting: injeta valor em expressão
var <- rlang::sym("mpg")   # símbolo R
dplyr::select(mtcars, !!var)   # seleciona coluna "mpg" programaticamente

# !!! (bang-bang-bang) — splice: injeta lista de expressões
filtros <- list(rlang::expr(cyl == 6), rlang::expr(hp > 110))
dplyr::filter(mtcars, !!!filtros)   # aplica todos os filtros

# Função que aceita coluna por nome (tidyeval pattern)
resumir_por <- function(df, grupo, valor) {
  df |>
    dplyr::group_by({{ grupo }}) |>
    dplyr::summarise(media = mean({{ valor }}, na.rm = TRUE))
}
resumir_por(mtcars, cyl, mpg)

Rcpp — Integração com C++

Rcpp permite escrever funções C++ que são compiladas e chamadas diretamente do R, eliminando gargalos de performance em laços computacionalmente intensos. Desenvolvido por Dirk Eddelbuettel e Romain François, é um dos pacotes mais baixados do CRAN.

rcpp_exemplo.R + rcpp_soma.cpp
# library(Rcpp)

# Compilação inline — cppFunction() define a função diretamente na string
Rcpp::cppFunction('
  double soma_c(NumericVector x) {
    double total = 0;
    for (int i = 0; i < x.size(); i++) {
      total += x[i];
    }
    return total;
  }
')

soma_c(c(1.0, 2.0, 3.0))   # 6 — executado em C++

# sourceCpp() — arquivo .cpp separado
# Arquivo: rcpp_soma.cpp
# #include <Rcpp.h>
# using namespace Rcpp;
#
# // [[Rcpp::export]]
# NumericVector cumsum_c(NumericVector x) {
#   int n = x.size();
#   NumericVector out(n);
#   out[0] = x[0];
#   for (int i = 1; i < n; i++) out[i] = out[i-1] + x[i];
#   return out;
# }

Rcpp::sourceCpp("rcpp_soma.cpp")
cumsum_c(c(1, 2, 3, 4))   # 1 3 6 10

# Benchmark: R vs C++ em laço intenso
# library(bench)
x <- runif(1e6)
bench::mark(
  r_puro  = cumsum(x),
  cpp_fn  = cumsum_c(x),
  check   = FALSE
)

Programação Funcional com purrr

purrr (tidyverse) fornece um conjunto consistente e type-safe de funções para programação funcional em R, substituindo e expandindo a família apply com APIs mais previsíveis e composíveis.

purrr.R
# library(purrr)

# map() — sempre retorna lista (equivalente a lapply)
purrr::map(1:5, \(x) x ^ 2)          # lista de 5 elementos

# map_dbl, map_int, map_chr, map_lgl — type-safe
purrr::map_dbl(1:5, \(x) x ^ 2)      # vetor double: 1 4 9 16 25
purrr::map_chr(c("a","b"), toupper)  # "A" "B"

# map2() — dois vetores em paralelo
purrr::map2_dbl(1:3, 10:12, \(x, y) x + y)   # 11 13 15

# pmap() — múltiplos vetores via lista
parametros <- list(
  media = c(0, 5, 10),
  sd    = c(1, 2, 3)
)
purrr::pmap(parametros, \(media, sd) rnorm(5, media, sd))

# walk() — efeitos colaterais, descarta resultado
purrr::walk(c("a.csv", "b.csv"), \(f) message("Processando:", f))

# reduce() e accumulate()
purrr::reduce(1:5, `+`)             # 15 = ((((1+2)+3)+4)+5)
purrr::accumulate(1:5, `+`)       # 1 3 6 10 15 — como cumsum

# safely() e possibly() — tratamento de erros funcional
safe_log <- purrr::safely(log)
safe_log(-1)   # list(result = NaN, error = NULL) — não lança erro

poss_log <- purrr::possibly(log, otherwise = NA_real_)
purrr::map_dbl(c(1, -1, 10), poss_log)   # 0 NA 2.30

Paralelismo: parallel, future e furrr

paralelismo.R
# parallel — biblioteca padrão do R
library("parallel")
cores <- detectCores() - 1L   # deixa 1 core para o SO

# mclapply — fork (Unix/macOS apenas)
resultado <- mclapply(1:8, function(x) x^2, mc.cores = cores)

# parLapply — cross-platform (Windows + Unix)
cl <- makeCluster(cores)
clusterExport(cl, c("minha_funcao"))
resultado <- parLapply(cl, 1:100, minha_funcao)
stopCluster(cl)

# future — abstração de back-ends de paralelismo
# library(future)
future::plan(future::multisession, workers = cores)

f1 <- future::future({ Sys.sleep(2); 42 })  # assíncrono
f2 <- future::future({ Sys.sleep(1); 99 })
future::value(f1)   # bloqueia até completar: 42
future::value(f2)   # 99

# furrr — purrr + future (drop-in replacement)
# library(furrr)
future::plan(future::multisession)
furrr::future_map_dbl(1:100, \(x) { Sys.sleep(0.01); x^2 })
# executa em paralelo — ~4x mais rápido com 4 workers
