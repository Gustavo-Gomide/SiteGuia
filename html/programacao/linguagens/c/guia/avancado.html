C — Tópicos Avançados

C Avançado

Ponteiros, memória dinâmica, preprocessador, function pointers e os segredos do comportamento indefinido.

É nos tópicos avançados que C revela sua real natureza: uma linguagem que confia plenamente no programador. Ponteiros dão acesso direto à memória; malloc e free exigem disciplina; macros transformam código em tempo de compilação. Dominar esses recursos é a diferença entre um programador C e um engenheiro de sistemas.

Ponteiros

Um ponteiro é uma variável que armazena um endereço de memória. O operador & obtém o endereço de uma variável; o operador * dereferencia (acessa o valor no endereço).

C — declaração, deref, aritmética de ponteiros
#include <stdio.h>

int main(void) {
    int  valor = 42;
    int *p     = &valor;   /* p aponta para valor */

    printf("valor   = %d\n",   valor);     /* 42      */
    printf("&valor  = %p\n",   (void*)&valor); /* endereço */
    printf("p       = %p\n",   (void*)p);  /* mesmo endereço */
    printf("*p      = %d\n",   *p);        /* 42 — deref */

    *p = 100;  /* modifica valor via ponteiro */
    printf("valor após *p=100: %d\n", valor); /* 100 */

    /* Aritmética de ponteiros em arrays */
    int arr[] = {10, 20, 30, 40};
    int *q = arr;   /* arr decai para &arr[0] */
    printf("arr[2] = %d, *(q+2) = %d\n", arr[2], *(q + 2)); /* ambos 30 */

    /* Ponteiro para ponteiro */
    int **pp = &p;
    printf("**pp = %d\n", **pp);  /* 100 */
    return 0;
}

Tipos de ponteiros importantes

DeclaraçãoSignificado
int *pponteiro para int
const int *pponteiro para int constante (não pode modificar o valor)
int * const pponteiro constante para int (não pode redirecionar)
const int * const pponteiro constante para int constante
void *pponteiro genérico — aceita qualquer tipo, exige cast
int **ppponteiro para ponteiro para int
int (*fp)(int, int)ponteiro para função que recebe dois int e retorna int

Ponteiro nulo e dangling pointer: Sempre inicialize ponteiros — ou com NULL, ou com um endereço válido. Verificar if (p != NULL) antes de dereferenciar evita segfaults. Nunca use um ponteiro após liberar a memória apontada.

Gerenciamento de Memória Dinâmica

A heap permite alocar memória em tempo de execução com tamanho variável ou longa duração. Toda alocação deve ter um free correspondente — sem exceções.

malloc

Aloca n bytes não inicializados. Retorna void* ou NULL em falha. malloc(n * sizeof(int))

calloc

Aloca n elementos de tamanho t e zera a memória. calloc(n, sizeof(int)) — mais seguro que malloc.

realloc

Redimensiona bloco alocado. Pode mover para novo endereço. Guarde o retorno em variável separada para não perder o ponteiro original se retornar NULL.

free

Libera memória alocada. Após o free, o ponteiro é inválido — atribua NULL imediatamente para evitar uso acidental (dangling pointer).

C — vetor dinâmico (array com realloc)
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
    int   *dados;
    int    tamanho;
    int    capacidade;
} Vetor;

Vetor *vetor_novo(int cap_inicial) {
    Vetor *v = malloc(sizeof(Vetor));
    if (!v) return NULL;
    v->dados = malloc(cap_inicial * sizeof(int));
    if (!v->dados) { free(v); return NULL; }
    v->tamanho   = 0;
    v->capacidade = cap_inicial;
    return v;
}

int vetor_push(Vetor *v, int valor) {
    if (v->tamanho == v->capacidade) {
        int nova_cap = v->capacidade * 2;
        int *tmp = realloc(v->dados, nova_cap * sizeof(int));
        if (!tmp) return -1;  /* falha: ponteiro original preservado */
        v->dados     = tmp;
        v->capacidade = nova_cap;
    }
    v->dados[v->tamanho++] = valor;
    return 0;
}

void vetor_free(Vetor *v) {
    if (!v) return;
    free(v->dados);
    free(v);
}

int main(void) {
    Vetor *v = vetor_novo(4);
    for (int i = 0; i < 10; i++) vetor_push(v, i * i);

    for (int i = 0; i < v->tamanho; i++)
        printf("%d ", v->dados[i]);
    puts("");

    vetor_free(v);
    v = NULL;  /* 'dangling pointer' prevention */
    return 0;
}
/* 0 1 4 9 16 25 36 49 64 81 */

Function Pointers e Callbacks

Ponteiros para funções permitem passar comportamento como argumento — a base de callbacks, tabelas de dispatch virtual (vtables) e plugins em C. É assim que qsort e bsearch da libc funcionam.

C — qsort, callbacks, tabela de dispatch
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

/* Assinatura de comparador para qsort */
int cmp_int_asc(const void *a, const void *b) {
    return *((const int*)a) - *((const int*)b);
}
int cmp_int_desc(const void *a, const void *b) {
    return *((const int*)b) - *((const int*)a);
}

/* Typedef para tornar a assinatura legível */
typedef double (*Operacao)(double, double);

double somar(double a, double b) { return a + b; }
double subtrair(double a, double b) { return a - b; }
double multiplicar(double a, double b) { return a * b; }
double dividir(double a, double b) { return b ? a / b : 0; }

/* Tabela de dispatch: associa char a função */
typedef struct { char op; Operacao fn; } Entrada;
Entrada tabela[] = {
    {'+', somar}, {'-', subtrair},
    {'*', multiplicar}, {'/', dividir}
};

int main(void) {
    int nums[] = {5, 3, 8, 1, 9, 2};
    int n = sizeof(nums) / sizeof(nums[0]);

    qsort(nums, n, sizeof(int), cmp_int_asc);
    printf("Asc: ");
    for(int i=0;i<n;i++) printf("%d ", nums[i]);

    qsort(nums, n, sizeof(int), cmp_int_desc);
    printf("\nDesc: ");
    for(int i=0;i<n;i++) printf("%d ", nums[i]);

    /* Tabela de dispatch */
    char op = '*';
    for (int i = 0; i < 4; i++) {
        if (tabela[i].op == op) {
            printf("\n6 %c 7 = %.0f\n", op, tabela[i].fn(6, 7));
            break;
        }
    }
    return 0;
}

Preprocessador — Macros e Guardas

O preprocessador C transforma código antes da compilação: expande #include, substitui macros, avalia condicionais. É poderoso mas perigoso se mal usado.

C — macros, guardas de header, X-macros
/* meu_header.h */
#ifndef MEU_HEADER_H   /* guarda de inclusão — evita dupla definição */
#define MEU_HEADER_H

/* Macro com argumento — adicione parênteses ao redor de cada uso! */
#define MAX(a, b)      ((a) > (b) ? (a) : (b))
#define ARRAY_LEN(arr) (sizeof(arr) / sizeof((arr)[0]))
#define SWAP(T, a, b)  do { T _t = (a); (a) = (b); (b) = _t; } while(0)

/* X-Macro: lista uma vez, usa múltiplas vezes */
#define CORES \
    X(VERMELHO, 0xFF0000) \
    X(VERDE,    0x00FF00) \
    X(AZUL,     0x0000FF)

typedef enum {
#define X(nome, val) CORE_##nome,
    CORES
#undef X
    CORE_TOTAL
} Cor;

#endif /* MEU_HEADER_H */

/* ---------------------------------- */
/* main.c */
#include <stdio.h>
/* #include "meu_header.h" — incluído acima para demonstração */

int main(void) {
    int x = 4, y = 7;
    printf("MAX = %d\n", MAX(x, y));  /* 7 */

    int arr[] = {1,2,3,4,5};
    printf("len = %zu\n", ARRAY_LEN(arr));  /* 5 */

    SWAP(int, x, y);
    printf("x=%d y=%d\n", x, y);  /* x=7 y=4 */

#ifdef DEBUG
    printf("[DEBUG] x=%d y=%d\n", x, y);
#endif

    return 0;
}

Macros sem parênteses causam bugs sutis: #define DOBRAR(x) x*2 resulta em DOBRAR(3+1) = 3+1*2 = 5 (errado!). Sempre: #define DOBRAR(x) ((x)*2). Para bloco de statements, use o idioma do { ... } while(0).

Comportamento Indefinido (UB)

C padrão lista centenas de situações de "comportamento indefinido" — onde o compilador pode fazer literalmente qualquer coisa: crashar, silenciosamente retornar valores errados, ou parecer funcionar e depois falhar em produção. Conhecer as armadilhas é obrigatório.

UBs mais comuns

  • Overflow de inteiro signed: INT_MAX + 1 — UB. Use unsigned ou verifique antes.
  • Deref de ponteiro nulo: int *p = NULL; *p = 1; — segfault ou pior.
  • Array out-of-bounds: arr[10] quando arr tem 5 elementos — corrompe memória.
  • Use-after-free: acessar ponteiro após free() — dados podem parecer válidos até garbage collection sobrescrever.
  • Shift além do tamanho: 1 << 32 em int de 32 bits — UB.
  • Data race: dois threads acessando a mesma variável sem sincronização — C11 introduziu <stdatomic.h>.

Como evitar

  • Compile sempre com -Wall -Wextra -Wpedantic -fsanitize=address,undefined
  • Use Valgrind para detectar vazamentos e acessos inválidos
  • Prefira calloc a malloc — memória inicializada com zero
  • Use size_t para índices e tamanhos — nunca negativo por acidente
  • Leia o relatório de -fsanitize=address line-by-line
  • Ferramentas estáticas: cppcheck, clang-static-analyzer
C — detectando UB com AddressSanitizer
/* Compilar: gcc -fsanitize=address,undefined -g ub_demo.c */
#include <stdio.h>
#include <stdlib.h>

int main(void) {
    int *buf = malloc(5 * sizeof(int));
    for (int i = 0; i <= 5; i++)   /* BUG: i=5 é out-of-bounds! */
        buf[i] = i;
    free(buf);
    buf[0] = 99;               /* BUG: use-after-free! */
    return 0;
}
/* ASan emite:
   heap-buffer-overflow on address 0x...
   READ of size 4 at offset 20 past end of region */

C Moderno: C99 a C23

O padrão C evoluiu significativamente. Cada edição adicionou recursos que tornam o código mais seguro, legível e alinhado com hardware moderno.

Principais adições por padrão

PadrãoDestaques
C99VLA, inline, restrict, // comentários, _Bool, stdint.h, stdbool.h, declarações mistas com código, literais compostos
C11Threads (_Thread_local), _Atomic, _Generic, _Static_assert, inicializador de membro anônimo, fopen modo "x" (exclusivo)
C17Correções de defeitos e clarificações do C11. Sem novos recursos maiores.
C23nullptr, bool/true/false como palavras-chave, [[nodiscard]], [[fallthrough]], typeof, _BitInt(N), binary literals (0b1010), constexpr
C — recursos C11/C23
#include <stdio.h>
#include <stdbool.h>
#include <stdatomic.h>    /* C11 */

/* _Static_assert: falha na compilação, não em runtime */
_Static_assert(sizeof(int) >= 4, "int muito pequeno");

/* _Generic: sobrecarga de funções simulada */
#define ABS(x) _Generic((x),    \
    int:    abs(x),            \
    long:   labs(x),           \
    double: fabs(x),           \
    default: x)

/* Atomic: operações thread-safe sem mutex */
atomic_int contador = 0;

int main(void) {
    /* bool nativo (C99 via stdbool.h; C23 como palavra-chave) */
    bool ativo = true;
    printf("ativo = %s\n", ativo ? "sim" : "não");

    printf("ABS(-3.14) = %.2f\n", ABS(-3.14)); /* usa fabs */
    printf("ABS(-7) = %d\n",    ABS(-7));     /* usa abs  */

    atomic_fetch_add(&contador, 1);
    printf("contador = %d\n", atomic_load(&contador));

    /* C23: binary literal e nullptr */
    int flags = 0b10110011;
    int *p = nullptr;   /* equivale a NULL mas tipado */
    return 0;
}
