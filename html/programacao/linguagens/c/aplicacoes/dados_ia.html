C para Dados e IA

BLAS/LAPACK para álgebra linear, TensorFlow C API, ONNX Runtime, OpenCL para GPU, e implementação de redes neurais do zero em C puro.

NumPy, PyTorch, TensorFlow — todas as bibliotecas de ML do Python são wrappers sobre código C/C++/CUDA. O núcleo de desempenho de toda IA moderna é escrito em C. Entender C para dados e IA é entender o que acontece por baixo do model.fit().

BLAS e LAPACK: Álgebra Linear de Alto Desempenho

BLAS (Basic Linear Algebra Subprograms) e LAPACK (Linear Algebra PACKage) são os padrões industriais para operações de álgebra linear. NumPy usa OpenBLAS internamente. Chamá-los diretamente em C elimina qualquer overhead.

C — BLAS: multiplicação de matrizes com cblas_dgemm
#include <cblas.h>
#include <stdio.h>

/*
 * C = alpha * A * B + beta * C
 * cblas_dgemm: double-precision General Matrix Multiply
 *
 * A[m×k], B[k×n] → C[m×n]
 */
int main(void) {
    const int M = 3, N = 3, K = 3;

    double A[9] = {1, 2, 3,
                    4, 5, 6,
                    7, 8, 9};

    double B[9] = {9, 8, 7,
                    6, 5, 4,
                    3, 2, 1};
    double C[9] = {0};

    cblas_dgemm(
        CblasRowMajor, CblasNoTrans, CblasNoTrans,
        M, N, K,
        1.0,   /* alpha */
        A, K,   /* A, lda */
        B, N,   /* B, ldb */
        0.0,    /* beta */
        C, N    /* C, ldc */
    );

    printf("C = A × B:\n");
    for (int i = 0; i < M; i++) {
        for (int j = 0; j < N; j++) printf("%8.2f ", C[i*N+j]);
        printf("\n");
    }
    return 0;
}
/* gcc -o blas_ex blas_ex.c -lopenblas  (ou -lblas -llapack) */

Rede Neural do Zero em C

Implementar um perceptron multicamadas (MLP) em C força o entendimento real das matemáticas de backpropagation, gradiente descendente e funções de ativação — sem abstrações de frameworks.

C — MLP com forward pass, sigmoid e backpropagation (XOR)
#include <stdio.h>
#include <math.h>
#include <stdlib.h>
#include <time.h>

#define ENTRADA  2
#define OCULTA   4
#define SAIDA    1
#define LR       0.5
#define EPOCAS   10000

typedef struct {
    double w1[ENTRADA][OCULTA];  /* pesos camada 1 */
    double b1[OCULTA];           /* bias camada 1 */
    double w2[OCULTA][SAIDA];   /* pesos camada 2 */
    double b2[SAIDA];            /* bias camada 2 */
} MLP;

double sigmoid(double x) { return 1.0 / (1.0 + exp(-x)); }
double dsigmoid(double y) { return y * (1.0 - y); }

double rand01(void) { return ((double)rand() / RAND_MAX) * 2.0 - 1.0; }

void inicializar(MLP *nn) {
    for (int i = 0; i < ENTRADA; i++)
        for (int j = 0; j < OCULTA; j++)
            nn->w1[i][j] = rand01() * 0.5;
    for (int j = 0; j < OCULTA; j++) nn->b1[j] = 0.0;
    for (int j = 0; j < OCULTA; j++)
        for (int k = 0; k < SAIDA; k++)
            nn->w2[j][k] = rand01() * 0.5;
    for (int k = 0; k < SAIDA; k++) nn->b2[k] = 0.0;
}

/* Forward pass + Backprop numa função só (para clareza pedagógica) */
void treinar(MLP *nn, double *x, double *y) {
    /* Forward */
    double h[OCULTA], o[SAIDA];

    for (int j = 0; j < OCULTA; j++) {
        double s = nn->b1[j];
        for (int i = 0; i < ENTRADA; i++) s += x[i] * nn->w1[i][j];
        h[j] = sigmoid(s);
    }
    for (int k = 0; k < SAIDA; k++) {
        double s = nn->b2[k];
        for (int j = 0; j < OCULTA; j++) s += h[j] * nn->w2[j][k];
        o[k] = sigmoid(s);
    }

    /* Backward (gradiente no output) */
    double delta_o[SAIDA];
    for (int k = 0; k < SAIDA; k++)
        delta_o[k] = (y[k] - o[k]) * dsigmoid(o[k]);

    double delta_h[OCULTA];
    for (int j = 0; j < OCULTA; j++) {
        double e = 0.0;
        for (int k = 0; k < SAIDA; k++) e += delta_o[k] * nn->w2[j][k];
        delta_h[j] = e * dsigmoid(h[j]);
    }

    /* Atualার pesos */
    for (int j = 0; j < OCULTA; j++)
        for (int k = 0; k < SAIDA; k++)
            nn->w2[j][k] += LR * delta_o[k] * h[j];
    for (int k = 0; k < SAIDA; k++) nn->b2[k] += LR * delta_o[k];

    for (int i = 0; i < ENTRADA; i++)
        for (int j = 0; j < OCULTA; j++)
            nn->w1[i][j] += LR * delta_h[j] * x[i];
    for (int j = 0; j < OCULTA; j++) nn->b1[j] += LR * delta_h[j];
}

double predizer(MLP *nn, double *x) {
    double h[OCULTA];
    for (int j = 0; j < OCULTA; j++) {
        double s = nn->b1[j];
        for (int i = 0; i < ENTRADA; i++) s += x[i] * nn->w1[i][j];
        h[j] = sigmoid(s);
    }
    double s = nn->b2[0];
    for (int j = 0; j < OCULTA; j++) s += h[j] * nn->w2[j][0];
    return sigmoid(s);
}

int main(void) {
    srand((unsigned)time(NULL));
    MLP nn; inicializar(&nn);

    double X[4][2] = {{0,0},{0,1},{1,0},{1,1}};
    double Y[4]    = {0,1,1,0};  /* XOR */

    for (int ep = 0; ep < EPOCAS; ep++)
        for (int i = 0; i < 4; i++)
            treinar(&nn, X[i], &Y[i]);

    printf("XOR após %d épocas:\n", EPOCAS);
    for (int i = 0; i < 4; i++)
        printf("  (%.0f, %.0f) → %.4f\n",
               X[i][0], X[i][1], predizer(&nn, X[i]));
    return 0;
}
/* Saída esperada: ~0.05, ~0.95, ~0.95, ~0.05 */

TensorFlow C API e ONNX Runtime

TensorFlow C API

TensorFlow fornece uma API C estável (tensorflow/c/c_api.h) que permite carregar modelos SavedModel e executar inferências sem Python. Ideal para servidores de produção de baixo overhead.

// carregar modelo SavedModel
TF_Graph *graph = TF_NewGraph();
TF_Status *status = TF_NewStatus();
TF_SessionOptions *opts = TF_NewSessionOptions();
const char *tags[] = {"serve"};
TF_Session *sess = TF_LoadSessionFromSavedModel(
    opts, NULL, "./modelo",
    tags, 1, graph, NULL, status);

ONNX Runtime C API

ONNX Runtime tem uma API C de primeira classe (onnxruntime_c_api.h). Suporta execução em CPU, CUDA, TensorRT, OpenVINO e CoreML via providers. É o runtime de inferência mais portável para C.

// inicializar ONNX Runtime
const OrtApi *g_ort = OrtGetApiBase()->GetApi(16);
OrtEnv *env;
g_ort->CreateEnv(ORT_LOGGING_LEVEL_WARNING,
                  "onnx_c", &env);
OrtSession *session;
g_ort->CreateSession(env, L"modelo.onnx",
                       opts, &session);

OpenCL: Computação em GPU

OpenCL permite executar kernels no GPU em C. Frameworks como TensorFlow e PyTorch usam OpenCL ou CUDA internamente para acelerar operações matriciais.

/* kernel em C string */
const char *src =
  "__kernel void add("
  "  __global float* a,"
  "  __global float* b,"
  "  __global float* c)"
  "{ int i = get_global_id(0);"
  "  c[i] = a[i] + b[i]; }";

llama.cpp e whisper.cpp: IA de Ponta em C puro

C como fundação de IA moderna

llama.cpp (Georgi Gerganov) implementa inferência de LLMs como LLaMA, Mistral e Phi inteiramente em C/C++ puramente — sem Python, sem CUDA obrigatório, rodando em CPUs de consumidor com quantização 4-bit (GGML). É o projeto Open Source de IA de crescimento mais rápido em 2024.

whisper.cpp do mesmo autor porta o modelo Whisper da OpenAI para C puro, com suporte a Apple Silicon (Metal), CUDA, OpenCL e CPU puro. Transcrição local de áudio sem nuvem.

Ambos usam GGML — uma biblioteca de tensores escrita em C sem dependências externas — como prova de que C é suficiente para IA de produção.
