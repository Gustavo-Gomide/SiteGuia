C na Web

libmicrohttpd, libevent, CGI/FastCGI, nginx modules e os fundamentos de servidores HTTP de alta performance escritos em C.

nginx e Apache HTTPD — os dois servidores web mais usados do planeta — são escritos em C. Redis, Memcached (que alimenta Facebook/Twitter em escala) e o runtime do PHP também. Entender C para web é entender a camada de infraestrutura que suporta toda a internet.

libmicrohttpd: Servidor HTTP Embutido

libmicrohttpd é uma biblioteca GNU para criar servidores HTTP dentro de qualquer aplicação C. Usada para REST APIs em dispositivos embarcados, microserviços de baixíssima latência e serviços internos em sistemas distribuídos.

C — servidor HTTP com roteamento manual
#include <microhttpd.h>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <stdint.h>

#define PORTA 8080

static enum MHD_Result handle_request(
    void *cls,
    struct MHD_Connection *conn,
    const char *url,
    const char *method,
    const char *version,
    const char *upload_data,
    size_t      *upload_size,
    void       **con_cls)
{
    (void)cls; (void)version; (void)upload_data;
    (void)upload_size; (void)con_cls;

    const char *body;
    unsigned int status = MHD_HTTP_OK;

    if (strcmp(url, "/") == 0) {
        body = "{\"msg\":\"Bem-vindo ao servidor C!\"}";
    } else if (strcmp(url, "/status") == 0) {
        body = "{\"status\":\"ok\",\"uptime\":true}";
    } else {
        body   = "{\"erro\":\"rota nao encontrada\"}";
        status = MHD_HTTP_NOT_FOUND;
    }

    struct MHD_Response *resp = MHD_create_response_from_buffer(
        strlen(body), (void *)body, MHD_RESPMEM_PERSISTENT
    );
    MHD_add_response_header(resp, "Content-Type", "application/json");
    MHD_add_response_header(resp, "Access-Control-Allow-Origin", "*");

    enum MHD_Result ret = MHD_queue_response(conn, status, resp);
    MHD_destroy_response(resp);
    (void)method;
    return ret;
}

int main(void) {
    struct MHD_Daemon *daemon = MHD_start_daemon(
        MHD_USE_INTERNAL_POLLING_THREAD | MHD_USE_EPOLL,
        PORTA, NULL, NULL,
        &handle_request, NULL,
        MHD_OPTION_THREAD_POOL_SIZE, (unsigned int)4,
        MHD_OPTION_END
    );
    if (!daemon) { fprintf(stderr, "Erro ao iniciar daemon\n"); return 1; }
    printf("Servidor em http://localhost:%d\n", PORTA);
    getchar();  /* aguardar Enter para encerrar */
    MHD_stop_daemon(daemon);
    return 0;
}
/* gcc -o srv servidor.c $(pkg-config --cflags --libs libmicrohttpd) */

libevent: I/O Assíncrono Event-Driven

libevent usa o melhor mecanismo de polling disponível no SO (epoll no Linux, kqueue no macOS, IOCP no Windows) de forma transparente. É a base do Tor, Memcached e muitos proxies de alta performance.

C — servidor TCP assíncrono com libevent (evento de leitura)
#include <event2/event.h>
#include <event2/bufferevent.h>
#include <event2/listener.h>
#include <arpa/inet.h>
#include <stdio.h>
#include <string.h>

static void cb_read(struct bufferevent *bev, void *ctx) {
    (void)ctx;
    char buf[512];
    size_t n;
    while ((n = bufferevent_read(bev, buf, sizeof(buf) - 1)) > 0) {
        buf[n] = '\0';
        fprintf(stdout, "Recebido: %s", buf);
        /* Echo de volta */
        bufferevent_write(bev, buf, n);
    }
}

static void cb_accept(struct evconnlistener *lsn,
                       evutil_socket_t fd,
                       struct sockaddr *addr, int len, void *ctx) {
    (void)lsn; (void)addr; (void)len;
    struct event_base *base = ctx;
    struct bufferevent *bev = bufferevent_socket_new(
        base, fd, BEV_OPT_CLOSE_ON_FREE);
    bufferevent_setcb(bev, cb_read, NULL, NULL, NULL);
    bufferevent_enable(bev, EV_READ | EV_WRITE);
    printf("Nova conexão aceita\n");
}

int main(void) {
    struct event_base *base = event_base_new();

    struct sockaddr_in sin = {
        .sin_family = AF_INET,
        .sin_port   = htons(9000),
        .sin_addr   = {.s_addr = INADDR_ANY}
    };
    struct evconnlistener *lsn = evconnlistener_new_bind(
        base, cb_accept, base,
        LEV_OPT_CLOSE_ON_FREE | LEV_OPT_REUSEABLE,
        128,
        (struct sockaddr *)&sin, sizeof(sin)
    );

    printf("Echo server em :9000\n");
    event_base_dispatch(base);  /* loop de eventos */
    evconnlistener_free(lsn);
    event_base_free(base);
    return 0;
}

CGI e FastCGI em C

C — CGI puro (stdin/stdout como protocolo)
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

/* Decodificar URL encoding: %2B → + */
void url_decode(char *dst, const char *src, size_t max) {
    size_t i = 0;
    while (*src && i < max - 1) {
        if (*src == '%' && src[1] && src[2]) {
            char hex[3] = {src[1], src[2], '\0'};
            dst[i++] = (char) strtol(hex, NULL, 16);
            src += 3;
        } else if (*src == '+') {
            dst[i++] = ' '; src++;
        } else {
            dst[i++] = *src++;
        }
    }
    dst[i] = '\0';
}

int main(void) {
    /* CGI: variáveis de ambiente fornecem metadados */
    const char *method   = getenv("REQUEST_METHOD");
    const char *qs       = getenv("QUERY_STRING");
    const char *clen_str = getenv("CONTENT_LENGTH");

    printf("Content-Type: text/html\r\n\r\n");
    printf("<!DOCTYPE html><html><body>\n");
    printf("<h1>Ola do CGI em C!</h1>\n");

    if (method) printf("<p>Method: %s</p>\n", method);
    if (qs && *qs) {
        char decoded[512];
        url_decode(decoded, qs, sizeof(decoded));
        printf("<p>Query: %s</p>\n", decoded);
    }

    if (method && strcmp(method, "POST") == 0 && clen_str) {
        int clen = atoi(clen_str);
        if (clen > 0 && clen < 4096) {
            char body[4096];
            int lido = (int) fread(body, 1, (size_t)clen, stdin);
            body[lido] = '\0';
            printf("<p>Body: %s</p>\n", body);
        }
    }

    printf("</body></html>\n");
    return 0;
}

Arquitetura do nginx: Módulo em C

Como o nginx usa C

nginx define sua arquitetura inteiramente em C. Um módulo nginx é uma ngx_module_t — uma struct com ponteiros de função. O framework chama esses hooks durante o ciclo de vida da requisição: postconfiguration, create_conf, merge_conf e o handler propriamente dito.

C — esqueleto de módulo nginx (handler simples)
#include <ngx_config.h>
#include <ngx_core.h>
#include <ngx_http.h>

static ngx_int_t ngx_hello_handler(ngx_http_request_t *r) {
    static u_char resp[] = "Ola do modulo C nginx!";

    r->headers_out.status           = NGX_HTTP_OK;
    r->headers_out.content_length_n = sizeof(resp) - 1;
    ngx_http_send_header(r);

    ngx_buf_t *b = ngx_pcalloc(r->pool, sizeof(ngx_buf_t));
    b->pos  = resp; b->last = resp + sizeof(resp) - 1;
    b->in_memory = 1; b->last_buf = 1;

    ngx_chain_t out = {b, NULL};
    return ngx_http_output_filter(r, &out);
}

static ngx_http_module_t ngx_hello_module_ctx = {
    NULL, NULL, NULL, NULL, NULL, NULL,
    NULL, NULL
};

ngx_module_t ngx_hello_module = {
    NGX_MODULE_V1,
    &ngx_hello_module_ctx,
    NULL,           /* commands */
    NGX_HTTP_MODULE,
    NULL, NULL, NULL, NULL, NULL, NULL, NULL,
    NGX_MODULE_V1_PADDING
};
