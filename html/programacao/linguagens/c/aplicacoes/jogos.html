Doom, Quake, Wolfenstein 3D, o interpretador do chip DMG (Game Boy) e emuladores como mGBA foram escritos em C. A linguagem oferece controle total sobre memória, tempo de execução e hardware gráfico — fundamental para jogos de alta performance.

SDL2: Inicialização e Game Loop

SDL2 (Simple DirectMedia Layer) fornece janela, renderização 2D acelerada por hardware, input de teclado/mouse/joystick e áudio — tudo em C puro, cross-platform (Windows, Linux, macOS, Android, iOS).

C — SDL2: game loop com delta time
#include <SDL2/SDL.h>
#include <stdio.h>
#include <stdbool.h>

#define LARGURA  800
#define ALTURA   600
#define FPS_ALVO 60

int main(void) {
    if (SDL_Init(SDL_INIT_VIDEO | SDL_INIT_AUDIO) != 0) {
        fprintf(stderr, "SDL_Init erro: %s\n", SDL_GetError());
        return 1;
    }

    SDL_Window *janela = SDL_CreateWindow(
        "Meu Jogo C",
        SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED,
        LARGURA, ALTURA,
        SDL_WINDOW_SHOWN
    );
    SDL_Renderer *ren = SDL_CreateRenderer(
        janela, -1,
        SDL_RENDERER_ACCELERATED | SDL_RENDERER_PRESENTVSYNC
    );

    bool rodando = true;
    Uint32 ultimo = SDL_GetTicks();

    while (rodando) {
        Uint32 agora = SDL_GetTicks();
        float  dt    = (agora - ultimo) / 1000.0f;  /* delta time em segundos */
        ultimo = agora;

        /* --- Processar eventos --- */
        SDL_Event ev;
        while (SDL_PollEvent(&ev)) {
            if (ev.type == SDL_QUIT) rodando = false;
            if (ev.type == SDL_KEYDOWN &&
                ev.key.keysym.sym == SDLK_ESCAPE) rodando = false;
        }

        /* --- Atualizar lógica (usando dt) --- */
        /* ex: player.x += velocidade * dt; */

        /* --- Renderizar --- */
        SDL_SetRenderDrawColor(ren, 30, 30, 46, 255);
        SDL_RenderClear(ren);

        /* Exemplo: retângulo em movimento virtual */
        SDL_Rect retangulo = {100, 100, 50, 50};
        SDL_SetRenderDrawColor(ren, 137, 220, 235, 255);
        SDL_RenderFillRect(ren, &retangulo);

        SDL_RenderPresent(ren);

        /* Limitar FPS caso VSync não esteja disponível */
        Uint32 frame_ms = SDL_GetTicks() - agora;
        if (frame_ms < 1000 / FPS_ALVO)
            SDL_Delay(1000 / FPS_ALVO - frame_ms);
    }

    SDL_DestroyRenderer(ren);
    SDL_DestroyWindow(janela);
    SDL_Quit();
    return 0;
}
/* gcc -o jogo main.c $(sdl2-config --cflags --libs) */

Pong: Jogo Completo em SDL2 + C

Implementação completa do Pong com física de bola, detecção de colisão AABB (Axis-Aligned Bounding Box), pontuação e controle por teclado.

C — Pong completo: structs, física, colisão e placar
#include <SDL2/SDL.h>
#include <stdbool.h>
#include <stdio.h>
#include <math.h>

#define W 800
#define H 600
#define PAD_L 10
#define PAD_A 80
#define BOLA_R 8
#define VEL_PAD 350.0f
#define VEL_BOLA 280.0f

typedef struct { float x, y, vy; int pts; } Raquete;
typedef struct { float x, y, vx, vy; } Bola;

bool aabb(float bx, float by, float rx, float ry) {
    return bx + BOLA_R > rx && bx - BOLA_R < rx + PAD_L &&
           by + BOLA_R > ry && by - BOLA_R < ry + PAD_A;
}

void resetar_bola(Bola *b, int direcao) {
    b->x = W / 2.0f; b->y = H / 2.0f;
    b->vx = direcao * VEL_BOLA;
    b->vy = VEL_BOLA * 0.5f;
}

void desenhar_raquete(SDL_Renderer *r, Raquete *p) {
    SDL_Rect rect = {(int)p->x, (int)p->y, PAD_L, PAD_A};
    SDL_SetRenderDrawColor(r, 255, 255, 255, 255);
    SDL_RenderFillRect(r, &rect);
}

int main(void) {
    SDL_Init(SDL_INIT_VIDEO);
    SDL_Window   *win = SDL_CreateWindow("Pong", SDL_WINDOWPOS_CENTERED,
                           SDL_WINDOWPOS_CENTERED, W, H, 0);
    SDL_Renderer *ren = SDL_CreateRenderer(win, -1,
                           SDL_RENDERER_ACCELERATED | SDL_RENDERER_PRESENTVSYNC);

    Raquete p1 = {15.0f,              H/2.0f - PAD_A/2, 0, 0};
    Raquete p2 = {W - 15.0f - PAD_L,  H/2.0f - PAD_A/2, 0, 0};
    Bola    b;  resetar_bola(&b, 1);

    const Uint8 *kb = SDL_GetKeyboardState(NULL);
    bool rodando = true;
    Uint32 ultimo = SDL_GetTicks();

    while (rodando) {
        float dt = (SDL_GetTicks() - ultimo) / 1000.0f;
        ultimo = SDL_GetTicks();

        SDL_Event ev;
        while (SDL_PollEvent(&ev))
            if (ev.type == SDL_QUIT) rodando = false;

        /* Controles: W/S para P1, Up/Down para P2 */
        if (kb[SDL_SCANCODE_W])  p1.y -= VEL_PAD * dt;
        if (kb[SDL_SCANCODE_S])  p1.y += VEL_PAD * dt;
        if (kb[SDL_SCANCODE_UP])   p2.y -= VEL_PAD * dt;
        if (kb[SDL_SCANCODE_DOWN]) p2.y += VEL_PAD * dt;

        /* Clampar raquetes nas bordas */
        if (p1.y < 0) p1.y = 0;  if (p1.y + PAD_A > H) p1.y = H - PAD_A;
        if (p2.y < 0) p2.y = 0;  if (p2.y + PAD_A > H) p2.y = H - PAD_A;

        /* Mover bola */
        b.x += b.vx * dt;
        b.y += b.vy * dt;

        /* Colisão com paredes superior/inferior */
        if (b.y - BOLA_R < 0 || b.y + BOLA_R > H) b.vy = -b.vy;

        /* Colisão com raquetes */
        if (aabb(b.x, b.y, p1.x, p1.y) && b.vx < 0) b.vx = -b.vx * 1.05f;
        if (aabb(b.x, b.y, p2.x, p2.y) && b.vx > 0) b.vx = -b.vx * 1.05f;

        /* Ponto marcado */
        if (b.x < 0) { p2.pts++;  resetar_bola(&b, 1); }
        if (b.x > W) { p1.pts++;  resetar_bola(&b, -1); }

        /* Renderizar */
        SDL_SetRenderDrawColor(ren, 20, 20, 20, 255);
        SDL_RenderClear(ren);

        desenhar_raquete(ren, &p1);
        desenhar_raquete(ren, &p2);

        /* Bola */
        SDL_Rect bola = {(int)(b.x - BOLA_R), (int)(b.y - BOLA_R),
                          BOLA_R*2, BOLA_R*2};
        SDL_SetRenderDrawColor(ren, 255, 200, 0, 255);
        SDL_RenderFillRect(ren, &bola);

        SDL_RenderPresent(ren);
    }
    SDL_Quit();
    return 0;
}

OpenGL com GLFW: Triângulo em 3D

C — OpenGL 3.3 core + GLAD + GLFW
#include <glad/glad.h>
#include <GLFW/glfw3.h>
#include <stdio.h>

static const char *vert_src =
    "#version 330 core\n"
    "layout(location=0) in vec3 aPos;\n"
    "void main(){ gl_Position = vec4(aPos, 1.0); }\n";

static const char *frag_src =
    "#version 330 core\n"
    "out vec4 FragColor;\n"
    "void main(){ FragColor = vec4(0.3,0.7,1.0,1.0); }\n";

static float vertices[] = {
    -0.5f, -0.5f, 0.0f,
     0.5f, -0.5f, 0.0f,
     0.0f,  0.5f, 0.0f
};

int main(void) {
    glfwInit();
    glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 3);
    glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 3);
    glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);

    GLFWwindow *win = glfwCreateWindow(800, 600, "OpenGL C", NULL, NULL);
    glfwMakeContextCurrent(win);
    gladLoadGLLoader((GLADloadproc)glfwGetProcAddress);

    /* Compilar shaders */
    GLuint vs = glCreateShader(GL_VERTEX_SHADER);
    glShaderSource(vs, 1, &vert_src, NULL);
    glCompileShader(vs);

    GLuint fs = glCreateShader(GL_FRAGMENT_SHADER);
    glShaderSource(fs, 1, &frag_src, NULL);
    glCompileShader(fs);

    GLuint prog = glCreateProgram();
    glAttachShader(prog, vs); glAttachShader(prog, fs);
    glLinkProgram(prog);

    /* VAO + VBO */
    GLuint vao, vbo;
    glGenVertexArrays(1, &vao);  glBindVertexArray(vao);
    glGenBuffers(1, &vbo);       glBindBuffer(GL_ARRAY_BUFFER, vbo);
    glBufferData(GL_ARRAY_BUFFER, sizeof(vertices), vertices, GL_STATIC_DRAW);
    glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 12, (void*)0);
    glEnableVertexAttribArray(0);

    while (!glfwWindowShouldClose(win)) {
        glClearColor(0.1f, 0.1f, 0.15f, 1.0f);
        glClear(GL_COLOR_BUFFER_BIT);
        glUseProgram(prog);
        glBindVertexArray(vao);
        glDrawArrays(GL_TRIANGLES, 0, 3);
        glfwSwapBuffers(win);
        glfwPollEvents();
    }
    glfwTerminate();
    return 0;
}
