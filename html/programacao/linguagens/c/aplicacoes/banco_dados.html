C — Bancos de Dados

C e Bancos de Dados

SQLite embutido, libpq do PostgreSQL, hiredis para Redis e padrões avançados de acesso a dados em C.

Os próprios bancos de dados são escritos em C — PostgreSQL, SQLite, MySQL, Redis e MongoDB. Para aplicações que precisam persistência, C oferece APIs nativas de máximo desempenho, sem camadas de abstração desnecessárias.

SQLite — Banco Embutido

SQLite é uma biblioteca C que implementa um SGBD completo em um único arquivo. Zero configuração, zero servidor, zero dependências — apenas incluir sqlite3.h e linkar -lsqlite3.

C — SQLite: CRUD completo com prepared statements
#include <stdio.h>
#include <stdlib.h>
#include <sqlite3.h>

#define CHECK(rc, db) \
    if ((rc) != SQLITE_OK) { \
        fprintf(stderr, "SQLite: %s\n", sqlite3_errmsg(db)); \
        goto cleanup; }

int main(void) {
    sqlite3      *db   = NULL;
    sqlite3_stmt *stmt = NULL;
    int rc;

    /* Abrir (ou criar) banco */
    rc = sqlite3_open("escola.db", &db);
    CHECK(rc, db);

    /* Criar tabela */
    rc = sqlite3_exec(db,
        "CREATE TABLE IF NOT EXISTS alunos ("
        "  id    INTEGER PRIMARY KEY AUTOINCREMENT,"
        "  nome  TEXT    NOT NULL,"
        "  nota  REAL    NOT NULL"
        ,
        NULL, NULL, NULL);
    CHECK(rc, db);

    /* INSERT com prepared statement (previne SQL injection) */
    rc = sqlite3_prepare_v2(db,
        "INSERT INTO alunos (nome, nota) VALUES (?, ?);",
        -1, &stmt, NULL);
    CHECK(rc, db);

    struct { const char *nome; double nota; } dados[] = {
        {"Ana", 9.5}, {"Bruno", 7.8}, {"Carla", 8.2}
    };

    for (int i = 0; i < 3; i++) {
        sqlite3_bind_text(stmt, 1, dados[i].nome, -1, SQLITE_STATIC);
        sqlite3_bind_double(stmt, 2, dados[i].nota);
        sqlite3_step(stmt);
        sqlite3_reset(stmt);
    }
    sqlite3_finalize(stmt);

    /* SELECT */
    rc = sqlite3_prepare_v2(db,
        "SELECT id, nome, nota FROM alunos ORDER BY nota DESC;",
        -1, &stmt, NULL);
    CHECK(rc, db);

    printf("%-4s %-15s %s\n", "ID", "Nome", "Nota");
    printf("%-4s %-15s %s\n", "--", "----", "----");
    while (sqlite3_step(stmt) == SQLITE_ROW) {
        printf("%-4d %-15s %.1f\n",
            sqlite3_column_int   (stmt, 0),
            sqlite3_column_text  (stmt, 1),
            sqlite3_column_double(stmt, 2));
    }

cleanup:
    sqlite3_finalize(stmt);
    sqlite3_close(db);
    return rc != SQLITE_OK ? 1 : 0;
}
/* gcc -o escola escola.c -lsqlite3 */

PostgreSQL — libpq

libpq é a API C oficial do PostgreSQL. Suporta queries síncronas e assíncronas, prepared statements, COPY de dados em bulk e notificações LISTEN/NOTIFY.

C — libpq: conexão, query e parâmetros
#include <stdio.h>
#include <stdlib.h>
#include <libpq-fe.h>

int main(void) {
    PGconn   *conn   = NULL;
    PGresult *result = NULL;

    /* Conectar */
    conn = PQconnectdb(
        "host=localhost port=5432 dbname=escola user=postgres password=secret"
    );
    if (PQstatus(conn) != CONNECTION_OK) {
        fprintf(stderr, "Conexão falhou: %s\n", PQerrorMessage(conn));
        goto cleanup;
    }

    /* SELECT com parâmetro ($1 previne SQL injection) */
    const char *param = "7.0";
    result = PQexecParams(conn,
        "SELECT nome, nota FROM alunos WHERE nota > $1 ORDER BY nota DESC",
        1,           /* nparams */
        NULL,        /* paramTypes (NULL = inferir) */
        &param,      /* paramValues */
        NULL, NULL,  /* paramLengths, paramFormats */
        0            /* resultFormat: 0=text, 1=binary */
    );

    if (PQresultStatus(result) != PGRES_TUPLES_OK) {
        fprintf(stderr, "Query falhou: %s\n", PQerrorMessage(conn));
        goto cleanup;
    }

    int nrows = PQntuples(result);
    printf("%-20s %s\n", "Nome", "Nota");
    for (int i = 0; i < nrows; i++) {
        printf("%-20s %s\n",
            PQgetvalue(result, i, 0),
            PQgetvalue(result, i, 1));
    }

cleanup:
    PQclear(result);
    PQfinish(conn);
    return 0;
}
/* gcc -o pg_app pg_app.c -lpq */

Redis — hiredis

hiredis é o cliente C oficial do Redis. Mínimo, rápido e direto — envia comandos Redis e recebe respostas tipadas. Usado internamente pelo próprio Redis source.

C — hiredis: SET, GET, LPUSH, EXPIRE
#include <stdio.h>
#include <hiredis/hiredis.h>

int main(void) {
    redisContext *c = redisConnect("127.0.0.1", 6379);
    if (!c || c->err) {
        fprintf(stderr, "Erro: %s\n", c ? c->errstr : "OOM");
        return 1;
    }

    /* SET com expiração */
    redisReply *r = redisCommand(c, "SET usuario:1 'Ana' EX 3600");
    printf("SET: %s\n", r->str);  /* OK */
    freeReplyObject(r);

    /* GET */
    r = redisCommand(c, "GET usuario:1");
    printf("GET: %s\n", r->str);
    freeReplyObject(r);

    /* LPUSH list + LRANGE */
    redisCommand(c, "RPUSH log 'evento1' 'evento2' 'evento3'");
    r = redisCommand(c, "LRANGE log 0 -1");
    printf("Log (%lld entradas):\n", r->elements);
    for (size_t i = 0; i < r->elements; i++)
        printf("  %s\n", r->element[i]->str);
    freeReplyObject(r);

    redisFree(c);
    return 0;
}
/* gcc -o redis_app redis_app.c -lhiredis */
