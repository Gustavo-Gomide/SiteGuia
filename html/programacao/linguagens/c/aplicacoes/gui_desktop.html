C — GUI Desktop

C para GUI Desktop

GTK4, GObject, Cairo 2D, GIMP e as bases da programação orientada a eventos em C puro com interfaces gráficas nativas.

O GIMP, Inkscape, Audacity, GNOME Shell e centenas de aplicações Linux foram construídas em C com GTK. A combinação GLib + GTK + Cairo oferece todo o poder de GUI nativa sem abandonar o controle absoluto que C proporciona.

GTK4: Janela, Botão e GObject

GTK4 é baseado no sistema de objetos GObject — um sistema OOP em C puro que usa macros para herança, sinais e propriedades. Toda interface GTK é um grafo de GObjects conectados por sinais.

C — GTK4: aplicação mínima com botão e callback
#include <gtk/gtk.h>

static int contador = 0;

static void ao_clicar(GtkButton *btn, gpointer label) {
    contador++;
    char texto[64];
    snprintf(texto, sizeof(texto), "Cliques: %d", contador);
    gtk_label_set_text(GTK_LABEL(label), texto);
    (void)btn;
}

static void ativar(GtkApplication *app, gpointer dados) {
    (void)dados;

    GtkWidget *janela = gtk_application_window_new(app);
    gtk_window_set_title(GTK_WINDOW(janela), "Contador GTK4");
    gtk_window_set_default_size(GTK_WINDOW(janela), 300, 200);

    /* Layout vertical (GtkBox substitui GtkVBox do GTK3) */
    GtkWidget *caixa = gtk_box_new(GTK_ORIENTATION_VERTICAL, 12);
    gtk_widget_set_margin_top(caixa, 20);
    gtk_widget_set_margin_bottom(caixa, 20);
    gtk_widget_set_margin_start(caixa, 20);
    gtk_widget_set_margin_end(caixa, 20);
    gtk_window_set_child(GTK_WINDOW(janela), caixa);

    GtkWidget *label = gtk_label_new("Cliques: 0");
    gtk_box_append(GTK_BOX(caixa), label);

    GtkWidget *botao = gtk_button_new_with_label("Clique aqui!");
    gtk_box_append(GTK_BOX(caixa), botao);

    /* Conectar sinal "clicked" ao callback */
    g_signal_connect(botao, "clicked", G_CALLBACK(ao_clicar), label);

    gtk_widget_show(janela);
}

int main(int argc, char *argv[]) {
    GtkApplication *app = gtk_application_new(
        "br.dev.exemplo.contador", G_APPLICATION_DEFAULT_FLAGS
    );
    g_signal_connect(app, "activate", G_CALLBACK(ativar), NULL);
    int status = g_application_run(G_APPLICATION(app), argc, argv);
    g_object_unref(app);
    return status;
}
/* gcc -o app app.c $(pkg-config --cflags --libs gtk4) */

Cairo: Desenho 2D em C

Cairo é uma biblioteca de gráficos vetoriais 2D com backends para X11, Win32, Quartz (macOS), PDF, SVG e PNG. O GTK usa Cairo internamente para renderizar todos os widgets.

C — Cairo: superfície PNG com bezier e gradiente
#include <cairo/cairo.h>
#include <math.h>

int main(void) {
    cairo_surface_t *surf = cairo_image_surface_create(
        CAIRO_FORMAT_ARGB32, 400, 300);
    cairo_t *cr = cairo_create(surf);

    /* Fundo gradiente */
    cairo_pattern_t *grad = cairo_pattern_create_linear(0, 0, 400, 300);
    cairo_pattern_add_color_stop_rgb(grad, 0,   0.06, 0.06, 0.15);
    cairo_pattern_add_color_stop_rgb(grad, 1.0, 0.15, 0.05, 0.30);
    cairo_set_source(cr, grad);
    cairo_paint(cr);
    cairo_pattern_destroy(grad);

    /* Círculo com borda */
    cairo_set_source_rgba(cr, 0.54, 0.86, 0.92, 0.9);
    cairo_arc(cr, 200, 150, 80, 0, 2 * M_PI);
    cairo_fill_preserve(cr);
    cairo_set_source_rgb(cr, 1, 1, 1);
    cairo_set_line_width(cr, 2);
    cairo_stroke(cr);

    /* Texto */
    cairo_set_source_rgb(cr, 1, 1, 1);
    cairo_select_font_face(cr, "Sans", CAIRO_FONT_SLANT_NORMAL,
                            CAIRO_FONT_WEIGHT_BOLD);
    cairo_set_font_size(cr, 20);
    cairo_move_to(cr, 130, 160);
    cairo_show_text(cr, "Cairo em C");

    cairo_surface_write_to_png(surf, "saida.png");
    cairo_destroy(cr);
    cairo_surface_destroy(surf);
    return 0;
}
/* gcc -o cairo_ex cairo_ex.c $(pkg-config --cflags --libs cairo) -lm */

GTK4 + Cairo: Widget de Desenho Personalizado

C — GtkDrawingArea com função de draw via Cairo
static void draw_func(GtkDrawingArea *da, cairo_t *cr,
                       int w, int h, gpointer dados) {
    (void)da; (void)dados;

    /* Grade de fundo */
    cairo_set_source_rgb(cr, 0.15, 0.15, 0.18);
    cairo_paint(cr);

    cairo_set_source_rgba(cr, 0.3, 0.3, 0.3, 0.5);
    cairo_set_line_width(cr, 0.5);
    for (int x = 0; x <= w; x += 20) {
        cairo_move_to(cr, x, 0);
        cairo_line_to(cr, x, h);
    }
    for (int y = 0; y <= h; y += 20) {
        cairo_move_to(cr, 0, y);
        cairo_line_to(cr, w, y);
    }
    cairo_stroke(cr);

    /* Curva de Bézier cúbica */
    cairo_set_source_rgb(cr, 0.54, 0.86, 0.92);
    cairo_set_line_width(cr, 2.5);
    cairo_move_to(cr, 30,   h/2);
    cairo_curve_to(cr, w/4,  20,
                       w*3/4, h - 20,
                       w - 30, h/2);
    cairo_stroke(cr);
}

static void ativar(GtkApplication *app, gpointer dados) {
    (void)dados;
    GtkWidget *win = gtk_application_window_new(app);
    gtk_window_set_default_size(GTK_WINDOW(win), 600, 400);

    GtkWidget *area = gtk_drawing_area_new();
    gtk_drawing_area_set_draw_func(GTK_DRAWING_AREA(area),
                                    draw_func, NULL, NULL);
    gtk_window_set_child(GTK_WINDOW(win), area);
    gtk_widget_show(win);
}
