C para Terminal e CLI

Parsing de argumentos, ncurses para TUI, pipes, sinais e os padrões de código das ferramentas Unix mais famosas do mundo.

ls, grep, awk, sed, cat, find, git, gcc — todas essas ferramentas são escritas em C. Criar ferramentas de linha de comando eficientes, composáveis e Unix-philosophic é onde C brilha de forma única.

Parsing de Argumentos: getopt e getopt_long

C — getopt_long: flags -v, --verbose, --output=FILE
#include <stdio.h>
#include <stdlib.h>
#include <getopt.h>
#include <string.h>

static int flag_verbose = 0;
static int flag_help    = 0;
static const char *output_file = NULL;

void uso(const char *prog) {
    fprintf(stderr,
        "Uso: %s [OPÇÕES] [ARQUIVO...]\n"
        "  -v, --verbose       saída detalhada\n"
        "  -o, --output=FILE   arquivo de saída\n"
        "  -h, --help          esta mensagem\n",
        prog);
}

int main(int argc, char *argv[]) {
    static const struct option opts_long[] = {
        {"verbose", no_argument,       &flag_verbose, 1},
        {"output",  required_argument, NULL,          'o'},
        {"help",    no_argument,       &flag_help,    1},
        {NULL,      0,                 NULL,          0}
    };

    int c, idx;
    while ((c = getopt_long(argc, argv, "vo:h", opts_long, &idx)) != -1) {
        switch (c) {
            case 'v': flag_verbose = 1; break;
            case 'o': output_file = optarg; break;
            case 'h': flag_help = 1; break;
            case  0: break; /* flag foi setado diretamente */
            default: uso(argv[0]); return 1;
        }
    }

    if (flag_help) { uso(argv[0]); return 0; }
    if (flag_verbose) printf("Modo verbose ativado\n");
    if (output_file)  printf("Saída: %s\n", output_file);

    /* Processar arquivos restantes (optind até argc) */
    for (int i = optind; i < argc; i++)
        printf("Processando: %s\n", argv[i]);

    return 0;
}

Pipes, Fork e Sinais

C — pipe entre processo pai e filho
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <sys/wait.h>
#include <signal.h>

static volatile int terminando = 0;

void handle_sigint(int sig) {
    (void)sig;
    terminando = 1;
    write(STDOUT_FILENO, "\nRecebido SIGINT — encerrando...\n", 33);
}

int main(void) {
    /* Registrar handler de SIGINT (Ctrl+C) */
    struct sigaction sa = {.sa_handler = handle_sigint};
    sigaction(SIGINT, &sa, NULL);

    int pipefd[2];  /* [0]=leitura, [1]=escrita */
    if (pipe(pipefd) == -1) { perror("pipe"); return 1; }

    pid_t pid = fork();
    if (pid == -1) { perror("fork"); return 1; }

    if (pid == 0) {
        /* Processo filho: lê do pipe */
        close(pipefd[1]);
        char buf[128];
        ssize_t n;
        while ((n = read(pipefd[0], buf, sizeof(buf) - 1)) > 0) {
            buf[n] = '\0';
            printf("Filho recebeu: %s\n", buf);
        }
        close(pipefd[0]);
        exit(0);

    } else {
        /* Processo pai: escreve no pipe */
        close(pipefd[0]);
        const char *msgs[] = {"Olá", "do", "processo pai!"};
        for (int i = 0; i < 3; i++) {
            write(pipefd[1], msgs[i], strlen(msgs[i]));
            sleep(1);
        }
        close(pipefd[1]);
        wait(NULL);  /* esperar filho terminar */
    }
    return 0;
}

Interface TUI com ncurses

ncurses permite criar interfaces de texto ricas no terminal: janelas, cores, teclado avançado e mouse. Usada no vim, mc, htop, midnight commander e bash readline.

C — ncurses: janela colorida com input
#include <ncurses.h>
#include <string.h>

int main(void) {
    initscr();           /* inicializar ncurses */
    cbreak();            /* sem buffer de linha */
    noecho();            /* não ecoar teclas */
    keypad(stdscr, TRUE); /* habilitar teclas especiais */
    curs_set(0);          /* esconder cursor */

    if (has_colors()) {
        start_color();
        init_pair(1, COLOR_GREEN,  COLOR_BLACK);
        init_pair(2, COLOR_YELLOW, COLOR_BLUE);
        init_pair(3, COLOR_WHITE,  COLOR_RED);
    }

    int rows, cols;
    getmaxyx(stdscr, rows, cols);

    /* Janela central */
    WINDOW *win = newwin(10, 40,
                          (rows - 10) / 2,
                          (cols - 40) / 2);
    wbkgd(win, COLOR_PAIR(2));
    box(win, 0, 0);
    wattron(win, A_BOLD | COLOR_PAIR(2));
    mvwprintw(win, 1, 2, "Pressione qualquer tecla...");
    wattroff(win, A_BOLD);
    wrefresh(win);

    mvwprintw(win, 5, 2, "Tecla: ");
    int ch = wgetch(win);
    mvwprintw(win, 5, 9, "%d ('%c')", ch, (ch > 31 && ch < 127) ? ch : '?');
    wrefresh(win);
    getch();

    delwin(win);
    endwin();  /* restaurar terminal */
    return 0;
}
