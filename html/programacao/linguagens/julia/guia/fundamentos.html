Tipos Primitivos e Compostos

Sistema de tipos de Julia
# Tipos numéricos
i::Int64    = 42
f::Float64  = 3.14
b::Bool     = true
c::Char     = 'A'
s::String   = "Julia"

# Inteiros com tamanhos explícitos
Int8, Int16, Int32, Int64, Int128
UInt8, UInt16, UInt32, UInt64, UInt128

# Tipos especiais matemáticos
r = 3//4               # Rational{Int64} — 3/4 exato
z = 1 + 2im            # Complex{Int64} — número complexo
big = BigInt(10)^100   # inteiro arbitrário
bigf = BigFloat(pi)    # ponto flutuante arbitrário

# Conversão de tipos
parse(Int, "42")       # String -> Int
parse(Float64, "3.14")
string(42)             # Int -> String
Float64(42)            # Int -> Float64
convert(Float64, 1)    # int -> float explícito

# Verificar tipo
typeof(42)             # Int64
isa(42, Integer)       # true
42 isa Number          # true (sintaxe alternativa)

# Hierarquia de tipos numéricos
# Number -> Real -> {AbstractFloat, Integer}
# AbstractFloat -> {Float16, Float32, Float64}
# Integer -> {Signed, Unsigned}

# structs — tipos compostos
struct Ponto
    x::Float64
    y::Float64
end

# mutable struct — mutável
mutable struct Retangulo
    largura::Float64
    altura::Float64
end

p = Ponto(3.0, 4.0)
println(p.x, ", ", p.y)

r = Retangulo(10.0, 5.0)
r.largura = 20.0        # ok — mutável

# Construtor interno personalizado
struct Temperatura
    celsius::Float64
    function Temperatura(c)
        c < -273.15 && error("Abaixo do zero absoluto!")
        new(c)
    end
end

Funções e Multiple Dispatch

Funções, dispatch e composição
# Função normal
function area(r::Float64)
    return π * r^2
end

# Sintaxe inline
area(r::Float64) = π * r^2

# Multiple dispatch — mesmo nome, tipos diferentes
function descrever(x::Int)
    "Inteiro: $x"
end
function descrever(x::Float64)
    "Float: $x"
end
function descrever(x::String)
    "Texto: \"$x\""
end
function descrever(x::Any)
    "Desconhecido: $(typeof(x))"
end

println(descrever(42))         # Inteiro: 42
println(descrever(3.14))       # Float: 3.14
println(descrever("oi"))       # Texto: "oi"

# Múltiplos métodos com múltiplos argumentos
function combinar(x::Int, y::Int)    = x + y
function combinar(x::String, y::String) = x * y
function combinar(x::Number, y::Number) = x * y

# Listar métodos de uma função
methods(descrever)

# Parâmetros padrão e variádicos
function cumprimentar(nome::String, saudacao::String = "Olá")
    "$saudacao, $nome!"
end

function somar(nums...)
    sum(nums)
end

# Argumentos keyword
function criar_janela(; largura::Int = 800, altura::Int = 600, titulo::String = "Janela")
    (largura, altura, titulo)
end
criar_janela(altura = 400, titulo = "Editor")

# Funções anônimas
quadrado = x -> x^2
soma = (a, b) -> a + b

# Composição de funções (operador ∘)
dobrar_depois_somar = x -> dobrar(x) + 1
composta = sqrt ∘ abs ∘ dobrar    # sqrt(abs(dobrar(x)))

# Pipe |>
resultado = -4 |> abs |> sqrt      # sqrt(abs(-4)) = 2.0

# Funções que alteram seus argumentos: convenção ! no nome
function dobrar!(v::Vector)
    for i in eachindex(v)
        v[i] *= 2
    end
end

Coleções: Array, Dict, Set e Tuple

Estruturas de dados nativas
## ARRAYS (1-indexados!)
v = [1, 2, 3, 4, 5]     # Vector{Int64}
m = [1 2 3; 4 5 6]      # Matrix{Int64} 2×3
v3 = Vector{Float64}(undef, 5)   # não inicializado
zeros(3, 3)              # matriz de zeros
ones(2, 4)               # matriz de uns
rand(3, 3)               # uniforme [0,1]
randn(3, 3)              # normal padrão

# Acesso (1-indexado!)
v[1]                     # primeiro elemento
v[end]                   # último elemento
v[2:4]                   # slice v[2], v[3], v[4]
v[end-1:end]             # dois últimos
m[1, :]                  # primeira linha
m[:, 2]                  # segunda coluna

# Operações
push!(v, 6)              # adicionar ao fim
pop!(v)                  # remover e retornar último
pushfirst!(v, 0)         # adicionar ao início
popfirst!(v)             # remover primeiro
append!(v, [7, 8])       # concatenar in-place
deleteat!(v, 2)          # remover índice 2
sort(v)                  # novo array ordenado
sort!(v)                 # ordenar in-place
length(v)                # tamanho
size(m)                  # dimensões (linhas, colunas)

## BROADCASTING — aplica operação elemento a elemento
v .* 2                   # multiplica cada elemento por 2
v .+ [10, 20, 30, 40, 50]     # soma elemento a elemento
sin.(v)                  # sen de cada elemento
2 .^ v                   # 2^cada_elemento
@. v^2 + 2*v + 1         # @. faz broadcast em toda expressão

## DICT
d = Dict("a" => 1, "b" => 2, "c" => 3)
d["a"]                   # 1
d["d"] = 4               # adicionar
get(d, "z", 0)           # 0 (default)
haskey(d, "a")           # true
delete!(d, "b")
keys(d), values(d), pairs(d)

## SET
s = Set([1, 2, 3, 2, 1])  # {1, 2, 3}
push!(s, 4)
4 in s                    # true
union(s, Set([5, 6]))
intersect(s, Set([2, 3, 5]))
setdiff(s, Set([1, 2]))

## TUPLE — imutável
t = (1, "Julia", 3.14)
t[1]                      # 1
a, b, c = t               # desestruturação

# NamedTuple
ponto = (x = 1.0, y = 2.0, z = 0.0)
ponto.x                   # 1.0
ponto[:y]                 # 2.0

Controle de Fluxo e Compreensões

Loops, condicionais, compreensões e iteradores
# if/elseif/else
function classificar(x::Int)
    if x < 0
        "negativo"
    elseif x == 0
        "zero"
    elseif x < 10
        "pequeno"
    else
        "grande"
    end
end

# Operador ternário
sinal = x >= 0 ? "positivo" : "negativo"

# for com range
for i in 1:5
    println(i)
end

# for com step
for i in 0:2:10       # 0, 2, 4, 6, 8, 10
    print("$i ")
end

# for com coleções
for (k, v) in Dict("a" => 1, "b" => 2)
    println("$k => $v")
end

# enumerate
frutas = ["maçã", "banana", "cereja"]
for (i, fruta) in enumerate(frutas)
    println("$i: $fruta")
end

# zip
for (a, b) in zip([1,2,3], ["a","b","c"])
    println("$a-$b")
end

# while
n = 1
while n <= 5
    print("$n ")
    n += 1
end

# break e continue
for i in 1:10
    i % 2 == 0 && continue    # pular pares
    i > 7 && break             # parar em 7
    print("$i ")
end

## COMPREENSÕES
quadrados = [x^2 for x in 1:10]
pares = [x for x in 1:20 if x % 2 == 0]
tabela = [(i, j) for i in 1:3, j in 1:3]  # 2D

# Generator — sem alocar array
soma_quadrados = sum(x^2 for x in 1:1000)

# Dict comprehension
quadrado_dict = Dict(x => x^2 for x in 1:5)

# map/filter/reduce
map(x -> x^2, 1:5)            # [1, 4, 9, 16, 25]
filter(iseven, 1:10)           # [2, 4, 6, 8, 10]
reduce(+, 1:100)               # 5050
foldl((acc, x) -> acc + x, 1:5, init=0)

Tratamento de Erros e I/O

try/catch, throw e operações de arquivo
# try/catch/finally
function dividir(a, b)
    try
        a / b
    catch e
        if isa(e, DivideError)
            println("Divisão por zero!")
            0
        else
            rethrow(e)
        end
    end
end

# Lançar exceções
function verificar_idade(idade::Int)
    idade < 0 && throw(ArgumentError("Idade negativa: $idade"))
    idade > 150 && throw(DomainError(idade, "Idade impossível"))
    idade
end

# finally sempre executa
try
    resultado = parse(Int, "abc")
catch e
    println("Erro: $(e.msg)")
finally
    println("Bloco finally sempre executa")
end

# Tipos de exceção padrão

# Exceção personalizada
struct ValidacaoError <: Exception
    campo::String
    mensagem::String
end
Base.showerror(io::IO, e::ValidacaoError) =
    print(io, "ValidacaoError no campo '$(e.campo)': $(e.mensagem)")

O broadcasting com . é a estrela da sintaxe Julia: sin.(v) aplica sin a cada elemento de v gerando código SIMD otimizado. E o @. transforma toda uma expressão em broadcasting: @. v^2 + 2v + 1 compila para um único loop sem alocações intermediárias.
