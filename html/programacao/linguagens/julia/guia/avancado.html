Julia — Tópicos Avançados

Metaprogramação com macros, tipos paramétricos, geração de código, concorrência com tasks e threads, análise de inferência de tipos e técnicas de performance que aproximam Julia de C e Fortran.

O diferencial de Julia em nível avançado é que a linguagem expõe quase tudo: AST, macros, inferência, código LLVM e assembly. Isso permite escrever software altamente expressivo e, ao mesmo tempo, entender exatamente por que uma função está rápida ou lenta.

Tipos Paramétricos e Hierarquia Abstrata

Modelagem genérica e especializada por tipo
# Tipos abstratos criam hierarquias sem instâncias diretas
abstract type Forma end
abstract type Forma2D <: Forma end
abstract type Forma3D <: Forma end

# Tipos concretos paramétricos
struct Circulo{T<:Real} <: Forma2D
	raio::T
end

struct Retangulo{T<:Real} <: Forma2D
	largura::T
	altura::T
end

struct Esfera{T<:Real} <: Forma3D
	raio::T
end

area(c::Circulo) = π * c.raio^2
area(r::Retangulo) = r.largura * r.altura
volume(e::Esfera) = (4 / 3) * π * e.raio^3

function descrever(f::Forma2D)
	"$(typeof(f)) com área $(round(area(f), digits = 2))"
end

# Tipos com múltiplos parâmetros
struct Par{A, B}
	primeiro::A
	segundo::B
end

Base.first(p::Par) = p.primeiro
Base.last(p::Par) = p.segundo

# where restringe e nomeia parâmetros
function soma_convertida(v::Vector{T}) where {T<:Real}
	total = zero(T)
	for x in v
		total += x
	end
	total
end

# Union para aceitar conjunto fechado de tipos
function processar(x::Union{Int, Float64, String})
	println("Recebido: $x")
end

# Observação importante: containers são invariantes
# Vector{Float64} não é subtipo de Vector{Real}
Vector{Float64} <: Vector{Real}         # false
Vector{Float64} <: AbstractVector{Float64}  # true

Metaprogramação e Macros

Código que gera código
# Expressões: código como estrutura de dados
expr = :(1 + 2 * 3)
dump(expr)
eval(expr)                     # 7

# quote cria blocos de AST
codigo = quote
	x = 10
	y = 20
	x + y
end

eval(codigo)                   # 30
Meta.parse("a = b + c")

# Macro simples
macro verificar(expr)
	quote
		local resultado = $(esc(expr))
		if resultado
			println("OK: ", $(string(expr)))
		else
			println("Falhou: ", $(string(expr)))
		end
		resultado
	end
end

@verificar 2 + 2 == 4

# Macro para medição manual
macro medir(expr)
	quote
		local inicio = time()
		local valor = $(esc(expr))
		local fim = time()
		println("Tempo: ", round(fim - inicio, digits = 6), " s")
		valor
	end
end

# @generated escolhe código conforme o tipo
@generated function construir_zeros(::Type{T}, n::Int) where {T}
	if T <: Integer
		:(fill(zero(T), n))
	else
		:(zeros(T, n))
	end
end

# Macros padrão úteis
@time sum(rand(10^6))
@elapsed sum(rand(10^6))
@allocated sum(rand(10^6))
@show sqrt(81)
@assert 1 + 1 == 2
@info "Executando pipeline"
@warn "Valor aproximado"

# Interpolação com $ insere valores/expressões no AST
nome = :ola
corpo = :(println("gerado dinamicamente"))
macro criar_funcao(nome, corpo)
	quote
		function $(esc(nome))()
			$(esc(corpo))
		end
	end
end

@criar_funcao ola corpo

Concorrência e Paralelismo

Tasks, Channels, Threads e Distributed
using Base.Threads
using Distributed

# Tasks cooperativas
function produtor(c::Channel)
	for i in 1:5
		put!(c, i)
		sleep(0.1)
	end
	close(c)
end

function consumidor(c::Channel)
	for valor in c
		println("Recebido: $valor")
	end
end

canal = Channel(10)
@sync begin
	@async produtor(canal)
	@async consumidor(canal)
end

# fetch aguarda o resultado de uma task
tarefa = @async begin
	sleep(1)
	42
end
resultado = fetch(tarefa)

# Threads para CPU-bound
dados = rand(1_000)
saida = zeros(length(dados))

@threads for i in eachindex(dados)
	saida[i] = sqrt(dados[i])
end

# Seção crítica com lock
lock_global = ReentrantLock()
contador = Ref(0)

@threads for _ in 1:1_000
	lock(lock_global) do
		contador[] += 1
	end
end

# Atomics para contadores simples
contador_atomic = Atomic{Int}(0)
@threads for _ in 1:1_000
	atomic_add!(contador_atomic, 1)
end

# Múltiplos processos
addprocs(4)

quadrados = @distributed (+) for i in 1:1_000
	i^2
end

raizes = pmap(sqrt, 1:20)

Performance e Ferramentas de Diagnóstico

Inferência de tipos e otimizações práticas
using BenchmarkTools
using StaticArrays

# Inspecionar inferência de tipos
function soma_vetor(v)
	s = 0.0
	for x in v
		s += x
	end
	s
end

@code_warntype soma_vetor(rand(10))
@code_llvm soma_vetor(rand(10))
@code_native soma_vetor(rand(10))

# Instabilidade de tipo prejudica desempenho
function instavel(x)
	if x > 0
		x
	else
		"negativo"
	end
end

function estavel(x::Int)
	x > 0 ? x : 0
end

# Globais devem ser const ou tipadas
const TAXA = 0.15
global limite::Int = 100

# Remover checagem de bounds
function soma_inbounds(v::Vector{Float64})
	total = 0.0
	@inbounds for i in eachindex(v)
		total += v[i]
	end
	total
end

# SIMD
function soma_simd(v::Vector{Float64})
	total = 0.0
	@inbounds @simd for i in eachindex(v)
		total += v[i]
	end
	total
end

# Pré-alocação evita garbage collector excessivo
function sqrt_buffer!(destino::Vector{Float64}, origem::Vector{Float64})
	@inbounds for i in eachindex(origem)
		destino[i] = sqrt(origem[i])
	end
	destino
end

buffer = Vector{Float64}(undef, 1_000)
origem = rand(1_000)
sqrt_buffer!(buffer, origem)

# Benchmark confiável
@benchmark soma_simd($origem)

# StaticArrays brilham para vetores pequenos e fixos
v3 = @SVector [1.0, 2.0, 3.0]
m2 = @SMatrix [1.0 0.0; 0.0 1.0]
m2 * v3[1:2]

# Dicas essenciais:
# 1. Coloque lógica quente dentro de funções.
# 2. Evite Any em estruturas críticas.
# 3. Prefira operações vetorizadas e broadcasting quando fizer sentido.
# 4. Meça com BenchmarkTools antes de otimizar.
