Organização Modular e Estilo

Separando API pública, implementação e scripts
# Boas práticas de módulos
module ProjetoJulia

export Cliente, Pedido, totalizar

include("modelos.jl")
include("negocio.jl")
include("formatacao.jl")

end

# modelos.jl
struct Cliente
		id::Int
		nome::String
end

struct Pedido
		id::Int
		itens::Vector{Float64}
end

# negocio.jl
function totalizar(pedido::Pedido)
		sum(pedido.itens)
end

# scripts devem ser finos: configuram e chamam funções
using ProjetoJulia

function main()
		pedido = Pedido(1, [10.0, 20.0, 5.0])
		println(totalizar(pedido))
end

abspath(PROGRAM_FILE) == @__FILE__ && main()

# Convenções úteis:
# 1. nomes_em_minusculo para funções e variáveis
# 2. CamelCase para tipos e módulos
# 3. funções mutáveis terminam com !
# 4. código quente sempre dentro de funções

Performance, Testes e Deploy

Fluxo mínimo de qualidade
using Test
using BenchmarkTools

@testset "regressão de negócio" begin
		@test totalizar(Pedido(1, [10.0, 5.0])) == 15.0
end

function processar!(destino, origem)
		@inbounds for i in eachindex(origem)
				destino[i] = origem[i] * 2
		end
		destino
end

origem = rand(10^5)
destino = similar(origem)

@benchmark processar!($destino, $origem)

# Deploy: PackageCompiler pode gerar sysimage
using PackageCompiler
create_sysimage([:DataFrames, :CSV]; sysimage_path = "projeto.so")

# Também é possível criar app nativo
create_app(".", "build/ProjetoJuliaApp")

Faça

  • Mantenha lógica de domínio em funções puras quando possível
  • Use testes automatizados desde cedo
  • Meça desempenho com cenários realistas
  • Considere sysimage para reduzir startup em produção

Evite

  • Grandes scripts monolíticos no REPL sem módulo
  • Estado global mutável espalhado
  • Otimizar antes de medir
  • Misturar notebooks exploratórios com API pública
