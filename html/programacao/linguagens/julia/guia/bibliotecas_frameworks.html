Dados, Estatística e Tabelas

DataFrames.jl, CSV.jl e Statistics
using CSV
using DataFrames
using Statistics

df = CSV.read("vendas.csv", DataFrame)
first(df, 5)
describe(df)

# Filtrar e transformar
df_filtrado = filter(:valor => >(100), df)
transform!(df, :valor => ByRow(x -> x * 1.1) => :valor_corrigido)

# Agrupar e agregar
resumo = combine(groupby(df, :categoria),
		:valor => mean => :media,
		:valor => sum => :total,
		nrow => :quantidade)

# Tabelas interoperáveis via Tables.jl
Tables.columntable(df)
CSV.write("saida.csv", resumo)

Machine Learning e Otimização

Flux.jl, MLJ.jl e JuMP.jl
using Flux

modelo = Chain(
		Dense(10 => 32, relu),
		Dense(32 => 1)
)

x = rand(Float32, 10, 128)
y = rand(Float32, 1, 128)

loss(m, x, y) = Flux.mse(m(x), y)
opt = Flux.setup(Adam(0.001), modelo)

for _ in 1:100
		grads = Flux.gradient(modelo) do m
				loss(m, x, y)
		end
		Flux.update!(opt, modelo, grads[1])
end

# MLJ: interface unificada para modelos
using MLJ
X, y = @load_iris
Tree = @load DecisionTreeClassifier pkg = DecisionTree
mach = machine(Tree(), X, y)
fit!(mach)

# JuMP para otimização matemática
using JuMP, HiGHS

model = Model(HiGHS.Optimizer)
@variable(model, x >= 0)
@variable(model, y >= 0)
@objective(model, Max, 5x + 3y)
@constraint(model, 2x + y <= 10)
@constraint(model, x + 3y <= 12)
optimize!(model)
value(x), value(y), objective_value(model)

Gráficos, Web e Simulação

Makie, HTTP.jl, Genie e DifferentialEquations
using CairoMakie

xs = 0:0.1:10
ys = sin.(xs)

fig = Figure()
ax = Axis(fig[1, 1], title = "Seno")
lines!(ax, xs, ys, color = :blue)
save("seno.png", fig)

# API simples com HTTP.jl
using HTTP, JSON3

HTTP.serve() do req
		if req.target == "/status"
				return HTTP.Response(200, JSON3.write((ok = true, tempo = time())))
		end
		HTTP.Response(404, "não encontrado")
end

# Framework web Genie.jl
using GenieFramework

@app begin
		@in nome = "Julia"
		@out saudacao = "Olá"
		@onbutton clicar begin
				saudacao = "Olá, $(nome)!"
		end
end

# Simulação científica
using DifferentialEquations

function f!(du, u, p, t)
		du[1] = 1.01u[1]
end

prob = ODEProblem(f!, [0.5], (0.0, 1.0))
sol = solve(prob)

Bibliotecas maduras

  • DataFrames.jl para análise tabular

Pontos de atenção

  • Algumas áreas não científicas têm menos opções que Python/JS
  • APIs mudam com mais frequência em pacotes pequenos
  • Stack web e mobile não é o principal foco da linguagem
