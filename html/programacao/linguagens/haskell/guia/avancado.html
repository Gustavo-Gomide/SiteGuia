Haskell — Avançado

Os tópicos avançados de Haskell incluem a hierarquia Functor → Applicative → Monad, criação de type classes personalizadas, transformadores de monad (ReaderT/StateT/ExceptT), Software Transactional Memory (STM) para concorrência e extensões de linguagem do GHC.

A abstração Monad Transformer permite combinar múltiplos efeitos: ReaderT config (ExceptT Erro IO) é um efeito que lê configuração somente leitura, pode falhar com Erro, e executa operações de I/O — tudo composto de forma type-safe. Este padrão é a base de frameworks como Servant e Yesod.

Type Classes — Definir e Instanciar

Polimorfismo ad-hoc: criar e implementar type classes
-- Definir type class
class Serializable a where
    serializar   :: a -> String
    deserializar :: String -> Maybe a

-- Implementar para tipos concretos
data Ponto = Ponto Double Double

instance Serializable Ponto where
    serializar (Ponto x y) = show x ++ "," ++ show y
    deserializar s = case break (==',') s of
        (xs, _:ys) -> Just (Ponto (read xs) (read ys))
        _          -> Nothing

-- Type class com default methods
class Container f where
    vazio    :: f a
    inserir  :: a -> f a -> f a
    remover  :: f a -> Maybe (a, f a)
    tamanho  :: f a -> Int
    tamanho c = case remover c of     -- default usando outros métodos
        Nothing    -> 0
        Just (_, r) -> 1 + tamanho r

-- Hierarquia de type classes
-- Eq     → Ord
-- Functor → Applicative → Monad

class (Applicative m) => Monad m where
    (>>=)  :: m a -> (a -> m b) -> m b
    return :: a -> m a
    return = pure    -- default implementation

-- Functor — mapear sobre estruturas
-- class Functor f where
--   fmap :: (a -> b) -> f a -> f b

-- Instância de Functor personalizada
data Arvore a = Folha | No a (Arvore a) (Arvore a)

instance Functor Arvore where
    fmap _ Folha      = Folha
    fmap f (No x l r) = No (f x) (fmap f l) (fmap f r)

-- Derivação automática de type classes comuns
data Cor = Vermelho | Verde | Azul
    deriving (Show, Read, Eq, Ord, Enum, Bounded)

minCor = minBound :: Cor    -- Vermelho
maxCor = maxBound :: Cor    -- Azul
cores  = [minBound..maxBound] :: [Cor]  -- [Vermelho, Verde, Azul]

Functor, Applicative e Monad

A hierarquia central do Haskell e suas leis
-- FUNCTOR: fmap / <$>
fmap (+1) (Just 5)          -- Just 6
fmap (+1) Nothing           -- Nothing
fmap (*2) [1,2,3]           -- [2,4,6]
fmap length (Right "abc")   -- Right 3
(*2) <$> [1,2,3]            -- [2,4,6]  (<$> é operador infix para fmap)

-- APPLICATIVE: pure / <*>
-- Funções TAMBÉM dentro do contexto
Just (+3) <*> Just 5        -- Just 8
Nothing   <*> Just 5        -- Nothing
[(+1),(+10)] <*> [1,2]      -- [2,3,11,12]  (produto cartesiano)

-- liftA2 — aplicar função binária dentro de contexto
import Control.Applicative (liftA2)
liftA2 (+) (Just 3) (Just 5)    -- Just 8
liftA2 (+) Nothing  (Just 5)    -- Nothing

-- MONAD: >>= / do notation
-- Maybe como monad (short-circuit em Nothing)
safeDiv :: Int -> Int -> Maybe Int
safeDiv _ 0 = Nothing
safeDiv a b = Just (a `div` b)

calcular :: Int -> Int -> Int -> Maybe Int
calcular a b c = do
    x <- safeDiv a b
    y <- safeDiv x c
    return (x + y)

-- List como monad (não-determinismo)
combinacoes :: [(Int, Int)]
combinacoes = do
    x <- [1..3]
    y <- [1..3]
    if x /= y then return (x, y) else []
-- [(1,2),(1,3),(2,1),(2,3),(3,1),(3,2)]

-- Estado com State monad
import Control.Monad.State

contar :: State Int [Int]
contar = do
    forM [1..5] $ \i -> do
        modify (+1)        -- incrementa estado
        n <- get           -- lê estado atual
        return (i * n)

resultado = runState contar 0    -- ([1,4,9,16,25], 5)

Monad Transformers — Combinar Efeitos

ReaderT + ExceptT + IO — padrão de aplicações reais
-- Stack de monad transformers
import Control.Monad.Reader
import Control.Monad.Except

data Config = Config
    { dbURL    :: String
    , maxConns :: Int
    } deriving Show

data AppError
    = BancoDadosErr String
    | ValidacaoErr String
    deriving Show

-- AppM combina: Reader Config + Either AppError + IO
type AppM a = ReaderT Config (ExceptT AppError IO) a

-- Funções que usam o stack
buscarConexao :: AppM String
buscarConexao = do
    config <- ask                  -- lê Config (from ReaderT)
    let url = dbURL config
    liftIO $ putStrLn $ "Conectando a: " ++ url
    when (null url) $ throwError (BancoDadosErr "URL vazia")
    return url

processar :: String -> AppM String
processar input = do
    when (null input) $ throwError (ValidacaoErr "input vazio")
    conn <- buscarConexao
    liftIO $ putStrLn $ "Processando via " ++ conn
    return $ map toUpper input

runApp :: Config -> AppM a -> IO (Either AppError a)
runApp config action = runExceptT (runReaderT action config)

main :: IO ()
main = do
    let config = Config { dbURL = "postgres://localhost/db", maxConns = 10 }
    resultado <- runApp config (processar "hello")
    case resultado of
        Right val -> putStrLn $ "Resultado: " ++ val
        Left err  -> putStrLn $ "Erro: " ++ show err

STM — Concorrência Transacional

Software Transactional Memory — sem locks, sem deadlocks
-- STM: transações compostas atomicamente
import Control.Concurrent.STM
import Control.Concurrent (forkIO, threadDelay)

-- TVar: variável transacional mutável
transferir :: TVar Int -> TVar Int -> Int -> STM ()
transferir origem destino valor = do
    saldo <- readTVar origem
    when (saldo < valor) $ retry   -- retry: bloqueia até condição mudar
    writeTVar origem   (saldo - valor)
    modifyTVar destino (+valor)

main :: IO ()
main = do
    contaA <- newTVarIO 1000
    contaB <- newTVarIO 500

    -- atomically garante que a transação é atômica
    atomically $ transferir contaA contaB 200

    saldoA <- readTVarIO contaA
    saldoB <- readTVarIO contaB
    putStrLn $ "A: " ++ show saldoA ++ " B: " ++ show saldoB

-- TQueue: fila transacional
produtor :: TQueue String -> IO ()
produtor q = do
    atomically $ writeTQueue q "mensagem 1"
    atomically $ writeTQueue q "mensagem 2"

consumidor :: TQueue String -> IO ()
consumidor q = do
    msg <- atomically $ readTQueue q    -- bloqueia se vazia
    putStrLn $ "Recebido: " ++ msg

-- Async / Concorrência de alto nível
import Control.Concurrent.Async

paralelo :: IO ()
paralelo = do
    (r1, r2) <- concurrently calcular1 calcular2
    print (r1, r2)

comTimeout :: IO (Maybe String)
comTimeout = race (threadDelay 1000000 >> return "timeout") buscarDados >>= \case
    Left timeout -> return Nothing
    Right dados  -> return (Just dados)

Engenharia de Sistemas em Haskell

No nivel avancado, a pergunta principal deixa de ser "como compilar" e passa a ser "como operar com seguranca em producao". Para isso, combine modelagem tipada com politicas de timeout, retry com backoff, idempotencia em comandos e rastreabilidade de eventos.

  • APIs e servidores: contratos tipados, validacao exaustiva e observabilidade por endpoint.
  • Bancos e filas: transacoes curtas, limites de concorrencia e tratamento consistente de falha parcial.
  • Dados e IA: pipelines deterministas e controle de versao de datasets/modelos.
  • IoT e edge: protocolos explicitos e reconexao resiliente em redes instaveis.

STM do Haskell é a implementação de referência de memória transacional em software: você escreve transações compostas que se comportam como transações de banco de dados — atômicas, isoladas e consistentes. A primitiva retry bloqueia a thread até que alguma TVar lida na transação mude, sem espera ativa e sem locks manuais. É a forma mais segura conhecida de programação concorrente com estado compartilhado.
