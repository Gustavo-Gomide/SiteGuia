Haskell — Terminal e CLI

Haskell é excelente para ferramentas CLI: optparse-applicative usa a estrutura de Applicative Functor para compor parsers de argumentos com auto-geração de help; turtle oferece uma API shell em Haskell idiomático; ansi-terminal para cores; e System.IO para manipulação de arquivos e processos.

optparse-applicative é único: parsers de argumentos são functores aplicativos que se compõem automaticamente gerando tanto o parser quanto o texto de ajuda. Adicionar um novo argumento ao tipo de dados gera automaticamente a flag e a documentação correspondente.

optparse-applicative — Parsers de Argumentos

CLI type-safe com help gerado automaticamente
{-# LANGUAGE RecordWildCards #-}
-- cabal install optparse-applicative
import Options.Applicative

-- Tipos de dados representam os argumentos
data Opcoes = Opcoes
    { entrada    :: FilePath
    , saida      :: FilePath
    , verbose    :: Bool
    , maxLinhas  :: Int
    , formato    :: String
    } deriving Show

data Comando
    = Converter Opcoes
    | Analisar  { arquivoAnalisar :: FilePath }
    | Validar   { arquivoValidar  :: FilePath, schema :: Maybe FilePath }
    deriving Show

-- Parser de Opcoes usando Applicative
parsOpcoes :: Parser Opcoes
parsOpcoes = Opcoes
    <$> strOption
            ( long "entrada"
           <> short 'i'
           <> metavar "ARQUIVO"
           <> help "Arquivo de entrada" )
    <*> strOption
            ( long "saida"
           <> short 'o'
           <> metavar "ARQUIVO"
           <> value "output.json"
           <> showDefault
           <> help "Arquivo de saída" )
    <*> switch
            ( long "verbose"
           <> short 'v'
           <> help "Mostrar detalhes" )
    <*> option auto
            ( long "max-linhas"
           <> value 1000
           <> showDefault
           <> help "Máximo de linhas a processar" )
    <*> strOption
            ( long "formato"
           <> value "json"
           <> showDefault
           <> help "Formato de saída: json|csv|yaml" )

-- Parser de subcomandos
parsComando :: Parser Comando
parsComando = subparser
    ( command "converter"
        (info (Converter <$> parsOpcoes)
              (progDesc "Converter arquivo de formato"))
   <> command "analisar"
        (info (Analisar <$> argument str (metavar "ARQUIVO"))
              (progDesc "Analisar estatísticas"))
   <> command "validar"
        (info (Validar
                <$> argument str (metavar "ARQUIVO")
                <*> optional (option str (long "schema" <> metavar "SCHEMA")))
              (progDesc "Validar arquivo contra schema"))
    )

main :: IO ()
main = do
    cmd <- execParser $ info (parsComando <**> helper)
        ( fullDesc
       <> progDesc "Ferramenta de processamento de dados"
       <> header "datatool - processa CSV, JSON e YAML" )
    case cmd of
        Converter opts -> executarConversor opts
        Analisar path  -> analisarArquivo path
        Validar p s    -> validarArquivo p s

turtle — Shell Scripting em Haskell

Manipulação de arquivos e processos de forma idiomática
-- cabal install turtle
import Turtle
import qualified Data.Text as T
import Prelude hiding (FilePath)

-- Listar arquivos recursivamente
listarHaskell :: FilePath -> Shell FilePath
listarHaskell dir = do
    f <- lstree dir      -- recursivo
    guard (extension f == Just "hs")
    return f

-- Contar linhas de código
contarLinhas :: IO ()
contarLinhas = do
    total <- fold (listarHaskell ".") (Fold (\ total _ -> total + 1) 0 id)
    printf ("Arquivos .hs encontrados: "%d%"\n") total

-- Executar comandos externos
gitStatus :: IO ()
gitStatus = do
    code <- shell "git status --short" empty
    case code of
        ExitSuccess   -> echo "Repositório limpo"
        ExitFailure n -> printf ("Git falhou com código "%d%"\n") n

-- Capturar saída de comando
obterBranch :: IO Text
obterBranch = do
    lines_ <- strict $ inshell "git rev-parse --abbrev-ref HEAD" empty
    return (T.strip lines_)

-- Script de backup
backup :: FilePath -> FilePath -> IO ()
backup src dst = do
    timestamp <- strict $ inshell "date +%Y%m%d_%H%M%S" empty
    let destDir = dst </> fromText (T.strip timestamp)
    mktree destDir
    cptree src destDir
    printf ("Backup criado em: "%fp%"\n") destDir

-- Ler e processar arquivos com Text
import qualified Data.Text.IO as TIO

processarTexto :: FilePath -> IO ()
processarTexto path = do
    conteudo <- TIO.readFile (T.unpack $ format fp path)
    let linhas    = T.lines conteudo
        semVazias = filter (not . T.null . T.strip) linhas
        ordenadas = sort semVazias
    TIO.writeFile "output.txt" (T.unlines ordenadas)
    printf (d%" linhas processadas\n") (length ordenadas)

Saída Colorida e Formatação

ansi-terminal e pretty-printer para output rico
-- cabal install ansi-terminal
import System.Console.ANSI

-- Colorir saída
sucesso :: String -> IO ()
sucesso msg = do
    setSGR [SetColor Foreground Vivid Green, SetConsoleIntensity BoldIntensity]
    putStr "✓ "
    setSGR [Reset]
    putStrLn msg

erro :: String -> IO ()
erro msg = do
    setSGR [SetColor Foreground Vivid Red]
    putStr "✗ "
    setSGR [Reset]
    putStrLn msg

aviso :: String -> IO ()
aviso msg = do
    setSGR [SetColor Foreground Vivid Yellow]
    putStr "⚠ "
    setSGR [Reset]
    putStrLn msg

info :: String -> IO ()
info msg = do
    setSGR [SetColor Foreground Vivid Cyan]
    putStr "ℹ "
    setSGR [Reset]
    putStrLn msg

-- Tabela simples
tabela :: [[String]] -> IO ()
tabela rows = do
    let widths = map (maximum . map length) (transpose rows)
    setSGR [SetColor Foreground Vivid White, SetConsoleIntensity BoldIntensity]
    mapM_ (putStr . padRight (head widths)) (head rows)
    putStrLn ""
    setSGR [Reset]
    putStrLn (replicate (sum widths + length widths) '-')
    mapM_ (\row -> do
        mapM_ (\(w, cell) -> putStr (padRight w cell)) (zip widths row)
        putStrLn "") (tail rows)
  where
    padRight n s = s ++ replicate (n - length s + 2) ' '
    transpose [] = []
    transpose ([] : _) = []
    transpose xss = map head xss : transpose (map tail xss)

-- Progress bar
progressBar :: Int -> Int -> IO ()
progressBar atual total = do
    clearLine
    let pct       = atual * 100 `div` total
        barWidth  = 40
        filled    = barWidth * atual `div` total
        bar       = replicate filled '█' ++ replicate (barWidth - filled) '░'
    setSGR [SetColor Foreground Vivid Cyan]
    putStr $ "\r[" ++ bar ++ "] " ++ show pct ++ "%"
    setSGR [Reset]
    hFlush stdout

Arquitetura de CLI Profissional

Uma CLI robusta separa parser de argumentos, casos de uso, adaptadores de IO e formatacao de saida. Esse desenho facilita testes automatizados, reduz regressao e permite reutilizar o mesmo nucleo em servicos ou jobs agendados.

  • Exemplo real: ferramentas internas de migracao de dados, validadores de contrato e automacao de deploy.
  • Boa pratica: saida estruturada opcional (JSON) para integracao com pipelines CI/CD.
  • Limitacao: para interacoes ricas em tempo real, TUI/GUI pode ser mais adequada do que CLI pura.
  • Trade-off: tipagem forte aumenta seguranca, mas exige maior disciplina na modelagem inicial.

optparse-applicative demonstra a elegância de usar abstrações matemáticas para resolver problemas reais: parsers de argumentos são Applicative Functors, então você os compõe com <$> e <*> exatamente como compõe qualquer Applicative. O resultado é que parsers complexos com subcomandos, flags opcionais e defaults são simplesmente registros Haskell — sem magic strings, sem configuração em runtime, erro de compilação se o tipo não bater.
