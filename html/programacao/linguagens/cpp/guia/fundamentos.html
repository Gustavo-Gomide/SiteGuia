C++ — Fundamentos

Fundamentos do C++

Ponteiros, referencias, classes com RAII, construtores e destrutores, operadores, templates basicos, STL containers (vector, map, set, list) e iteradores. O nucleo que todo programador C++ precisa dominar antes de partir para recursos avancados.

RAII — Resource Acquisition Is Initialization — e o idioma central do C++ moderno: todo recurso (memoria, arquivo, mutex, socket) deve ser de propriedade de um objeto, cujo destrutor libera o recurso automaticamente.

Ponteiros e Referencias

C++ — ponteiros brutos, referencias e smart pointers
#include <memory>
#include <print>

int main() {
  // Ponteiro bruto — use apenas quando necessario (interop C, data race)
  int  valor  = 42;
  int* ptr    = &valor;       // endereco de valor
  *ptr = 100;               // derreferencia
  std::println("*ptr = {}", *ptr);    // 100

  // Referencia — alias permanente, nunca nula, nao reassociavel
  int& ref = valor;
  ref = 200;
  std::println("valor = {}", valor);  // 200

  // unique_ptr — propriedade exclusiva (move-only)
  auto u = std::make_unique<int>(10);
  std::println("unique = {}", *u);
  // delete automatico quando sai do escopo

  // shared_ptr — propriedade compartilhada (ref-count)
  auto s1 = std::make_shared<int>(20);
  auto s2 = s1;               // copia — contagem vai a 2
  std::println("use_count = {}", s1.use_count());  // 2

  // weak_ptr — referencia que nao incrementa contagem (quebra ciclos)
  std::weak_ptr<int> w = s1;
  if (auto tmp = w.lock()) {
    std::println("weak = {}", *tmp);
  }
}
Dangling pointer / dangling reference — um ponteiro ou referencia que aponta para objeto ja destruido causa Undefined Behaviour (UB). Sempre prefira smart pointers a ponteiros brutos proprietarios. Use -fsanitize=address em debug para detectar acessos invalidos.

Classes, RAII e Rule of Five

C++ — classe Buffer com RAII e Rule of Five
#include <cstddef>
#include <algorithm>  // std::copy_n
#include <stdexcept>

// Buffer de bytes com propriedade RAII
class Buffer {
  std::byte* dados_;
  std::size_t capacidade_;

public:
  // 1. Construtor — adquire recurso
  explicit Buffer(std::size_t n)
    : dados_{ new std::byte[n] }, capacidade_{ n } {}

  // 2. Destrutor — libera recurso (nunca pode lancar excecao)
  ~Buffer() noexcept { delete[] dados_; }

  // 3. Construtor de copia — copia profunda
  Buffer(const Buffer& outro)
    : dados_{ new std::byte[outro.capacidade_] },
      capacidade_{ outro.capacidade_ } {
    std::copy_n(outro.dados_, capacidade_, dados_);
  }

  // 4. Operador de atribuicao por copia (copy-and-swap)
  Buffer& operator=(Buffer outro) noexcept {
    swap(*this, outro);
    return *this;
  }

  // 5. Construtor de movimento — transfere propriedade
  Buffer(Buffer&& outro) noexcept
    : dados_{ outro.dados_ }, capacidade_{ outro.capacidade_ } {
    outro.dados_       = nullptr;
    outro.capacidade_  = 0;
  }

  // 6. Operador de atribuicao por movimento
  Buffer& operator=(Buffer&& outro) noexcept {
    if (this != &outro) {
      delete[] dados_;
      dados_      = outro.dados_;
      capacidade_ = outro.capacidade_;
      outro.dados_      = nullptr;
      outro.capacidade_ = 0;
    }
    return *this;
  }

  friend void swap(Buffer& a, Buffer& b) noexcept {
    using std::swap;
    swap(a.dados_,       b.dados_);
    swap(a.capacidade_,  b.capacidade_);
  }

  std::size_t size()  const noexcept { return capacidade_;  }
  std::byte*  data()  noexcept       { return dados_;        }
};

STL Containers

C++ — vector, map, set, unordered_map, list
#include <vector>
#include <map>
#include <set>
#include <unordered_map>
#include <string>
#include <print>
#include <algorithm>
#include <ranges>

int main() {
  // vector — array dinamico contiguous na memoria
  std::vector<int> v = { 3, 1, 4, 1, 5, 9 };
  v.push_back(26);
  v.emplace_back(53);                   // constroi no lugar — eficiente
  v.reserve(100);                       // pre-aloca sem mudar size
  std::sort(v.begin(), v.end());

  // map — arvore balanceada (chave ordenada), O(log n)
  std::map<std::string, int> freq;
  freq["cpp"]++;
  freq["java"]++;
  freq["cpp"]++;
  if (auto it = freq.find("cpp"); it != freq.end()) {
    std::println("cpp: {}", it->second);     // 2
  }

  // unordered_map — hash table, O(1) medio
  std::unordered_map<std::string, std::string> cache;
  cache.emplace("key", "value");
  cache.try_emplace("key2", "value2");  // nao sobrescreve se existir

  // set — elementos unicos e ordenados
  std::set<int> s = { 3, 1, 4, 1, 5 };  // {1, 3, 4, 5}
  auto [it, ok] = s.insert(4);            // ok=false, nao inseriu (duplicata)

  // Ranges (C++20) — pipeline funcional sobre containers
  auto pares_grandes = v
    | std::views::filter([](int x) { return x > 3 && x % 2 == 0; })
    | std::views::transform([](int x) { return x * 2; });

  for (int n : pares_grandes)
    std::print("{} ", n);
  std::println();
}

Funcoes, Lambdas e std::function

C++ — overload, default args, lambda, std::function
#include <functional>
#include <vector>
#include <algorithm>
#include <print>

// Overload de funcao — mesmo nome, assinaturas distintas
double area(double lado) { return lado * lado; }
double area(double l, double h) { return l * h; }

// Template de funcao — gerado em compile time
template<typename T>
T maximo(T a, T b) { return a > b ? a : b; }

int main() {
  // Lambda com captura por valor [=] e por referencia [&]
  int limite = 5;
  auto filtrar = [limite](int x) { return x > limite; };
  auto imprimir = [&](int x) { std::println("limite={} x={}", limite, x); };

  // Lambda generica (C++14) — template implicito
  auto dobrar = [](auto x) { return x * 2; };
  std::println("{}", dobrar(3));     // 6
  std::println("{:.2f}", dobrar(3.14)); // 6.28

  // std::function — wrapper polimorfico para qualquer callable
  std::function<int(int)> operacao;
  operacao = filtrar;
  // Troca em tempo de execucao:
  operacao = [](int x) { return x * x; };

  std::vector<int> v = { 1, 2, 3, 4, 5, 6 };
  std::vector<int> resultado;
  std::copy_if(v.begin(), v.end(), std::back_inserter(resultado), filtrar);

  for (int n : resultado) std::print("{} ", n);   // 6
  std::println();
}

Excecoes e Tratamento de Erros

C++ — try/catch, hierarquia de excecoes, std::expected (C++23)
#include <stdexcept>
#include <expected>    // C++23
#include <string>
#include <print>

// Excecao customizada herdando std::runtime_error
class ValidationError : public std::runtime_error {
  int codigo_;
public:
  ValidationError(std::string msg, int cod)
    : std::runtime_error(std::move(msg)), codigo_{ cod } {}
  int codigo() const noexcept { return codigo_; }
};

void validar(int age) {
  if (age < 0)  throw ValidationError("Idade negativa", 400);
  if (age > 150) throw ValidationError("Idade improvavel", 422);
}

// std::expected (C++23) — retorna valor OU erro sem excecoes
std::expected<double, std::string> dividir(double a, double b) {
  if (b == 0.0) return std::unexpected("divisao por zero");
  return a / b;
}

int main() {
  try {
    validar(-5);
  } catch (const ValidationError& e) {
    std::println("Validacao [{}]: {}", e.codigo(), e.what());
  } catch (const std::exception& e) {
    std::println("Erro generico: {}", e.what());
  }

  // expected — zero-overhead error path
  auto r = dividir(10.0, 0.0);
  if (r) {
    std::println("Resultado: {}", *r);
  } else {
    std::println("Erro: {}", r.error());
  }
  // ou com value_or:
  double val = dividir(1.0, 2.0).value_or(0.0);
}

Heranca, Polimorfismo e Virtual

C++ — classe base abstrata, override, final, dynamic_cast
#include <memory>
#include <vector>
#include <print>
#include <numbers>   // std::numbers::pi

class Forma {                             // classe base abstrata
public:
  virtual double area()  const = 0;    // metodo puramente virtual
  virtual double perimetro() const = 0;
  virtual ~Forma() = default;          // destrutor virtual obrigatorio
};

class Circulo final : public Forma {   // final — nao pode ser herdada
  double r_;
public:
  explicit Circulo(double r) : r_{ r } {}
  double area()      const override { return std::numbers::pi * r_ * r_; }
  double perimetro() const override { return 2.0 * std::numbers::pi * r_; }
};

class Retangulo : public Forma {
  double l_, h_;
public:
  Retangulo(double l, double h) : l_{ l }, h_{ h } {}
  double area()      const override { return l_ * h_; }
  double perimetro() const override { return 2.0 * (l_ + h_); }
};

int main() {
  std::vector<std::unique_ptr<Forma>> formas;
  formas.emplace_back(std::make_unique<Circulo>(5.0));
  formas.emplace_back(std::make_unique<Retangulo>(4.0, 3.0));

  for (const auto& f : formas) {
    std::println("Area={:.2f}  Perimetro={:.2f}",
      f->area(), f->perimetro());
  }

  // dynamic_cast — cast seguro com RTTI (retorna nullptr se falhar)
  if (auto* c = dynamic_cast<Circulo*>(formas[0].get())) {
    std::println("e um Circulo");
  }
}
