Boost

C++ + Boost — Boost.Asio, Boost.Beast (HTTP) e Boost.JSON
#include <boost/asio.hpp>
#include <boost/beast.hpp>
#include <boost/json.hpp>
#include <print>

namespace asio  = boost::asio;
namespace beast = boost::beast;
namespace http  = beast::http;
namespace json  = boost::json;

int main() {
  // --- Boost.JSON: parse e serialize ---
  json::value jv = json::parse(R"({"nome":"C++","versao":23,"ativo":true})");
  auto& obj = jv.as_object();
  std::println("nome: {}", json::value_to<std::string>(obj.at("nome")));

  // Construir JSON
  json::object saida;
  saida["resultado"] = 42;
  saida["itens"]     = json::array{ 1, 2, 3 };
  std::println("{}", json::serialize(saida));

  // --- Boost.Asio: TCP echo server (coroutine) ---
  asio::io_context ctx;
  asio::co_spawn(ctx, [&]() -> asio::awaitable<void> {
    asio::ip::tcp::acceptor acc{ ctx,
      { asio::ip::tcp::v4(), 8080 } };
    while (true) {
      auto sock = co_await acc.async_accept(asio::use_awaitable);
      asio::co_spawn(ctx, [s = std::move(sock)]() mutable
          -> asio::awaitable<void> {
        std::array<char, 1024> buf;
        for (;;) {
          auto n = co_await s.async_read_some(
            asio::buffer(buf), asio::use_awaitable);
          co_await asio::async_write(
            s, asio::buffer(buf, n), asio::use_awaitable);
        }
      }, asio::detached);
    }
  }, asio::detached);
  // ctx.run();  // inicia event loop
}

nlohmann/json — JSON moderno

C++ + nlohmann/json — parse, serialize e struct mapping
#include <nlohmann/json.hpp>
#include <fstream>
#include <print>

using json = nlohmann::json;
using namespace nlohmann::literals;

struct Pessoa {
  std::string nome;
  int         idade;
  std::vector<std::string> hobbies;
};

// Macro de serialization automatica (NLOHMANN_DEFINE_TYPE_INTRUSIVE)
NLOHMANN_DEFINE_TYPE_NON_INTRUSIVE(Pessoa, nome, idade, hobbies)

int main() {
  // Literal operator"_json
  auto j = R"({ "nome": "Ana", "idade": 30, "hobbies": ["cpp", "musica"] })"_json;

  Pessoa p = j.get<Pessoa>();
  std::println("Nome: {}, Idade: {}", p.nome, p.idade);

  // Serialize de volta para JSON string
  p.hobbies.push_back("games");
  json j2 = p;
  std::println("{}", j2.dump(2));    // indent=2

  // Acesso seguro com value() e contains()
  if (j2.contains("ativo")) {
    bool ativo = j2["ativo"].get<bool>();
  }
  int versao = j2.value("versao", 1);  // default=1 se ausente

  // Ler de arquivo
  std::ifstream f("config.json");
  if (f.is_open()) {
    json cfg = json::parse(f);
  }
}

spdlog — Logging de Alta Performance

C++ + spdlog — loggers, sinks e formatacao
#include <spdlog/spdlog.h>
#include <spdlog/sinks/rotating_file_sink.h>
#include <spdlog/sinks/stdout_color_sinks.h>
#include <spdlog/async.h>

int main() {
  // Sink para console colorido
  auto console = std::make_shared<spdlog::sinks::stdout_color_sink_mt>();
  console->set_level(spdlog::level::debug);

  // Sink para arquivo rotativo (10MB, maximo 5 arquivos)
  auto fileSink = std::make_shared<spdlog::sinks::rotating_file_sink_mt>(
    "logs/app.log", 10 * 1024 * 1024, 5);

  // Logger com multiplos sinks (multi-sink)
  spdlog::logger logger{ "app",
    { console, fileSink } };
  logger.set_level(spdlog::level::trace);
  logger.set_pattern("[%Y-%m-%d %H:%M:%S.%e] [%l] [%n] %v");

  logger.trace("trace — detalhe maximo");
  logger.debug("debug — usuario {}", "admin");
  logger.info("servidor ouvindo em :{}", 8080);
  logger.warn("conexao lenta: {} ms", 950);
  logger.error("falha ao abrir arquivo: {}", "config.json");
  logger.critical("banco de dados indisponivel!");

  // Logger assincrono com thread pool
  spdlog::init_thread_pool(8192, 1);  // fila de 8k, 1 thread
  auto async_logger = std::make_shared<spdlog::async_logger>(
    "async", fileSink,
    spdlog::thread_pool(),
    spdlog::async_overflow_policy::block);
  spdlog::register_logger(async_logger);

  auto alog = spdlog::get("async");
  alog->info("mensagem assincrona");
  spdlog::shutdown();              // flush e encerra thread pool
}

Eigen — Algebra Linear

C++ + Eigen — matrizes, vetores, decomposicoes e solvers
#include <Eigen/Dense>
#include <print>

using namespace Eigen;

int main() {
  // Matriz e vetor de tamanho fixo (alocados na stack)
  Matrix3d A;
  A << 2, -1,  0,
      -1,  2, -1,
       0, -1,  2;
  Vector3d b(1, 0, 0);

  // Resolver sistema linear A*x = b (LLT — Cholesky: A deve ser SPD)
  Vector3d x = A.llt().solve(b);
  std::println("x = [{:.4f}, {:.4f}, {:.4f}]", x(0), x(1), x(2));

  // Valores e vetores proprios
  SelfAdjointEigenSolver<Matrix3d> solver(A);
  std::println("eigenvalues: {}", solver.eigenvalues().transpose());

  // Decomposicao SVD
  MatrixXd M = MatrixXd::Random(4, 3);
  JacobiSVD<MatrixXd> svd(M, ComputeThinU | ComputeThinV);
  std::println("Singular values:\n{}", svd.singularValues().transpose());

  // Operacoes element-wise (Array)
  ArrayXd arr = ArrayXd::LinSpaced(5, 1, 5);
  std::println("arr^2 = {}", (arr * arr).transpose());   // element-wise
}

Catch2 — Testes Unitarios

C++ + Catch2 — TEST_CASE, SECTION, matchers e benchmarks
#include <catch2/catch_test_macros.hpp>
#include <catch2/matchers/catch_matchers_vector.hpp>
#include <catch2/benchmark/catch_benchmark.hpp>
#include "calculadora.hpp"

TEST_CASE("Calculadora basica", "[calculadora][basico]") {
  Calculadora calc;

  SECTION("soma") {
    REQUIRE(calc.somar(2, 3) == 5);
    REQUIRE(calc.somar(-1, 1) == 0);
  }
  SECTION("divisao") {
    REQUIRE(calc.dividir(10.0, 4.0) == Approx(2.5));
    REQUIRE_THROWS_AS(calc.dividir(1.0, 0.0), std::domain_error);
  }
}

TEST_CASE("Ordenacao", "[algoritmos]") {
  std::vector<int> v = { 5, 3, 1, 4 };
  std::sort(v.begin(), v.end());

  using namespace Catch::Matchers;
  REQUIRE_THAT(v, Equals(std::vector<int>{ 1, 3, 4, 5 }));
  REQUIRE_THAT(v, Contains(std::vector<int>{ 1, 4 }));
}

TEST_CASE("Benchmark soma", "[!benchmark]") {
  BENCHMARK("somar 1000 elementos") {
    Calculadora c;
    for (int i = 0; i < 1000; ++i) c.somar(i, i + 1);
  };
}

Visao Geral das Principais Bibliotecas

Qt 6

Framework completo para GUI cross-platform (Windows/Linux/macOS/Android/iOS). QML para interfaces declarativas, Qt Network, Qt SQL, Qt Multimedia. Licenca dual (GPL/comercial).

SFML / SDL2

SFML — biblioteca 2D de alto nivel (window, graphics, audio, network). SDL2 — mais baixo nivel, suporte a OpenGL/Vulkan, ampla adocao em jogos indie. Ambas zlib/LGPL.

OpenCV 4

Biblioteca de visao computacional: leitura/escrita de imagens e video, algoritmos de deteccao, segmentacao, calibracao de camera, DNN module (redes neurais via ONNX).

fmt / std::format

Biblioteca de formatacao de strings que inspirou std::format do C++20 e std::println do C++23. API type-safe, mais rapida que printf e iostreams.

GoogleTest (gtest)

Framework de testes unitarios do Google. ASSERT_*/EXPECT_*, test fixtures (TEST_F), mocking com gMock, integracao nativa com CMake via FetchContent.

Abseil (absl)

Biblioteca base da Google: absl::flat_hash_map (open-addressing), absl::string_view, logging (absl::log), flags, status codes (absl::Status). Usada internamente no Chromium e TensorFlow.
