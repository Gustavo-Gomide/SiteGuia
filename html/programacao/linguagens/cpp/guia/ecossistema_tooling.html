C++ — Ecossistema e Tooling

Ecossistema e Tooling C++

CMake como meta-sistema de build, Ninja para compilacao rapida, vcpkg e Conan para gerenciamento de pacotes, Clang-Tidy e sanitizers para qualidade, GDB e AddressSanitizer para debugging, e integracao continua com GitHub Actions.

Um projeto C++ bem configurado com CMake + vcpkg + Clang-Tidy + ASan e tao produtivo quanto qualquer outra linguagem moderna. A configuracao inicial compensa em manutencao e qualidade.

CMake — Sistema de Build

CMake — CMakeLists.txt moderno com targets, tests e vcpkg
# CMakeLists.txt — projeto moderno C++23
cmake_minimum_required(VERSION 3.27)
project(MeuApp VERSION 1.0.0 LANGUAGES CXX)

# Padrao da linguagem — aplicado a todos os targets
set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)         # sem extensoes GCC/Clang
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)  # para clangd/clang-tidy

# Dependencias via vcpkg (toolchain passada no preset ou CLI)
find_package(nlohmann_json CONFIG REQUIRED)
find_package(spdlog         CONFIG REQUIRED)
find_package(Catch2         CONFIG REQUIRED)

# Biblioteca principal (separada do executavel — testavel)
add_library(core STATIC
  src/calculadora.cpp
  src/parser.cpp)
target_include_directories(core PUBLIC include)
target_link_libraries(core
  PUBLIC  nlohmann_json::nlohmann_json
  PRIVATE spdlog::spdlog)
target_compile_options(core PRIVATE
  $<$<CXX_COMPILER_ID:GNU,Clang>:-Wall -Wextra -Wpedantic>
  $<$<CXX_COMPILER_ID:MSVC>:/W4>)

# Executavel
add_executable(app src/main.cpp)
target_link_libraries(app PRIVATE core)

# Testes com CTest + Catch2
enable_testing()
add_executable(tests
  tests/calculadora_test.cpp
  tests/parser_test.cpp)
target_link_libraries(tests PRIVATE core Catch2::Catch2WithMain)
include(Catch)
catch_discover_tests(tests)

# target de instalacao
install(TARGETS app DESTINATION bin)
Shell — build com CMake Presets e Ninja
# CMakePresets.json — configura builds de forma padronizada
# (arquivo JSON na raiz do projeto)

# Configurar (debug)
cmake --preset debug

# Compilar
cmake --build --preset debug -- -j$(nproc)

# Rodar testes
ctest --preset debug --output-on-failure

# Build release com Ninja
cmake -B build/release -G Ninja \
  -DCMAKE_BUILD_TYPE=Release \
  -DCMAKE_TOOLCHAIN_FILE="$VCPKG_ROOT/scripts/buildsystems/vcpkg.cmake"
cmake --build build/release

vcpkg — Gerenciador de Pacotes

Shell + vcpkg.json — manifest mode
# vcpkg.json — dependencias declarativas no projeto
{
  "name": "meu-app",
  "version": "1.0.0",
  "dependencies": [
    "nlohmann-json",
    "spdlog",
    "eigen3",
    { "name": "boost-asio",    "version>=": "1.83.0" },
    { "name": "catch2",        "features": ["with-main"] }
  ],
  "overrides": [
    { "name": "openssl", "version": "3.2.2" }
  ]
}

# Instalar as dependencias (manifest mode)
vcpkg install

# Instalar biblioteca especifica sem manifest
vcpkg install opencv4 --triplet x64-linux

# Listar instalados
vcpkg list

# Atualizar tudo
vcpkg upgrade --no-dry-run

Sanitizers e Analise Estatica

Shell — AddressSanitizer, UBSan, ThreadSanitizer e Valgrind
# AddressSanitizer (ASan) — detecta use-after-free, heap overflow, stack overflow
g++ -std=c++23 -g -fsanitize=address \
    -fno-omit-frame-pointer -o app_asan main.cpp
./app_asan

# Undefined Behaviour Sanitizer (UBSan)
g++ -std=c++23 -g -fsanitize=undefined -o app_ub main.cpp
./app_ub

# Combinar ASan + UBSan (mais comum em CI)
g++ -std=c++23 -g -fsanitize=address,undefined -o app_san main.cpp

# ThreadSanitizer (TSan) — detecta data races
g++ -std=c++23 -g -fsanitize=thread -o app_tsan main.cpp

# Valgrind Memcheck — analise de memoria (mais lento, mas sem recompilacao)
valgrind --leak-check=full --show-leak-kinds=all --track-origins=yes ./app

# Clang-Tidy — analise estatica com auto-fix
clang-tidy src/*.cpp -- -std=c++23 -Iinclude
clang-tidy --fix src/*.cpp -- -std=c++23 -Iinclude

# .clang-tidy (configuracao de checks)
# Checks: cppcoreguidelines-*, modernize-*, performance-*, bugprone-*

# cppcheck — analise estatica alternativa
cppcheck --enable=all --std=c++23 --suppress=missingInclude src/

GDB e Profiling

GDB — comandos essenciais e perf para profiling
# Compilar com simbolos de debug
g++ -std=c++23 -g3 -O0 -o app main.cpp

# Iniciar GDB
gdb ./app

# Dentro do GDB:
(gdb) break main              # breakpoint na funcao main
(gdb) break calculadora.cpp:42 # breakpoint na linha 42
(gdb) run                     # executar
(gdb) next    (ou n)          # proxima linha (nao entra em funcao)
(gdb) step    (ou s)          # entra na funcao
(gdb) print valor             # imprime variavel
(gdb) print *ptr              # derreferencia ponteiro
(gdb) backtrace               # stack trace atual
(gdb) frame 2                 # navega para frame 2 do stack trace
(gdb) watch variavel          # watchpoint: para quando variavel muda
(gdb) continue (ou c)         # continua ate proximo breakpoint
(gdb) quit                    # sair

# Profiling com perf (Linux)
perf record -g ./app
perf report --stdio

# Profiling com gprof
g++ -std=c++23 -pg -O2 -o app main.cpp
./app
gprof app gmon.out | head -50

CI/CD com GitHub Actions

YAML — .github/workflows/ci.yml para C++ com CMake e vcpkg
name: CI C++

on: [push, pull_request]

env:
  VCPKG_ROOT: ${{ github.workspace }}/vcpkg
  CMAKE_BUILD_TYPE: Release

jobs:
  build:
    strategy:
      matrix:
        os: [ubuntu-24.04, windows-2022, macos-14]
        compiler: [gcc, clang]
        exclude:
          - os: windows-2022
            compiler: gcc

    runs-on: ${{ matrix.os }}

    steps:
      - uses: actions/checkout@v4
        with:
          submodules: true

      - name: Cache vcpkg
        uses: actions/cache@v4
        with:
          path: ${{ env.VCPKG_ROOT }}/installed
          key: vcpkg-${{ matrix.os }}-${{ hashFiles('vcpkg.json') }}

      - name: Install vcpkg
        run: |
          git clone https://github.com/microsoft/vcpkg $VCPKG_ROOT
          $VCPKG_ROOT/bootstrap-vcpkg.sh -disableMetrics

      - name: Configure
        run: |
          cmake -B build -G Ninja \
            -DCMAKE_BUILD_TYPE=$CMAKE_BUILD_TYPE \
            -DCMAKE_TOOLCHAIN_FILE=$VCPKG_ROOT/scripts/buildsystems/vcpkg.cmake

      - name: Build
        run: cmake --build build --parallel

      - name: Test
        run: ctest --test-dir build --output-on-failure

      - name: Upload coverage (opcional)
        uses: codecov/codecov-action@v4
