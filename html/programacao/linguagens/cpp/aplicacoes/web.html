C++ para Web

C++ e uma escolha incomum mas poderosa para backends de alta performance: Drogon (um dos frameworks HTTP mais rapidos do mundo), Crow (leve e expressivo), Boost.Beast para controle total do protocolo, e Emscripten para compilar C++ para WebAssembly e rodar no browser.

O Drogon consistentemente aparece no top-3 de benchmarks TechEmpower para linguagens nativas. Ideal para proxies, microsservicos de latencia critica e processamento de dados em tempo real.

Drogon — Framework HTTP Assincrono

C++ + Drogon — REST API com controllers, filtros e middlewares
#include <drogon/drogon.h>
#include <nlohmann/json.hpp>

using namespace drogon;
using json = nlohmann::json;

// Controller de produtos — herada de HttpController
class ProdutosController : public HttpController<ProdutosController> {
public:
  METHOD_LIST_BEGIN
    ADD_METHOD_TO(ProdutosController::listar,  "/api/v1/produtos",      Get);
    ADD_METHOD_TO(ProdutosController::criar,   "/api/v1/produtos",      Post);
    ADD_METHOD_TO(ProdutosController::buscar,  "/api/v1/produtos/{id}", Get);
    ADD_METHOD_TO(ProdutosController::remover, "/api/v1/produtos/{id}", Delete);
  METHOD_LIST_END

  // GET /api/v1/produtos
  void listar(const HttpRequestPtr& req,
              std::function<void(HttpResponsePtr)>&& cb) {
    json body = json::array();
    body.push_back({ {"id", 1}, {"nome", "C++ Book"}, {"preco", 89.90} });

    auto resp = HttpResponse::newHttpResponse();
    resp->setStatusCode(HttpStatusCode::k200OK);
    resp->setContentTypeCode(ContentType::CT_APPLICATION_JSON);
    resp->setBody(body.dump());
    cb(resp);
  }

  // POST /api/v1/produtos
  void criar(const HttpRequestPtr& req,
             std::function<void(HttpResponsePtr)>&& cb) {
    try {
      auto body = json::parse(req->body());
      auto nome  = body.at("nome").get<std::string>();
      auto preco = body.at("preco").get<double>();

      json respBody = { {"id", 99}, {"nome", nome}, {"preco", preco} };
      auto resp = HttpResponse::newHttpResponse();
      resp->setStatusCode(HttpStatusCode::k201Created);
      resp->setContentTypeCode(ContentType::CT_APPLICATION_JSON);
      resp->setBody(respBody.dump());
      cb(resp);
    } catch (const json::exception& e) {
      auto resp = HttpResponse::newHttpResponse();
      resp->setStatusCode(HttpStatusCode::k400BadRequest);
      resp->setBody(json{{ "erro", e.what() }}.dump());
      cb(resp);
    }
  }

  void buscar(const HttpRequestPtr& req,
              std::function<void(HttpResponsePtr)>&& cb,
              std::string id) {
    json respBody = { {"id", std::stoi(id)}, {"nome", "Produto"} };
    auto resp = HttpResponse::newHttpResponse();
    resp->setBody(respBody.dump());
    cb(resp);
  }

  void remover(const HttpRequestPtr&,
               std::function<void(HttpResponsePtr)>&& cb,
               std::string) {
    auto resp = HttpResponse::newHttpResponse();
    resp->setStatusCode(HttpStatusCode::k204NoContent);
    cb(resp);
  }
};

int main() {
  app().setLogLevel(trantor::Logger::kInfo)
       .addListener("0.0.0.0", 8080)
       .setThreadNum(std::thread::hardware_concurrency())
       .run();
}

Crow — Microframework Expressivo

C++ + Crow — rotas, middleware de autenticacao e WebSocket
#include <crow.h>

// Middleware de autenticacao Bearer
struct AuthMiddleware {
  struct context { std::string usuario; };

  void before_handle(crow::request& req, crow::response& res, context& ctx) {
    auto auth = req.get_header_value("Authorization");
    if (auth.substr(0, 7) != "Bearer ") {
      res.code = 401;
      res.end(R"({"erro":"nao autorizado"})");
    } else {
      ctx.usuario = auth.substr(7);   // simplificado
    }
  }
  void after_handle(crow::request&, crow::response&, context&) {}
};

int main() {
  crow::App<AuthMiddleware> app;
  app.loglevel(crow::LogLevel::Warning);

  // Rota publica
  CROW_ROUTE(app, "/health").methods("GET"_method)([] {
    return crow::response(200, R"({"status":"ok"})");
  });

  // Rota protegida — acessa contexto do middleware
  CROW_ROUTE(app, "/api/perfil").methods("GET"_method)
    ([&](crow::request& req) {
      auto& ctx = app.get_context<AuthMiddleware>(req);
      crow::json::wvalue body;
      body["usuario"] = ctx.usuario;
      return crow::response(body);
    });

  // WebSocket echo
  CROW_WEBSOCKET_ROUTE(app, "/ws")
    .onmessage([](crow::websocket::connection& conn, const std::string& msg, bool) {
      conn.send_text("echo: " + msg);
    })
    .onerror([](crow::websocket::connection&, const std::string& reason) {
      CROW_LOG_WARNING << "WS error: " << reason;
    });

  app.port(8080).multithreaded().run();
}

WebAssembly com Emscripten

C++ + Emscripten — compilar para WASM e expor funcoes ao JavaScript
// crypto.cpp — biblioteca de cifras compilada para WASM
#include <emscripten/emscripten.h>
#include <emscripten/bind.h>
#include <string>
#include <algorithm>

// XOR simples (exemplo didatico — nao use em producao)
std::string cifrar(std::string texto, char chave) {
  std::transform(texto.begin(), texto.end(), texto.begin(),
    [chave](char c) { return c ^ chave; });
  return texto;
}

// Fibonacci otimizado — exposto para JS
extern "C" {
  EMSCRIPTEN_KEEPALIVE
  long long fibonacci(int n) {
    if (n <= 1) return n;
    long long a = 0, b = 1;
    for (int i = 2; i <= n; ++i) {
      auto c = a + b; a = b; b = c;
    }
    return b;
  }
}

// Embind — expoe classes e funcoes C++ com binding automatico
EMSCRIPTEN_BINDINGS(crypto_module) {
  emscripten::function("cifrar", &cifrar);
}

// em++ -O3 --bind -o crypto.js crypto.cpp
//    -s WASM=1 -s EXPORTED_RUNTIME_METHODS=['ccall','cwrap']
