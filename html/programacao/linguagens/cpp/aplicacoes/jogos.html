C++ para Jogos

Desenvolvimento de jogos em C++ com SDL2 + OpenGL para renderizacao customizada, SFML para jogos 2D rapidos, Box2D para fisicas rigorosas, EnTT como arquitetura ECS (Entity Component System) de alta performance, e game loop profissional com timestep fixo.

C++ e a linguagem de jogos por excelencia — Unreal Engine, id Software (DOOM, Quake), Valve Source Engine e a maioria dos AAA sao escritos em C++. Controle de alocacao de memoria, cache-friendly data-oriented design (DOD) e SIMD sao diferenciais criticos nessa area.

SDL2 + OpenGL — Game Loop e Renderizacao

C++ + SDL2 — game loop com timestep fixo e input system
#include <SDL.h>
#include <SDL_mixer.h>
#include <array>
#include <chrono>
#include <print>
using namespace std::chrono;

class InputManager {
  std::array<bool, SDL_NUM_SCANCODES> curr_{}, prev_{};
public:
  void update() {
    prev_ = curr_;
    const uint8_t* state = SDL_GetKeyboardState(nullptr);
    for (int i = 0; i < SDL_NUM_SCANCODES; ++i)
      curr_[i] = state[i];
  }
  bool pressionado(SDL_Scancode k) const { return curr_[k]; }
  bool acabouDePressionar(SDL_Scancode k) const { return curr_[k] && !prev_[k]; }
  bool acabouDeSoltar(SDL_Scancode k) const { return !curr_[k] && prev_[k]; }
};

int main() {
  SDL_Init(SDL_INIT_VIDEO | SDL_INIT_AUDIO);
  Mix_OpenAudio(44100, MIX_DEFAULT_FORMAT, 2, 2048);
  Mix_AllocateChannels(16);  // 16 canais de audio simultaneos

  SDL_Window* win = SDL_CreateWindow(
    "Jogo C++", SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED,
    1280, 720, SDL_WINDOW_SHOWN);
  SDL_Renderer* ren = SDL_CreateRenderer(
    win, -1, SDL_RENDERER_ACCELERATED | SDL_RENDERER_PRESENTVSYNC);

  // Carregar assets
  Mix_Chunk* sfxPulo  = Mix_LoadWAV("pulo.wav");
  Mix_Music* bgmusic  = Mix_LoadMUS("musica.ogg");
  Mix_PlayMusic(bgmusic, -1);  // loop infinito

  InputManager input;
  bool rodando = true;

  // Fixed Timestep Loop — fisica a 60Hz, render desacoplado
  constexpr double DT    = 1.0 / 60.0;
  double acumulado       = 0.0;
  auto   tAnterior       = high_resolution_clock::now();

  float jogadorX = 100.0f, jogadorY = 400.0f;
  float velX = 0.0f, velY = 0.0f;
  bool  noChao = true;

  while (rodando) {
    auto tAgora = high_resolution_clock::now();
    double frameTempo = duration_cast<duration<double>>(tAgora - tAnterior).count();
    tAnterior = tAgora;
    if (frameTempo > 0.05) frameTempo = 0.05;  // spiral of death cap
    acumulado += frameTempo;

    // Processar eventos
    SDL_Event ev;
    while (SDL_PollEvent(&ev)) {
      if (ev.type == SDL_QUIT) rodando = false;
    }
    input.update();

    // Updates de fisica em passo fixo
    while (acumulado >= DT) {
      float spd = 200.0f * static_cast<float>(DT);
      velX = 0.0f;
      if (input.pressionado(SDL_SCANCODE_A)) velX = -spd;
      if (input.pressionado(SDL_SCANCODE_D)) velX =  spd;
      if (input.acabouDePressionar(SDL_SCANCODE_SPACE) && noChao) {
        velY = -500.0f;
        noChao = false;
        Mix_PlayChannel(-1, sfxPulo, 0);
      }
      velY += 1200.0f * static_cast<float>(DT);  // gravidade
      jogadorX += velX;
      jogadorY += velY * static_cast<float>(DT);
      if (jogadorY >= 400.0f) { jogadorY = 400.0f; velY = 0.0f; noChao = true; }
      acumulado -= DT;
    }

    // Render
    SDL_SetRenderDrawColor(ren, 15, 15, 30, 255);
    SDL_RenderClear(ren);
    SDL_Rect player{ static_cast<int>(jogadorX),
                      static_cast<int>(jogadorY), 40, 60 };
    SDL_SetRenderDrawColor(ren, 100, 180, 255, 255);
    SDL_RenderFillRect(ren, &player);
    SDL_RenderPresent(ren);
  }

  Mix_FreeChunk(sfxPulo);
  Mix_FreeMusic(bgmusic);
  Mix_CloseAudio();
  SDL_DestroyRenderer(ren);
  SDL_DestroyWindow(win);
  SDL_Quit();
}

EnTT — Entity Component System (ECS)

C++ + EnTT — registry, components POD, views e sistemas
#include <entt/entt.hpp>
#include <print>

// Components sao structs POD — sem heranca, sem virtual
struct Posicao  { float x, y; };
struct Velocidade{ float dx, dy; };
struct Saude    { int hp, maxHp; };
struct Inimigo  {};   // tag component — sem dados
struct Sprite   { int texId; float escala; };

// Sistema de movimento
void sistemaMovimento(entt::registry& reg, float dt) {
  // View: itera somente entidades com Posicao E Velocidade
  auto view = reg.view<Posicao, Velocidade>();
  for (auto [entity, pos, vel] : view.each()) {
    pos.x += vel.dx * dt;
    pos.y += vel.dy * dt;
  }
}

// Sistema de colisao com borda
void sistemaBorda(entt::registry& reg) {
  reg.view<Posicao, Velocidade>().each(
    [](Posicao& pos, Velocidade& vel) {
      if (pos.x < 0.0f)    { pos.x = 0.0f;   vel.dx *= -1.0f; }
      if (pos.x > 1280.0f) { pos.x = 1280.0f; vel.dx *= -1.0f; }
      if (pos.y < 0.0f)    { pos.y = 0.0f;   vel.dy *= -1.0f; }
      if (pos.y > 720.0f)  { pos.y = 720.0f; vel.dy *= -1.0f; }
    });
}

// Sistema de dano em area
void aplicarDanoAoRedor(entt::registry& reg, float cx, float cy, float raio, int dano) {
  auto view = reg.view<Saude, Posicao, Inimigo>();
  for (auto [e, saude, pos] : view.each()) {
    float dx = pos.x - cx, dy = pos.y - cy;
    if (dx * dx + dy * dy <= raio * raio) {
      saude.hp -= dano;
      if (saude.hp <= 0) reg.destroy(e);  // remove entidade morta
    }
  }
}

int main() {
  entt::registry reg;

  // Criar entidades
  auto jogador = reg.create();
  reg.emplace<Posicao>  (jogador, 640.0f, 360.0f);
  reg.emplace<Velocidade>(jogador, 150.0f, 0.0f);
  reg.emplace<Saude>    (jogador, 100, 100);
  reg.emplace<Sprite>   (jogador, 0, 1.0f);

  // Criar 100 inimigos de uma vez com construct_batch
  for (int i = 0; i < 100; ++i) {
    auto e = reg.create();
    reg.emplace<Posicao>  (e, static_cast<float>(i * 12), 200.0f);
    reg.emplace<Velocidade>(e, 50.0f, 30.0f);
    reg.emplace<Saude>    (e, 30, 30);
    reg.emplace<Inimigo>  (e);
  }

  // Game loop ECS
  float dt = 1.0f / 60.0f;
  for (int frame = 0; frame < 120; ++frame) {
    sistemaMovimento(reg, dt);
    sistemaBorda(reg);
    if (frame == 60)
      aplicarDanoAoRedor(reg, 640, 360, 200, 50);
  }

  std::println("Inimigos vivos: {}",
    reg.view<Inimigo>().size());
}

Box2D — Fisica 2D

C++ + Box2D v2 — mundo, corpos dinamicos/estaticos, joint e raycast
#include <box2d/box2d.h>
#include <print>

// Callback de contato — detecta colisoes
class MyContactListener : public b2ContactListener {
public:
  void BeginContact(b2Contact* c) override {
    void* a = c->GetFixtureA()->GetBody()->GetUserData().pointer;
    void* b = c->GetFixtureB()->GetBody()->GetUserData().pointer;
    std::println("Colisao entre {} e {}",
      reinterpret_cast<const char*>(a),
      reinterpret_cast<const char*>(b));
  }
};

int main() {
  // Mundo com gravidade
  b2World mundo({ 0.0f, -9.8f });
  MyContactListener cl;
  mundo.SetContactListener(&cl);

  // Chao estatico
  b2BodyDef chaoDef;
  chaoDef.position.Set(0.0f, -10.0f);
  chaoDef.userData.pointer = reinterpret_cast<uintptr_t>("chao");
  b2Body* chao = mundo.CreateBody(&chaoDef);
  b2PolygonShape chaoShape;
  chaoShape.SetAsBox(50.0f, 10.0f);
  chao->CreateFixture(&chaoShape, 0.0f);  // densidade 0 = estatico

  // Cubo dinamico
  b2BodyDef cuboDef;
  cuboDef.type = b2_dynamicBody;
  cuboDef.position.Set(0.0f, 10.0f);
  cuboDef.userData.pointer = reinterpret_cast<uintptr_t>("cubo");
  b2Body* cubo = mundo.CreateBody(&cuboDef);
  b2PolygonShape cuboShape;
  cuboShape.SetAsBox(1.0f, 1.0f);
  b2FixtureDef fixDef;
  fixDef.shape       = &cuboShape;
  fixDef.density     = 1.0f;
  fixDef.friction    = 0.3f;
  fixDef.restitution = 0.5f;  // elasticidade
  cubo->CreateFixture(&fixDef);

  // Simular 3 segundos (180 steps a 60hz)
  for (int i = 0; i < 180; ++i) {
    mundo.Step(1.0f / 60.0f, 8, 3);  // velocity/position iterations
  }

  b2Vec2 pos = cubo->GetPosition();
  std::println("Posicao final: ({:.2f}, {:.2f})", pos.x, pos.y);

  // Raycast — verifica se algo esta abaixo de um ponto
  struct RayHit : b2RayCastCallback {
    bool acertou = false;
    float ReportFixture(b2Fixture*, const b2Vec2&, const b2Vec2&, float) override {
      acertou = true; return 0.0f;  // para no primeiro hit
    }
  } rh;
  mundo.RayCast(&rh, { 0.0f, 50.0f }, { 0.0f, -50.0f });
  std::println("Raycast acertou: {}", rh.acertou);
}

SFML — Jogo 2D Rapido

C++ + SFML 2.6 — sprites, animacoes, tilemap e colisao AABB
#include <SFML/Graphics.hpp>
#include <SFML/Audio.hpp>
#include <print>

class Animacao {
  sf::Sprite&      sprite_;
  sf::IntRect       quadro_;
  int               nQuadros_, quadroAtual_{ 0 };
  float             intervalo_, tempoAcum_{ 0.0f };
public:
  Animacao(sf::Sprite& s, sf::IntRect inicio, int n, float fps)
    : sprite_(s), quadro_(inicio), nQuadros_(n), intervalo_(1.0f / fps) {}

  void update(float dt) {
    tempoAcum_ += dt;
    if (tempoAcum_ >= intervalo_) {
      tempoAcum_ -= intervalo_;
      quadroAtual_ = (quadroAtual_ + 1) % nQuadros_;
      quadro_.left = quadro_.width * quadroAtual_;
      sprite_.setTextureRect(quadro_);
    }
  }
};

int main() {
  sf::RenderWindow win({ 1280, 720 }, "SFML Game");
  win.setFramerateLimit(60);

  sf::Texture txPlayer, txTiles;
  txPlayer.loadFromFile("player.png");
  txTiles.loadFromFile("tileset.png");

  sf::Sprite sprPlayer(txPlayer);
  sprPlayer.setOrigin(16, 16);  // pivô no centro do sprite 32x32
  Animacao anim_idle (sprPlayer, { 0, 0, 32, 32 }, 4, 8.0f);
  Animacao anim_walk (sprPlayer, { 0, 32, 32, 32 }, 6, 12.0f);
  Animacao* animAtual = &anim_idle;

  sf::SoundBuffer buf;
  buf.loadFromFile("passo.wav");
  sf::Sound som(buf);

  sf::Clock clock;
  sf::Vector2f pos(200, 360);

  while (win.isOpen()) {
    float dt = clock.restart().asSeconds();

    sf::Event ev;
    while (win.pollEvent(ev)) {
      if (ev.type == sf::Event::Closed) win.close();
    }

    bool movendo = false;
    if (sf::Keyboard::isKeyPressed(sf::Keyboard::Left))  {
      pos.x -= 200.0f * dt; sprPlayer.setScale(-1.0f, 1.0f); movendo = true;
    }
    if (sf::Keyboard::isKeyPressed(sf::Keyboard::Right)) {
      pos.x += 200.0f * dt; sprPlayer.setScale( 1.0f, 1.0f); movendo = true;
    }
    animAtual = movendo ? &anim_walk : &anim_idle;
    animAtual->update(dt);
    sprPlayer.setPosition(pos);

    win.clear({ 20, 20, 40 });
    win.draw(sprPlayer);
    win.display();
  }
}
