C++ — Terminal e CLI

C++ para Terminal e CLI

Ferramentas de linha de comando em C++: CLI11 para parsing de argumentos complexos com subcomandos, FTXUI para interfaces de terminal (TUI) interativas e animadas, ncurses para controle total do terminal, termios para modo raw e leitura de teclas individuais, e ANSI escape codes para cor e formatacao.

C++ produz executaveis standalone sem runtime externo — ideal para ferramentas de sistema, compiladores, gerenciadores de pacotes e scripts de build. CLI11, FTXUI e cxxopts sao header-only ou de compilacao trivial, facilitando distribuicao.

CLI11 — Parsing de Argumentos

C++ + CLI11 — flags, opcoes, subcomandos, validadores e help automatico
#include <CLI/CLI.hpp>
#include <print>
#include <filesystem>

int main(int argc, char** argv) {
  CLI::App app{"dataproc — processador de dados v1.0"};
  app.set_version_flag("--version", "1.0.0");
  app.require_subcommand(1);  // exige exatamente 1 subcomando

  // === Subcomando: convert ===
  std::string inputFile, outputFile, formato = "json";
  bool        verboso = false;
  int         workers = 4;

  auto* conv = app.add_subcommand("convert", "Converter formato de arquivo");
  conv->add_option("-i,--input", inputFile, "Arquivo de entrada")
      ->required()
      ->check(CLI::ExistingFile);      // valida que o arquivo existe
  conv->add_option("-o,--output", outputFile, "Arquivo de saida")
      ->required();
  conv->add_option("-f,--format", formato, "Formato: json|csv|parquet")
      ->check(CLI::IsMember({"json", "csv", "parquet"}));
  conv->add_option("-w,--workers", workers, "Threads de trabalho [1-32]")
      ->check(CLI::Range(1, 32));
  conv->add_flag("-v,--verbose", verboso, "Saida detalhada");

  // === Subcomando: stats ===
  std::vector<std::string> arquivos;
  bool                    incluirMedia = true;

  auto* stats = app.add_subcommand("stats", "Calcular estatisticas");
  stats->add_option("arquivos", arquivos, "Arquivos de entrada (1+)")
       ->required()->check(CLI::ExistingFile);
  stats->add_flag("!--no-mean", incluirMedia, "Omitir media");

  // Callback executado se subcomando marcado
  conv->callback([&]() {
    std::println("Convertendo {} → {} (formato={}, workers={})",
      inputFile, outputFile, formato, workers);
    if (verboso) std::println("[verbose] Threads inicializadas");
  });

  stats->callback([&]() {
    for (const auto& f : arquivos)
      std::println("Analisando {} (media={})", f, incluirMedia);
  });

  CLI11_PARSE(app, argc, argv);
  return 0;
}

// Uso:
// ./dataproc convert -i dados.csv -o saida.json -f json -w 8 -v
// ./dataproc stats arquivo1.csv arquivo2.csv --no-mean
// ./dataproc --help | --version

FTXUI — Interface TUI Moderna

C++ + FTXUI — menu, input, progress bar, canvas e layout
#include <ftxui/component/component.hpp>
#include <ftxui/component/screen_interactive.hpp>
#include <ftxui/dom/elements.hpp>
#include <functional>
#include <string>
#include <thread>
using namespace ftxui;

int main() {
  auto screen = ScreenInteractive::Fullscreen();

  // Input de texto
  std::string entrada;
  auto inputComp = Input(&entrada, "Digite algo...");

  // Menu de selecao
  std::vector<std::string> itens = { "Abrir arquivo", "Processar", "Salvar", "Sair" };
  int menuSelecionado = 0;
  auto menuComp = Menu(&itens, &menuSelecionado);

  // Checkboxes
  bool optA = true, optB = false;
  auto cb1 = Checkbox("Compressao", &optA);
  auto cb2 = Checkbox("Criptografia", &optB);

  // Slider
  int  nivelSlider = 50;
  auto sliderComp  = Slider("Nivel: ", &nivelSlider, 0, 100, 1);

  // Progress bar animada em background thread
  float progresso = 0.0f;
  std::atomic<bool> rodando{ true };
  std::thread worker([&] {
    while (rodando && progresso < 1.0f) {
      progresso += 0.01f;
      screen.PostEvent(Event::Custom);  // redesenhar a UI
      std::this_thread::sleep_for(std::chrono::milliseconds(50));
    }
  });

  // Componente raiz com layout
  auto layout = Container::Vertical({
    inputComp,
    menuComp,
    Container::Horizontal({ cb1, cb2 }),
    sliderComp,
  });

  auto renderer = Renderer(layout, [&] {
    return vbox({
      text("=== DataProc TUI ===") | bold | center,
      separator(),
      hbox({ text(" Busca: "), inputComp->Render() | flex }),
      separator(),
      hbox({
        vbox({ text("Acoes") | bold, menuComp->Render() }) | flex,
        separator(),
        vbox({
          text("Opcoes") | bold,
          cb1->Render(), cb2->Render(),
          separator(),
          sliderComp->Render(),
          text("Progresso:"),
          gauge(progresso) | color(Color::GreenLight),
        }) | flex
      }),
      separator(),
      text("[q] sair  [Tab] navegar  [Enter] selecionar")
        | dim | center,
    }) | border;
  });

  auto comp = CatchEvent(renderer, [&](Event ev) {
    if (ev == Event::Character('q')) { rodando = false; screen.ExitLoopClosure()(); return true; }
    return false;
  });

  screen.Loop(comp);
  if (worker.joinable()) worker.join();
}

termios — Raw Mode e ANSI

C++ + termios — leitura tecla a tecla, ANSI colors e cursor
#include <termios.h>
#include <unistd.h>
#include <print>
#include <string_view>

// RAII para ativar/desativar raw mode
class RawMode {
  termios orig_;
public:
  RawMode() {
    tcgetattr(STDIN_FILENO, &orig_);
    termios raw = orig_;
    raw.c_lflag &= ~(ICANON | ECHO);  // sem buffer de linha, sem echo
    raw.c_cc[VMIN]  = 1;  // bloqueia ate 1 caracter
    raw.c_cc[VTIME] = 0;
    tcsetattr(STDIN_FILENO, TCSANOW, &raw);
  }
  ~RawMode() { tcsetattr(STDIN_FILENO, TCSANOW, &orig_); }
};

// ANSI escape helper
namespace ansi {
  constexpr std::string_view reset   = "\033[0m";
  constexpr std::string_view bold    = "\033[1m";
  constexpr std::string_view vermelho = "\033[31m";
  constexpr std::string_view verde   = "\033[32m";
  constexpr std::string_view amarelo = "\033[33m";
  constexpr std::string_view azul    = "\033[34m";
  constexpr std::string_view ciano   = "\033[36m";
  constexpr std::string_view clearScr = "\033[2J\033[H";
  constexpr std::string_view cursorHide = "\033[?25l";
  constexpr std::string_view cursorShow = "\033[?25h";

  inline void moveto(int row, int col) {
    std::print("\033[{};{}H", row, col);
  }
  inline void cores256(int fg, int bg) {
    std::print("\033[38;5;{}m\033[48;5;{}m", fg, bg);
  }
}

int main() {
  std::print("{}", ansi::clearScr);
  std::print("{}", ansi::cursorHide);

  // Gradiente de cor 256
  for (int i = 0; i < 36; ++i) {
    ansi::cores256(255, 16 + i);
    std::print("  ");
  }
  std::println("{}", ansi::reset);

  std::println("{}{}Bem-vindo ao raw mode! Pressione 'q' para sair.{}{}",
    ansi::bold, ansi::ciano, ansi::reset, ansi::cursorShow);

  RawMode rm;  // ativa raw mode — desativado no destrutor
  char ch;
  while (std::read(STDIN_FILENO, &ch, 1) == 1 && ch != 'q') {
    if (ch == '\033') {
      // Sequencia de escape (setas: \033[A/B/C/D)
      char seq[3]{};
      read(STDIN_FILENO, seq, 2);
      if (seq[0] == '[') {
        switch (seq[1]) {
          case 'A': std::println("{}Seta para cima{}",   ansi::verde,   ansi::reset); break;
          case 'B': std::println("{}Seta para baixo{}",  ansi::amarelo, ansi::reset); break;
          case 'C': std::println("{}Seta direita{}",     ansi::azul,    ansi::reset); break;
          case 'D': std::println("{}Seta esquerda{}",    ansi::ciano,   ansi::reset); break;
        }
      }
    } else {
      std::println("{}'{}' (ASCII {}){}",
        ansi::vermelho, ch, static_cast<int>(ch), ansi::reset);
    }
  }
  println("{}Saindo...{}", ansi::amarelo, ansi::reset);
}

Subprocessos e Pipes

C++ — popen, pipe() + fork(), e captura de output de processo filho
#include <cstdio>
#include <array>
#include <string>
#include <memory>
#include <stdexcept>
#include <print>

// Executar comando e capturar stdout como string
[[nodiscard]]
std::string exec(std::string_view cmd) {
  // NOTA: nunca passe entrada do usuario diretamente aqui
  // Valide e sanitize antes de montar o comando
  std::unique_ptr<FILE, decltype(&pclose)> pipe{
    popen(cmd.data(), "r"),
    &pclose
  };
  if (!pipe) throw std::runtime_error("popen falhou");

  std::string resultado;
  std::array<char, 4096> buf;
  while (fgets(buf.data(), buf.size(), pipe.get()) != nullptr)
    resultado += buf.data();
  return resultado;
}

// Obter versao do compilador — chamada com argumentos fixos (seguro)
void infoSistema() {
  std::print("GCC: {}", exec("g++ --version 2>&1 | head -1"));
  std::print("OS : {}", exec("uname -sr"));
  std::print("CPU: {}", exec("nproc"));
}

// Versao segura com execv sem shell (evita injection)
#include <unistd.h>
#include <sys/wait.h>

int executarSeguro(const char* prog, std::vector<const char*> args) {
  args.push_back(nullptr);  // NULL sentinel do execv
  pid_t pid = fork();
  if (pid == 0) {
    // processo filho
    execv(prog, const_cast<char* const*>(args.data()));
    _exit(127);  // execv so retorna em falha
  }
  int status;
  waitpid(pid, &status, 0);
  return WEXITSTATUS(status);
}

int main() {
  infoSistema();
  // Compilar e executar arquivo — sem shell, sem injection
  int r = executarSeguro("/usr/bin/g++",
    { "/usr/bin/g++", "-std=c++23", "-O2", "main.cpp", "-o", "main" });
  std::println("g++ saiu com codigo {}", r);
}
Injecao de comandos: Nunca passe dados de entrada do usuario para popen(), system() ou execlp() sem validacao rigorosa. Use execv() com vetor de argumentos para executar processos sem invocar o shell — isso elimina o risco de injecao de comandos via caracteres especiais como ;, & ou `.
