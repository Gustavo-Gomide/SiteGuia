C++ — Mobile

C++ para Mobile

Desenvolvimento mobile nativo em C++: Android NDK com JNI para expor logica C++ ao Java/Kotlin, iOS com Objective-C++ (.mm) para bridge com Swift/Obj-C, SDL2 para apps crossplatform Android + iOS a partir do mesmo codigo C++, e boas praticas de bridge de tipos e gerenciamento de memoria.

C++ em mobile e mais comum do que parece: jogos Unity incluem plugins nativos em C++, aplicativos de processamento de imagem usam OpenCV via NDK, e motores como Godot, Cocos2d-x e Defold compilam para Android e iOS com C++ nativo.

Android NDK — JNI

C++ + JNI — expondo funcoes C++ para Java/Kotlin no Android
// native-lib.cpp
#include <jni.h>
#include <string>
#include <cmath>
#include <vector>
#include <android/log.h>

#define LOGI(...) __android_log_print(ANDROID_LOG_INFO, "NativeLib", __VA_ARGS__)
#define LOGE(...) __android_log_print(ANDROID_LOG_ERROR, "NativeLib", __VA_ARGS__)

// Nomenclatura obrigatoria: Java_PACOTE_CLASSE_METODO
extern "C" {

// String simples
JNIEXPORT jstring JNICALL
Java_com_exemplo_app_MainViewModel_versaoNativa(
    JNIEnv*  env,
    jobject  /* this */)
{
  return env->NewStringUTF("Nativo C++17 — NDK r26");
}

// Operacao intensiva: FFT simplificada, retorna array de floats
JNIEXPORT jfloatArray JNICALL
Java_com_exemplo_app_AudioProcessor_calcularEspectro(
    JNIEnv*    env,
    jobject    /* this */,
    jfloatArray amostras,
    jint       tamanho)
{
  if (!amostras || tamanho <= 0) return nullptr;

  // Acesso aos bytes Java sem copia (GetPrimitiveArrayCritical)
  jfloat* data = static_cast<jfloat*>(
    env->GetPrimitiveArrayCritical(amostras, nullptr));
  if (!data) return nullptr;

  std::vector<float> espectro(tamanho / 2);
  for (int k = 0; k < tamanho / 2; ++k) {
    double re = 0, im = 0;
    for (int n = 0; n < tamanho; ++n) {
      double theta = 2.0 * M_PI * k * n / tamanho;
      re += data[n] * std::cos(theta);
      im -= data[n] * std::sin(theta);
    }
    espectro[k] = static_cast<float>(std::sqrt(re * re + im * im));
  }
  // Liberar ponteiro critico ASAP (nao alocar objetos Java entre Get/Release)
  env->ReleasePrimitiveArrayCritical(amostras, data, JNI_ABORT);

  jfloatArray resultado = env->NewFloatArray(tamanho / 2);
  env->SetFloatArrayRegion(resultado, 0, tamanho / 2, espectro.data());
  LOGI("Espectro calculado: %d bins", tamanho / 2);
  return resultado;
}

// Passar objeto Java para C++ e acessar campos com reflection
JNIEXPORT jdouble JNICALL
Java_com_exemplo_app_Calculator_computar(
    JNIEnv* env, jobject, jobject config)
{
  jclass  cls = env->GetObjectClass(config);
  jfieldID fX = env->GetFieldID(cls, "valorX", "D");  // D = double
  jfieldID fY = env->GetFieldID(cls, "valorY", "D");
  double x = env->GetDoubleField(config, fX);
  double y = env->GetDoubleField(config, fY);
  return std::hypot(x, y);
}

} // extern "C"

iOS — Objective-C++ Bridge

Objective-C++ (.mm) — expondo C++ para Swift/Objective-C
// ImageProcessor.hpp — interface limpa, sem dependencias C++
#pragma once
#include <stdint.h>

#ifdef __cplusplus
extern "C" {
#endif

typedef struct { float r, g, b; } RGBColor;

// Calcular cor media de uma imagem RGBA
RGBColor calcularCorMedia(const uint8_t* pixels, int largura, int altura);

#ifdef __cplusplus
}
#endif

// --- ImageProcessor.mm (compila C++ + Obj-C) ---
#include "ImageProcessor.hpp"
#import <Foundation/Foundation.h>
#include <numeric>

RGBColor calcularCorMedia(const uint8_t* pixels, int w, int h) {
  long long sr = 0, sg = 0, sb = 0;
  int total = w * h;
  for (int i = 0; i < total; ++i) {
    sr += pixels[i * 4 + 0];  // R
    sg += pixels[i * 4 + 1];  // G
    sb += pixels[i * 4 + 2];  // B
    // ignora canal Alpha [3]
  }
  return {
    static_cast<float>(sr) / total / 255.0f,
    static_cast<float>(sg) / total / 255.0f,
    static_cast<float>(sb) / total / 255.0f
  };
}

// Wrapper Obj-C — expoe para Swift sem expor header C++
@interface ImageProcessorBridge : NSObject
+ (NSArray<NSNumber*>*)corMediaDePixels:(const uint8_t*)pixels
                                  largura:(int)w
                                   altura:(int)h;
@end

@implementation ImageProcessorBridge
+ (NSArray<NSNumber*>*)corMediaDePixels:(const uint8_t*)pixels
                                  largura:(int)w
                                   altura:(int)h {
  RGBColor cor = calcularCorMedia(pixels, w, h);
  return @[@(cor.r), @(cor.g), @(cor.b)];
}
@end
Swift — chamando C++ via Objective-C bridge
// ViewModel.swift
import UIKit

class ImageViewModel: ObservableObject {
  @Published var corMedia: UIColor = .gray

  func processarImagem(_ image: UIImage) {
    guard let cgImage = image.cgImage else { return }
    let w = cgImage.width, h = cgImage.height
    var pixels = [UInt8](repeating: 0, count: w * h * 4)

    pixels.withUnsafeMutableBytes { ptr in
      let ctx = CGContext(
        data: ptr.baseAddress, width: w, height: h,
        bitsPerComponent: 8, bytesPerRow: w * 4,
        space: CGColorSpaceCreateDeviceRGB(),
        bitmapInfo: CGImageAlphaInfo.premultipliedLast.rawValue)
      ctx?.draw(cgImage, in: CGRect(x: 0, y: 0, width: w, height: h))
    }

    pixels.withUnsafeBytes { rawBuf in
      let rgba = rawBuf.bindMemory(to: UInt8.self).baseAddress!
      let rgb = ImageProcessorBridge.corMedia(dePixels: rgba,
                   largura: Int32(w), altura: Int32(h))
      self.corMedia = UIColor(
        red:   CGFloat(rgb[0].floatValue),
        green: CGFloat(rgb[1].floatValue),
        blue:  CGFloat(rgb[2].floatValue), alpha: 1)
    }
  }
}

SDL2 — Crossplatform Android + iOS

C++ + SDL2 — main portavel que roda em Android, iOS, Desktop
#include <SDL.h>
#include <SDL_ttf.h>
#include <print>

// SDL2 redireciona main() para SDL_main em Android/iOS
int SDL_main(int argc, char* argv[]) {
  SDL_Init(SDL_INIT_VIDEO | SDL_INIT_EVENTS);
  TTF_Init();

  // SDL_WINDOW_FULLSCREEN_DESKTOP: adapta ao tamanho da tela do dispositivo
  SDL_Window* win = SDL_CreateWindow(
    "App", SDL_WINDOWPOS_UNDEFINED, SDL_WINDOWPOS_UNDEFINED,
    0, 0, SDL_WINDOW_FULLSCREEN_DESKTOP | SDL_WINDOW_ALLOW_HIGHDPI);
  SDL_Renderer* ren = SDL_CreateRenderer(
    win, -1, SDL_RENDERER_ACCELERATED | SDL_RENDERER_PRESENTVSYNC);

  // Escala logica independente de DPI — 1280x720 logico, qualquer tela
  SDL_RenderSetLogicalSize(ren, 1280, 720);

  int largura, altura;
  SDL_GetWindowSize(win, &largura, &altura);
  std::println("Tela: {}x{}", largura, altura);

  bool rodando = true;
  float btnX = 540, btnY = 300;  // botao central em coords logicas

  while (rodando) {
    SDL_Event ev;
    while (SDL_PollEvent(&ev)) {
      switch (ev.type) {
        case SDL_QUIT:
          rodando = false;
          break;
        case SDL_APP_WILLENTERBACKGROUND:
          // Pausar audio, salvar estado
          std::println("App: background");
          break;
        case SDL_APP_DIDENTERFOREGROUND:
          std::println("App: foreground");
          break;
        case SDL_FINGERDOWN: {
          // Touch: coordenadas normalizadas 0..1, converter para logico
          float tx = ev.tfinger.x * 1280;
          float ty = ev.tfinger.y * 720;
          if (tx > btnX && tx < btnX + 200 && ty > btnY && ty < btnY + 80)
            std::println("Botao tocado!");
          break;
        }
      }
    }
    SDL_SetRenderDrawColor(ren, 20, 20, 40, 255);
    SDL_RenderClear(ren);

    SDL_FRect botao{ btnX, btnY, 200, 80 };
    SDL_SetRenderDrawColor(ren, 80, 150, 255, 255);
    SDL_RenderFillRectF(ren, &botao);
    SDL_RenderPresent(ren);
  }

  TTF_Quit();
  SDL_DestroyRenderer(ren);
  SDL_DestroyWindow(win);
  SDL_Quit();
  return 0;
}

JNI — Boas Praticas e Armadilhas

Correto

  • Usar GetPrimitiveArrayCritical para arrays grandes sem copia
  • Liberar referencias com DeleteLocalRef em loops longos
  • Cachear jclass e jmethodID no JNI_OnLoad
  • Usar extern "C" para evitar name mangling
  • Validar todos os parametros vindos do Java antes de usar

Evitar

  • Alocar objetos Java entre GetCritical/ReleaseCritical
  • Chamar JNI de threads criadas em C++ sem AttachCurrentThread
  • Passar ponteiros C++ como long sem verificar ABI (32 vs 64 bit)
  • Usar strings Java como char* sem GetStringUTFChars
  • Esqueceu ReleaseStringUTFChars — memory leak garantido
C++ + JNI — cache de IDs e AttachCurrentThread em std::thread
static JavaVM*   gJVM  = nullptr;
static jmethodID gOnDadoID = nullptr;
static jclass    gCallbackClass = nullptr;

// Chamado pelo Android quando a biblioteca e carregada
JNIEXPORT jint JNICALL JNI_OnLoad(JavaVM* vm, void*) {
  gJVM = vm;
  JNIEnv* env;
  vm->GetEnv((void**)&env, JNI_VERSION_1_6);

  // Cachear class/methodID — chamadas de Get*ID sao lentas
  jclass local = env->FindClass("com/exemplo/app/NativeCallback");
  gCallbackClass = (jclass)env->NewGlobalRef(local);
  gOnDadoID = env->GetMethodID(gCallbackClass, "onDado", "([F)V");
  env->DeleteLocalRef(local);
  return JNI_VERSION_1_6;
}

// Chamar de background thread C++ (exige attach/detach)
void notificarDeThread(jobject cb, std::vector<float> dados) {
  JNIEnv* env;
  bool attached = false;
  jint res = gJVM->GetEnv((void**)&env, JNI_VERSION_1_6);
  if (res == JNI_EDETACHED) {
    gJVM->AttachCurrentThread(&env, nullptr);
    attached = true;
  }

  jfloatArray arr = env->NewFloatArray(static_cast<jsize>(dados.size()));
  env->SetFloatArrayRegion(arr, 0, dados.size(), dados.data());
  env->CallVoidMethod(cb, gOnDadoID, arr);
  env->DeleteLocalRef(arr);

  if (attached) gJVM->DetachCurrentThread();
}
