C++ para GUI Desktop

Desenvolvimento de interfaces graficas desktop em C++: Qt6 com QML (frontend declarativo) e Widgets (API classica), Dear ImGui para paineis de debug e ferramentas internas, wxWidgets para apps nativas multiplataformas, e FLTK para UIs ultra-leves em sistemas com poucos recursos.

Qt6 e o framework mais abrangente para C++ desktop: suporta Windows, macOS, Linux, iOS, Android e WebAssembly a partir do mesmo codigo. ImGui e o padrao de fato para paineis de depuracao em motores de jogos e editores de ferramentas.

Qt6 — Widgets e Signals/Slots

C++ + Qt6 Widgets — MainWindow, actions, model-view, threading
// mainwindow.h
#pragma once
#include <QMainWindow>
#include <QTableView>
#include <QStandardItemModel>
#include <QPushButton>
#include <QFuture>

class MainWindow : public QMainWindow {
  Q_OBJECT
  QTableView*         tabela_;
  QStandardItemModel* modelo_;
  QPushButton*        btnCarregar_;

public:
  explicit MainWindow(QWidget* parent = nullptr);

private slots:
  void onCarregarClicado();
  void onDadosCarregados(const QList<QStringList>&);
};

// mainwindow.cpp
#include "mainwindow.h"
#include <QVBoxLayout>
#include <QProgressDialog>
#include <QtConcurrent>
#include <QMessageBox>

MainWindow::MainWindow(QWidget* parent) : QMainWindow(parent) {
  auto* central = new QWidget(this);
  auto* layout  = new QVBoxLayout(central);

  btnCarregar_ = new QPushButton("Carregar dados", central);
  tabela_      = new QTableView(central);
  modelo_      = new QStandardItemModel(this);

  tabela_->setModel(modelo_);
  tabela_->horizontalHeader()->setStretchLastSection(true);
  layout->addWidget(btnCarregar_);
  layout->addWidget(tabela_);
  setCentralWidget(central);

  // Lambda como slot (C++14 lambda em connects)
  connect(btnCarregar_, &QPushButton::clicked,
          this, &MainWindow::onCarregarClicado);

  // Ação no menu
  auto* menuArq = menuBar()->addMenu("Arquivo");
  auto* actSair = menuArq->addAction("Sair", QKeySequence::Quit);
  connect(actSair, &QAction::triggered, this, &QWidget::close);
}

void MainWindow::onCarregarClicado() {
  btnCarregar_->setEnabled(false);

  // Carregar CSV em background thread via QtConcurrent::run
  QFuture<QList<QStringList>> futuro = QtConcurrent::run([]() {
    QList<QStringList> dados;
    QFile f("dados.csv");
    if (f.open(QIODevice::ReadOnly)) {
      QTextStream in(&f);
      while (!in.atEnd())
        dados << in.readLine().split(',');
    }
    return dados;
  });

  auto* watcher = new QFutureWatcher<QList<QStringList>>(this);
  connect(watcher, &QFutureWatcher<QList<QStringList>>::finished, this,
    [this, watcher]() {
      onDadosCarregados(watcher->result());
      watcher->deleteLater();
      btnCarregar_->setEnabled(true);
    });
  watcher->setFuture(futuro);
}

void MainWindow::onDadosCarregados(const QList<QStringList>& linhas) {
  modelo_->clear();
  for (const auto& linha : linhas) {
    QList<QStandardItem*> items;
    for (const auto& cel : linha)
      items.append(new QStandardItem(cel));
    modelo_->appendRow(items);
  }
}

Qt6 QML — Frontend Declarativo

C++ + QML — QAbstractListModel exposto para QML, propriedades Q_PROPERTY
// backend.h — Model exposto para QML
#pragma once
#include <QAbstractListModel>
#include <QQmlEngine>
#include <vector>

struct Tarefa { QString titulo; bool feita; };

class TarefaModel : public QAbstractListModel {
  Q_OBJECT
  QML_ELEMENT  // registra automaticamente para QML (Qt6)
  Q_PROPERTY(int pendentes READ pendentes NOTIFY pendentesChanged)
  std::vector<Tarefa> tarefas_;

public:
  enum Roles { TituloRole = Qt::UserRole + 1, FeitaRole };
  int rowCount(const QModelIndex&) const override { return tarefas_.size(); }
  QVariant data(const QModelIndex& index, int role) const override;
  QHash<int, QByteArray> roleNames() const override;
  int pendentes() const;

public slots:
  void adicionar(const QString& titulo);
  void alternarFeita(int idx);

signals:
  void pendentesChanged();
};

// main.cpp
#include <QGuiApplication>
#include <QQmlApplicationEngine>
#include "backend.h"

int main(int argc, char** argv) {
  QGuiApplication app(argc, argv);
  QQmlApplicationEngine engine;
  qmlRegisterType<TarefaModel>(
    "myapp", 1, 0, "TarefaModel");
  engine.loadFromModule("MinhaApp", "Main");
  return app.exec();
}

// Main.qml
import QtQuick 6.6
import QtQuick.Controls 6.6
import myapp 1.0

ApplicationWindow {
  title: "Tarefas"; width: 400; height: 600; visible: true

  TarefaModel { id: modelo }

  ColumnLayout {
    anchors.fill: parent; spacing: 8; padding: 12

    RowLayout {
      TextField {
        id: campo; Layout.fillWidth: true
        placeholderText: "Nova tarefa..."
        Keys.onReturnPressed: {
          modelo.adicionar(campo.text)
          campo.clear()
        }
      }
      Button { text: "+"; onClicked: { modelo.adicionar(campo.text); campo.clear() } }
    }

    Label {
      text: "Pendentes: " + modelo.pendentes
      color: modelo.pendentes > 0 ? "#ff6b6b" : "#6bcf7f"
      font.bold: true
    }

    ListView {
      Layout.fillWidth: true; Layout.fillHeight: true
      model: modelo; clip: true; spacing: 4

      delegate: Rectangle {
        width: ListView.view.width; height: 48
        color: feita ? "#1a2a1a" : "#1a1a2a"
        radius: 6

        RowLayout {
          anchors.fill: parent; anchors.margins: 8
          CheckBox {
            checked: feita
            onToggled: modelo.alternarFeita(index)
          }
          Label {
            text: titulo; Layout.fillWidth: true
            font.strikeout: feita; opacity: feita ? 0.5 : 1.0
          }
        }
      }
    }
  }
}

Dear ImGui — UI Imediata

C++ + Dear ImGui + SDL2 + OpenGL3 — painel de ferramentas completo
#include "imgui.h"
#include "imgui_impl_sdl2.h"
#include "imgui_impl_opengl3.h"
#include <SDL.h>
#include <print>

int main() {
  SDL_Init(SDL_INIT_VIDEO);
  SDL_GL_SetAttribute(SDL_GL_CONTEXT_VERSION_MAJOR, 3);
  SDL_GL_SetAttribute(SDL_GL_CONTEXT_VERSION_MINOR, 3);
  SDL_GL_SetAttribute(SDL_GL_CONTEXT_PROFILE_MASK,
    SDL_GL_CONTEXT_PROFILE_CORE);

  SDL_Window* win = SDL_CreateWindow(
    "Editor", SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED,
    1280, 720, SDL_WINDOW_OPENGL | SDL_WINDOW_RESIZABLE);
  SDL_GLContext gl_ctx = SDL_GL_CreateContext(win);

  IMGUI_CHECKVERSION();
  ImGui::CreateContext();
  ImGuiIO& io = ImGui::GetIO();
  io.ConfigFlags |= ImGuiConfigFlags_DockingEnable;
  io.ConfigFlags |= ImGuiConfigFlags_ViewportsEnable;
  ImGui::StyleColorsDark();

  ImGui_ImplSDL2_InitForOpenGL(win, gl_ctx);
  ImGui_ImplOpenGL3_Init("#version 330");

  bool rodando    = true;
  bool show_demo  = false;
  float cor[4]  = { 0.26f, 0.59f, 0.98f, 1.0f };
  char texto[256] = "Ola ImGui";
  float slider   = 0.5f;

  while (rodando) {
    SDL_Event ev;
    while (SDL_PollEvent(&ev)) {
      ImGui_ImplSDL2_ProcessEvent(&ev);
      if (ev.type == SDL_QUIT) rodando = false;
    }
    ImGui_ImplOpenGL3_NewFrame();
    ImGui_ImplSDL2_NewFrame();
    ImGui::NewFrame();

    // Docking fullscreen
    ImGui::DockSpaceOverViewport(ImGui::GetMainViewport());

    // Painel principal
    ImGui::Begin("Ferramenta");
    ImGui::Text("FPS: %.1f", io.Framerate);
    ImGui::SameLine();
    ImGui::Checkbox("Demo Window", &show_demo);

    ImGui::Separator();
    ImGui::InputText("Texto", texto, sizeof(texto));
    ImGui::ColorEdit4("Cor", cor);
    ImGui::SliderFloat("Valor", &slider, 0.0f, 1.0f);

    if (ImGui::Button("Aplicar"))
      std::println("Texto: {} Slider: {:.2f}", texto, slider);
    ImGui::End();

    if (show_demo) ImGui::ShowDemoWindow(&show_demo);

    // Plot de linha simples
    static float valores[90]{};
    static int  offset = 0;
    valores[offset] = io.Framerate;
    offset = (offset + 1) % 90;
    ImGui::Begin("Performance");
    ImGui::PlotLines("FPS", valores, 90, offset,
      nullptr, 0.0f, 120.0f, {0, 80});
    ImGui::End();

    ImGui::Render();
    glClearColor(0.1f, 0.1f, 0.1f, 1.0f);
    glClear(GL_COLOR_BUFFER_BIT);
    ImGui_ImplOpenGL3_RenderDrawData(ImGui::GetDrawData());
    SDL_GL_SwapWindow(win);
  }

  ImGui_ImplOpenGL3_Shutdown();
  ImGui_ImplSDL2_Shutdown();
  ImGui::DestroyContext();
  SDL_GL_DeleteContext(gl_ctx);
  SDL_DestroyWindow(win);
  SDL_Quit();
}

wxWidgets — Nativo Multiplataforma

C++ + wxWidgets — app completa com frame, menus, sizer e dialogo
#include <wx/wx.h>
#include <wx/filedlg.h>
#include <wx/spinctrl.h>

class MeuFrame : public wxFrame {
  wxTextCtrl* editor_;
  wxSpinCtrl* spin_;

public:
  MeuFrame() : wxFrame(nullptr, wxID_ANY, "Editor C++", 
      wxDefaultPosition, { 800, 600 }) {

    // Menu bar
    auto* mbar = new wxMenuBar;
    auto* mArq = new wxMenu;
    mArq->Append(wxID_OPEN,  "Abrir\tCtrl+O");
    mArq->Append(wxID_SAVE,  "Salvar\tCtrl+S");
    mArq->AppendSeparator();
    mArq->Append(wxID_EXIT,  "Sair");
    mbar->Append(mArq, "Arquivo");
    SetMenuBar(mbar);

    // Layout com BoxSizer
    auto* sizer = new wxBoxSizer(wxVERTICAL);

    editor_ = new wxTextCtrl(this, wxID_ANY, "",
      wxDefaultPosition, wxDefaultSize,
      wxTE_MULTILINE | wxTE_RICH2);
    sizer->Add(editor_, 1, wxEXPAND | wxALL, 5);

    auto* row = new wxBoxSizer(wxHORIZONTAL);
    row->Add(new wxStaticText(this, wxID_ANY, "Tamanho:"), 0, wxALIGN_CENTER);
    spin_ = new wxSpinCtrl(this, wxID_ANY, "12", wxDefaultPosition,
      { 60, -1 }, wxSP_ARROW_KEYS, 8, 72);
    row->Add(spin_, 0, wxLEFT, 5);
    auto* btnAplicar = new wxButton(this, wxID_APPLY, "Aplicar");
    row->Add(btnAplicar, 0, wxLEFT, 5);
    sizer->Add(row, 0, wxALIGN_RIGHT | wxALL, 5);

    SetSizerAndFit(sizer);
    CreateStatusBar(2);

    // Eventos
    Bind(wxEVT_MENU, [this](wxCommandEvent&) {
      wxFileDialog dlg(this, "Abrir arquivo", "", "",
        "Texto (*.txt)|*.txt", wxFD_OPEN | wxFD_FILE_MUST_EXIST);
      if (dlg.ShowModal() == wxID_OK)
        editor_->LoadFile(dlg.GetPath());
    }, wxID_OPEN);

    Bind(wxEVT_MENU, [this](wxCommandEvent&) {
      editor_->SaveFile();
    }, wxID_SAVE);

    Bind(wxEVT_MENU, [this](wxCommandEvent&) { Close(true); }, wxID_EXIT);
  }
};

class MinhaApp : public wxApp {
public:
  bool OnInit() override {
    (new MeuFrame)->Show();
    return true;
  }
};
wxIMPLEMENT_APP(MinhaApp);
