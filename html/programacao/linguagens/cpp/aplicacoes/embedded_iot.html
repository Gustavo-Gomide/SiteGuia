C++ — Embedded e IoT

C++ para Embedded e IoT

Programacao de sistemas embarcados e dispositivos IoT em C++: controle de GPIO com libgpiod, comunicacao serial POSIX (termios), protocolos I²C e SPI via ioctl, mensageria IoT com MQTT Paho C++, memory-mapped I/O e introducao ao FreeRTOS em ARM Cortex-M.

C++ e amplamente usado em sistemas embarcados por seu controle preciso de memoria, custo zero de abstracoes (zero-cost abstractions) e suporte a placas como Raspberry Pi, STM32 e ESP32. Templates e constexpr permitem configuracao em compile-time sem overhead de runtime.

GPIO com libgpiod

C++ + libgpiod — RAII wrapper, leitura de pino e polling de eventos
#include <gpiod.hpp>
#include <chrono>
#include <print>
using namespace std::chrono_literals;

int main() {
  // Abrir chip GPIO (Raspberry Pi 4: gpiochip0)
  gpiod::chip chip("gpiochip0");

  // Configurar pino 17 como saida (LED)
  gpiod::line led = chip.get_line(17);
  led.request({
    .consumer = "blink",
    .request_type = gpiod::line_request::DIRECTION_OUTPUT,
    .flags = 0
  }, 0);  // valor inicial = 0 (desligado)

  // Configurar pino 18 como entrada com pull-up (botao)
  gpiod::line btn = chip.get_line(18);
  btn.request({
    .consumer = "button",
    .request_type = gpiod::line_request::DIRECTION_INPUT,
    .flags = gpiod::line_request::FLAG_BIAS_PULL_UP
  });

  // Piscar LED 5 vezes, 500ms on/off
  for (int i = 0; i < 5; ++i) {
    led.set_value(1);
    std::this_thread::sleep_for(500ms);
    led.set_value(0);
    std::this_thread::sleep_for(500ms);
  }

  // Ler estado do botao
  int estado = btn.get_value();
  std::println("Botao: {}", estado == 0 ? "PRESSIONADO" : "SOLTO");

  // Configurar interrupcao de borda (rising + falling)
  btn.release();
  btn.request({
    .consumer = "button_irq",
    .request_type = gpiod::line_request::EVENT_BOTH_EDGES,
    .flags = gpiod::line_request::FLAG_BIAS_PULL_UP
  });

  while (true) {
    if (btn.event_wait(1s)) {
      auto event = btn.event_read();
      std::println("Evento: {} @ {} ns",
        event.type == gpiod::line_event::RISING_EDGE ? "RISING" : "FALLING",
        event.timestamp.count());
    }
  }
}

Comunicacao Serial (POSIX termios)

C++ + termios — UART raw mode, baud rate, leitura nao-bloqueante
#include <termios.h>
#include <fcntl.h>
#include <unistd.h>
#include <sys/select.h>
#include <stdexcept>
#include <span>
#include <print>

class SerialPort {
  int fd_{ -1 };

public:
  SerialPort(const char* device, int baud) {
    fd_ = open(device, O_RDWR | O_NOCTTY | O_NONBLOCK);
    if (fd_ < 0) throw std::runtime_error("Nao foi possivel abrir porta serial");

    termios tty{};
    tcgetattr(fd_, &tty);

    // Baud rate
    speed_t speed = (baud == 115200) ? B115200
                  : (baud == 9600)   ? B9600
                  : throw std::invalid_argument("baud nao suportado");
    cfsetispeed(&tty, speed);
    cfsetospeed(&tty, speed);

    // Raw mode: sem echo, sem sinais, 8N1
    cfmakeraw(&tty);
    tty.c_cflag |= (CLOCAL | CREAD);
    tty.c_cflag &= ~CSIZE;
    tty.c_cflag |= CS8;     // 8 bits
    tty.c_cflag &= ~PARENB; // sem paridade
    tty.c_cflag &= ~CSTOPB; // 1 stop bit

    // VMIN=0, VTIME=1 — leitura retorna em 100ms mesmo sem dados
    tty.c_cc[VMIN]  = 0;
    tty.c_cc[VTIME] = 1;

    if (tcsetattr(fd_, TCSANOW, &tty) != 0)
      throw std::runtime_error("tcsetattr falhou");
  }

  ~SerialPort() { if (fd_ >= 0) close(fd_); }

  [[nodiscard]] int ler(std::span<uint8_t> buf) {
    return static_cast<int>(read(fd_, buf.data(), buf.size()));
  }
  void escrever(std::span<const uint8_t> data) {
    write(fd_, data.data(), data.size());
  }
};

int main() {
  SerialPort porta("/dev/ttyAMA0", 115200);

  // Enviar comando AT para modulo GPS
  const uint8_t cmd[] = { 'A', 'T', '\r', '\n' };
  porta.escrever(cmd);

  std::array<uint8_t, 256> buf{};
  int n = porta.ler(buf);
  if (n > 0)
    std::println("Resposta ({} bytes): {:s}", n,
      std::string_view(reinterpret_cast<char*>(buf.data()), n));
}

I²C e SPI via ioctl

C++ + Linux ioctl — I2C e SPI em Raspberry Pi
#include <linux/i2c-dev.h>
#include <linux/spi/spidev.h>
#include <fcntl.h>
#include <unistd.h>
#include <sys/ioctl.h>
#include <print>

// ===================== I2C =====================
class I2CDevice {
  int fd_;
public:
  I2CDevice(const char* bus, uint8_t addr) {
    fd_ = open(bus, O_RDWR);
    if (fd_ < 0) throw std::runtime_error("I2C: falha ao abrir");
    if (ioctl(fd_, I2C_SLAVE, addr) < 0)
      throw std::runtime_error("I2C: falha ao setar slave");
  }
  ~I2CDevice() { close(fd_); }

  void escreverReg(uint8_t reg, uint8_t val) {
    uint8_t buf[2] = { reg, val };
    write(fd_, buf, sizeof(buf));
  }
  uint8_t lerReg(uint8_t reg) {
    write(fd_, &reg, 1);
    uint8_t val{};
    read(fd_, &val, 1);
    return val;
  }
};

// ===================== SPI =====================
class SPIDevice {
  int      fd_;
  uint32_t speed_;
public:
  SPIDevice(const char* dev, uint32_t hz)
    : fd_(open(dev, O_RDWR)), speed_(hz) {
    if (fd_ < 0) throw std::runtime_error("SPI: falha ao abrir");
    uint8_t mode  = SPI_MODE_0;
    uint8_t bits  = 8;
    ioctl(fd_, SPI_IOC_WR_MODE,           &mode);
    ioctl(fd_, SPI_IOC_WR_BITS_PER_WORD, &bits);
    ioctl(fd_, SPI_IOC_WR_MAX_SPEED_HZ,  &hz);
  }
  ~SPIDevice() { close(fd_); }

  void transferir(std::span<const uint8_t> tx, std::span<uint8_t> rx) {
    spi_ioc_transfer tr{
      .tx_buf = reinterpret_cast<uint64_t>(tx.data()),
      .rx_buf = reinterpret_cast<uint64_t>(rx.data()),
      .len    = static_cast<uint32_t>(tx.size()),
      .speed_hz = speed_,
      .bits_per_word = 8
    };
    ioctl(fd_, SPI_IOC_MESSAGE(1), &tr);
  }
};

int main() {
  // BME280: sensore I2C de temperatura/umidade/pressao
  I2CDevice bme("/dev/i2c-1", 0x76);
  uint8_t chip_id = bme.lerReg(0xD0);  // reg ID do BME280 = 0x60
  std::println("BME280 chip_id = {:#x}", chip_id);

  // MCP3204: ADC SPI de 4 canais
  SPIDevice adc("/dev/spidev0.0", 1'000'000);
  std::array<uint8_t, 3> tx{ 0x06, 0x00, 0x00 }; // canal 0 single-ended
  std::array<uint8_t, 3> rx{};
  adc.transferir(tx, rx);
  int valor = ((rx[1] & 0x0F) << 8) | rx[2];
  std::println("ADC CH0 = {} ({:.2f}V)", valor, valor * 3.3 / 4095.0);
}

MQTT com Eclipse Paho C++

C++ + Paho MQTT — publisher TLS, subscriber assíncrono e reconnect
#include <mqtt/async_client.h>
#include <print>
#include <thread>

class Callback : public mqtt::callback,
                   public mqtt::iaction_listener {
  mqtt::async_client& client_;
  mqtt::connect_options opts_;

  void connection_lost(const std::string& cause) override {
    std::println("Conexao perdida: {}. Reconectando...", cause);
    client_.reconnect();
  }
  void message_arrived(mqtt::const_message_ptr msg) override {
    std::println("[{}] {}", msg->get_topic(), msg->get_payload_str());
  }
  void on_failure(const mqtt::token& tok) override {
    std::println("Acao falhou, code={}", tok.get_return_code());
  }
  void on_success(const mqtt::token&) override {}

public:
  Callback(mqtt::async_client& c, mqtt::connect_options o)
    : client_(c), opts_(o) {}
  void connected(const std::string&) override {
    std::println("Conectado ao broker");
    client_.subscribe("sensores/#", 1);
  }
};

int main() {
  const std::string BROKER = "ssl://mqtt.exemplo.com:8883";
  const std::string ID     = "sensor-quarto-01";

  mqtt::async_client client(BROKER, ID);

  // Conexao com TLS
  mqtt::ssl_options ssl;
  ssl.set_trust_store("/etc/ssl/certs/ca-certificates.crt");
  ssl.set_verify(true);

  mqtt::connect_options opts;
  opts.set_ssl(ssl);
  opts.set_user_name("usuario");
  opts.set_password("senha_secreta");
  opts.set_keep_alive_interval(30);
  opts.set_automatic_reconnect(2, 30);  // min 2s, max 30s

  // LWT — Last Will Testament: broker publica se dispositivo desconectar abruptamente
  opts.set_will_message(mqtt::make_message(
    "sensores/quarto/status", "OFFLINE", 1, true));

  Callback cb{ client, opts };
  client.set_callback(cb);
  client.connect(opts, nullptr, cb);

  std::this_thread::sleep_for(std::chrono::seconds(1));

  // Publicar dado de sensor
  auto msg = mqtt::make_message(
    "sensores/quarto/temperatura", "{\"temp\": 23.5, \"umid\": 60}", 1, false);
  client.publish(msg)->wait_for(std::chrono::seconds(5));

  std::this_thread::sleep_for(std::chrono::seconds(10));
  client.disconnect()->wait();
}

Memory-Mapped I/O e Bare-Metal

C++ + mmap — acesso direto a registradores de hardware
#include <cstdint>
#include <fcntl.h>
#include <sys/mman.h>
#include <unistd.h>

// Registrador GPIO do Raspberry Pi 4 (BCM2711, base 0xFE200000)
static constexpr uint32_t GPIO_BASE    = 0xFE200000;
static constexpr uint32_t BLOCK_SIZE  = 0x1000;
static constexpr uint32_t GPFSEL1     = 0x04;  // funcao pinos 10-19
static constexpr uint32_t GPSET0      = 0x1C;  // set output high
static constexpr uint32_t GPCLR0      = 0x28;  // set output low

class DirectGPIO {
  volatile uint32_t* gpio_ = nullptr;
  int mem_fd_{ -1 };
  void* map_{ MAP_FAILED };

public:
  DirectGPIO() {
    mem_fd_ = open("/dev/gpiomem", O_RDWR | O_SYNC);
    if (mem_fd_ < 0) throw std::runtime_error("gpiomem nao disponivel");
    map_ = mmap(nullptr, BLOCK_SIZE,
                 PROT_READ | PROT_WRITE, MAP_SHARED,
                 mem_fd_, GPIO_BASE);
    if (map_ == MAP_FAILED) throw std::runtime_error("mmap falhou");
    gpio_ = static_cast<volatile uint32_t*>(map_);
  }
  ~DirectGPIO() {
    if (map_ != MAP_FAILED) munmap(map_, BLOCK_SIZE);
    if (mem_fd_ >= 0)   close(mem_fd_);
  }

  void setOutput(int pin) {
    int reg    = pin / 10;         // GPFSEL0..5
    int offset = (pin % 10) * 3;  // 3 bits por pino
    gpio_[reg] &= ~(7 << offset);
    gpio_[reg] |=  (1 << offset); // 001 = output
  }

  void high(int pin) { gpio_[GPSET0 / 4] = (1u << pin); }
  void low(int pin)  { gpio_[GPCLR0 / 4] = (1u << pin); }
};

int main() {
  DirectGPIO gpio;
  gpio.setOutput(17);    // PIN 17 como saida
  gpio.high(17);          // LED aceso
  usleep(500'000);
  gpio.low(17);           // LED apagado
}
Atencao: Acesso direto via /dev/mem ao endereco fisico exige privilegio root e pode travar o sistema. Prefira /dev/gpiomem (seguro, sem root) ou use libgpiod sempre que possivel.
