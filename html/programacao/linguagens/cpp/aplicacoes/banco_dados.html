C++ — Banco de Dados

C++ e Banco de Dados

Acesso a banco de dados em C++ com a biblioteca nativa do SQLite (C API + RAII wrapper), PostgreSQL via libpqxx, ORM com ODB, e integracoes nativas do framework Drogon para conexao com pool automatico e transacoes assincronas.

Em C++, o padrao e usar prepared statements para prevenir SQL injection — nunca concatenar strings de usuario em queries SQL. A biblioteca libpqxx e a soluicao moderna para PostgreSQL com suporte a C++17/20.

SQLite — RAII Wrapper

C++ + sqlite3 — wrapper RAII, prepared statements e transacoes
#include <sqlite3.h>
#include <memory>
#include <string>
#include <vector>
#include <stdexcept>
#include <print>

// RAII wrapper para sqlite3*
class Database {
  struct Closer {
    void operator()(sqlite3* p) const noexcept { sqlite3_close(p); }
  };
  std::unique_ptr<sqlite3, Closer> db_;

public:
  explicit Database(const std::string& path) {
    sqlite3* raw;
    if (sqlite3_open(path.c_str(), &raw) != SQLITE_OK) {
      throw std::runtime_error(sqlite3_errmsg(raw));
    }
    db_.reset(raw);
    // WAL mode — mais performance para leituras concorrentes
    executar("PRAGMA journal_mode = WAL");
    executar("PRAGMA foreign_keys = ON");
  }

  void executar(const std::string& sql) {
    char* errMsg = nullptr;
    if (sqlite3_exec(db_.get(), sql.c_str(), nullptr, nullptr, &errMsg) != SQLITE_OK) {
      std::string err{ errMsg };
      sqlite3_free(errMsg);
      throw std::runtime_error(err);
    }
  }

  sqlite3* get() const noexcept { return db_.get(); }
};

// RAII para sqlite3_stmt*
class Statement {
  struct Finalizer {
    void operator()(sqlite3_stmt* p) const noexcept { sqlite3_finalize(p); }
  };
  std::unique_ptr<sqlite3_stmt, Finalizer> stmt_;

public:
  Statement(Database& db, std::string_view sql) {
    sqlite3_stmt* raw;
    if (sqlite3_prepare_v2(db.get(), sql.data(),
        static_cast<int>(sql.size()), &raw, nullptr) != SQLITE_OK) {
      throw std::runtime_error(sqlite3_errmsg(db.get()));
    }
    stmt_.reset(raw);
  }

  void bind(int idx, std::string_view v) {
    sqlite3_bind_text(stmt_.get(), idx, v.data(), static_cast<int>(v.size()),
      SQLITE_TRANSIENT);  // SQLITE_TRANSIENT == SQLite copia o dado
  }
  void bind(int idx, double v)  { sqlite3_bind_double(stmt_.get(), idx, v); }
  void bind(int idx, int v)     { sqlite3_bind_int(stmt_.get(), idx, v); }

  bool step() { return sqlite3_step(stmt_.get()) == SQLITE_ROW; }
  void reset() { sqlite3_reset(stmt_.get()); }

  std::string texto(int col) {
    return reinterpret_cast<const char*>(sqlite3_column_text(stmt_.get(), col));
  }
  double real(int col)  { return sqlite3_column_double(stmt_.get(), col); }
  int    inteiro(int col) { return sqlite3_column_int(stmt_.get(), col); }
};

int main() {
  Database db(  "loja.db");
  db.executar(R"(
    CREATE TABLE IF NOT EXISTS produtos (
      id     INTEGER PRIMARY KEY AUTOINCREMENT,
      nome   TEXT    NOT NULL,
      preco  REAL    NOT NULL CHECK(preco > 0)
    )
  )");

  // INSERT com prepared statement — parametros via ?
  Statement ins{ db, "INSERT INTO produtos (nome, preco) VALUES (?, ?)" };
  ins.bind(1, "C++ Book");
  ins.bind(2, 89.90);
  ins.step();

  // SELECT com iteracao
  Statement sel{ db, "SELECT id, nome, preco FROM produtos" };
  while (sel.step()) {
    std::println("[{}] {} — R${:.2f}",
      sel.inteiro(0), sel.texto(1), sel.real(2));
  }

  // Transacao explicita
  db.executar("BEGIN");
  try {
    Statement upd{ db, "UPDATE produtos SET preco = preco * 1.1 WHERE preco < 100" };
    upd.step();
    db.executar("COMMIT");
  } catch (...) {
    db.executar("ROLLBACK");
    throw;
  }
}

PostgreSQL com libpqxx

C++ + libpqxx — conexao, transacao, prepared statements e pool
#include <pqxx/pqxx>
#include <print>
#include <vector>

struct Produto {
  int         id;
  std::string nome;
  double      preco;
};

int main() {
  // Connection string — nunca coloque credenciais em codigo
  pqxx::connection conn{
    "host=localhost port=5432 dbname=loja user=app password=senha sslmode=require" };

  // Preparar statement uma vez, reutilizar N vezes
  conn.prepare("inserir_produto",
    "INSERT INTO produtos (nome, preco) VALUES ($1, $2) RETURNING id");
  conn.prepare("listar_porpreco",
    "SELECT id, nome, preco FROM produtos WHERE preco < $1 ORDER BY preco");

  // Transacao de escrita (work)
  {
    pqxx::work txn{ conn };
    auto r = txn.exec_prepared("inserir_produto", "Effective C++", 120.0);
    int novoId = r[0][0].as<int>();
    std::println("Produto inserido com id={}", novoId);
    txn.commit();              // sem commit = rollback automatico no destrutor
  }

  // Transacao de leitura (nontransaction — sem overhead de transacao)
  {
    pqxx::nontransaction ntxn{ conn };
    auto rows = ntxn.exec_prepared("listar_porpreco", 200.0);
    std::vector<Produto> produtos;
    produtos.reserve(rows.size());
    for (const auto& row : rows) {
      produtos.push_back({
        row["id"].as<int>(),
        row["nome"].as<std::string>(),
        row["preco"].as<double>()
      });
    }
    for (const auto& [id, nome, preco] : produtos) {
      std::println("[{}] {} = R${:.2f}", id, nome, preco);
    }
  }

  // Transacao com savepoint (nested)
  {
    pqxx::work txn{ conn };
    txn.exec("UPDATE produtos SET preco = preco * 0.9");
    pqxx::subtransaction sub{ txn, "ajuste" };
    try {
      sub.exec("UPDATE produtos SET estoque = -1");  // viola CHECK
      sub.commit();
    } catch (...) {
      sub.abort();   // rollback so do savepoint, txn continua
    }
    txn.commit();
  }
}

Redis com hiredis

C++ + hiredis — SET/GET, pipelines e pub/sub
#include <hiredis/hiredis.h>
#include <memory>
#include <print>

int main() {
  // Conexao com RAII
  std::unique_ptr<redisContext,
    decltype(&redisFree)> ctx{
      redisConnect("127.0.0.1", 6379),
      &redisFree };

  if (ctx == nullptr || ctx->err) {
    throw std::runtime_error(ctx ? ctx->errstr : "falha ao alocar");
  }

  // SET com TTL
  std::unique_ptr<redisReply,
    decltype(&freeReplyObject)> reply{
      static_cast<redisReply*>(redisCommand(
        ctx.get(), "SET sessao:%s %s EX %d", "abc123", "user42", 3600)),
      &freeReplyObject };

  // GET
  reply.reset(static_cast<redisReply*>(
    redisCommand(ctx.get(), "GET sessao:abc123")));
  if (reply && reply->type == REDIS_REPLY_STRING) {
    std::println("sessao: {}", reply->str);
  }

  // Pipeline — envia multiplos comandos sem esperar resposta
  redisAppendCommand(ctx.get(), "INCR visitas");
  redisAppendCommand(ctx.get(), "EXPIRE visitas 86400");
  redisAppendCommand(ctx.get(), "GET visitas");

  redisReply* r;
  for (int i = 0; i < 3; ++i) {
    redisGetReply(ctx.get(), reinterpret_cast<void**>(&r));
    freeReplyObject(r);
  }
}
