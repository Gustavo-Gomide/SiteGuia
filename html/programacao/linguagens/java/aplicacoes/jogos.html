Java — Jogos

Java para Desenvolvimento de Jogos

LibGDX com architetura ApplicationListener, SpriteBatch, camera orto/perspectiva, fisica com Box2D, cenas com Scene2D UI, udio com Sound/Music, e grafico de baixo nivel com LWJGL3 e OpenGL moderno via shaders GLSL.

LibGDX e o framework Java mais usado para jogos 2D e permite deploy em Desktop (Windows/Linux/macOS), Android, iOS (via RoboVM) e HTML5 (via GWT) com o mesmo codigo-base.

Estrutura e Game Loop

Java + LibGDX — ApplicationListener e game loop principal
import com.badlogic.gdx.*;
import com.badlogic.gdx.graphics.*;
import com.badlogic.gdx.graphics.g2d.*;
import com.badlogic.gdx.utils.ScreenUtils;

public class MeuJogo implements ApplicationListener {

  private SpriteBatch        batch;
  private OrthographicCamera camera;
  private Texture            texNave;
  private Sprite             spriteNave;
  private float              x = 400, y = 300;

  @Override
  public void create() {
    // Chamado uma vez ao iniciar
    batch  = new SpriteBatch();
    camera = new OrthographicCamera();
    camera.setToOrtho(false, 800, 600);   // viewport 800x600 unidades logicas

    texNave     = new Texture("nave.png");
    spriteNave  = new Sprite(texNave);
    spriteNave.setSize(64, 64);
    spriteNave.setOriginCenter();
  }

  @Override
  public void render() {
    // Chamado ~60 vezes por segundo
    float delta = Gdx.graphics.getDeltaTime();   // tempo desde o ultimo frame

    processarEntrada(delta);

    ScreenUtils.clear(0, 0, 0.1f, 1);   // limpa buffer com azul escuro
    camera.update();
    batch.setProjectionMatrix(camera.combined);

    batch.begin();
    spriteNave.setPosition(x - spriteNave.getWidth()  / 2,
                             y - spriteNave.getHeight() / 2);
    spriteNave.draw(batch);
    batch.end();
  }

  private void processarEntrada(float delta) {
    float vel = 200 * delta;  // 200 unidades/segundo independente de FPS
    if (Gdx.input.isKeyPressed(Input.Keys.LEFT)  || Gdx.input.isKeyPressed(Input.Keys.A)) x -= vel;
    if (Gdx.input.isKeyPressed(Input.Keys.RIGHT) || Gdx.input.isKeyPressed(Input.Keys.D)) x += vel;
    if (Gdx.input.isKeyPressed(Input.Keys.UP)   || Gdx.input.isKeyPressed(Input.Keys.W)) y += vel;
    if (Gdx.input.isKeyPressed(Input.Keys.DOWN)  || Gdx.input.isKeyPressed(Input.Keys.S)) y -= vel;
    // Toque em dispositivos moveis
    if (Gdx.input.isTouched()) {
      var touch = camera.unproject(new Vector3(
        Gdx.input.getX(), Gdx.input.getY(), 0));
      x = touch.x; y = touch.y;
    }
  }

  @Override
  public void resize(int w, int h) {
    camera.setToOrtho(false, w, h);
  }

  @Override
  public void dispose() {
    // OBRIGATORIO — libera recursos de GPU/disco
    batch.dispose();
    texNave.dispose();
  }

  @Override public void pause()  {}
  @Override public void resume() {}
}

// Launcher Desktop
public class DesktopLauncher {
  public static void main(String[] args) {
    new Lwjgl3Application(new MeuJogo(), new Lwjgl3ApplicationConfiguration() {{
      setTitle("Meu Jogo");
      setWindowedMode(800, 600);
      setForegroundFPS(60);
      useVsync(true);
    }});
  }
}

Fisica com Box2D

Java + LibGDX Box2D — corpos fisicos, colisao e debug
import com.badlogic.gdx.physics.box2d.*;

public class FisicaJogo {

  private static final float PPM = 32f;   // pixels por metro
  private World            world;
  private Box2DDebugRenderer debugRenderer;
  private Body             jogador, chao;

  public void inicializar() {
    // Gravidade 9.8 m/s² para baixo
    world = new World(new Vector2(0, -9.8f), true);
    debugRenderer = new Box2DDebugRenderer();

    // Corpo estatico — chao
    var defChao = new BodyDef();
    defChao.type = BodyDef.BodyType.StaticBody;
    defChao.position.set(400 / PPM, 16 / PPM);
    chao = world.createBody(defChao);
    var formaChao = new PolygonShape();
    formaChao.setAsBox(400 / PPM, 16 / PPM);
    chao.createFixture(formaChao, 0f);
    formaChao.dispose();

    // Corpo dinamico — jogador
    var defJog = new BodyDef();
    defJog.type = BodyDef.BodyType.DynamicBody;
    defJog.position.set(400 / PPM, 200 / PPM);
    defJog.fixedRotation = true;   // personagem nao gira ao colidir
    jogador = world.createBody(defJog);
    var circulo = new CircleShape();
    circulo.setRadius(16 / PPM);
    var fd = new FixtureDef();
    fd.shape       = circulo;
    fd.density     = 1f;
    fd.friction    = 0.5f;
    fd.restitution = 0;   // sem quique
    jogador.createFixture(fd);
    circulo.dispose();

    // Deteccao de colisao com listener
    world.setContactListener(new ContactListener() {
      @Override
      public void beginContact(Contact c) {
        // Identifica os fixtures que colidiram
        var a = c.getFixtureA().getBody().getUserData();
        var b = c.getFixtureB().getBody().getUserData();
        System.out.println("Colisao: " + a + " x " + b);
      }
      @Override public void endContact(Contact c)                             {}
      @Override public void preSolve(Contact c, Manifold m)                  {}
      @Override public void postSolve(Contact c, ContactImpulse impulso)      {}
    });
  }

  public void atualizar(float delta) {
    // Passo de simulacao (8 iteracoes de velocidade, 3 de posicao = boa precisao)
    world.step(delta, 8, 3);

    // Pulo — so se estiver no chao
    if (Gdx.input.isKeyJustPressed(Input.Keys.SPACE)
        && Math.abs(jogador.getLinearVelocity().y) < 0.1f) {
      jogador.applyLinearImpulse(
        new Vector2(0, 500 / PPM), jogador.getWorldCenter(), true);
    }
  }
}

Scene2D — UI e Gerenciamento de Cenas

Java + LibGDX Scene2D — menu com Stage, botoes e transicoes
import com.badlogic.gdx.scenes.scene2d.*;
import com.badlogic.gdx.scenes.scene2d.ui.*;

public class MenuPrincipal extends ScreenAdapter {

  private Stage  stage;
  private Skin   skin;

  @Override
  public void show() {
    stage = new Stage(new FitViewport(800, 600));
    skin  = new Skin(Gdx.files.internal("ui/skin.json"));
    Gdx.input.setInputProcessor(stage);

    var tabela = new Table(skin);
    tabela.setFillParent(true);
    tabela.center();

    var titulo = new Label("Nave Espacial", skin, "title");
    var btnJogar  = new TextButton("Jogar",     skin);
    var btnOpcoes = new TextButton("Opcoes",    skin);
    var btnSair   = new TextButton("Sair",      skin);

    btnJogar.addListener(new ChangeListener() {
      @Override
      public void changed(ChangeEvent e, Actor a) {
        MeuJogo.instancia().setScreen(new TelaJogo());
      }
    });
    btnSair.addListener(new ChangeListener() {
      @Override
      public void changed(ChangeEvent e, Actor a) {
        Gdx.app.exit();
      }
    });

    tabela.add(titulo).padBottom(40).row();
    tabela.add(btnJogar).width(200).padBottom(10).row();
    tabela.add(btnOpcoes).width(200).padBottom(10).row();
    tabela.add(btnSair).width(200);
    stage.addActor(tabela);
  }

  @Override
  public void render(float delta) {
    ScreenUtils.clear(0, 0, 0, 1);
    stage.act(delta);
    stage.draw();
  }

  @Override
  public void dispose() {
    stage.dispose();
    skin.dispose();
  }
}

Assets, Audio e Gerenciamento de Recursos

Java + LibGDX — AssetManager, atlas de sprites e audio
public class GerenciadorAssets {

  private static final AssetManager manager = new AssetManager();

  public static void carregarTodos() {
    // Texturas
    manager.load("texturas/personagens.atlas", TextureAtlas.class);
    manager.load("texturas/tiles.atlas",         TextureAtlas.class);
    // Audio
    manager.load("audio/trilha.mp3", Music.class);
    manager.load("audio/tiro.wav",    Sound.class);
    // Bitmap font
    manager.load("fontes/ui.fnt", BitmapFont.class);
  }

  // Retorna progresso (0..1) p/ tela de carregamento
  public static float progresso() { return manager.getProgress(); }
  public static boolean atualizar() { return manager.update(); }

  public static <T> T get(String nome, Class<T> tipo) {
    return manager.get(nome, tipo);
  }
}

// Animacao a partir de TextureAtlas
var atlas  = GerenciadorAssets.get("texturas/personagens.atlas", TextureAtlas.class);
var frames = atlas.findRegions("correndo");  // correndo_00, correndo_01 ...
var animCorrer = new Animation<TextureRegion>(1f/12f, frames, Animation.PlayMode.LOOP);

// No render: acumula tempo e consulta o frame atual
private float tempoAnim = 0;
// dentro de render():
tempoAnim += delta;
TextureRegion frame = animCorrer.getKeyFrame(tempoAnim);
batch.draw(frame, x, y, 64, 64);

// Musica em loop
var musica = GerenciadorAssets.get("audio/trilha.mp3", Music.class);
musica.setLooping(true);
musica.setVolume(0.7f);
musica.play();

// Efeito sonoro (polux — pode tocar varios simultaneamente)
var sfxTiro = GerenciadorAssets.get("audio/tiro.wav", Sound.class);
sfxTiro.play(1f);   // volume 100%
