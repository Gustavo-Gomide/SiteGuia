Java — Terminal e CLI

Java para Terminal e CLI

Picocli para CLIs profissionais com subcomandos e autocompletion, JLine3 para REPLs interativos com historico e navegacao, Process API para executar e controlar processos externos, e boas praticas de I/O com fluxos tipados.

Com GraalVM Native Image, um CLI Java pode ser compilado para binario nativo com tempo de startup < 10ms, competindo diretamente com Go e Rust para ferramentas de linha de comando.

Picocli — CLI com Subcomandos

Java + Picocli — CLI com subcomandos, validacao e ajuda automatica
import picocli.CommandLine;
import picocli.CommandLine.*;

@Command(
  name = "loja",
  version = "1.0.0",
  mixinStandardHelpOptions = true,
  description = "CLI de gerenciamento da loja",
  subcommands = {
    ProdutoCmd.class,
    PedidoCmd.class,
    CommandLine.HelpCommand.class
  }
)
public class LojaApp implements Runnable {

  public static void main(String[] args) {
    int code = new CommandLine(new LojaApp())
      .setExecutionExceptionHandler((ex, cmd, pr) -> {
        cmd.getErr().println("Erro: " + ex.getMessage());
        return 1;
      })
      .execute(args);
    System.exit(code);
  }

  @Override
  public void run() {
    CommandLine.usage(this, System.out);
  }
}

@Command(name = "produto", description = "Gerencia produtos",
         subcommands = { ProdutoCmd.Listar.class, ProdutoCmd.Criar.class })
public class ProdutoCmd {

  @Command(name = "listar", description = "Lista produtos com filtros opcionais")
  static class Listar implements Runnable {
    @Option(names = { "-c", "--categoria" }, description = "Filtrar por categoria")
    String categoria;

    @Option(names = "--min-preco", defaultValue = "0")
    double minPreco;

    @Option(names = { "-o", "--output" },
           description = "Formato de saida: ${COMPLETION-CANDIDATES}")
    OutputFormat formato = OutputFormat.TABLE;

    @Spec CommandLine.Model.CommandSpec spec;

    @Override
    public void run() {
      var produtos = ProdutoService.listar(categoria, minPreco);
      if (formato == OutputFormat.JSON) {
        spec.commandLine().getOut().println(JsonUtil.toJson(produtos));
      } else {
        imprimirTabela(produtos, spec.commandLine().getOut());
      }
    }
  }

  @Command(name = "criar", description = "Cria um novo produto")
  static class Criar implements Runnable {
    @Parameters(index = "0", description = "Nome do produto")
    String nome;

    @Parameters(index = "1", description = "Preco (ex: 29.90)")
    BigDecimal preco;

    @Option(names = { "-e", "--estoque" }, defaultValue = "0")
    int estoque;

    @Override
    public void run() {
      var produto = ProdutoService.criar(nome, preco, estoque);
      System.out.printf("Produto criado: %s (id=%s)%n",
        produto.getNome(), produto.getId());
    }
  }
}
// loja produto listar --categoria eletronicos -o json
// loja produto criar "Notebook Pro" 4999.90 --estoque 10

JLine3 — REPL Interativo

Java + JLine3 — REPL com historico, autocompletion e colorido
import org.jline.reader.*;
import org.jline.reader.impl.completer.*;
import org.jline.terminal.*;
import org.jline.utils.AttributedStringBuilder;
import org.jline.utils.AttributedStyle;

public class LojaRepl {

  public static void main(String[] args) throws Exception {
    try (Terminal terminal = TerminalBuilder.terminal()) {

      // Completador com palavras-chave do sistema
      var completer = new AggregateCompleter(
        new StringsCompleter("buscar", "criar", "listar", "remover", "sair"),
        new ArgumentCompleter(
          new StringsCompleter("buscar"),
          new StringsCompleter("produto", "pedido", "cliente")
        )
      );

      LineReader reader = LineReaderBuilder.builder()
        .terminal(terminal)
        .completer(completer)
        .option(LineReader.Option.AUTO_FRESH_LINE, true)
        .variable(LineReader.HISTORY_FILE, ".loja_history")
        .build();

      // Prompt colorido com AttributedString
      String prompt = new AttributedStringBuilder()
        .style(AttributedStyle.DEFAULT.foreground(AttributedStyle.GREEN).bold())
        .append("loja")
        .style(AttributedStyle.DEFAULT)
        .append(" > ")
        .toAnsi();

      String linha;
      while (true) {
        try {
          linha = reader.readLine(prompt).trim();
        } catch (UserInterruptException e) {
          break;   // Ctrl+C
        } catch (EndOfFileException e) {
          break;   // Ctrl+D / EOF
        }

        if (linha.isBlank()) continue;
        if (linha.equalsIgnoreCase("sair")) break;

        var partes = linha.split("\\s+", 3);
        executarComando(partes, terminal.writer());
      }
    }
  }
}

Process API — Executar Processos Externos

Java — ProcessBuilder, captura de saida e timeout seguro
public class ExecutorComando {

  public record ResultadoProcesso(int codigo, String stdout, String stderr) {}

  public static ResultadoProcesso executar(List<String> comando)
      throws IOException, InterruptedException {

    // NUNCA use Runtime.exec(String) — vulneravel a injecao de comando
    // Sempre passe uma lista de argumentos separados
    var pb = new ProcessBuilder(comando)
      .redirectErrorStream(false);

    // Herda variaveis de ambiente do processo pai (nao expoe segredos extras)
    pb.environment().put("LANG", "pt_BR.UTF-8");
    pb.directory(new File(System.getProperty("user.dir")));

    var processo = pb.start();

    // Le stdout e stderr em threads separadas para nao bloquear
    var futureOut = CompletableFuture.supplyAsync(() -> {
      try (var br = new BufferedReader(
            new InputStreamReader(processo.getInputStream()))) {
        return br.lines().collect(Collectors.joining("\n"));
      } catch (IOException e) { return ""; }
    });
    var futureErr = CompletableFuture.supplyAsync(() -> {
      try (var br = new BufferedReader(
            new InputStreamReader(processo.getErrorStream())) ) {
        return br.lines().collect(Collectors.joining("\n"));
      } catch (IOException e) { return ""; }
    });

    // Timeout de seguranca — processo nao pode travar a JVM
    boolean concluiu = processo.waitFor(30, TimeUnit.SECONDS);
    if (!concluiu) {
      processo.destroyForcibly();
      throw new ProcessoExcedeuTimeoutException("Processo excedeu 30 segundos");
    }

    return new ResultadoProcesso(
      processo.exitValue(),
      futureOut.join(),
      futureErr.join()
    );
  }

  // Inspecionar processos em execucao (Java 9+)
  public static void listarProcessos() {
    ProcessHandle.allProcesses()
      .filter(ph -> ph.info().command().map(c -> c.contains("java")).orElse(false))
      .forEach(ph -> System.out.printf("PID: %d CMD: %s%n",
        ph.pid(),
        ph.info().commandLine().orElse("?")));
  }
}

Injecao de Comando: Nunca concatene entrada do usuario em strings de comando (ex: Runtime.exec("ls " + userInput)). Sempre use List<String> separando programa e argumentos. Valide e sanitize entradas antes de passa-las como argumento de processo.

I/O Padrao e Leitura de Arquivos

Java — Scanner, BufferedReader, Files NIO e pipes
public class IoExemplos {

  // Detectar se stdin e um pipe ou terminal interativo
  public static boolean ehPipe() {
    try {
      return System.in.available() > 0
        || System.console() == null;
    } catch (IOException e) { return false; }
  }

  // Processar linhas de stdin (pipe) de forma eficiente
  public static void processarStdin() throws IOException {
    try (var br = new BufferedReader(new InputStreamReader(System.in))) {
      br.lines()
        .filter(l -> !l.isBlank())
        .map(String::strip)
        .forEach(System.out::println);
    }
  }

  // Ler arquivo grande com Stream — nao carrega tudo na memoria
  public static long contarLinhas(Path arquivo) throws IOException {
    try (var linhas = Files.lines(arquivo, StandardCharsets.UTF_8)) {
      return linhas.count();
    }
  }

  // Escrever arquivo atomicamente (evita arquivo corrompido em crash)
  public static void escreverAtomic(Path destino, String conteudo)
      throws IOException {
    var tmp = Files.createTempFile(
      destino.getParent(), ".tmp_", ".atomic");
    try {
      Files.writeString(tmp, conteudo, StandardCharsets.UTF_8);
      Files.move(tmp, destino,
        StandardCopyOption.REPLACE_EXISTING,
        StandardCopyOption.ATOMIC_MOVE);
    } catch (Exception e) {
      Files.deleteIfExists(tmp);
      throw e;
    }
  }

  // Entrada de senha sem eco (Console.readPassword)
  public static char[] lerSenha(String prompt) {
    var console = System.console();
    if (console == null) {
      throw new IllegalStateException("Nao ha console interativo disponivel");
    }
    return console.readPassword("%s: ", prompt);
  }
}

GraalVM Native Image para CLI

Shell + Maven — compilar CLI Java para binario nativo com GraalVM
<!-- pom.xml — plugin graalvm-native -->
<plugin>
  <groupId>org.graalvm.buildtools</groupId>
  <artifactId>native-maven-plugin</artifactId>
  <version>0.10.2</version>
  <configuration>
    <imageName>loja-cli</imageName>
    <mainClass>com.loja.LojaApp</mainClass>
    <buildArgs>
      <buildArg>--no-fallback</buildArg>
      <buildArg>-H:+ReportExceptionStackTraces</buildArg>
    </buildArgs>
  </configuration>
</plugin>

# Compilar para nativo (requer GraalVM JDK 21+)
./mvnw -Pnative native:compile

# Binario gerado em target/loja-cli (~30MB, <10ms startup)
./target/loja-cli produto listar --categoria eletronicos

Arquitetura CLI para Ferramentas Profissionais

Padrao que funciona em equipe

  • Comandos pequenos e composiveis, com contrato claro de entrada e saida.
  • Camada de dominio separada da camada de terminal para reaproveitar logica em API ou job.
  • Codigo de erro padronizado e mensagens orientadas a automacao de scripts.
  • Modo verbose para diagnostico e modo silencioso para pipelines CI/CD.
