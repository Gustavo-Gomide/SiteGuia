REST API com Paginacao e HATEOAS

Java + Spring Boot 3 — Controller com paginacao e links hipermidia
import org.springframework.data.domain.*;
import org.springframework.hateoas.*;
import static org.springframework.hateoas.server.mvc.WebMvcLinkBuilder.*;

// DTO de resposta com suporte a HATEOAS
public class ProdutoModel extends RepresentationModel<ProdutoModel> {
  private UUID       id;
  private String     nome;
  private BigDecimal preco;
  private int        estoque;
  // getters + setters
}

@RestController
@RequestMapping("/api/v1/produtos")
public class ProdutoController {

  private final ProdutoService   service;
  private final ProdutoAssembler assembler;

  public ProdutoController(ProdutoService s, ProdutoAssembler a) {
    this.service   = s;
    this.assembler = a;
  }

  // GET /api/v1/produtos?page=0&size=20&sort=nome,asc
  @GetMapping
  public PagedModel<ProdutoModel> listar(
      @PageableDefault(size = 20, sort = "nome") Pageable pageable) {
    return assembler.toPagedModel(service.listarPaginado(pageable));
  }

  // GET /api/v1/produtos/{id}
  @GetMapping("/{id}")
  public EntityModel<ProdutoModel> buscar(@PathVariable UUID id) {
    var produto = service.buscarOuLancar(id);
    var model   = assembler.toModel(produto);
    model.add(linkTo(methodOn(ProdutoController.class).listar(null)).withRel("produtos"));
    return EntityModel.of(model);
  }

  @PostMapping
  @ResponseStatus(HttpStatus.CREATED)
  public ResponseEntity<ProdutoModel> criar(@RequestBody @Valid CriarProdutoRequest req) {
    var produto = service.criar(req);
    var uri = linkTo(methodOn(ProdutoController.class)
      .buscar(produto.getId())).toUri();
    return ResponseEntity.created(uri).body(assembler.toModel(produto));
  }

  @PutMapping("/{id}")
  public ProdutoModel atualizar(
      @PathVariable UUID id,
      @RequestBody @Valid AtualizarProdutoRequest req) {
    return assembler.toModel(service.atualizar(id, req));
  }

  @DeleteMapping("/{id}")
  @ResponseStatus(HttpStatus.NO_CONTENT)
  public void remover(@PathVariable UUID id) {
    service.remover(id);
  }
}

Spring Security com JWT

Java + Spring Security 6 — Geração e validação de JWT
import com.auth0.jwt.JWT;
import com.auth0.jwt.algorithms.Algorithm;

@Service
public class TokenService {

  @Value("${app.jwt.secret}")       // variavel de ambiente — nunca hardcode
  private String secret;

  @Value("${app.jwt.expiracao-horas:2}")
  private long expiracaoHoras;

  public String gerarToken(Usuario usuario) {
    var alg = Algorithm.HMAC256(secret);
    return JWT.create()
      .withIssuer("loja-api")
      .withSubject(usuario.getEmail())
      .withClaim("roles", usuario.getRoles().stream().map(Role::name).toList())
      .withExpiresAt(Instant.now().plus(expiracaoHoras, ChronoUnit.HOURS))
      .sign(alg);
  }

  public String verificarToken(String token) {
    try {
      return JWT.require(Algorithm.HMAC256(secret))
        .withIssuer("loja-api")
        .build()
        .verify(token)
        .getSubject();
    } catch (JWTVerificationException e) {
      throw new TokenInvalidoException(e);
    }
  }
}

// Filtro que intercepta cada requisicao
@Component
public class JwtAuthFilter extends OncePerRequestFilter {

  private final TokenService      tokenService;
  private final UsuarioRepository usuarioRepo;

  @Override
  protected void doFilterInternal(HttpServletRequest req,
      HttpServletResponse resp, FilterChain chain)
      throws ServletException, IOException {

    var token = extrairBearerToken(req);
    if (token != null) {
      var email   = tokenService.verificarToken(token);
      var usuario = usuarioRepo.findByEmail(email).orElseThrow();
      var auth    = new UsernamePasswordAuthenticationToken(
        usuario, null, usuario.getAuthorities());
      SecurityContextHolder.getContext().setAuthentication(auth);
    }
    chain.doFilter(req, resp);
  }

  private String extrairBearerToken(HttpServletRequest request) {
    var header = request.getHeader("Authorization");
    if (header == null || !header.startsWith("Bearer ")) return null;
    return header.substring(7);
  }
}

<dependency>
  <groupId>org.springdoc</groupId>
  <artifactId>springdoc-openapi-starter-webmvc-ui</artifactId>
  <version>2.5.0</version>
</dependency>

// URL gerada: http://localhost:8080/swagger-ui.html

// Configuracao OpenAPI global
@Configuration
public class OpenApiConfig {
  @Bean
  public OpenAPI openAPI() {
    return new OpenAPI()
      .info(new Info().title("Loja API").version("v1")
        .description("API REST da loja virtual")
        .license(new License().name("MIT")))
      .addSecurityItem(new SecurityRequirement().addList("bearerAuth"))
      .components(new Components()
        .addSecuritySchemes("bearerAuth", new SecurityScheme()
          .type(SecurityScheme.Type.HTTP).scheme("bearer").bearerFormat("JWT")));
  }
}

Testes de Integracao com Testcontainers

Java + Testcontainers + MockMvc — Teste de ponta a ponta
import org.testcontainers.containers.PostgreSQLContainer;
import org.testcontainers.junit.jupiter.Container;
import org.testcontainers.junit.jupiter.Testcontainers;

@SpringBootTest(webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT)
@AutoConfigureMockMvc
@Testcontainers
@Transactional
class ProdutoApiIntegrationTest {

  @Container
  static PostgreSQLContainer<?> postgres =
    new PostgreSQLContainer<>("postgres:16-alpine")
      .withDatabaseName("loja_test")
      .withUsername("test")
      .withPassword("test");

  @DynamicPropertySource   // injeta URL do container no contexto Spring
  static void configDB(DynamicPropertyRegistry registry) {
    registry.add("spring.datasource.url",        postgres::getJdbcUrl);
    registry.add("spring.datasource.username",   postgres::getUsername);
    registry.add("spring.datasource.password",   postgres::getPassword);
  }

  @Autowired MockMvc            mockMvc;
  @Autowired ObjectMapper       mapper;
  @Autowired ProdutoRepository  repo;

  @Test
  @WithMockUser(roles = "ADMIN")
  void criarProdutoRetorna201() throws Exception {
    var req = new CriarProdutoRequest("Cafe 500g", BigDecimal.valueOf(25.90), 100);

    mockMvc.perform(post("/api/v1/produtos")
        .contentType(MediaType.APPLICATION_JSON)
        .content(mapper.writeValueAsString(req)))
      .andExpect(status().isCreated())
      .andExpect(jsonPath("$.nome").value("Cafe 500g"))
      .andExpect(jsonPath("$.preco").value(25.90))
      .andExpect(header().exists("Location"));

    assertThat(repo.count()).isEqualTo(1);
  }
}

Arquitetura Web Java em Escala

Camadas claras

Controller valida e traduz protocolo HTTP; servico implementa regra de negocio; repositorio encapsula persistencia para manter coesao e testabilidade.

Observabilidade nativa

Logs estruturados + metricas + tracing desde o inicio reduzem tempo de diagnostico e permitem evoluir arquitetura com dados reais de uso.

Trade-off principal

Produtividade do Spring e alta, mas exige disciplina de limites entre modulos para evitar monolito acoplado por anotacoes e dependencias circulares.
