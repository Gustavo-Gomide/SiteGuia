Java + DL4J — rede neural densa para classificacao binaria

      <version>1.0.0-M2.1</version>
    </dependency>
  </dependencies>
</dependencyManagement>

import org.deeplearning4j.nn.conf.*;
import org.deeplearning4j.nn.conf.layers.*;
import org.deeplearning4j.nn.multilayer.MultiLayerNetwork;
import org.nd4j.linalg.activations.Activation;
import org.nd4j.linalg.lossfunctions.LossFunctions;

public class ExemploRedeDensa {

  public static void main(String[] args) {

    // Configuracao da arquitetura
    var conf = new NeuralNetConfiguration.Builder()
      .seed(42)
      .updater(new Adam(0.001))
      .weightInit(WeightInit.XAVIER)
      .list()
        .layer(new DenseLayer.Builder()
          .nIn(16).nOut(64)
          .activation(Activation.RELU)
          .build())
        .layer(new DenseLayer.Builder()
          .nOut(32)
          .activation(Activation.RELU)
          .dropOut(0.5)   // regularizacao
          .build())
        .layer(new OutputLayer.Builder(LossFunctions.LossFunction.XENT)
          .nOut(1)
          .activation(Activation.SIGMOID)
          .build())
      .build();

    var model = new MultiLayerNetwork(conf);
    model.init();

    // Carrega dados com DataSetIterator (CSV, MNIST, etc.)
    DataSetIterator treino = carregarDados("treino.csv");
    DataSetIterator teste  = carregarDados("teste.csv");

    // Treinamento
    for (int epoch = 0; epoch < 20; epoch++) {
      model.fit(treino);
      treino.reset();
      var eval = model.evaluate(teste);
      System.out.printf("Epoch %2d | Acc: %.4f | F1: %.4f%n",
        epoch, eval.accuracy(), eval.f1());
      teste.reset();
    }

    // Inferencia em novos dados
    var entrada  = Nd4j.create(new double[][] {{ 0.1, 0.9, ... }});
    var saida    = model.output(entrada);
    boolean positivo = saida.getDouble(0) > 0.5;

    // Serializar modelo
    ModelSerializer.writeModel(model, new File("modelo.zip"), true);
  }
}

Tribuo — Machine Learning Tipado

Java + Tribuo — classificacao com Random Forest e avaliacao
import org.tribuo.*;
import org.tribuo.classification.*;
import org.tribuo.classification.ensemble.*;
import org.tribuo.data.csv.CSVLoader;

public class ExemploTribuo {

  public static void main(String[] args) throws IOException {

    // Carrega CSV com coluna alvo "label"
    var factory = new LabelFactory();
    var csvLoader = new CSVLoader<>(factory);

    var dados = csvLoader.loadDataSource(
      Paths.get("iris.csv"), "species");

    // Divide em treino/teste (80/20)
    var split = new TrainTestSplitter<>(dados, 0.8, 42L);
    var treino = new MutableDataset<>(split.getTrain());
    var teste  = new ImmutableDataset<>(split.getTest());

    // Random Forest — nao precisa de normalizacao de features
    var treinador = new RandomForestTrainer<>(
      new ClassificationTreeTrainer(),   // arvore base
      new VotingCombiner(),               // combinacao por votacao
      100                                  // numero de arvores
    );

    var modelo = treinador.train(treino);

    // Avaliacao com metricas multiclasse
    var avaliador   = new LabelEvaluator();
    var avaliacao   = avaliador.evaluate(modelo, teste);
    System.out.println(avaliacao);
    // Class          n  tp   fn   fp  Recall  Prec   F1
    // setosa        10  10    0    0  1.000  1.000  1.000
    // versicolor    10   9    1    0  0.900  1.000  0.947
    // ...

    // Predicao individual
    var amostra = ArrayExample.createWithWeight(factory, 1.0f,
      new String[]{ "sepalLength", "sepalWidth", "petalLength", "petalWidth" },
      new double[]{ 5.1,  3.5,  1.4,  0.2 }
    );
    var pred = modelo.predict(amostra);
    System.out.printf("Predicao: %s (%.2f%%)%n",
      pred.getOutput().getLabel(),
      pred.getOutput().getScore() * 100);

    // Serializar com Protobuf (nao usa Java serialization)
    modelo.serializeToFile(Paths.get("iris_rf.model"));
  }
}

Apache Spark MLlib

Java + Spark MLlib — pipeline de regressao com transformacoes
import org.apache.spark.ml.Pipeline;
import org.apache.spark.ml.regression.LinearRegression;
import org.apache.spark.ml.feature.*;
import org.apache.spark.sql.SparkSession;

public class SparkMLExemplo {

  public static void main(String[] args) {

    var spark = SparkSession.builder()
      .appName("PrevisaoPreco")
      .master("local[*]")  // todos os nucleos locais
      .getOrCreate();

    // Carrega CSV para DataFrame
    var df = spark.read()
      .option("header", "true")
      .option("inferSchema", "true")
      .csv("dados/casas.csv");

    // Indexar coluna categorica (bairro)
    var indexer = new StringIndexer()
      .setInputCol("bairro")
      .setOutputCol("bairroIdx");

    // One-hot encoding
    var encoder = new OneHotEncoder()
      .setInputCol("bairroIdx")
      .setOutputCol("bairroVec");

    // Montar vetor de features
    var assembler = new VectorAssembler()
      .setInputCols(new String[]{ "area", "quartos", "bairroVec" })
      .setOutputCol("features");

    // Normalizar
    var scaler = new StandardScaler()
      .setInputCol("features")
      .setOutputCol("scaledFeatures")
      .setWithStd(true).setWithMean(true);

    // Modelo
    var lr = new LinearRegression()
      .setFeaturesCol("scaledFeatures")
      .setLabelCol("preco")
      .setMaxIter(100)
      .setElasticNetParam(0.8);   // 80% L1, 20% L2

    // Pipeline encadeia todas as etapas
    var pipeline = new Pipeline()
      .setStages(new PipelineStage[]{ indexer, encoder, assembler, scaler, lr });

    var[] splits = df.randomSplit(new double[]{ 0.8, 0.2 }, 42);
    var model   = pipeline.fit(splits[0]);
    var pred    = model.transform(splits[1]);

    var eval = new RegressionEvaluator()
      .setLabelCol("preco").setMetricName("rmse");
    System.out.printf("RMSE: %.2f%n", eval.evaluate(pred));

    model.save("modelos/regressao_casas");
    spark.stop();
  }
}

Inferencia com ONNX Runtime

Java + ONNX Runtime — usar modelo treinado em Python
// pom.xml
// <dependency>
//   <groupId>com.microsoft.onnxruntime</groupId>
//   <artifactId>onnxruntime</artifactId>
//   <version>1.18.0</version>
// </dependency>

// No Python: model = sklearn.train(...)
// import skl2onnx; skl2onnx.convert_sklearn(model, ...) -> modelo.onnx

import ai.onnxruntime.*;
import java.nio.FloatBuffer;

public class InferenciaONNX {

  private final OrtEnvironment env;
  private final OrtSession    session;

  public InferenciaONNX(String caminhoModelo) throws OrtException {
    env     = OrtEnvironment.getEnvironment();
    session = env.createSession(caminhoModelo,
      new OrtSession.SessionOptions());
  }

  public float[] prever(float[] featuras) throws OrtException {
    // Cria tensor de entrada: forma [1, n_features]
    var shape  = new long[]{ 1, featuras.length };
    var buffer = FloatBuffer.wrap(featuras);
    var tensor = OnnxTensor.createTensor(env, buffer, shape);

    // Descobre nome da entrada pelo grafo do modelo
    var nomeEntrada = session.getInputNames().iterator().next();

    try (var result = session.run(Map.of(nomeEntrada, tensor))) {
      var output = (OnnxTensor) result.get(0);
      return output.getFloatBuffer().array();
    }
  }

  public void fechar() throws OrtException {
    session.close();
    env.close();
  }
}

Weka API — Prototipagem Rapida

Java + Weka — classificador J48 com validacao cruzada
import weka.classifiers.trees.J48;
import weka.classifiers.Evaluation;
import weka.core.Instances;
import weka.core.converters.ConverterUtils.DataSource;

public class WekaExemplo {

  public static void main(String[] args) throws Exception {

    // Carrega arquivo ARFF ou CSV
    var source = new DataSource("dados/iris.arff");
    Instances dados = source.getDataSet();
    dados.setClassIndex(dados.numAttributes() - 1);   // ultima coluna = classe

    // Arvore de decisao C4.5 (J48)
    var arvore = new J48();
    arvore.setOptions(new String[]{ "-C", "0.25", "-M", "2" });

    // Validacao cruzada 10-folds
    var eval = new Evaluation(dados);
    eval.crossValidateModel(arvore, dados, 10, new Random(1));

    System.out.printf("Acuracia: %.2f%%%n", eval.pctCorrect());
    System.out.println(eval.toClassDetailsString());
    System.out.println(eval.toMatrixString("Matriz de confusao"));

    // Treina no conjunto completo para producao
    arvore.buildClassifier(dados);
    System.out.println(arvore.toString());  // mostra a arvore textual

    // Serializa o modelo
    SerializationHelper.write("iris_j48.model", arvore);
  }
}

Deeplearning4j

Redes neurais completas na JVM. Suporte a GPU via CUDA. Interoperabilidade com Keras importando modelos HDF5. Ideal para NLP, imagens e series temporais em producao Java.

Tribuo

Biblioteca de ML do Oracle focada em reproducibilidade. Provenance tracking rastreia exatamente quais dados e parametros geraram cada modelo. Type-safe por design.

Apache Spark MLlib

Escala para petabytes em clusters. Pipeline API compativel com sklearn. Suporte a streaming com Spark Structured Streaming. Ideal para dados que nao cabem em RAM.

ONNX Runtime

Executa modelos treinados em Python (PyTorch, TensorFlow, scikit-learn) diretamente em Java sem reescrever nada. Otimizacoes automaticas de grafo em tempo de carregamento.
