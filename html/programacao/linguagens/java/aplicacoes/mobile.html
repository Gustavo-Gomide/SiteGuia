Arquitetura MVVM com ViewModel

Java + Android Jetpack — ViewModel com LiveData e Repository
// Model — dados e logica de negocio
@Entity(tableName = "produtos")
public class Produto {
  @PrimaryKey(autoGenerate = true)
  public int    id;
  @NonNull
  public String nome;
  public double preco;
  public int    estoque;
}

// Repository — unica fonte de verdade
public class ProdutoRepository {
  private final ProdutoDao       dao;
  private final ProdutoApiService api;
  private final Executor         io = Executors.newSingleThreadExecutor();

  public ProdutoRepository(Application app) {
    var db = ProdutoDB.getInstance(app);
    dao = db.produtoDao();
    api = new Retrofit.Builder()
      .baseUrl("https://loja.exemplo.com/")
      .addConverterFactory(GsonConverterFactory.create())
      .build()
      .create(ProdutoApiService.class);
  }

  // LiveData do Room — observa mudancas automaticamente
  public LiveData<List<Produto>> getTodosOsProdutos() {
    return dao.listarTodos();
  }

  public void sincronizarDoServidor() {
    io.execute(() -> {
      try {
        var resp = api.listar().execute();
        if (resp.isSuccessful() && resp.body() != null) {
          dao.upsertTodos(resp.body());
        }
      } catch (Exception e) {
        Log.e("Repo", "Sync falhou", e);
      }
    });
  }
}

// ViewModel — sobrevive a rotacao de tela
public class ProdutoViewModel extends AndroidViewModel {

  private final ProdutoRepository          repo;
  private final LiveData<List<Produto>>   produtos;
  private final MutableLiveData<String>   statusMensagem = new MutableLiveData<>();

  public ProdutoViewModel(@NonNull Application app) {
    super(app);
    repo     = new ProdutoRepository(app);
    produtos = repo.getTodosOsProdutos();
  }

  public LiveData<List<Produto>> getProdutos()     { return produtos;       }
  public LiveData<String>           getStatus()        { return statusMensagem; }

  public void sincronizar() {
    statusMensagem.setValue("Sincronizando...");
    repo.sincronizarDoServidor();
  }
}

Room 2.6 — Banco de Dados Local

Java + Room — DAO, @Query e migrations
@Dao
public interface ProdutoDao {

  @Query("SELECT * FROM produtos ORDER BY nome ASC")
  LiveData<List<Produto>> listarTodos();

  @Query("SELECT * FROM produtos WHERE id = :id")
  Produto buscarPorId(int id);

  @Query("SELECT * FROM produtos WHERE nome LIKE '%' || :termo || '%'")
  List<Produto> buscar(String termo);

  @Insert(onConflict = OnConflictStrategy.REPLACE)
  long inserir(Produto produto);

  @Insert(onConflict = OnConflictStrategy.REPLACE)
  void upsertTodos(List<Produto> produtos);

  @Update
  void atualizar(Produto produto);

  @Delete
  void remover(Produto produto);

  @Query("DELETE FROM produtos WHERE estoque = 0")
  int removerSemEstoque();
}

@Database(entities = { Produto.class }, version = 2, exportSchema = true)
public abstract class ProdutoDB extends RoomDatabase {

  private static volatile ProdutoDB INSTANCE;

  public abstract ProdutoDao produtoDao();

  public static ProdutoDB getInstance(@NonNull Context ctx) {
    if (INSTANCE == null) {
      synchronized (ProdutoDB.class) {
        if (INSTANCE == null) {
          INSTANCE = Room.databaseBuilder(ctx.getApplicationContext(),
              ProdutoDB.class, "loja.db")
            .addMigrations(MIGRATION_1_2)
            .build();
        }
      }
    }
    return INSTANCE;
  }

  // Migracao de v1 para v2: adicionou coluna estoque
  static final Migration MIGRATION_1_2 = new Migration(1, 2) {
    @Override
    public void migrate(@NonNull SupportSQLiteDatabase db) {
      db.execSQL("ALTER TABLE produtos ADD COLUMN estoque INTEGER NOT NULL DEFAULT 0");
    }
  };
}

Retrofit2 e OkHttp

Java + Retrofit2 — interface API, interceptor de autenticacao e Call
// Interface que descreve a API REST
public interface ProdutoApiService {

  @GET("api/v1/produtos")
  Call<List<Produto>> listar(
    @Query("page") int page,
    @Query("size") int tamanho);

  @GET("api/v1/produtos/{id}")
  Call<Produto> buscarPorId(@Path("id") String id);

  @POST("api/v1/produtos")
  Call<Produto> criar(@Body CriarProdutoRequest req);

  @Multipart
  @POST("api/v1/produtos/{id}/imagem")
  Call<Void> enviarImagem(
    @Path("id") String id,
    @Part("imagem") MultipartBody.Part imagem);
}

// Construcao com OkHttp + interceptors
public class ApiClient {

  public static ProdutoApiService criar(String token) {
    var okHttp = new OkHttpClient.Builder()
      .connectTimeout(10, TimeUnit.SECONDS)
      .readTimeout(30,    TimeUnit.SECONDS)
      .addInterceptor(chain -> {
        // Injeta token JWT em cada requisicao
        var req = chain.request().newBuilder()
          .header("Authorization", "Bearer " + token)
          .build();
        return chain.proceed(req);
      })
      .addNetworkInterceptor(new HttpLoggingInterceptor()
        .setLevel(BuildConfig.DEBUG
          ? HttpLoggingInterceptor.Level.BODY
          : HttpLoggingInterceptor.Level.NONE))
      .build();

    return new Retrofit.Builder()
      .baseUrl("https://loja.exemplo.com/")
      .client(okHttp)
      .addConverterFactory(GsonConverterFactory.create())
      .build()
      .create(ProdutoApiService.class);
  }
}

// Chamada assincrona (no ViewModel)
api.listar(0, 20).enqueue(new Callback<List<Produto>>() {
  @Override
  public void onResponse(@NonNull Call<List<Produto>> call,
      @NonNull Response<List<Produto>> resp) {
    if (resp.isSuccessful() && resp.body() != null) {
      dao.upsertTodos(resp.body());
    }
  }
  @Override
  public void onFailure(@NonNull Call<List<Produto>> call, @NonNull Throwable t) {
    Log.e("API", "Falha na chamada", t);
  }
});

Activity, Fragment e Ciclo de Vida

Java + Android — Activity com ViewModel e observacao de LiveData
public class ProdutosActivity extends AppCompatActivity {

  private ActivityProdutosBinding binding;
  private ProdutoViewModel        viewModel;
  private ProdutoAdapter          adapter;

  @Override
  protected void onCreate(Bundle savedInstanceState) {
    super.onCreate(savedInstanceState);
    // View Binding — sem findViewById, null-safe
    binding = ActivityProdutosBinding.inflate(getLayoutInflater());
    setContentView(binding.getRoot());

    // ViewModel vinculado ao ciclo de vida da Activity
    viewModel = new ViewModelProvider(this).get(ProdutoViewModel.class);

    // RecyclerView com DiffUtil
    adapter = new ProdutoAdapter(produto -> {
      // Clique no item: abre tela de detalhe
      var intent = new Intent(this, DetalheActivity.class);
      intent.putExtra("produto_id", produto.getId());
      startActivity(intent);
    });
    binding.recyclerProdutos.setAdapter(adapter);
    binding.recyclerProdutos.setLayoutManager(new LinearLayoutManager(this));

    // Observa LiveData — atualiza UI automaticamente
    viewModel.getProdutos().observe(this, adapter::submitList);
    viewModel.getStatus().observe(this, msg ->
      binding.txtStatus.setText(msg));

    binding.fabSync.setOnClickListener(v -> viewModel.sincronizar());
  }

  @Override
  protected void onDestroy() {
    super.onDestroy();
    binding = null;   // evita memory leak se Activity for destruida
  }
}

Jetpack Compose

Kit de UI declarativo moderno do Android. Integra com Java via interop: @Composable e KTX sao Kotlin, mas ViewModels e Room em Java funcionam perfeitamente como camada de dados.

WorkManager

Tarefas em background garantidas (sobrevive a reinicializacao e ao app fechado). Ideal para sincronizacao periodica, upload de imagens em fila e notificacoes agendadas.

Navigation Component

Gerencia navegacao entre Fragments com back stack automatico. NavGraph define rotas em XML. safe-args plugin gera classes tipadas para passar argumentos entre destinos.

Hilt (DI)

Injecao de dependencias sobre Dagger2. @AndroidEntryPoint em Activity/Fragment; @HiltViewModel para ViewModels. Reduz boilerplate de Dagger e integra com ciclo de vida Android.
