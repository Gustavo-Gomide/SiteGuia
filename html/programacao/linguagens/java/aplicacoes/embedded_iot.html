Java para Embedded e IoT

Pi4J v2 para controle de GPIO, I2C e SPI no Raspberry Pi, Eclipse Paho para MQTT publish/subscribe com broker, comunicacao serial RS-232/485 com jSerialComm, e padrao de projeto para dispositivos IoT com reconnect automatico.

Pi4J v2 — GPIO, I2C e SPI

Java + Pi4J v2 — GPIO digital, PWM e leitura de sensor I2C
import com.pi4j.Pi4J;
import com.pi4j.io.gpio.digital.*;
import com.pi4j.io.pwm.*;
import com.pi4j.io.i2c.*;

public class ControladorHardware {

  public static void main(String[] args) throws Exception {

    // Contexto Pi4J — auto-detecta plataforma (Raspberry Pi, etc.)
    try (var pi4j = Pi4J.newAutoContext()) {

      // ===  GPIO DIGITAL DE SAIDA  ===
      // Pino GPIO 18 (BCM) ligado a um LED
      var cfgLed = DigitalOutput.newConfigBuilder(pi4j)
        .id("led")
        .name("LED Status")
        .address(18)
        .shutdown(DigitalState.LOW)
        .initial(DigitalState.LOW)
        .provider("pigpio-digital-output")
        .build();

      var led = pi4j.create(cfgLed);
      led.on();
      Thread.sleep(500);
      led.off();

      // ===  GPIO DIGITAL DE ENTRADA (botao com pull-up)  ===
      var cfgBtn = DigitalInput.newConfigBuilder(pi4j)
        .id("botao")
        .name("Botao A")
        .address(23)
        .pull(PullResistance.PULL_UP)
        .debounce(3000L)   // debounce 3ms
        .provider("pigpio-digital-input")
        .build();

      var botao = pi4j.create(cfgBtn);
      botao.addListener(e -> {
        if (e.state() == DigitalState.LOW) {   // LOW = pressionado (pull-up)
          System.out.println("Botao pressionado!");
          led.toggle();
        }
      });

      // ===  PWM — controle de servo motor  ===
      // Servo: 50Hz, duty cycle 5%=0°, 7.5%=90°, 10%=180°
      var cfgServo = Pwm.newConfigBuilder(pi4j)
        .id("servo")
        .address(12)
        .frequency(50)
        .dutyCycle(7.5)
        .provider("pigpio-pwm")
        .build();

      var servo = pi4j.create(cfgServo);
      servo.on(50);           // frequencia Hz
      servo.dutyCycle(5.0);  // 0 graus
      Thread.sleep(1000);
      servo.dutyCycle(10.0); // 180 graus
      Thread.sleep(1000);
      servo.off();

      // ===  I2C — sensor BME280 (temperatura, pressao, umidade)  ===
      var cfgI2c = I2C.newConfigBuilder(pi4j)
        .id("bme280")
        .bus(1)               // I2C bus 1 (SDA=GPIO2, SCL=GPIO3)
        .device(0x76)         // endereco do BME280
        .provider("pigpio-i2c")
        .build();

      var bme280 = pi4j.create(cfgI2c);
      // Ler registrador do chip_id (deve retornar 0x60 para BME280)
      int chipId = bme280.readRegister(0xD0);
      System.out.printf("BME280 chip_id: 0x%02X%n", chipId);

      var dados = bme280.readRegister(0xF7, 8);   // burst read 8 bytes
      double temperatura = parsarTemperatura(dados);
      System.out.printf("Temperatura: %.2f C%n", temperatura);
    }
  }
}

MQTT com Eclipse Paho

Java + Eclipse Paho — publisher/subscriber MQTT com reconnect
import org.eclipse.paho.client.mqttv3.*;
import org.eclipse.paho.client.mqttv3.persist.MemoryPersistence;

public class ClienteMQTT {

  private static final String BROKER   = "ssl://broker.exemplo.com:8883";
  private static final String CLIENT_ID = "sensor-temperatura-01";

  private MqttClient client;

  public void conectar(String usuario, char[] senha) throws MqttException {
    client = new MqttClient(BROKER, CLIENT_ID, new MemoryPersistence());

    var opts = new MqttConnectOptions();
    opts.setUserName(usuario);
    opts.setPassword(senha);
    opts.setCleanSession(false);         // persiste subscricoes entre reconexoes
    opts.setAutomaticReconnect(true);
    opts.setConnectionTimeout(10);
    opts.setKeepAliveInterval(30);
    // TLS — somente conectar se o certificado for valido
    opts.setSocketFactory(SSLContext.getDefault().getSocketFactory());

    client.setCallback(new MqttCallbackExtended() {
      @Override
      public void connectComplete(boolean reconnect, String serverURI) {
        System.out.println("Conectado: " + serverURI);
        try {
          // Re-subscrever no connectComplete para sobreviver a reconexoes
          client.subscribe("sensores/+/comando", 1, (topico, msg) ->
            processarComando(topico, msg.toString()));
        } catch (MqttException e) {
          System.err.println("Erro ao subscrever: " + e.getMessage());
        }
      }
      @Override
      public void connectionLost(Throwable cause) {
        System.out.println("Conexao perdida: " + cause.getMessage());
        // automaticReconnect cuida da reconexao
      }
      @Override
      public void messageArrived(String topico, MqttMessage msg) {}
      @Override
      public void deliveryComplete(IMqttDeliveryToken t) {}
    });

    client.connect(opts);
  }

  public void publicarLeitura(String sensorId, double temperatura) throws Exception {
    if (!client.isConnected()) throw new IllegalStateException("Nao conectado");
    var topico  = "sensores/" + sensorId + "/temperatura";
    var payload = String.format("{\"temp\":%.2f,\"ts\":%d}",
      temperatura, System.currentTimeMillis());
    var msg = new MqttMessage(payload.getBytes(StandardCharsets.UTF_8));
    msg.setQos(1);        // at-least-once
    msg.setRetained(true); // broker armazena ultimo valor
    client.publish(topico, msg);
  }
}

Comunicacao Serial com jSerialComm

Java + jSerialComm — comunicacao RS-232/485 com Arduino
import com.fazecast.jSerialComm.SerialPort;

public class ComunicacaoSerial {

  public static void main(String[] args) throws InterruptedException {

    // Listar portas disponiveis
    for (var p : SerialPort.getCommPorts()) {
      System.out.printf("%-20s | %s%n", p.getSystemPortName(), p.getDescriptivePortName());
    }

    var porta = SerialPort.getCommPort("/dev/ttyUSB0");   // ou "COM3" no Windows
    porta.setBaudRate(9600);
    porta.setNumDataBits(8);
    porta.setNumStopBits(SerialPort.ONE_STOP_BIT);
    porta.setParity(SerialPort.NO_PARITY);
    porta.setComPortTimeouts(
      SerialPort.TIMEOUT_READ_BLOCKING, 2000, 0);

    if (!porta.openPort()) {
      throw new RuntimeException("Nao foi possivel abrir a porta serial");
    }

    try {
      // Enviar comando ao Arduino (ex: "LED:ON\n")
      var cmd = "LED:ON\n".getBytes(StandardCharsets.US_ASCII);
      int escritos = porta.writeBytes(cmd, cmd.length);
      System.out.println("Bytes enviados: " + escritos);

      // Ler resposta linha a linha
      try (var br = new BufferedReader(
            new InputStreamReader(porta.getInputStream()))) {
        String linha;
        while ((linha = br.readLine()) != null) {
          System.out.println("Arduino: " + linha);
          if (linha.startsWith("OK")) break;
        }
      }
    } finally {
      porta.closePort();
    }
  }
}

// Listener assincrono (nao bloqueia thread principal)
porta.addDataListener(new SerialPortDataListener() {
  @Override
  public int getListeningEvents() {
    return SerialPort.LISTENING_EVENT_DATA_AVAILABLE;
  }
  @Override
  public void serialEvent(SerialPortEvent event) {
    int bytesDisponiveis = porta.bytesAvailable();
    byte[] buffer = new byte[bytesDisponiveis];
    porta.readBytes(buffer, bytesDisponiveis);
    System.out.println(new String(buffer, StandardCharsets.US_ASCII));
  }
});

Padroes de Projeto para IoT

Device Shadow

Armazena o ultimo estado conhecido do dispositivo em nuvem (AWS IoT Core / Azure IoT Hub). Permite consultar e controlar o dispositivo mesmo quando ele esta offline, sincronizando ao reconectar.

Telemetria com Buffer Local

Quando sem conexao, armazena leituras em SQLite local (via JDBC). Ao reconectar ao broker MQTT, reenvia os dados em ordem cronologica e limpa o buffer.

Heartbeat / Watchdog

Daemon thread que publica estado "ONLINE" a cada 30s no topico sensores/{id}/status com retained=true. Se o broker nao recebe por 90s, LWT (Last Will Testament) publica "OFFLINE" automaticamente.

OTA Update

Dispositivo subscreve ota/{id}/firmware. Backend publica URL de novo JAR assinado. Dispositivo baixa, verifica hash SHA-256, substitui o JAR atual e reinicia via watchdog do SO.
