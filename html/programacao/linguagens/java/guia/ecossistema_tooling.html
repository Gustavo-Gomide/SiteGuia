Java — Ecossistema e Tooling

Ecossistema e Tooling Java

Maven e Gradle para gerenciamento de build, JUnit 5 com parametrize e extensoes, Mockito para mocking, JaCoCo para cobertura e pipeline CI/CD completo com GitHub Actions.

Maven e Gradle resolvem o mesmo problema — gerenciar dependencias e build — com filosofias diferentes. Maven usa convencao-sobre-configuracao com XML declarativo; Gradle usa DSL Kotlin/Groovy imperativa com build incremental mais rapido.

Maven — Ciclo de Vida e Plugins

Shell — Comandos Maven essenciais
# Fases do ciclo de vida padrao (executam em ordem)
# validate → compile → test → package → verify → install → deploy

mvn compile            # Compila src/main/java
mvn test               # Compila e executa src/test/java
mvn package            # Gera .jar em target/
mvn package -DskipTests # Empacota sem rodar testes (evite em CI)
mvn verify             # Roda integration tests e checks de qualidade
mvn install            # Instala no repositorio local (~/.m2/)
mvn clean package      # Limpa e empacota — padrao em CI

# Arvore de dependencias
mvn dependency:tree
mvn dependency:tree -Dincludes=org.springframework.boot

# Verificar atualizacoes
mvn versions:display-dependency-updates
mvn versions:display-plugin-updates
XML — pom.xml avancado com Spring Boot, JaCoCo e Surefire
<build>
  <plugins>
    <!-- Spring Boot plugin: gera fat JAR executavel -->
    <plugin>
      <groupId>org.springframework.boot</groupId>
      <artifactId>spring-boot-maven-plugin</artifactId>
      <configuration>
        <excludes>
          <exclude>
            <groupId>org.projectlombok</groupId>
            <artifactId>lombok</artifactId>
          </exclude>
        </excludes>
      </configuration>
    </plugin>

    <!-- Surefire: executa testes JUnit 5 -->
    <plugin>
      <artifactId>maven-surefire-plugin</artifactId>
      <version>3.3.0</version>
      <configuration>
        <includes>
          <include>**/*Test.java</include>
          <include>**/*Tests.java</include>
        </includes>
      </configuration>
    </plugin>

    <!-- JaCoCo: cobertura de codigo -->
    <plugin>
      <groupId>org.jacoco</groupId>
      <artifactId>jacoco-maven-plugin</artifactId>
      <version>0.8.12</version>
      <executions>
        <execution>
          <goals><goal>prepare-agent</goal></goals>
        </execution>
        <execution>
          <id>report</id>
          <phase>verify</phase>
          <goals><goal>report</goal></goals>
        </execution>
        <execution>
          <id>check</id>
          <goals><goal>check</goal></goals>
          <configuration>
            <rules>
              <rule>
                <limits>
                  <limit>
                    <counter>LINE</counter>
                    <minimum>0.80</minimum>  <!-- 80% de cobertura -->
                  </limit>
                </limits>
              </rule>
            </rules>
          </configuration>
        </execution>
      </executions>
    </plugin>
  </plugins>
</build>

Gradle com Kotlin DSL

Kotlin DSL — build.gradle.kts completo
plugins {
  java
  id("org.springframework.boot") version "3.3.0"
  id("io.spring.dependency-management") version "1.1.5"
  jacoco
}

group   = "br.com.guia"
version = "0.0.1-SNAPSHOT"

java {
  toolchain { languageVersion = JavaLanguageVersion.of(21) }
}

repositories { mavenCentral() }

dependencies {
  implementation("org.springframework.boot:spring-boot-starter-web")
  implementation("org.springframework.boot:spring-boot-starter-data-jpa")
  runtimeOnly("com.h2database:h2")
  testImplementation("org.springframework.boot:spring-boot-starter-test")
}

tasks.test {
  useJUnitPlatform()
  finalizedBy(tasks.jacocoTestReport)
}

tasks.jacocoTestReport {
  dependsOn(tasks.test)
  reports { xml.required = true }  // para SonarQube / Codecov
}

tasks.jacocoTestCoverageVerification {
  violationRules {
    rule {
      limit { minimum = "0.80".toBigDecimal() }
    }
  }
}

JUnit 5 e Mockito

Java + JUnit 5 — Testes parametrizados e extensoes
import org.junit.jupiter.api.*;
import org.junit.jupiter.params.ParameterizedTest;
import org.junit.jupiter.params.provider.*;
import static org.assertj.core.api.Assertions.*;

@DisplayName("ProdutoService")
class ProdutoServiceTest {

  private ProdutoService service;

  @BeforeEach
  void setUp() {
    service = new ProdutoService(/* dependencias */);
  }

  @Test
  @DisplayName("Deve calcular preco com desconto")
  void calcularPrecoDesconto() {
    var produto = new Produto("Cafe", BigDecimal.valueOf(100.0));

    var precoFinal = service.aplicarDesconto(produto, 0.1);

    assertThat(precoFinal)
      .isEqualByComparingTo(BigDecimal.valueOf(90.0));
  }

  // @ParameterizedTest com valores inline
  @ParameterizedTest
  @CsvSource({
    "100.0, 0.1, 90.0",
    "200.0, 0.2, 160.0",
    "50.0,  0.5, 25.0"
  })
  void descontosVariados(double preco, double desconto, double esperado) {
    var p = new Produto("X", BigDecimal.valueOf(preco));
    assertThat(service.aplicarDesconto(p, desconto))
      .isEqualByComparingTo(BigDecimal.valueOf(esperado));
  }

  // @EnumSource — testa cada valor de um enum
  @ParameterizedTest
  @EnumSource(StatusPedido.class)
  void statusPedidoTemDescricao(StatusPedido status) {
    assertThat(status.getDescricao()).isNotBlank();
  }

  // Excecao esperada
  @Test
  void descontoInvalidoLancaExcecao() {
    var p = new Produto("Y", BigDecimal.TEN);

    assertThatThrownBy(() -> service.aplicarDesconto(p, -0.1))
      .isInstanceOf(IllegalArgumentException.class)
      .hasMessageContaining("desconto");
  }
}

// Mockito — isolando dependencias
@ExtendWith(MockitoExtension.class)
class PedidoServiceTest {

  @Mock    PedidoRepository  pedidoRepo;
  @Mock    ProdutoRepository produtoRepo;
  @InjectMocks PedidoService service;

  @Test
  void processarPedidoExistente() {
    var pedido = new Pedido(1L, StatusPedido.PENDENTE);
    when(pedidoRepo.findById(1L)).thenReturn(Optional.of(pedido));

    service.processar(1L);

    verify(pedidoRepo).save(argThat(p -> p.getStatus() == StatusPedido.PROCESSADO));
    verifyNoMoreInteractions(pedidoRepo);
  }

  @Test
  void pedidoInexistenteDeveLancarExcecao() {
    when(pedidoRepo.findById(99L)).thenReturn(Optional.empty());

    assertThatExceptionOfType(EntidadeNaoEncontradaException.class)
      .isThrownBy(() -> service.processar(99L))
      .withMessageContaining("99");
  }
}

CI/CD com GitHub Actions

YAML — .github/workflows/ci.yml — Java build e testes
name: CI

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  build:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        java: [21]

    steps:
      - uses: actions/checkout@v4

      - name: Setup Java ${{ matrix.java }}
        uses: actions/setup-java@v4
        with:
          java-version: ${{ matrix.java }}
          distribution: temurin
          cache: maven          # cache do ~/.m2

      - name: Build e testes
        run: mvn -B verify

      - name: Publicar cobertura
        if: always()
        uses: codecov/codecov-action@v4
        with:
          token: ${{ secrets.CODECOV_TOKEN }}
          files: target/site/jacoco/jacoco.xml

      - name: Upload artefato
        uses: actions/upload-artifact@v4
        with:
          name: app-jar
          path: target/*.jar
          retention-days: 30

  release:
    needs: build
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-java@v4
        with:
          java-version: 21
          distribution: temurin
          server-id: github
          settings-path: ${{ github.workspace }}
      - run: mvn -B deploy -P release
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}

Qualidade automatica

Bloqueie merge sem testes, cobertura minima e linters. Regra clara evita discussao subjetiva em review e reduz regressao.

Documente startup, rollback, rotinas de incidentes e comandos de observabilidade para reduzir MTTR em producao.
