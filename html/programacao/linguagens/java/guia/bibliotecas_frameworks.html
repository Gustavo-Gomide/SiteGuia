Java + Spring Boot — Controller REST com validacao e tratamento de erro
// DTO de requisicao com Bean Validation
public record CriarProdutoRequest(
  @NotBlank(message = "Nome obrigatorio")
  @Size(max = 120)
  String nome,

  @NotNull @Positive
  BigDecimal preco,

  @Min(0)
  int estoque
) {}

// Entidade e repositorio
@Entity
public class Produto {
  @Id @GeneratedValue(strategy = GenerationType.IDENTITY)
  private Long id;
  private String nome;
  private BigDecimal preco;
  private int estoque;
  // getters + setters omitidos
}

@Repository
public interface ProdutoRepository extends JpaRepository<Produto, Long> {
  List<Produto> findByNomeContainingIgnoreCase(String termo);
  @Query("SELECT p FROM Produto p WHERE p.estoque > 0 ORDER BY p.preco")
  List<Produto> findEmEstoqueOrdenadoPorPreco();
}

// Service com logica de negocio
@Service
@Transactional
public class ProdutoService {
  private final ProdutoRepository repo;

  public ProdutoService(ProdutoRepository repo) { this.repo = repo; }

  public Produto criar(CriarProdutoRequest req) {
    var p = new Produto();
    p.setNome(req.nome());
    p.setPreco(req.preco());
    p.setEstoque(req.estoque());
    return repo.save(p);
  }
}

// Controller REST
@RestController
@RequestMapping("/api/produtos")
public class ProdutoController {
  private final ProdutoService service;

  public ProdutoController(ProdutoService service) { this.service = service; }

  @GetMapping
  public List<Produto> listar() { return service.listarTodos(); }

  @PostMapping
  @ResponseStatus(HttpStatus.CREATED)
  public Produto criar(@RequestBody @Valid CriarProdutoRequest req) {
    return service.criar(req);
  }

  @GetMapping("/{id}")
  public ResponseEntity<Produto> buscar(@PathVariable Long id) {
    return service.buscarOuLancar(id);
  }
}

// Handler global de excecoes
@RestControllerAdvice
public class GlobalExceptionHandler {

  @ExceptionHandler(MethodArgumentNotValidException.class)
  @ResponseStatus(HttpStatus.UNPROCESSABLE_ENTITY)
  public Map<String, String> handleValidation(MethodArgumentNotValidException ex) {
    return ex.getBindingResult().getFieldErrors().stream()
      .collect(Collectors.toMap(
        FieldError::getField,
        FieldError::getDefaultMessage));
  }
}

Jackson — Serializacao JSON

Java + Jackson — ObjectMapper, anotacoes e customizacao
import com.fasterxml.jackson.databind.*;
import com.fasterxml.jackson.annotation.*;

// Configuracao do ObjectMapper (Spring Boot configura automaticamente)
ObjectMapper mapper = new ObjectMapper()
  .registerModule(new JavaTimeModule())  // suporte a java.time
  .disable(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS)
  .configure(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES, false);

// Serializar
String json   = mapper.writeValueAsString(objeto);
String pretty = mapper.writerWithDefaultPrettyPrinter().writeValueAsString(objeto);

// Desserializar
Produto p   = mapper.readValue(json, Produto.class);
List<Produto> lista = mapper.readValue(json,
  mapper.getTypeFactory().constructCollectionType(List.class, Produto.class));

// Anotacoes de mapeamento
public class PedidoDto {
  @JsonProperty("numero_pedido")         // nome diferente no JSON
  public String numeroPedido;

  @JsonIgnore                             // nunca serializa
  public String senhaInterna;

  @JsonFormat(pattern = "dd/MM/yyyy HH:mm", timezone = "America/Sao_Paulo")
  public LocalDateTime criadoEm;

  @JsonInclude(JsonInclude.Include.NON_NULL)  // omite campos null
  public String observacao;
}

// Polimorfismo com @JsonTypeInfo
@JsonTypeInfo(use = JsonTypeInfo.Id.NAME, property = "tipo")
@JsonSubTypes({
  @JsonSubTypes.Type(value = PagamentoCartao.class, name = "cartao"),
  @JsonSubTypes.Type(value = PagamentoPix.class,   name = "pix")
})
public abstract class Pagamento {}

// Agora {"tipo":"pix",...} desserializa diretamente para PagamentoPix

Logging com SLF4J e Logback

Java — SLF4J / Logback com MDC e structured logging
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.slf4j.MDC;

@Service
public class PedidoService {
  // Logger estatico — convenção universal
  private static final Logger log = LoggerFactory.getLogger(PedidoService.class);

  public Pedido processar(Long pedidoId, String usuarioId) {
    // MDC — Mapped Diagnostic Context — enriquece todos os logs do thread
    try (var ignored = MDC.putCloseable("pedidoId", pedidoId.toString());
         var ignored2 = MDC.putCloseable("usuario", usuarioId)) {

      log.info("Iniciando processamento do pedido");

      // Parametrizacao — nao concatenar strings em log!
      log.debug("Buscando pedido id={} para usuario={}", pedidoId, usuarioId);

      try {
        var pedido = executarProcessamento(pedidoId);
        log.info("Pedido {} processado com sucesso, total={}", pedidoId, pedido.getTotal());
        return pedido;
      } catch (PagamentoRecusadoException e) {
        log.warn("Pagamento recusado para pedido {}: {}", pedidoId, e.getCodigo());
        throw e;
      } catch (Exception e) {
        log.error("Erro inesperado ao processar pedido {}", pedidoId, e); // ex como ultimo arg
        throw e;
      }
    }
  }
}

Estrategia de Stack para Projetos Java

Stack minima para backend empresarial

  • API: Spring Boot + Bean Validation + OpenAPI para contrato e documentacao.
