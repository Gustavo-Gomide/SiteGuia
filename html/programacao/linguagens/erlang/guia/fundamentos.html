Erlang — Fundamentos

Erlang é uma linguagem de tipagem dinâmica, funcional e puramente expressional — não há statements, apenas expressões que retornam valores. Variáveis começam com maiúscula e são vinculadas uma única vez (single assignment). Atoms são identificadores globais sem conteúdo. Tudo é imutável, recursão substitui loops, e pattern matching é onipresente.

Em Erlang, a variável X não pode mudar de valor. Se X = 5, X sempre será 5 naquele escopo. Esta restrição elimina uma classe inteira de bugs e torna raciocinar sobre código concorrente drasticamente mais simples.

Tipos de Dados Fundamentais

Todos os tipos básicos de Erlang
% Inteiros e floats
X = 42.
Y = 3.14.
Hex = 16#FF.       % 255 em hexadecimal
Bin = 2#1010.      % 10 em binário
Big = 999999999999999999999999.  % bignum nativo — sem overflow

% Atoms — identificadores globais imutáveis
ok.
error.
nome_composto.
'Com espaços'.
true.              % atom (não keyword)
false.             % atom
undefined.

% Strings — listas de inteiros (codepoints)
S1 = "hello".         % equivale a [104,101,108,108,111]
S2 = <<"hello">>.     % binary string (mais eficiente)
B1 = <<1,2,3>>.       % binary literal
B2 = <<X:16,Y:8>>.    % binary com tamanho de bit especificado

% Tuples — tamanho fixo, elementos heterogêneos
Ponto = {3, 7}.
Resultado = {ok, 42}.
Erro = {error, not_found}.
{Tag, Valor} = Resultado.   % pattern matching — Tag=ok, Valor=42

% Listas — singly-linked, cabeça+cauda
L1 = [1, 2, 3, 4, 5].
[H|T] = L1.         % H=1, T=[2,3,4,5]  — head|tail
L2 = [0|L1].        % prepend: [0,1,2,3,4,5]
L3 = L1 ++ [6,7].   % concatenar: O(length(L1)) — caro!
L4 = L1 -- [2,4].   % diferença: [1,3,5]

% Maps (dict moderno, desde OTP 17)
M = #{nome => "João", idade => 30, ativo => true}.
Nome = maps:get(nome, M).
M2 = M#{idade => 31}.  % update imutável

Pattern Matching — O Coração do Erlang

Matching em funções, case e receive
% Pattern matching em atribuição
{ok, Resultado} = {ok, 42}.    % Resultado = 42
{ok, _} = {ok, ignorado}.      % _ ignora o valor
[A, B | _] = [1, 2, 3, 4].    % A=1, B=2
<<H:8, Resto/binary>> = <<"hello">>.  % H=104 (h), Resto=<<"ello">>

% Múltiplas cláusulas de função (pattern matching)
-module(fib).
-export([fib/1]).

fib(0) -> 0;                   % cláusula 1: quando N=0
fib(1) -> 1;                   % cláusula 2: quando N=1
fib(N) when N > 1 ->           % cláusula 3: com guard
    fib(N-1) + fib(N-2).

% case — matching com múltiplos padrões
classificar(N) ->
    case N of
        0          -> zero;
        X when X < 0  -> negativo;
        X when X =:= 42 -> resposta;
        _          -> positivo
    end.

% Tupla de resultado — idioma Erlang fundamental
abrir_arquivo(Nome) ->
    case file:read_file(Nome) of
        {ok, Conteudo} ->
            {ok, byte_size(Conteudo)};
        {error, enoent} ->
            {error, arquivo_nao_encontrado};
        {error, Motivo} ->
            {error, Motivo}
    end.

Funções — Guards e Funções Anônimas

Guards, funções de alta ordem e list comprehensions
% Guards — condições adicionais após when
max(X, Y) when X >= Y -> X;
max(_, Y) -> Y.

% Guards permitidos: comparações, is_*, arithmetic, abs, length, etc.
valido(X) when is_integer(X), X > 0, X < 100 -> true;
valido(_) -> false.

% Funções anônimas (fun)
Dobro = fun(X) -> X * 2 end.
Dobro(5).  % 10

Soma = fun(A, B) -> A + B end.
lists:foldl(Soma, 0, [1,2,3,4,5]).  % 15

% Capture de função (referência)
F = fun lists:map/2.
F(fun(X) -> X * 2 end, [1,2,3]).  % [2,4,6]

% Funções de alta ordem
lists:map(fun(X) -> X * X end, [1,2,3,4]).   % [1,4,9,16]
lists:filter(fun(X) -> X rem 2 =:= 0 end, [1,2,3,4,5]).  % [2,4]
lists:foldl(fun(X, Acc) -> Acc + X end, 0, [1,2,3]).     % 6

% List comprehensions
Quadrados = [X*X || X <- lists:seq(1,5)].        % [1,4,9,16,25]
Pares   = [X || X <- lists:seq(1,10), X rem 2 =:= 0].  % [2,4,6,8,10]
Pares_Produtos = [X*Y || X <- [1,2,3], Y <- [10,20]].  % produto cartesiano

% Binary comprehensions
<<(X*2) || <<X>> <= <<1,2,3,4>>>>.  % <<2,4,6,8>>

Recursão — O Loop do Erlang

Recursão de cauda, acumuladores e listas
% Recursão simples — NÃO tail-recursive (stack cresce)
soma_lista([]) -> 0;
soma_lista([H|T]) -> H + soma_lista(T).

% Tail-recursive com acumulador — CORRETO (não usa stack)
soma_lista_tr(Lista) -> soma_lista_tr(Lista, 0).
soma_lista_tr([], Acc) -> Acc;
soma_lista_tr([H|T], Acc) -> soma_lista_tr(T, Acc + H).

% Inverter lista — clássico com acumulador
inverter(L) -> inverter(L, []).
inverter([], Acc) -> Acc;
inverter([H|T], Acc) -> inverter(T, [H|Acc]).

% Flatten de lista aninhada
flatten([]) -> [];
flatten([H|T]) when is_list(H) -> flatten(H) ++ flatten(T);
flatten([H|T]) -> [H|flatten(T)].

% Quicksort funcional
qsort([]) -> [];
qsort([H|T]) ->
    Menor  = [X || X <- T, X =< H],
    Maior  = [X || X <- T, X > H],
    qsort(Menor) ++ [H] ++ qsort(Maior).

A single assignment (vinculação única de variáveis) em Erlang não é uma limitação — é uma garantia de segurança. Se X foi vinculado a 5, qualquer tentativa de X = 6 lança um erro imediato em vez de silenciosamente sobrescrever — catching um bug que em linguagens mutáveis causaria um comportamento imprevisível horas depois.
