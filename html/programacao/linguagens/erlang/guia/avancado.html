Erlang — Avançado

Além dos fundamentos, Erlang oferece behaviours especializados (gen_statem para máquinas de estado), armazenamento em memória compartilhada via ETS, banco de dados distribuído Mnesia, mecanismos de introspecção de processos e hot code upgrade — substituição de código em produção sem parar o sistema.

ETS (Erlang Term Storage) é a exceção à regra de imutabilidade: uma tabela ETS é mutável e acessível por múltiplos processos simultaneamente sem cópia de dados — ideal para cache e contadores de alta frequência. O processo dono da tabela define permissões de acesso.

gen_statem — Máquina de Estados

Semáforo com gen_statem (callback_mode: state_functions)
-module(semaforo).
-behaviour(gen_statem).
-export([start_link/0, avancar/0, estado/0]).
-export([init/1, callback_mode/0, vermelho/3, verde/3, amarelo/3]).

start_link() -> gen_statem:start_link({local, ?MODULE}, ?MODULE, [], []).
avancar()    -> gen_statem:cast(?MODULE, avancar).
estado()     -> gen_statem:call(?MODULE, estado).

callback_mode() -> state_functions.  % nome-da-função == estado

init([]) -> {ok, vermelho, 0}.       % estado inicial, dados iniciais

%% Estado: vermelho
vermelho(cast, avancar, Ciclos) ->
    io:format("Vermelho → Verde~n"),
    {next_state, verde, Ciclos};
vermelho({call, From}, estado, Ciclos) ->
    {keep_state, Ciclos, [{reply, From, vermelho}]}.

%% Estado: verde
verde(cast, avancar, Ciclos) ->
    io:format("Verde → Amarelo~n"),
    {next_state, amarelo, Ciclos};
verde({call, From}, estado, Ciclos) ->
    {keep_state, Ciclos, [{reply, From, verde}]}.

%% Estado: amarelo
amarelo(cast, avancar, Ciclos) ->
    io:format("Amarelo → Vermelho (ciclo ~p completo)~n", [Ciclos + 1]),
    {next_state, vermelho, Ciclos + 1};
amarelo({call, From}, estado, Ciclos) ->
    {keep_state, Ciclos, [{reply, From, amarelo}]}.

%% Uso no shell:
%% semaforo:start_link().
%% semaforo:estado().  % vermelho
%% semaforo:avancar(). % → verde
%% semaforo:avancar(). % → amarelo
%% semaforo:avancar(). % → vermelho (ciclo 1 completo)

ETS — Erlang Term Storage

Tipos de tabela, CRUD e operações atômicas
%% Criar tabela — tipos:
%% set: chave única, melhor para lookup
%% ordered_set: chave única, itreração ordenada (menos eficiente)
%% bag: múltiplos valores por chave (tuplas duplicadas retornadas)
%% duplicate_bag: permite duplicatas exatas
T = ets:new(minha_tabela, [set, named_table, public]).
% named_table: acessar pelo nome em vez do TID
% public/protected/private: quem pode ler/escrever

%% Inserir
ets:insert(minha_tabela, {joao, 30, engenheiro}).
ets:insert(minha_tabela, [{maria, 25, designer}, {pedro, 35, dev}]).

%% Lookup por chave (elemento 1 da tupla por padrão)
ets:lookup(minha_tabela, joao).     % [{joao, 30, engenheiro}]
ets:member(minha_tabela, joao).     % true

%% Match — padrão com _
ets:match(minha_tabela, {joao, '$1', '_'}).  % [[30]]
ets:match_object(minha_tabela, {'_', '_', dev}).

%% Select com matchspecs (mais poderoso)
MS = ets:fun2ms(fun({Nome, Idade, _}) when Idade > 28 -> Nome end).
ets:select(minha_tabela, MS).     % [joao, pedro]

%% Update atômico — incrementa contador sem race condition
ets:insert(minha_tabela, {visitas, 0}).
ets:update_counter(minha_tabela, visitas, 1).  % retorna novo valor
ets:update_counter(minha_tabela, visitas, {2, 1}).  % {pos, incremento}

%% Deletar
ets:delete(minha_tabela, joao).
ets:delete_all_objects(minha_tabela).
ets:delete(minha_tabela).  % apaga a tabela

%% Tamanho e informações
ets:info(minha_tabela, size).     % número de objetos
ets:info(minha_tabela, memory).   % words usadas

Mnesia — Banco de Dados Distribuído

Schema, tabelas e transações Mnesia
%% Schema e inicialização (uma vez, antes de start)
mnesia:create_schema([node()]).   % criar schema no disco
mnesia:start().

%% Definir registro
-record(usuario, {id, nome, email, criado_em}).

%% Criar tabela
mnesia:create_table(usuario, [
    {attributes, record_info(fields, usuario)},
    {disc_copies, [node()]},      % disco + RAM (vs ram_copies ou disc_only)
    {type, set}
]).

%% Transação — ACID garantido
F = fun() ->
    U = #usuario{id = 1, nome = "João", email = "jo@ex.com",
                 criado_em = erlang:system_time(second)},
    mnesia:write(U)
end,
mnesia:transaction(F).    % {atomic, ok} ou {aborted, Reason}

%% Leitura transacional
Ler = fun() -> mnesia:read(usuario, 1) end,
{atomic, [Registro]} = mnesia:transaction(Ler).

%% Query com QLC (Query List Comprehension)
Q = fun() ->
    qlc:e(qlc:q([U#usuario.nome || U <- mnesia:table(usuario),
                                    U#usuario.id > 0]))
end,
{atomic, Nomes} = mnesia:transaction(Q).

%% dirty_read/write — sem transação, mais rápido (sem garantias)
mnesia:dirty_write(#usuario{id = 2, nome = "Ana", email = "ana@ex.com"}).
mnesia:dirty_read(usuario, 2).

%% Mnesia em múltiplos nodes
mnesia:add_table_copy(usuario, outro_node@host, disc_copies).

Introspecção e Diagnóstico de Processos

process_info, erlang:system_info e ferramentas de diagnóstico
%% Informações de um processo
erlang:process_info(Pid).                        % tudo
erlang:process_info(Pid, message_queue_len).     % tamanho da mailbox
erlang:process_info(Pid, memory).                % bytes usados
erlang:process_info(Pid, heap_size).             % words do heap
erlang:process_info(Pid, reductions).            % reduções executadas
erlang:process_info(Pid, current_function).      % onde está agora
erlang:process_info(Pid, registered_name).       % nome, se registrado
erlang:process_info(Pid, links).                 % PIDs linkados
erlang:process_info(Pid, monitors).              % PIDs monitorados

%% Informações do sistema
erlang:system_info(schedulers).         % número de schedulers
erlang:system_info(process_count).      % processos ativos
erlang:system_info(process_limit).      % máximo permitido
erlang:system_info(atom_count).         % átomos criados

%% Listar todos os processos
Pids = erlang:processes().
GrandePid = lists:max([{element(2, erlang:process_info(P, memory)), P}
                        || P <- Pids]).

%% Inspecionar estado de um gen_server em produção
sys:get_state(contador).              % retorna estado interno
sys:get_status(contador).             % status completo com histórico
sys:suspend(contador).                % pausar processamento
sys:resume(contador).                 % retomar

%% Rastrear mensagens de um processo em tempo real
dbg:tracer().
dbg:p(Pid, [send, receive]).
% ... executar ações ...
dbg:stop().

Hot Code Upgrade — Atualização sem Parar

Atualizar módulos em produção via code:load_file e code_change/3
%% A BEAM mantém 2 versões de cada módulo: "current" e "old"
%% Processos em code_change migram dados do estado antigo para novo

%% No shell — recompilar e carregar novo módulo
c(contador).              % recompila e carrega no shell
l(contador).              % só carrega (sem compilar)
code:load_file(contador). % semelhante a l/1
code:which(contador).     % caminho do .beam carregado
code:purge(contador).     % remove versão "old"

%% Chamada qualificada — usa versão "current" imediatamente
%% (loop normal usa versão já carregada pelo processo)
loop(Estado) ->
    receive
        Msg -> handle(Msg, Estado)
    end,
    ?MODULE:loop(Estado).   % chamada qualificada = hot swap ponto de entrada

%% code_change/3 em gen_server — migrar estado entre versões
code_change(VersaoAntiga, EstadoAntigo, _Extra) ->
    %% converter {estado_v1, Contador} → {estado_v2, Contador, Historico}
    {contador_v1, N} = EstadoAntigo,
    NovoEstado = {contador_v2, N, []},
    io:format("Migrado de ~p para v2~n", [VersaoAntiga]),
    {ok, NovoEstado}.

%% Upgrade coordenado via release handler (systools/appup)
%%   1. Criar arquivo .appup com instruções update/load/purge
%%   2. systools:make_relup/3 gera relup file
%%   3. release_handler:install_release/1 aplica em produção

Hot code upgrade é viável em Erlang porque processos não compartilham estado: a troca de versão de um módulo afeta apenas novas chamadas. O estado persiste dentro de cada processo e é migrado explicitamente via code_change/3 — o desenvolvedor controla completamente a transformação de dados durante o upgrade, mesmo enquanto o sistema serve requisições.
