Erlang — Atores e Concorrência

O modelo de atores de Erlang é a implementação mais madura e testada em produção do paradigma: cada processo é um ator com mailbox privado, comunicação apenas por mensagens assíncronas, sem memória compartilhada. A BEAM schedula milhões destes atores em múltiplos núcleos com preempção cooperativa — nenhum processo pode monopolizar a CPU.

A BEAM usa redução contável como unidade de preempção: a cada ~2000 reduções (chamadas de função), o scheduler ativa o próximo processo. Isso garante que um loop infinito em um processo não trava outros — diferente de threads do SO onde um processo pode monopolizar um core.

spawn, send ( ! ) e receive

Criar processos e comunicar por mensagens
% Criar um processo — spawn retorna PID
Pid = spawn(fun() ->
    io:format("Processo: ~p~n", [self()])
end).

% Spawn com módulo e função
Pid2 = spawn(meu_modulo, funcao, [Arg1, Arg2]).

% Enviar mensagem — ! é o operador send
Pid ! {mensagem, "olá"}.
Pid ! ping.
self() ! resposta.    % enviar para si mesmo

% receive — consumir mensagem da mailbox
receive
    ping ->
        io:format("Recebi ping~n");
    {mensagem, Texto} ->
        io:format("Mensagem: ~s~n", [Texto]);
    _ ->
        io:format("Mensagem desconhecida~n")
after 5000 ->            % timeout em ms
    io:format("Timeout!~n")
end.

% Servidor eco — receive em loop
eco() ->
    receive
        {De, Mensagem} ->
            De ! {resposta, Mensagem},
            eco();
        parar ->
            ok
    end.

% Uso:
EcoPid = spawn(fun eco/0).
EcoPid ! {self(), "testando"}.
receive {resposta, R} -> io:format("Eco: ~s~n", [R]) end.

Links, Monitors e Tratamento de Falhas

Detectar crashes com links e monitors
% spawn_link — se filho crasha, pai recebe EXIT signal
process_flag(trap_exit, true).  % capturar EXIT como mensagem
FilhoPid = spawn_link(fun() ->
    timer:sleep(1000),
    exit(motivo_qualquer)      % crash intencional
end).

receive
    {'EXIT', FilhoPid, normal} ->
        io:format("Filho terminou normalmente~n");
    {'EXIT', FilhoPid, Motivo} ->
        io:format("Filho crashou: ~p~n", [Motivo])
end.

% monitor — unidirecional, não propaga exit
Ref = erlang:monitor(process, AlgumPid).
receive
    {'DOWN', Ref, process, AlgumPid, normal} ->
        io:format("Processo terminou~n");
    {'DOWN', Ref, process, AlgumPid, Motivo} ->
        io:format("Processo caiu: ~p~n", [Motivo])
end.

% Diferença: link bidirecional (ambos morrem), monitor unidirecional

% Registro de processos por nome
register(servidor_eco, spawn(fun eco/0)).
servidor_eco ! {self(), "olá"}.   % usar nome em vez de PID
whereis(servidor_eco).            % retorna PID ou undefined
unregister(servidor_eco).

OTP gen_server — Servidor Genérico

gen_server completo com call, cast e info
-module(contador).
-behaviour(gen_server).

%% API pública
-export([start_link/0, incrementar/0, decrementar/1, valor/0]).
%% Callbacks obrigatórios do gen_server
-export([init/1, handle_call/3, handle_cast/2, handle_info/2,
         terminate/2, code_change/3]).

%% --- API ---
start_link() ->
    gen_server:start_link({local, ?MODULE}, ?MODULE, [], []).

incrementar() ->
    gen_server:cast(?MODULE, incrementar).   % assíncrono

decrementar(N) ->
    gen_server:cast(?MODULE, {decrementar, N}).

valor() ->
    gen_server:call(?MODULE, valor).          % síncrono — aguarda resposta

%% --- Callbacks ---
init([]) ->
    {ok, 0}.   % estado inicial

handle_call(valor, _From, Estado) ->
    {reply, Estado, Estado};   % {reply, Resposta, NovoEstado}
handle_call(_Req, _From, Estado) ->
    {reply, {error, desconhecido}, Estado}.

handle_cast(incrementar, Estado) ->
    {noreply, Estado + 1};     % {noreply, NovoEstado}
handle_cast({decrementar, N}, Estado) ->
    {noreply, Estado - N};
handle_cast(_Msg, Estado) ->
    {noreply, Estado}.

handle_info(timeout, Estado) ->
    io:format("Timeout periódico, valor atual: ~p~n", [Estado]),
    {noreply, Estado};
handle_info(_Info, Estado) ->
    {noreply, Estado}.

terminate(_Motivo, _Estado) -> ok.
code_change(_VersaoAntiga, Estado, _Extra) -> {ok, Estado}.

OTP supervisor — Árvore de Supervisão

supervisor com estratégias de restart
-module(meu_app_sup).
-behaviour(supervisor).
-export([start_link/0, init/1]).

start_link() ->
    supervisor:start_link({local, ?MODULE}, ?MODULE, []).

init([]) ->
    %% SupFlags: estratégia, max_restarts, max_seconds
    SupFlags = #{
        strategy  => one_for_one,  % reiniciar apenas o processo que crashou
        intensity => 10,           % máximo 10 restarts
        period    => 60            % em 60 segundos
    },

    %% childSpec: id, start, restart, shutdown, type, modules
    Filhos = [
        #{id       => contador,
          start    => {contador, start_link, []},
          restart  => permanent,   % sempre reiniciar
          shutdown => 5000,        % timeout para shutdown limpo
          type     => worker,
          modules  => [contador]},

        #{id       => cache,
          start    => {cache_server, start_link, [#{max_size => 1000}]},
          restart  => transient,   % reiniciar só se crash anormal
          shutdown => 5000,
          type     => worker,
          modules  => [cache_server]}
    ],
    {ok, {SupFlags, Filhos}}.

%% Estratégias de restart:
%% one_for_one    — reinicia apenas o processo morto
%% one_for_all    — reinicia todos se qualquer um morrer
%% rest_for_one   — reinicia o morto e todos iniciados depois dele
%% simple_one_for_one — template para processos dinâmicos

A filosofia "let it crash" parece contraintuitiva mas é mais robusta que tentar recuperar estado corrompido: um processo Erlang que crasha sai de um estado desconhecido, e o Supervisor reinicia do zero em estado limpo — garantido. Tratar _todos_ os erros possíveis em código defensivo é impossível; supervison é uma garantia incondicional.
