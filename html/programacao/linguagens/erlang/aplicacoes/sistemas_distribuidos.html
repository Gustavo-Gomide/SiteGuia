Erlang — Sistemas Distribuídos

Erlang foi projetado para ser distribuído desde a versão 1: enviar uma mensagem para um PID remoto tem a mesma sintaxe que enviar localmente — Pid ! Mensagem. O net_kernel gerencia conexões entre nós, o cookie autentica nós, e o epmd (Erlang Port Mapper Daemon) resolve endereços. Formar um cluster de nós BEAM é trivialmente fácil.

O modelo de distribuição Erlang é share-nothing por rede: nós se comunicam exclusivamente por troca de mensagens. Isso significa que uma falha de rede isola o nó mas não corrompe estado. O desafio real é lidar com partições de rede (netsplit) — quando dois nós perdem contato mas continuam rodando independentemente.

Clustering de Nós Erlang

Conectar nós e enviar mensagens remotas
%% Iniciar nós com nome e cookie
%% Terminal 1:
%%   erl -name no1@192.168.1.10 -setcookie servidor_secreto
%% Terminal 2:
%%   erl -name no2@192.168.1.11 -setcookie servidor_secreto

%% Conectar nós
net_adm:ping('no2@192.168.1.11').  % pong = ok, pang = falhou

%% Listar nós conectados
nodes().              % ['no2@192.168.1.11']
nodes(connected).     % visible nodes
nodes(hidden).        % nós ocultos (sem participar do sistema global)

%% Monitorar nós
net_kernel:monitor_nodes(true).
receive
    {nodeup,   No} -> io:format("Nó conectado: ~p~n",    [No]);
    {nodedown, No} -> io:format("Nó desconectou: ~p~n", [No])
end.

%% Enviar mensagem para processo em outro nó
%% (via PID remoto)
{servidor, 'no2@192.168.1.11'} ! {self(), ping}.

%% (via PID obtido de registro global — ver seção global)
Pid = global:whereis_name(servidor_global).
Pid ! {self(), ola}.

%% spawn em outro nó
NoPid = spawn('no2@192.168.1.11', fun() ->
    io:format("Rodando em ~p~n", [node()])
end).

RPC — Remote Procedure Call

rpc:call, cast, multicall e pmap
%% rpc:call — síncrono, bloqueia até resultado
Resultado = rpc:call('no2@192.168.1.11', string, to_upper, ["hello"]).
%% Resultado = "HELLO" ou {badrpc, Motivo}

%% Sempre verificar badrpc
case rpc:call(No, M, F, A) of
    {badrpc, nodedown} -> {error, no_down};
    {badrpc, timeout}  -> {error, timeout};
    {badrpc, Outro}    -> {error, Outro};
    Resp               -> {ok, Resp}
end.

%% rpc:call com timeout
rpc:call('no2@192.168.1.11', modulo, funcao, [Args], 5000).

%% rpc:cast — assíncrono, fire and forget
rpc:cast('no2@192.168.1.11', io, format, ["executando em no2~n"]).

%% rpc:multicall — chamar mesmo RPC em múltiplos nós
Nos = ['no1@h1', 'no2@h2', 'no3@h3'],
{Resultados, Falhos} = rpc:multicall(Nos, erlang, node, []).
%% Resultados = [no1@h1, no2@h2, no3@h3]
%% Falhos = [] (nós que não responderam)

%% rpc:pmap — parallel map em nós
Nos2 = nodes(),
Resultados2 = rpc:pmap({modulo, funcao}, [arg_extra], Nos2).

%% Async call — não bloqueia
Key = rpc:async_call('no2@192.168.1.11', modulo, funcao, [args]).
Resp = rpc:yield(Key).           % bloqueia até resultado chegar
Resp2 = rpc:nb_yield(Key, 2000). % non-blocking yield com timeout

Registro Global de Processos

global e pg — coordenação de processos no cluster
%% global:register_name — nome único em todo o cluster
%% Só um processo pode ter o nome — resolve conflitos com Resolver/2
global:register_name(servidor_primario, self()).
global:whereis_name(servidor_primario).   % PID ou undefined
global:unregister_name(servidor_primario).

%% Enviar mensagem via global
global:send(servidor_primario, {self(), ping}).

%% gen_server com registro global — start_link
gen_server:start_link({global, servidor_primario}, ?MODULE, [], []).
%% chamadas
gen_server:call({global, servidor_primario}, Pedido).
gen_server:cast({global, servidor_primario}, Notificacao).

%% pg — Process Groups (OTP 24+, substitui pg2)
%% Grupos distribuídos de processos — pub/sub no cluster
pg:start_link().
pg:join(grupo_processadores, self()).
pg:get_members(grupo_processadores).  % [Pid1, Pid2, ...] de todos os nós
pg:get_local_members(grupo_processadores).  % só nó local
pg:leave(grupo_processadores, self()).

%% Broadcast para todos do grupo
broadcast(Grupo, Mensagem) ->
    Pids = pg:get_members(Grupo),
    lists:foreach(fun(Pid) -> Pid ! Mensagem end, Pids).

ETS Distribuído e Mnesia em Cluster

Replicar dados entre nós com Mnesia
%% ETS não é distribuído nativamente — dono é um único nó
%% Para cache distribuído, use um gen_server com ETS por nó
%% + pg para broadcast de invalidações

invalidar_cache(Chave) ->
    ets:delete(cache_local, Chave),
    broadcast(grupo_cache, {invalidar, Chave}).

%% Mnesia distribuído — réplica em múltiplos nós
%% Passo 1: criar schema em todos os nós do cluster
todos_os_nos() -> [node() | nodes()].

inicializar_mnesia() ->
    Nos = todos_os_nos(),
    mnesia:create_schema(Nos),
    mnesia:start(),
    mnesia:create_table(sessao, [
        {attributes, record_info(fields, sessao)},
        {ram_copies, Nos}     % réplica em RAM em todos os nós
    ]),
    mnesia:create_table(pedido, [
        {attributes, record_info(fields, pedido)},
        {disc_copies, Nos}    % réplica em disco em todos os nós
    ]).

%% Mnesia replica automaticamente escritas para todos os nós com cópia
%% Leituras são locais (fast) — sem round trip

%% Aguardar Mnesia sincronizar após reconexão
mnesia:wait_for_tables([sessao, pedido], 30000).

%% Checar nós com cópia
mnesia:table_info(sessao, all_nodes).    % todos os nós com cópia
mnesia:table_info(sessao, active_replicas). % nós ativos agora

Partições de Rede e Estratégias de Resolução

Detectar e tratar netsplit
%% Monitorar desconexões de nós
net_kernel:monitor_nodes(true).

handle_info({nodedown, No}, Estado) ->
    ?LOG_WARNING(#{evento => nodedown, no => No}),
    %% escalar para nó secundário se No era o primário
    case maps:get(primario, Estado) of
        No ->
            Novo = eleger_novo_primario(maps:get(nos, Estado), No),
            ?LOG_INFO(#{evento => novo_primario, no => Novo}),
            {noreply, Estado#{primario => Novo}};
        _ ->
            {noreply, maps:update_with(nos, fun(Ns) -> lists:delete(No, Ns) end, Estado)}
    end;

handle_info({nodeup, No}, Estado) ->
    ?LOG_INFO(#{evento => nodeup, no => No}),
    %% reconciliar estado — depende da lógica da aplicação
    rpc:cast(No, meu_modulo, sincronizar_estado, [self()]),
    NovoNos = [No | maps:get(nos, Estado)],
    {noreply, Estado#{nos => NovoNos}}.

A distribuição em Erlang é transparente na sintaxe mas não na semântica: enviar uma mensagem local é confiável; enviar para um nó remoto pode falhar silenciosamente se o nó cair antes de processar. O modelo correto é monitorar os processos remotos (erlang:monitor) e tratar o DOWN como parte normal do fluxo — não como exceção.
