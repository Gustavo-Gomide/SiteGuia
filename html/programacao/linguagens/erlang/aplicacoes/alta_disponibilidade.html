Erlang — Alta Disponibilidade

Erlang foi construído para os switches AXD301 da Ericsson que alcançaram 99.9999999% de uptime (nove noves) — menos de 31 milissegundos de downtime por ano. Isso não é acidente: é o resultado de releases bem estruturados, árvores de supervisão, hot upgrades e monitoramento de saúde embutidos no OTP.

Alta disponibilidade em Erlang não vem de hardware redundante ou de código sem bugs — vem da capacidade de detectar falhas e se recuperar automaticamente, atualizar código sem parar, e distribuir carga por múltiplos nós. O hardware pode falhar; o sistema continua.

OTP Releases — Empacotamento para Produção

Estrutura de release com rebar3
%% rebar.config — configurar release
{relx, [
    {release, {meu_sistema, "1.0.0"},
     [meu_sistema,     % aplicação principal
      sasl,            % System Architecture Support Libraries (logs)
      runtime_tools,   % rastreamento em produção
      observer]},      % GUI (opcional, dev)

    {mode, prod},
    {extended_start_script, true},  % gera bin/meu_sistema start|stop|console|attach

    {sys_config, "config/sys.config"},
    {vm_args,    "config/vm.args"}
]}.

Árvores de Supervisão em Produção

Hierarquia de supervisores para máxima resiliência
-module(meu_sistema_app).
-behaviour(application).
-export([start/2, stop/1]).

start(_Type, _Args) ->
    meu_sistema_sup:start_link().

stop(_State) -> ok.

%%====================================================================
-module(meu_sistema_sup).
-behaviour(supervisor).
-export([start_link/0, init/1]).

start_link() ->
    supervisor:start_link({local, ?MODULE}, ?MODULE, []).

init([]) ->
    SupFlags = #{strategy => one_for_one, intensity => 3, period => 10},
    Filhos = [
        %% Supervisor de conexões HTTP
        #{id      => http_sup,
          start   => {http_sup, start_link, []},
          type    => supervisor,     % supervisor filho!
          restart => permanent,
          shutdown => infinity},

        %% Pool de workers de banco de dados
        #{id      => db_pool_sup,
          start   => {db_pool_sup, start_link, [#{size => 10}]},
          type    => supervisor,
          restart => permanent,
          shutdown => infinity},

        %% Cache em memória
        #{id      => cache,
          start   => {cache_server, start_link, []},
          type    => worker,
          restart => permanent,
          shutdown => 5000}
    ],
    {ok, {SupFlags, Filhos}}.

Logging com OTP logger

Logger OTP 21+ — configuração e uso em produção
%% Usar logger (OTP 21+) — substitui error_logger e lager
-include_lib("kernel/include/logger.hrl").

%% Macros disponíveis: ?LOG_DEBUG, ?LOG_INFO, ?LOG_NOTICE,
%%                     ?LOG_WARNING, ?LOG_ERROR, ?LOG_CRITICAL

processar_pedido(Pedido) ->
    ?LOG_INFO(#{evento => pedido_recebido, id => maps:get(id, Pedido)}),
    try
        Resultado = executar(Pedido),
        ?LOG_INFO(#{evento => pedido_ok, id => maps:get(id, Pedido)}),
        {ok, Resultado}
    catch
        error:Motivo:Stack ->
            ?LOG_ERROR(#{evento => pedido_falhou,
                         id    => maps:get(id, Pedido),
                         motivo => Motivo,
                         stack => Stack}),
            {error, Motivo}
    end.

Health Checks e Auto-Recuperação

Watchdog, heartbeat e monitoramento de saúde
-module(health_monitor).
-behaviour(gen_server).
-export([start_link/0, status/0]).
-export([init/1, handle_call/3, handle_cast/2, handle_info/2,
         terminate/2, code_change/3]).

-define(INTERVALO, 30000).  % checar a cada 30s

start_link() ->
    gen_server:start_link({local, ?MODULE}, ?MODULE, [], []).

status() ->
    gen_server:call(?MODULE, status).

init([]) ->
    self() ! checar,
    {ok, #{cheques => 0, falhas => 0}}.

handle_info(checar, Estado) ->
    NovoEstado = executar_checagens(Estado),
    erlang:send_after(?INTERVALO, self(), checar),
    {noreply, NovoEstado}.

executar_checagens(Estado) ->
    Checks = [
        checar_processo(banco_conexoes),
        checar_processo(cache_server),
        checar_memoria(200 * 1024 * 1024),   % 200 MB máximo
        checar_mailbox(banco_conexoes, 1000)  % mailbox não deve acumular
    ],
    Falhas = [F || {falha, F} <- Checks],
    case Falhas of
        [] -> ok;
        _  -> ?LOG_WARNING(#{evento => health_check_falhou, falhas => Falhas})
    end,
    Estado#{cheques := maps:get(cheques, Estado) + 1,
            falhas  := maps:get(falhas, Estado) + length(Falhas)}.

checar_processo(Nome) ->
    case whereis(Nome) of
        undefined -> {falha, {processo_morto, Nome}};
        Pid       ->
            case erlang:process_info(Pid, status) of
                {status, _} -> ok;
                _            -> {falha, {processo_inativo, Nome}}
            end
    end.

checar_memoria(MaxBytes) ->
    [{total, Total} | _] = erlang:memory(),
    case Total > MaxBytes of
        true  -> {falha, {memoria_alta, Total, MaxBytes}};
        false -> ok
    end.

checar_mailbox(Nome, Max) ->
    case whereis(Nome) of
        undefined -> ok;
        Pid       ->
            {message_queue_len, N} = erlang:process_info(Pid, message_queue_len),
            case N > Max of
                true  -> {falha, {mailbox_cheia, Nome, N}};
                false -> ok
            end
    end.

handle_call(status, _From, Estado) ->
    {reply, Estado, Estado};
handle_call(_, _, Estado) -> {reply, ok, Estado}.
handle_cast(_, Estado) -> {noreply, Estado}.
terminate(_, _) -> ok.
code_change(_, Estado, _) -> {ok, Estado}.

A filosofia "let it crash" significa que você escreve código para o caminho feliz, sem tratar cada erro possível, e deixa o Supervisor reconstruir o estado. Mas isso só funciona se as árvores de supervisão forem bem projetadas: granularidade adequada (um processo por conexão, não um por aplicação), intensity/period calibrados para não entrar em restart loop, e shutdown ordenado para liberar recursos limpos.
