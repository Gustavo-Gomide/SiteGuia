React: UI Declarativa com Hooks

JavaScript + React 18 — Hooks: useState, useEffect, useReducer, custom hook
// App.jsx — React 18 com hooks modernos
import { useState, useEffect, useReducer, useCallback, useMemo } from "react";

// Custom hook: busca com loading/error/data
function useFetch(url) {
  const [estado, dispatch] = useReducer(
    (s, action) => ({ ...s, ...action }),
    { loading: true, data: null, error: null }
  );

  useEffect(() => {
    const controller = new AbortController();
    dispatch({ loading: true, error: null });
    fetch(url, { signal: controller.signal })
      .then(r => r.json())
      .then(data => dispatch({ data, loading: false }))
      .catch(e => {
        if (e.name !== "AbortError")
          dispatch({ error: e.message, loading: false });
      });
    return () => controller.abort();
  }, [url]);

  return estado;
}

// Componente lista de usuários
function ListaUsuarios() {
  const [pagina, setPagina] = useState(1);
  const { data, loading, error } = useFetch(
    `/api/usuarios?page=${pagina}&limit=10`
  );

  const usuariosAtivos = useMemo(
    () => data?.items?.filter(u => u.ativo) ?? [],
    [data]
  );

  const proximaPagina = useCallback(
    () => setPagina(p => p + 1),
    []
  );

  if (loading) return <p>Carregando...</p>;
  if (error)   return <p className="erro">{error}</p>;

  return (
    <section>
      <h2>Usuários Ativos ({usuariosAtivos.length})</h2>
      <ul>
        {usuariosAtivos.map(u => (
          <li key={u.id}>{u.nome} — {u.email}</li>
        ))}
      </ul>
      <button onClick={proximaPagina} disabled={!data?.hasNext}>
        Página {pagina + 1} →
      </button>
    </section>
  );
}

Vue 3: Composition API e Reactivity

JavaScript + Vue 3 — Composition API: ref, computed, watch, composable
<!-- ListaProdutos.vue -->
<script setup>
import { ref, computed, watch, onMounted } from "vue";
import { useProdutos } from "./composables/useProdutos.js";

const busca      = ref("");
const { produtos, carregando, recarregar } = useProdutos();

const filtrados = computed(() =>
  produtos.value.filter(p =>
    p.nome.toLowerCase().includes(busca.value.toLowerCase())
  )
);

watch(busca, (nova) => {
  if (nova.length > 2) console.log("Busca:", nova);
});
</script>

<template>
  <input v-model="busca" placeholder="Buscar produto" />
  <p v-if="carregando">Carregando...</p>
  <ul v-else>
    <li v-for="p in filtrados" :key="p.id">
      {{ p.nome }} — R$ {{ p.preco.toFixed(2) }}
    </li>
  </ul>
</template>

// composables/useProdutos.js
import { ref, onMounted } from "vue";
export function useProdutos() {
  const produtos    = ref([]);
  const carregando  = ref(true);

  async function recarregar() {
    carregando.value = true;
    const res = await fetch("/api/produtos");
    produtos.value = await res.json();
    carregando.value = false;
  }

  onMounted(recarregar);
  return { produtos, carregando, recarregar };
}

Zod: Validacao de Schema em Runtime

JavaScript + Zod — Validacao, parse seguro e mensagens customizadas
import { z } from "zod";

const UsuarioSchema = z.object({
  id:       z.string().uuid(),
  nome:     z.string().min(2).max(100),
  email:    z.string().email({ message: "Email inválido" }),
  idade:    z.number().int().min(18).max(120).optional(),
  perfil:   z.enum(["admin", "editor", "leitor"]).default("leitor"),
  criado_em: z.string().datetime(),
});

type Usuario = z.infer<typeof UsuarioSchema>; // TypeScript: tipo derivado

// safeParse: sem exceção
const resultado = UsuarioSchema.safeParse({
  id:        "não-é-uuid",
  nome:      "Ana",
  email:     "ana@exemplo.com",
  criado_em: new Date().toISOString(),
});
if (!resultado.success) {
  for (const issue of resultado.error.issues) {
    console.error(issue.path.join("."), "→", issue.message);
  }
}

// Schema de formulário com refinement custom
const CadastroSchema = z.object({
  senha:          z.string().min(8),
  confirmarSenha: z.string(),
}).refine(d => d.senha === d.confirmarSenha, {
  message: "Senhas não coincidem",
  path: ["confirmarSenha"],
});

// Transformacao com .transform()
const PrecoSchema = z
  .string()
  .regex(/^\d+([.,]\d{1,2})?$/, "Formato inválido")
  .transform(v => parseFloat(v.replace(",", ".")));

console.log(PrecoSchema.parse("12,90"));  // 12.9 (number)

JavaScript — date-fns: manipulacao de datas sem mutacao
import {
  format, addDays, differenceInDays,
  isWeekend, startOfMonth, endOfMonth,
  parseISO, isBefore, isAfter,
} from "date-fns";
import { ptBR } from "date-fns/locale";

const hoje  = new Date();
const amanha = addDays(hoje, 1);
console.log(format(hoje, "dd/MM/yyyy HH:mm", { locale: ptBR }));
console.log(differenceInDays(addDays(hoje, 30), hoje));  // 30

// Dias úteis no mês atual (sem finais de semana)
function diasUteisNoMes(ref = new Date()) {
  let  atual = startOfMonth(ref);
  const fim  = endOfMonth(ref);
  let  count = 0;
  while (!isAfter(atual, fim)) {
    if (!isWeekend(atual)) count++;
    atual = addDays(atual, 1);
  }
  return count;
}
