JavaScript — Fundamentos

Fundamentos do JavaScript

Variáveis, funções, arrays, objetos, desestruturação, Promises, fetch e manipulação do DOM com JavaScript moderno (ES2022+).

Dominar os fundamentos do JavaScript significa entender escopo léxico, closures, o event loop e o modelo de herança prototipal — pilares sobre os quais todos os frameworks modernos são construídos.

Variáveis, Escopo e Hoisting

JavaScript — var × let × const e hoisting
// var: escopo de função, sofre hoisting
console.log(x); // undefined (hoisted, não ReferenceError)
var x = 10;

// let: escopo de bloco, TDZ (Temporal Dead Zone)
{
  let y = 20;
  console.log(y); // 20
}
// console.log(y); // ReferenceError — fora do bloco

// const: escopo de bloco, ligação imutável (não o valor)
const obj = { valor: 1 };
obj.valor = 2;     // OK — mutando a propriedade
// obj = {};       // TypeError — reatribuição proibida

// Escopo léxico: funções enxergam o escopo onde foram definidas
function outer() {
  const msg = "olá";
  return function inner() {
    console.log(msg);  // closure — msg "lembrada"
  };
}
outer()();  // "olá"

Funções: Declarativas, Arrow e Higher-Order

JavaScript — Formas de declarar funções e HoF
// Declaração (hoistável)
function somar(a, b) { return a + b; }

// Expressão (não hoistável)
const multiplicar = function(a, b) { return a * b; };

// Arrow function: sem próprio 'this', sempre expression
const dobrar = (n) => n * 2;
const cumprimentar = nome => `Olá, ${nome}!`;

// Parâmetros com valor padrão e rest
function log(nivel = "info", ...msgs) {
  console.log(`[${nivel.toUpperCase()}]`, ...msgs);
}
log("warn", "disco cheio", "backup falhou");

// Higher-Order Functions — map/filter/reduce
const nums = [1, 2, 3, 4, 5];

const pares     = nums.filter(n => n % 2 === 0);   // [2, 4]
const quadrados = nums.map(n => n ** 2);           // [1,4,9,16,25]
const soma      = nums.reduce((acc, n) => acc + n, 0); // 15

// Composição funcional manual
const pipe = (...fns) => x => fns.reduce((v, f) => f(v), x);
const processar = pipe(
  n => n * 2,
  n => n + 1,
  n => `resultado: ${n}`,
);
console.log(processar(5)); // "resultado: 11"

Objetos, Arrays e Desestruturação

JavaScript — Destructuring, spread, optional chaining
// Desestruturação de objeto com renomeação e default
const usuario = { nome: "Ana", idade: 30, cidade: "SP" };
const { nome, idade: anos = 0, cidade, pais = "Brasil" } = usuario;
console.log(nome, anos, pais); // "Ana" 30 "Brasil"

// Desestruturação de array
const [primeiro, segundo, ...resto] = [10, 20, 30, 40];
console.log(primeiro, segundo, resto); // 10 20 [30, 40]

// Spread: copiar e mesclar
const base  = { ativo: true, nivel: 1 };
const extra = { nivel: 5, admin: true };
const merged = { ...base, ...extra }; // { ativo: true, nivel: 5, admin: true }

// Optional chaining (?.) e nullish coalescing (??)
const cfg = { db: { host: "localhost" } };
const port = cfg?.db?.port ?? 5432;  // 5432 — default seguro
const tag  = cfg?.server?.tag;         // undefined (sem throw)

// Métodos funcionais de array
const pessoas = [
  { nome: "Carlos", nota: 8 },
  { nome: "Diana",  nota: 6 },
  { nome: "Eva",    nota: 9 },
];

const aprovados = pessoas
  .filter(p => p.nota >= 7)
  .map(p => ({ ...p, status: "aprovado" }))
  .sort((a, b) => b.nota - a.nota);

// Array.at() — indexação negativa (ES2022)
console.log(aprovados.at(-1)); // último aprovado

// Object.groupBy (ES2024)
const porStatus = Object.groupBy(pessoas, p =>
  p.nota >= 7 ? "aprovado" : "reprovado"
);

Assincronismo: Callbacks, Promises e async/await

JavaScript — Promises encadeadas e async/await com fetch
// Promise: representação de valor futuro
function esperar(ms) {
  return new Promise(resolve => setTimeout(resolve, ms));
}

// Promise.all — paralelo, falha se qualquer falhar
const [a, b] = await Promise.all([
  esperar(100).then(() => 1),
  esperar(200).then(() => 2),
]);

// Promise.allSettled — aguarda todas independente de erros
const resultados = await Promise.allSettled([
  fetch("/api/usuario"),
  fetch("/api/config"),
]);
for (const r of resultados) {
  if (r.status === "fulfilled") console.log(await r.value.json());
  else console.error(r.reason);
}

// async/await com tratamento de erro e AbortController
async function buscarUsuario(id) {
  const controller = new AbortController();
  const timer = setTimeout(() => controller.abort(), 5000);
  try {
    const res = await fetch(`/api/usuarios/${id}`, {
      signal: controller.signal,
    });
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    return await res.json();
  } catch (e) {
    if (e.name === "AbortError") throw new Error("Timeout");
    throw e;
  } finally {
    clearTimeout(timer);
  }
}

Manipulação do DOM

JavaScript (Browser) — querySelector, events e template literals
// Seleção
const btn   = document.querySelector("#btn-enviar");
const lista = document.querySelector(".lista-itens");
const todos = [...document.querySelectorAll("li")]; // NodeList → Array

// Criar e inserir elemento
function adicionarItem(texto) {
  const li = document.createElement("li");
  li.textContent = texto;
  li.setAttribute("data-criado", new Date().toISOString());
  lista.append(li);
}

// Event delegation — um listener na lista inteira
lista.addEventListener("click", (e) => {
  if (e.target.matches("li")) {
    e.target.classList.toggle("concluido");
  }
});

// IntersectionObserver: lazy load / animações ao entrar na tela
const observer = new IntersectionObserver((entries) => {
  for (const entry of entries) {
    if (entry.isIntersecting) {
      entry.target.classList.add("visivel");
      observer.unobserve(entry.target);
    }
  }
}, { threshold: 0.1 });

document.querySelectorAll(".animar").forEach(el => observer.observe(el));

Classes e Herança Prototipal

JavaScript — Classes ES2022: private fields, static, getter/setter
class Conta {
  // Private fields (ES2022)
  #saldo = 0;
  #historico = [];

  static #totalContas = 0;

  constructor(public titular, valorInicial = 0) {
    this.titular = titular;
    this.#depositar(valorInicial);
    Conta.#totalContas++;
  }

  get saldo() { return this.#saldo; }

  #depositar(valor) {
    if (valor <= 0) throw new RangeError("Valor inválido");
    this.#saldo += valor;
    this.#historico.push({ tipo: "deposito", valor, data: new Date() });
  }

  depositar(valor) { return this.#depositar(valor); }

  sacar(valor) {
    if (valor > this.#saldo) throw new RangeError("Saldo insuficiente");
    this.#saldo -= valor;
    this.#historico.push({ tipo: "saque", valor, data: new Date() });
  }

  get extrato() { return [...this.#historico]; }

  static get total() { return Conta.#totalContas; }
}

class ContaCorrente extends Conta {
  #limite;
  constructor(titular, valorInicial, limite = 1000) {
    super(titular, valorInicial);
    this.#limite = limite;
  }
  sacarComLimite(valor) {
    if (valor > this.saldo + this.#limite)
      throw new RangeError("Excede limite");
    if (valor <= this.saldo) return this.sacar(valor);
    this.sacar(this.saldo); // esgota saldo, usa limite
  }
}
