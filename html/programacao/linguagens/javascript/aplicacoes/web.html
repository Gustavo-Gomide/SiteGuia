Express: API REST com JWT e Middleware

Node.js + Express — API REST autenticada com JWT e Zod
// src/app.js
import express   from "express";
import helmet    from "helmet";
import cors      from "cors";
import jwt       from "jsonwebtoken";
import bcrypt    from "bcrypt";
import { z }     from "zod";

const app    = express();
const SECRET = process.env.JWT_SECRET; // nunca hardcoded
const db     = new Map(); // substituir por banco real

app.use(helmet());
app.use(cors({ origin: process.env.ALLOWED_ORIGIN }));
app.use(express.json({ limit: "50kb" }));

// Middleware de autenticacao
function autenticar(req, res, next) {
  const auth = req.headers.authorization;
  if (!auth?.startsWith("Bearer "))
    return res.sendStatus(401);
  try {
    req.usuario = jwt.verify(auth.slice(7), SECRET);
    next();
  } catch {
    res.sendStatus(401);
  }
}

// Rota de login
const LoginSchema = z.object({
  email: z.string().email(),
  senha: z.string().min(8),
});

app.post("/auth/login", async (req, res) => {
  const resultado = LoginSchema.safeParse(req.body);
  if (!resultado.success)
    return res.status(400).json(resultado.error.flatten());

  const { email, senha } = resultado.data;
  const usuario = db.get(email);
  if (!usuario || !(await bcrypt.compare(senha, usuario.hash)))
    return res.sendStatus(401);

  const token = jwt.sign(
    { sub: usuario.id, email },
    SECRET,
    { expiresIn: "2h", algorithm: "HS256" }
  );
  res.json({ token });
});

// CRUD protegido
app.get("/api/perfil", autenticar, (req, res) => {
  res.json({ usuario: req.usuario });
});

// Tratamento global de erros
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(500).json({ erro: "Erro interno" }); // sem stack em produção
});

app.listen(3000);

Fastify: Performance e Schema Nativo

Node.js + Fastify — Plugin, schema JSON, hooks e serialização rápida
import Fastify      from "fastify";
import fastifyJwt   from "@fastify/jwt";
import fastifyCors  from "@fastify/cors";
import fastifySensible from "@fastify/sensible";

const app = Fastify({ logger: true });
await app.register(fastifyCors, { origin: true });
await app.register(fastifyJwt, { secret: process.env.JWT_SECRET });
await app.register(fastifySensible);

// Rota com schema JSON (validação + serialização automática)
app.post("/api/tarefas", {
  schema: {
    body: {
      type: "object",
      required: ["titulo"],
      properties: {
        titulo:    { type: "string", minLength: 1, maxLength: 200 },
        descricao: { type: "string" },
        prioridade: { type: "integer", minimum: 1, maximum: 5, default: 3 },
      },
      additionalProperties: false,  // rejeita campos não declarados
    },
    response: {
      201: {
        type: "object",
        properties: {
          id:     { type: "string", format: "uuid" },
          titulo: { type: "string" },
        },
      },
    },
  },
  async handler(req, reply) {
    const tarefa = { id: crypto.randomUUID(), ...req.body };
    return reply.code(201).send(tarefa);
  },
});

await app.listen({ port: 3000, host: "0.0.0.0" });

WebSockets: Chat em Tempo Real com ws

Node.js + ws — Chat com rooms e heartbeat
import { WebSocketServer, WebSocket } from "ws";
import { createServer } from "node:http";

const httpServer = createServer();
const wss = new WebSocketServer({ server: httpServer });
const salas = new Map(); // sala → Set<WebSocket>

wss.on("connection", (ws) => {
  ws.isAlive = true;
  ws.on("pong", () => { ws.isAlive = true; });
  ws.on("message", (dados) => {
    let msg;
    try { msg = JSON.parse(dados); }
    catch { return; }

    if (msg.tipo === "entrar") {
      ws.sala = msg.sala;
      if (!salas.has(msg.sala)) salas.set(msg.sala, new Set());
      salas.get(msg.sala).add(ws);
    } else if (msg.tipo === "mensagem" && ws.sala) {
      const payload = JSON.stringify({ de: msg.usuario, texto: msg.texto });
      for (const cliente of salas.get(ws.sala) ?? []) {
        if (cliente !== ws && cliente.readyState === WebSocket.OPEN)
          cliente.send(payload);
      }
    }
  });
  ws.on("close", () => salas.get(ws.sala)?.delete(ws));
});

// Heartbeat: remover clientes inativos a cada 30s
setInterval(() => {
  wss.clients.forEach((ws) => {
    if (!ws.isAlive) return ws.terminate();
    ws.isAlive = false;
    ws.ping();
  });
}, 30_000);

httpServer.listen(8080);

Deploy com Docker Multi-Stage

Dockerfile — Node.js multi-stage com usuario nao-root
FROM node:22-alpine AS deps
WORKDIR /app
COPY package*.json ./
RUN npm ci --only=production

FROM node:22-alpine AS runner
WORKDIR /app
RUN addgroup -S nodejs && adduser -S node -G nodejs
COPY --from=deps /app/node_modules ./node_modules
COPY src ./src
USER node
ENV NODE_ENV=production PORT=3000
EXPOSE 3000
HEALTHCHECK --interval=30s --timeout=5s \
  CMD wget -qO- http://localhost:3000/health || exit 1
CMD ["node", "src/app.js"]
