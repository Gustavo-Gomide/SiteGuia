JavaScript para Desktop

Electron empacota uma aplicacao web como app nativo. Tauri usa um backend Rust leve com WebView do SO, gerando binarios muito menores.

Electron e usado por VS Code, Slack, Figma e Discord. Tauri e a alternativa moderna: backend Rust com seguranca de tipos, instalador de 2-8 MB, API de sistema via comandos, e acesso opcional a Node.js via sidecar.

Electron: Main, Renderer e contextBridge

Electron — main.js: criacao de janela e IPC seguro
// main.js (processo principal Node.js)
import { app, BrowserWindow, ipcMain, dialog, shell } from "electron";
import { join } from "node:path";
import { readFile, writeFile } from "node:fs/promises";

let win;

function criarJanela() {
  win = new BrowserWindow({
    width:  1200,
    height: 800,
    webPreferences: {
      preload:          join(__dirname, "preload.js"),
      contextIsolation: true,   // OBRIGATORIO para seguranca
      nodeIntegration:  false,  // nao expor Node diretamente
    },
  });

  win.loadFile("dist/index.html");

  // Abrir devtools so em desenvolvimento
  if (process.env.NODE_ENV === "development")
    win.webContents.openDevTools();
}

app.whenReady().then(() => {
  criarJanela();

  app.on("activate", () => {
    if (BrowserWindow.getAllWindows().length === 0) criarJanela();
  });
});

app.on("window-all-closed", () => {
  if (process.platform !== "darwin") app.quit();
});

// Handler IPC: abrir arquivo
ipcMain.handle("abrir-arquivo", async () => {
  const { canceled, filePaths } = await dialog.showOpenDialog(win, {
    properties: ["openFile"],
    filters: [{ name: "Texto", extensions: ["txt", "md"] }],
  });
  if (canceled) return null;
  return readFile(filePaths[0], "utf8");
});

// Handler IPC: salvar arquivo
ipcMain.handle("salvar-arquivo", async (_event, conteudo) => {
  const { canceled, filePath } = await dialog.showSaveDialog(win, {
    filters: [{ name: "Texto", extensions: ["txt"] }],
  });
  if (canceled) return false;
  await writeFile(filePath, conteudo, "utf8");
  return true;
});

// Abrir links externos no browser padrao
ipcMain.on("abrir-url-externa", (_e, url) => shell.openExternal(url));
Electron — preload.js e uso no renderer
// preload.js — ponte segura entre main e renderer
const { contextBridge, ipcRenderer } = require("electron");

contextBridge.exposeInMainWorld("api", {
  abrirArquivo:   ()       => ipcRenderer.invoke("abrir-arquivo"),
  salvarArquivo:  (texto)  => ipcRenderer.invoke("salvar-arquivo", texto),
  abrirUrl:       (url)    => ipcRenderer.send("abrir-url-externa", url),
  onProgresso:    (cb)     => ipcRenderer.on("progresso", (_e, p) => cb(p)),
});

// renderer.js — acesso via window.api (sem Node.js diretto)
const btn = document.getElementById("abrir");
btn.addEventListener("click", async () => {
  const conteudo = await window.api.abrirArquivo();
  if (conteudo) document.getElementById("editor").value = conteudo;
});

window.api.onProgresso((pct) => {
  document.getElementById("barra").style.width = pct + "%";
});

Tauri: WebView Nativo + Backend Rust

Tauri — invoke comando Rust + event listen no frontend JS
// frontend/main.js — chamada de comando Rust via @tauri-apps/api
import { invoke }  from "@tauri-apps/api/core";
import { listen }  from "@tauri-apps/api/event";
import { open }    from "@tauri-apps/plugin-dialog";
import { readTextFile, writeTextFile } from "@tauri-apps/plugin-fs";

// Chamar comando Rust (type-safe com TypeScript)
const resultado = await invoke("greet", { name: "Mundo" });
console.log(resultado); // "Ola, Mundo! (de Rust)"

// Escutar evento emitido pelo backend Rust
await listen("progresso-download", (event) => {
  const { pct, arquivo } = event.payload;
  document.getElementById("barra").style.width = pct + "%";
  document.getElementById("status").textContent = arquivo;
});

// Abrir dialogo nativo do SO e ler arquivo
const caminho = await open({ multiple: false, directory: false });
if (caminho) {
  const conteudo = await readTextFile(caminho);
  document.getElementById("editor").value = conteudo;
}
