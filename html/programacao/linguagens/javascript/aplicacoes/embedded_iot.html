Johnny-Five: Arduino com Node.js

JavaScript + Johnny-Five — LED, Sensor e Servo
// Instalar: npm i johnny-five
// Arduino precisa ter Firmata carregado (arquivo > exemplos > StandardFirmata)
import { Board, Led, Button, Sensor, Servo, Thermometer } from "johnny-five";

const board = new Board({ port: "/dev/ttyUSB0" }); // ou "COM3" no Windows

board.on("ready", () => {
  console.log("Board pronta!");

  // LED no pino 13
  const led = new Led(13);
  led.blink(500); // 500ms on/off

  // Botao no pino 7 com pull-up interno
  const botao = new Button({ pin: 7, isPullup: true });
  botao.on("press",   () => led.on());
  botao.on("release", () => led.off());

  // Sensor analogico (potenciometro no A0)
  const pot = new Sensor({ pin: "A0", freq: 250 });
  pot.on("change", function() {
    const pct = Math.round(this.scaleTo(0, 100));
    console.log(`Potenciometro: ${pct}%`);
  });

  // Servo no pino 9
  const servo = new Servo({ pin: 9, startAt: 90 });
  let angulo = 0;
  setInterval(() => {
    angulo = angulo === 0 ? 180 : 0;
    servo.to(angulo);
  }, 2000);

  // Termometro LM35 no A1
  const termo = new Thermometer({
    controller: "LM35", pin: "A1", freq: 1000,
  });
  termo.on("change", function() {
    console.log(`Temperatura: ${this.celsius.toFixed(1)} C`);
  });
});

MQTT.js: Mensageria IoT

JavaScript + MQTT.js — Publisher e Subscriber com TLS
import mqtt from "mqtt";
import { readFileSync } from "node:fs";

// Publisher: sensor publicando telemetria
const publisher = mqtt.connect("mqtts://broker.example.com:8883", {
  clientId:  `sensor-${crypto.randomUUID()}`,
  username:  process.env.MQTT_USER,
  password:  process.env.MQTT_PASS,
  ca:        [readFileSync("ca.crt")],
  rejectUnauthorized: true,
  clean:     true,
  keepalive: 60,
  reconnectPeriod: 5000,
  will: {
    topic:   "sensores/sala/status",
    payload: "offline",
    qos:     1,
    retain:  true,
  },
});

publisher.on("connect", () => {
  console.log("Conectado ao broker MQTT");
  publisher.publish("sensores/sala/status", "online", { retain: true });

  setInterval(() => {
    const payload = JSON.stringify({
      temperatura: (20 + Math.random() * 10).toFixed(1),
      umidade:     (40 + Math.random() * 30).toFixed(1),
      ts:          new Date().toISOString(),
    });
    publisher.publish("sensores/sala/telemetria", payload, { qos: 1 });
  }, 5000);
});

publisher.on("error", (err) => console.error("Erro MQTT:", err));

// Subscriber: dashboard consumindo telemetria
const sub = mqtt.connect("mqtts://broker.example.com:8883", {
  clientId: `dashboard-${crypto.randomUUID()}`,
  ca: [readFileSync("ca.crt")],
});

sub.on("connect", () => {
  // Subscrever multiplos topicos com wildcard
  sub.subscribe(["sensores/+/telemetria", "sensores/+/status"], { qos: 1 });
});

sub.on("message", (topic, payload) => {
  console.log(`[${topic}]`, payload.toString());
});

// Desconectar graciosamente
process.once("SIGINT", () => {
  sub.end();
  publisher.end();
});

Node-RED: Fluxos Visuais

JavaScript — Node-RED: node personalizado + function node
// Instalar e iniciar: npm i -g node-red && node-red
// Dashboard: http://localhost:1880

// Criar um node personalizado: my-sensor/my-sensor.js
module.exports = function(RED) {
  function MySensorNode(config) {
    RED.nodes.createNode(this, config);
    const node    = this;
    const intervalo = config.intervalo || 5000;

    const timer = setInterval(() => {
      const leitura = (20 + Math.random() * 10).toFixed(2);
      node.send({
        topic:   "temperatura",
        payload: Number(leitura),
      });
    }, intervalo);

    node.on("close", () => clearInterval(timer));
  }

  RED.nodes.registerType("my-sensor", MySensorNode);
};
