React Native: Componentes Nativos

React Native — App com navegacao Stack e FlatList
// App.tsx — navegacao com React Navigation
import React, { useState, useEffect } from "react";
import { NavigationContainer }           from "@react-navigation/native";
import { createNativeStackNavigator }     from "@react-navigation/native-stack";
import {
  View, Text, FlatList, TouchableOpacity,
  StyleSheet, ActivityIndicator, RefreshControl,
} from "react-native";

const Stack = createNativeStackNavigator();

// Tela Lista
function ListaScreen({ navigation }) {
  const [posts, setPosts]           = useState([]);
  const [carregando, setCarregando] = useState(true);
  const [refreshing, setRefreshing] = useState(false);

  async function carregar() {
    const res  = await fetch("https://jsonplaceholder.typicode.com/posts?_limit=20");
    const data = await res.json();
    setPosts(data);
    setCarregando(false);
    setRefreshing(false);
  }

  useEffect(() => { carregar(); }, []);

  if (carregando) return (
    <View style={styles.centro}>
      <ActivityIndicator size="large" color="#7c3aed" />
    </View>
  );

  return (
    <FlatList
      data={posts}
      keyExtractor={(item) => String(item.id)}
      refreshControl={<RefreshControl refreshing={refreshing} onRefresh={() => { setRefreshing(true); carregar(); }} />}
      renderItem={({ item }) => (
        <TouchableOpacity
          style={styles.card}
          onPress={() => navigation.navigate("Detalhe", { post: item })}
        >
          <Text style={styles.titulo} numberOfLines={1}>{item.title}</Text>
          <Text style={styles.corpo}  numberOfLines={2}>{item.body}</Text>
        </TouchableOpacity>
      )}
    />
  );
}

// Tela Detalhe
function DetalheScreen({ route }) {
  const { post } = route.params;
  return (
    <View style={styles.detalhe}>
      <Text style={styles.titulo}>{post.title}</Text>
      <Text style={styles.corpo}>{post.body}</Text>
    </View>
  );
}

// App principal
export default function App() {
  return (
    <NavigationContainer>
      <Stack.Navigator initialRouteName="Lista">
        <Stack.Screen name="Lista"   component={ListaScreen}  options={{ title: "Posts" }} />
        <Stack.Screen name="Detalhe" component={DetalheScreen} options={{ title: "Detalhe" }} />
      </Stack.Navigator>
    </NavigationContainer>
  );
}

const styles = StyleSheet.create({
  centro:  { flex: 1, justifyContent: "center", alignItems: "center" },
  card:    { padding: 16, borderBottomWidth: 1, borderColor: "#333" },
  titulo:  { fontSize: 16, fontWeight: "bold", color: "#e2e8f0" },
  corpo:   { fontSize: 14, color: "#94a3b8", marginTop: 4 },
  detalhe: { flex: 1, padding: 20 },
});

Expo: Managed Workflow

Expo — Camara, notificacoes e EAS Build
// npx create-expo-app meu-app --template tabs
// npx expo start      → QR code para Expo Go
// npx eas build --platform android   → APK/AAB na nuvem EAS

import { Camera, CameraType }         from "expo-camera";
import * as Notifications               from "expo-notifications";
import * as Location                    from "expo-location";
import { useState, useRef, useEffect } from "react";
import { View, Button, Image, Text }   from "react-native";

export default function CameraScreen() {
  const [permissao, pedirPermissao] = Camera.useCameraPermissions();
  const [foto, setFoto]              = useState(null);
  const camRef                        = useRef(null);

  useEffect(() => {
    async function init() {
      // Permissao de notificacoes
      const { status } = await Notifications.requestPermissionsAsync();
      if (status === "granted")
        await Notifications.scheduleNotificationAsync({
          content:  { title: "Lembrete!", body: "Nao se esqueca de tirar fotos." },
          trigger:  { seconds: 60 },
        });

      // Localizacao
      const locPerm = await Location.requestForegroundPermissionsAsync();
      if (locPerm.status === "granted") {
        const loc = await Location.getCurrentPositionAsync({});
        console.log(loc.coords.latitude, loc.coords.longitude);
      }
    }
    init();
  }, []);

  if (!permissao?.granted) return (
    <View><Button title="Permitir camera" onPress={pedirPermissao} /></View>
  );

  return (
    <View style={{ flex: 1 }}>
      {!foto ? (
        <>
          <Camera style={{ flex: 1 }} type={CameraType.back} ref={camRef} />
          <Button title="Fotografar" onPress={async () => {
            const img = await camRef.current.takePictureAsync({ quality: 0.8 });
            setFoto(img.uri);
          }} />
        </>
      ) : (
        <>
          <Image source={{ uri: foto }} style={{ flex: 1 }} />
          <Button title="Nova foto" onPress={() => setFoto(null)} />
        </>
      )}
    </View>
  );
}
