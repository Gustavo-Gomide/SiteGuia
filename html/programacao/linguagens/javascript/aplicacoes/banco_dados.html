Prisma: ORM com Auto-complete e Migrations

Prisma Schema — PostgreSQL com relações e enums
// prisma/schema.prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")  // nunca hardcoded
}

model Usuario {
  id        String    @id @default(uuid())
  email     String    @unique
  nome      String
  perfil    Perfil    @default(LEITOR)
  criado_em DateTime  @default(now())
  posts     Post[]
  @@index([email])
}

model Post {
  id        String    @id @default(uuid())
  titulo    String    @db.VarChar(200)
  corpo     String
  publicado Boolean   @default(false)
  autor     Usuario   @relation(fields: [autorId], references: [id])
  autorId   String
  criado_em DateTime  @default(now())
  @@index([autorId, publicado])
}

enum Perfil { ADMIN EDITOR LEITOR }
JavaScript + Prisma — CRUD com transacao e include
import { PrismaClient } from "@prisma/client";

// Singleton — evitar múltiplas conexões em dev (HMR)
const prisma = globalThis.__prisma ?? new PrismaClient();
if (process.env.NODE_ENV !== "production") globalThis.__prisma = prisma;

// Busca com relacionamento e paginacao cursor-based
async function listarPosts({ cursor, limite = 10 } = {}) {
  return prisma.post.findMany({
    where:   { publicado: true },
    include: { autor: { select: { id: true, nome: true } } },
    orderBy: { criado_em: "desc" },
    take:    limite,
    ...(cursor && { cursor: { id: cursor }, skip: 1 }),
  });
}

// Transação: mover créditos entre usuários
async function transferirCreditos(deId, paraId, valor) {
  return prisma.$transaction(async (tx) => {
    const [de, para] = await Promise.all([
      tx.usuario.findUniqueOrThrow({ where: { id: deId } }),
      tx.usuario.findUniqueOrThrow({ where: { id: paraId } }),
    ]);
    if (de.creditos < valor) throw new Error("Saldo insuficiente");
    await tx.usuario.update({
      where: { id: deId }, data: { creditos: { decrement: valor } },
    });
    await tx.usuario.update({
      where: { id: paraId }, data: { creditos: { increment: valor } },
    });
  });
}

Drizzle ORM: SQL Type-Safe sem Magia

JavaScript + Drizzle — Schema, query builder e migrations
// db/schema.js — Drizzle com PostgreSQL
import { pgTable, text, integer, boolean, timestamp, uuid }
  from "drizzle-orm/pg-core";
import { relations } from "drizzle-orm";

export const usuarios = pgTable("usuarios", {
  id:       uuid("id").primaryKey().defaultRandom(),
  email:    text("email").notNull().unique(),
  nome:     text("nome").notNull(),
  ativo:    boolean("ativo").default(true),
  criado_em: timestamp("criado_em").defaultNow(),
});

export const usuariosRelations = relations(usuarios, ({ many }) => ({
  posts: many(posts),
}));

// db/index.js
import { drizzle }    from "drizzle-orm/postgres-js";
import postgres       from "postgres";
import * as schema   from "./schema.js";

const client = postgres(process.env.DATABASE_URL);
export const db = drizzle(client, { schema });

// Queries type-safe
import { eq, and, gt, desc } from "drizzle-orm";

const ativos = await db
  .select({ id: usuarios.id, nome: usuarios.nome })
  .from(usuarios)
  .where(eq(usuarios.ativo, true))
  .orderBy(desc(usuarios.criado_em))
  .limit(20);

Redis com ioredis: Cache e Rate Limiting

Node.js + ioredis — Cache, pub/sub e Lua rate limiter
import Redis from "ioredis";

const redis = new Redis(process.env.REDIS_URL);

// Decorator de cache genérico
function comCache(fn, { chave, ttlSegundos = 300 }) {
  return async (...args) => {
    const key = typeof chave === "function" ? chave(...args) : chave;
    const cached = await redis.get(key);
    if (cached) return JSON.parse(cached);
    const result = await fn(...args);
    await redis.setex(key, ttlSegundos, JSON.stringify(result));
    return result;
  };
}

const buscarUsuario = comCache(
  (id) => db.usuario.findUnique({ where: { id } }),
  { chave: (id) => `usuario:${id}`, ttlSegundos: 60 }
);

// Rate limiting com sliding window (script Lua atômico)
const rateLimitScript = `
  local key  = KEYS[1]
  local now  = tonumber(ARGV[1])
  local win  = tonumber(ARGV[2])
  local max  = tonumber(ARGV[3])
  redis.call('zremrangebyscore', key, 0, now - win)
  local count = redis.call('zcard', key)
  if count >= max then return 0 end
  redis.call('zadd', key, now, now)
  redis.call('expire', key, math.ceil(win / 1000))
  return 1
`;

async function permitir(ip, { janela = 60000, max = 100 } = {}) {
  const ok = await redis.eval(
    rateLimitScript, 1,
    `rl:${ip}`, Date.now(), janela, max
  );
  return ok === 1;
}

// Pub/Sub para invalidação de cache
const sub = new Redis(process.env.REDIS_URL);
sub.subscribe("cache:invalidar");
sub.on("message", (_, key) => redis.del(key));
