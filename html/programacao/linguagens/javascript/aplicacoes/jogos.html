JavaScript para Jogos

Phaser 3 para jogos 2D, Canvas API puro para controle total e Three.js para cenas 3D no browser.

O browser e uma plataforma de jogos madura: Canvas 2D, WebGL/WebGL2, WebGPU, WebAssembly (para ports de C++/Rust), Web Audio API e GamePad API. Phaser 3 abstrai tudo isso com um game loop, fisica arcade e gerenciamento de cenas.

Phaser 3: Framework 2D Completo

JavaScript + Phaser 3 — Plataformer com fisica arcade e tilemap
import Phaser from "phaser";

class Jogo extends Phaser.Scene {
  constructor() { super({ key: "Jogo" }); }

  preload() {
    // Carregar assets
    this.load.image("fundo",     "assets/fundo.png");
    this.load.image("chao-tiles","assets/tileset.png");
    this.load.tilemapTiledJSON("mapa", "assets/mapa.json");
    this.load.spritesheet("player", "assets/heroi.png", {
      frameWidth: 32, frameHeight: 48,
    });
    this.load.audio("pulo", "assets/pulo.ogg");
  }

  create() {
    // Tilemap
    const mapa   = this.make.tilemap({ key: "mapa" });
    const tiles  = mapa.addTilesetImage("tileset", "chao-tiles");
    const chao   = mapa.createLayer("Plataformas", tiles, 0, 0);
    chao.setCollisionByProperty({ colide: true });

    // Player com fisica arcade
    this.player = this.physics.add.sprite(100, 300, "player");
    this.player.body.setGravityY(400);
    this.player.body.setCollideWorldBounds(true);

    // Animacoes
    this.anims.create({
      key:        "correr",
      frames:     this.anims.generateFrameNumbers("player", { start: 0, end: 7 }),
      frameRate:  12,
      repeat:     -1,
    });
    this.anims.create({
      key:       "parado",
      frames:    this.anims.generateFrameNumbers("player", { start: 8, end: 11 }),
      frameRate: 6,
      repeat:    -1,
    });

    // Colisao player x tilemap
    this.physics.add.collider(this.player, chao);

    // Teclado
    this.cursors = this.input.keyboard.createCursorKeys();
    this.pulo    = this.sound.add("pulo");

    // Pontuacao
    this.score   = 0;
    this.scoreTxt = this.add.text(16, 16, "Pontos: 0", {
      fontSize: "18px", fill: "#fff",
    }).setScrollFactor(0); // fixo na camera

    // Camera segue player
    this.cameras.main.startFollow(this.player, true, 0.08, 0.08);
    this.cameras.main.setBounds(0, 0, mapa.widthInPixels, mapa.heightInPixels);
  }

  update() {
    const { left, right, up } = this.cursors;
    const noChao = this.player.body.blocked.down;

    if (left.isDown()) {
      this.player.body.setVelocityX(-160);
      this.player.anims.play("correr", true);
      this.player.setFlipX(true);
    } else if (right.isDown()) {
      this.player.body.setVelocityX(160);
      this.player.anims.play("correr", true);
      this.player.setFlipX(false);
    } else {
      this.player.body.setVelocityX(0);
      this.player.anims.play("parado", true);
    }

    if (Phaser.Input.Keyboard.JustDown(up) && noChao) {
      this.player.body.setVelocityY(-440);
      this.pulo.play();
    }
  }
}

new Phaser.Game({
  type:   Phaser.AUTO,
  width:  800,
  height: 480,
  physics: { default: "arcade", arcade: { gravity: { y: 0 }, debug: false } },
  scene:   [Jogo],
});

Canvas API: Game Loop Manual

JavaScript + Canvas 2D — Asteroids simples com colisao AABB
const canvas = document.getElementById("game");
const ctx    = canvas.getContext("2d");
const W = canvas.width  = 800;
const H = canvas.height = 600;

const nave = { x: W/2, y: H/2, vx: 0, vy: 0, w: 24, h: 24, tiros: [] };
const asteroids = Array.from({ length: 5 }, () => ({
  x:  Math.random() * W,
  y:  Math.random() * H,
  vx: (Math.random() - 0.5) * 2,
  vy: (Math.random() - 0.5) * 2,
  r:  30 + Math.random() * 20,
}));

const keys = {};
document.addEventListener("keydown", e => { keys[e.code] = true; });
document.addEventListener("keyup",   e => { keys[e.code] = false; });

function aabb(a, b) {
  return a.x < b.x + b.w && a.x + a.w > b.x &&
         a.y < b.y + b.h && a.y + a.h > b.y;
}

function loop() {
  ctx.fillStyle = "#0a0a0a";
  ctx.fillRect(0, 0, W, H);

  // Input
  if (keys["ArrowLeft"])  nave.vx -= 0.3;
  if (keys["ArrowRight"]) nave.vx += 0.3;
  if (keys["ArrowUp"])    nave.vy -= 0.3;
  nave.vx *= 0.98; nave.vy *= 0.98;

  // Atualizar nave
  nave.x = (nave.x + nave.vx + W) % W;
  nave.y = (nave.y + nave.vy + H) % H;

  // Tiro
  if (keys["Space"] && nave.tiros.length < 5)
    nave.tiros.push({ x: nave.x, y: nave.y, vx: 6, vy: -6, w: 4, h: 4 });

  // Desenhar e atualizar tiros
  nave.tiros = nave.tiros
    .map(t => ({ ...t, x: t.x + t.vx, y: t.y + t.vy }))
    .filter(t => t.x > 0 && t.x < W && t.y > 0 && t.y < H);

  nave.tiros.forEach(t => {
    ctx.fillStyle = "#ff0";
    ctx.fillRect(t.x, t.y, t.w, t.h);
  });

  // Asteroids
  asteroids.forEach(a => {
    a.x = (a.x + a.vx + W) % W;
    a.y = (a.y + a.vy + H) % H;
    ctx.beginPath();
    ctx.arc(a.x, a.y, a.r, 0, Math.PI * 2);
    ctx.strokeStyle = "#888";
    ctx.stroke();
  });

  // Nave
  ctx.fillStyle = "#0f0";
  ctx.fillRect(nave.x - nave.w/2, nave.y - nave.h/2, nave.w, nave.h);

  requestAnimationFrame(loop);
}
loop();

Three.js: Cenas 3D no Browser

JavaScript + Three.js — Cena com luz, PBR e animacao
import * as THREE from "three";
import { OrbitControls } from "three/addons/controls/OrbitControls.js";
import { GLTFLoader }    from "three/addons/loaders/GLTFLoader.js";

// Renderer
const renderer = new THREE.WebGLRenderer({ antialias: true });
renderer.setSize(window.innerWidth, window.innerHeight);
renderer.shadowMap.enabled = true;
renderer.shadowMap.type    = THREE.PCFSoftShadowMap;
renderer.toneMapping       = THREE.ACESFilmicToneMapping;
document.body.appendChild(renderer.domElement);

// Cena e camera
const scene  = new THREE.Scene();
scene.background = new THREE.Color("#1a1a2e");
const camera = new THREE.PerspectiveCamera(60, innerWidth / innerHeight, 0.1, 100);
camera.position.set(0, 3, 5);

// Controls
const controls     = new OrbitControls(camera, renderer.domElement);
controls.enableDamping = true;

// Iluminacao PBR
scene.add(new THREE.AmbientLight("#ffffff", 0.4));
const sol = new THREE.DirectionalLight("#ffe0b0", 2);
sol.position.set(5, 10, 5);
sol.castShadow = true;
scene.add(sol);

// Chao com MeshStandardMaterial
const chao = new THREE.Mesh(
  new THREE.PlaneGeometry(20, 20),
  new THREE.MeshStandardMaterial({ color: "#334", roughness: 0.9 })
);
chao.rotation.x   = -Math.PI / 2;
chao.receiveShadow = true;
scene.add(chao);

// Cubo animado
const cubo = new THREE.Mesh(
  new THREE.BoxGeometry(1, 1, 1),
  new THREE.MeshStandardMaterial({ color: "#e040fb", metalness: 0.3, roughness: 0.5 })
);
cubo.position.y    = 1;
cubo.castShadow    = true;
scene.add(cubo);

// Carregar modelo GLTF
const loader = new GLTFLoader();
loader.load("modelo.glb", (gltf) => {
  scene.add(gltf.scene);
  // Animacoes
  const mixer = new THREE.AnimationMixer(gltf.scene);
  gltf.animations.forEach(clip => mixer.clipAction(clip).play());
});

// Loop de renderizacao
const clock = new THREE.Clock();
function animar() {
  requestAnimationFrame(animar);
  const delta = clock.getDelta();
  cubo.rotation.x += delta * 0.8;
  cubo.rotation.y += delta * 1.2;
  controls.update();
  renderer.render(scene, camera);
}
animar();

// Responsivo
window.addEventListener("resize", () => {
  camera.aspect = innerWidth / innerHeight;
  camera.updateProjectionMatrix();
  renderer.setSize(innerWidth, innerHeight);
});
