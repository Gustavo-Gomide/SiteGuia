JavaScript permite executar modelos de machine learning diretamente no browser via WebGL/WebGPU, garantindo privacidade total dos dados do usuário. O Transformers.js da Hugging Face porta a API popular do Python para JS com suporte a WebAssembly e WebGPU.

TensorFlow.js: Redes Neurais no Browser

JavaScript + TensorFlow.js — MLP de classificacao treinado no browser
import * as tf from "@tensorflow/tfjs";

// Dataset XOR
const xs = tf.tensor2d([[0,0],[0,1],[1,0],[1,1]]);
const ys = tf.tensor2d([[0],[1],[1],[0]]);

// Construindo modelo sequencial
const modelo = tf.sequential({
  layers: [
    tf.layers.dense({ inputShape: [2], units: 8, activation: "relu" }),
    tf.layers.dense({ units: 4, activation: "relu" }),
    tf.layers.dense({ units: 1, activation: "sigmoid" }),
  ],
});

modelo.compile({
  optimizer: tf.train.adam(0.01),
  loss:      "binaryCrossentropy",
  metrics:   ["accuracy"],
});

// Treinar com callback de progresso
await modelo.fit(xs, ys, {
  epochs:   500,
  verbose:  0,
  callbacks: {
    onEpochEnd(epoch, logs) {
      if (epoch % 100 === 0)
        console.log(`Época ${epoch}: loss=${logs.loss.toFixed(4)}`);
    },
  },
});

// Predição
const pred = modelo.predict(tf.tensor2d([[0,1],[1,1]]));
pred.print(); // ~[[0.99], [0.01]]

// Salvar modelo (IndexedDB no browser)
await modelo.save("indexeddb://meu-modelo");
// Carregar modelo salvo
const carregado = await tf.loadLayersModel("indexeddb://meu-modelo");

// Transferir aprendizado: MobileNet para classificação
const mobilenet = await tf.loadGraphModel(
  "https://tfhub.dev/google/tfjs-model/imagenet/mobilenet_v3_small_100_224/classification/5/default/1",
  { fromTFHub: true }
);
// Extrair features de uma imagem
const img    = document.getElementById("img");
const tensor = tf.browser.fromPixels(img)
  .resizeBilinear([224, 224])
  .toFloat()
  .div(255)
  .expandDims(0);
const logits = mobilenet.predict(tensor);

Transformers.js: LLMs no Browser

JavaScript + Transformers.js — NLP: sentimento, traducao e embeddings
import { pipeline, env } from "@xenova/transformers";

// Configurar cache local (Node.js)
env.cacheDir = "./.cache";

// Análise de sentimento no browser (sem servidor!)
const sentimento = await pipeline(
  "sentiment-analysis",
  "Xenova/distilbert-base-uncased-finetuned-sst-2-english"
);

const resultado = await sentimento("I love JavaScript!");
console.log(resultado); // [{ label: "POSITIVE", score: 0.9997 }]

// Embeddings para busca semântica
const extractor = await pipeline(
  "feature-extraction",
  "Xenova/all-MiniLM-L6-v2"
);

async function embedding(texto) {
  const saida = await extractor(texto, { pooling: "mean", normalize: true });
  return Array.from(saida.data);
}

function similaridadeCosseno(a, b) {
  const dot    = a.reduce((s, ai, i) => s + ai * b[i], 0);
  const normA  = Math.sqrt(a.reduce((s, ai) => s + ai * ai, 0));
  const normB  = Math.sqrt(b.reduce((s, bi) => s + bi * bi, 0));
  return dot / (normA * normB);
}

const [e1, e2] = await Promise.all([
  embedding("Como fazer café?"),
  embedding("Qual a receita para preparar café?"),
]);
console.log(similaridadeCosseno(e1, e2).toFixed(4)); // ~0.92

Analise de Dados com Arquero

JavaScript + Arquero — DataFrame com groupby, join e agg
import aq from "arquero";

// Criar DataFrame de array de objetos
const vendas = aq.from([
  { produto: "A", regiao: "Sul",   valor: 120, mes: 1 },
  { produto: "B", regiao: "Norte", valor: 80,  mes: 1 },
  { produto: "A", regiao: "Sul",   valor: 200, mes: 2 },
  { produto: "B", regiao: "Norte", valor: 150, mes: 2 },
  { produto: "A", regiao: "Norte", valor: 90,  mes: 2 },
]);

// Agrupamento e agregação
const resumo = vendas
  .groupby("produto", "regiao")
  .rollup({
    total_vendas: aq.op().sum("valor"),
    media_mensal: aq.op().mean("valor"),
    contagem:     aq.op().count(),
  })
  .orderby(aq.desc("total_vendas"));

resumo.print();

// Filtro e derivação de coluna
const top = vendas
  .filter(aq.escape(d => d.valor > 100))
  .derive({ valor_com_iva: aq.escape(d => d.valor * 1.12) });

top.print();

// Ler CSV (Node.js)
import { readFileSync } from "node:fs";
const dados = aq.fromCSV(readFileSync("dados.csv", "utf8"));
console.log(dados.numRows(), dados.numCols());
