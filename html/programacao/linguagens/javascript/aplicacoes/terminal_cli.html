JavaScript para Terminal e CLI

Crie ferramentas de linha de comando profissionais com Commander.js, Inquirer.js, Chalk, Ora e Listr2 no ecossistema Node.js.

Node.js transformou JavaScript em uma plataforma primeira classe para CLIs. Ferramentas como npm, pnpm, Vite, ESLint e Prettier sao todas escritas em JS. O ecossistema oferece bibliotecas maduras para parsing de argumentos, prompts interativos e output colorido e estruturado.

Commander.js: Estrutura de Comandos

JavaScript + Commander.js — CLI com subcomandos, opcoes e acoes
#!/usr/bin/env node
// arquivo: cli.mjs
import { Command, Option } from "commander";
import { readFileSync, writeFileSync } from "node:fs";
import { resolve } from "node:path";

const program = new Command();

program
  .name("minha-cli")
  .description("Ferramenta de processamento de texto")
  .version("1.0.0");

// Subcomando: contar palavras
program
  .command("contar")
  .description("Conta palavras em um arquivo")
  .argument("<arquivo>", "caminho do arquivo de entrada")
  .addOption(new Option("-f, --formato <tipo>", "formato de saida")
    .choices(["texto", "json"])
    .default("texto"))
  .option("-v, --verbose", "exibe informacoes detalhadas")
  .action((arquivo, opcoes) => {
    const conteudo = readFileSync(resolve(arquivo), "utf8");
    const palavras = conteudo.trim().split(/\s+/).filter(Boolean);
    const linhas   = conteudo.split("\n").length;
    const chars    = conteudo.length;

    if (opcoes.verbose)
      console.log(`Arquivo: ${arquivo}`);

    if (opcoes.formato === "json") {
      console.log(JSON.stringify({ palavras: palavras.length, linhas, chars }, null, 2));
    } else {
      console.log(`Palavras: ${palavras.length}  Linhas: ${linhas}  Chars: ${chars}`);
    }
  });

// Subcomando: transformar (maiúsculas / minúsculas)
program
  .command("transformar")
  .description("Transforma o texto de um arquivo")
  .argument("<arquivo>", "arquivo de entrada")
  .argument("[saida]",   "arquivo de saida (padrao: stdout)")
  .option("-u, --upper", "converter para maiusculas")
  .option("-l, --lower", "converter para minusculas")
  .action((entrada, saida, { upper, lower }) => {
    let texto = readFileSync(resolve(entrada), "utf8");
    if (upper) texto = texto.toUpperCase();
    if (lower) texto = texto.toLowerCase();
    if (saida) writeFileSync(saida, texto);
    else process.stdout.write(texto);
  });

program.parse(process.argv);

Inquirer.js: Prompts Interativos

JavaScript + Inquirer.js — Wizard interativo para criar projeto
import { input, select, confirm, checkbox, password } from "@inquirer/prompts";
import { mkdirSync, writeFileSync } from "node:fs";

async function criarProjeto() {
  const nome = await input({
    message: "Nome do projeto:",
    validate(v) {
      if (!v.trim()) return "Nome obrigatorio";
      if (/\s/.test(v)) return "Sem espacos — use kebab-case";
      return true;
    },
  });

  const tipo = await select({
    message: "Tipo de projeto:",
    choices: [
      { name: "API REST (Express)",   value: "express"  },
      { name: "API REST (Fastify)",  value: "fastify"  },
      { name: "CLI Tool (Commander)", value: "cli"      },
      { name: "Biblioteca (ESM)",     value: "lib"      },
    ],
  });

  const features = await checkbox({
    message: "Funcionalidades extras:",
    choices: [
      { name: "TypeScript", value: "ts",   checked: true  },
      { name: "Vitest",      value: "test", checked: true  },
      { name: "ESLint",      value: "lint", checked: false },
      { name: "Docker",      value: "docker"              },
    ],
  });

  const ok = await confirm({
    message: `Criar "${nome}" (${tipo})?`,
    default: true,
  });

  if (!ok) { console.log("Cancelado."); return; }

  mkdirSync(nome, { recursive: true });
  writeFileSync(`${nome}/package.json`, JSON.stringify({
    name: nome, type: "module", version: "0.1.0",
    features,
  }, null, 2));

  console.log(`Projeto criado em ./${nome}`);
}

criarProjeto();

Output com Chalk, Ora e Listr2

JavaScript + Chalk + Ora + Listr2 — Feedback visual rico
import chalk from "chalk";
import ora   from "ora";
import { Listr } from "listr2";

// Chalk: cores e estilos no terminal
console.log(chalk.green.bold("Sucesso!"));
console.log(chalk.red("Erro:"), chalk.white("mensagem detalhada"));
console.log(chalk.bgBlue.black(" INFO "), "texto normal");

// Nivel de suporte de cores
console.log(`Chalk suporta ${chalk.level === 3 ? "16M cores" : "256 cores"}`);

// Ora: spinner animado
const spinner = ora({ text: "Buscando dados...", color: "cyan" }).start();
try {
  await fetch("https://api.exemplo.com/dados");
  spinner.succeed(chalk.green("Dados obtidos!"));
} catch (err) {
  spinner.fail(chalk.red(`Falha: ${err.message}`));
  process.exit(1);
}

// Listr2: lista de tarefas com estado visual
const tarefas = new Listr([
  {
    title: "Instalar dependencias",
    async task(ctx, task) {
      await sleep(1000);
      ctx.depsInstalados = true;
      task.title = "Dependencias instaladas";
    },
  },
  {
    title: "Compilar TypeScript",
    skip(ctx) { return !ctx.depsInstalados; },
    async task(ctx, task) {
      task.output = "tsc --build...";
      await sleep(800);
    },
  },
  {
    title: "Executar testes",
    async task() {
      await sleep(500);
    },
  },
], { concurrent: false, rendererOptions: { collapseErrors: false } });

await tarefas.run();
console.log(chalk.bold.green("\nBuild concluido!"));

const sleep = (ms) => new Promise(res => setTimeout(res, ms));
