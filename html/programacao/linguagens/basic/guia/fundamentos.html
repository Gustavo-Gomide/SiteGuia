BASIC — Fundamentos

O BASIC moderno (FreeBASIC, QB64, VB.NET) mantém a legibilidade que tornou a linguagem famosa — palavras-chave em inglês natural como PRINT, IF, FOR, INPUT — enquanto adiciona tipagem estática, subprogramas com parâmetros, e estruturas de controle sem GOTO. Esta página cobre os blocos fundamentais da linguagem usando sintaxe FreeBASIC/QB64.

O BASIC usa sufixos de tipo em nomes de variáveis ($ para String, % para Integer, ! para Single, # para Double) na sintaxe clássica — ou declaração explícita com DIM variavel AS Tipo em dialetos modernos.

Variáveis e Tipos de Dados

Declaração e Tipos — FreeBASIC / QB64
' Sintaxe moderna com DIM (recomendada)
Dim inteiro   As Integer      ' 32-bit signed (-2B a +2B)
Dim longo     As Long         ' 32-bit ou 64-bit dependendo do dialeto
Dim simples   As Single       ' float 32-bit
Dim duplo     As Double       ' float 64-bit
Dim texto     As String       ' string de tamanho variável
Dim fixo      As String * 10  ' string de tamanho fixo (10 chars)
Dim flag      As Boolean      ' True ou False (FreeBASIC)
Dim byte_val  As Byte         ' 0 a 255
Dim ubyte_val As UByte        ' igual a Byte sem sinal

' Inicialização na declaração
Dim pi        As Double = 3.14159265358979
Dim nome      As String = "Ana"
Dim ativo     As Boolean = True

' Sufixos clássicos (QBasic/GW-BASIC)
Dim n%    ' Integer
Dim x#    ' Double
Dim s$    ' String
n% = 42
x# = 2.718
s$ = "Olá"

' Constantes
Const MAX_ITEMS = 100
Const APP_NAME = "Meu Programa"

Operadores

Operadores aritméticos, relacionais e lógicos
' ── Aritméticos ──────────────────────────────────────────
Dim a As Integer = 10
Dim b As Integer = 3

Print a + b    ' 13  — adição
Print a - b    ' 7   — subtração
Print a * b    ' 30  — multiplicação
Print a / b    ' 3.333... — divisão real (retorna Double)
Print a \ b    ' 3   — divisão inteira (descarta fração)
Print a Mod b  ' 1   — resto da divisão
Print a ^ b    ' 1000 — exponenciação (a elevado a b)

' ── Concatenação de strings ──────────────────────────────
Dim s As String
s = "Olá" & ", " & "Mundo!"  ' & é o operador de concatenação
s = "Olá" + " Mundo"        ' + também funciona, mas & é mais seguro

' ── Relacionais ──────────────────────────────────────────
Print (a = b)   ' False — igualdade (= é comparação E atribuição)
Print (a <> b)  ' True  — diferente
Print (a > b)   ' True  — maior
Print (a < b)   ' False — menor
Print (a >= b)  ' True  — maior ou igual
Print (a <= b)  ' False — menor ou igual

' ── Lógicos ──────────────────────────────────────────────
Print (a > 5) And (b < 10)  ' True  — E lógico
Print (a > 5) Or  (b > 10)  ' True  — OU lógico
Print Not (a = b)            ' True  — NÃO lógico
Print (a > 5) Xor (b > 5)  ' True  — OU exclusivo

Estruturas de Controle

IF / THEN / ELSEIF / ELSE / END IF
Dim nota As Integer
Input "Digite sua nota (0-100): ", nota

' Forma de uma linha (inline)
If nota >= 90 Then Print "Excelente!"

' Forma de bloco (multi-linha)
If nota >= 90 Then
    Print "Conceito A"
ElseIf nota >= 75 Then
    Print "Conceito B"
ElseIf nota >= 60 Then
    Print "Conceito C"
Else
    Print "Reprovado"
End If

' SELECT CASE (equivalente ao switch/match)
Select Case nota
    Case 90 To 100
        Print "A"
    Case 75, 80, 85, 86, 87, 88, 89
        Print "B"
    Case Is >= 60
        Print "C"
    Case Else
        Print "Reprovado"
End Select
Loops — FOR/NEXT, WHILE/WEND, DO/LOOP
' ── FOR / NEXT ───────────────────────────────────────────
Dim i As Integer

' Contar de 1 a 10
For i = 1 To 10
    Print i
Next i

' Com STEP (passo)
For i = 10 To 1 Step -1   ' countdown
    Print i
Next i

For i = 0 To 100 Step 5
    Print i   ' 0, 5, 10, 15...
Next i

' ── WHILE / WEND ─────────────────────────────────────────
Dim contador As Integer = 1
While contador <= 5
    Print "Contador: "; contador
    contador = contador + 1
Wend

' ── DO / LOOP ────────────────────────────────────────────
Dim n As Integer

' DO WHILE — verifica antes
Do While n < 10
    n = n + 1
    Print n
Loop

' DO ... LOOP UNTIL — verifica depois (executa ao menos 1x)
Do
    Input "Digite um número positivo: ", n
Loop Until n > 0

' EXIT FOR / EXIT DO — sair antecipadamente
For i = 1 To 100
    If i = 7 Then Exit For
    Print i
Next i

Subprogramas: SUB e FUNCTION

SUB (procedimento sem retorno) e FUNCTION (com retorno)
' ── SUB (não retorna valor) ──────────────────────────────
Sub Saudar(nome As String)
    Print "Olá, " & nome & "!"
End Sub

' Chamando
Call Saudar("Maria")
Saudar "João"             ' sem parênteses (sintaxe alternativa)

' ── FUNCTION (retorna valor) ─────────────────────────────
Function Fatorial(n As Long) As Long
    If n <= 1 Then
        Fatorial = 1           ' atribuir ao nome da função = retorno
        Exit Function
    End If
    Fatorial = n * Fatorial(n - 1)  ' recursão
End Function

' Função com múltiplos parâmetros
Function Media(a As Double, b As Double, c As Double) As Double
    Media = (a + b + c) / 3.0
End Function

' Parâmetro por referência (ByRef) vs por valor (ByVal)
Sub Trocar(ByRef x As Integer, ByRef y As Integer)
    Dim tmp As Integer = x
    x = y
    y = tmp
End Sub

' Usando
Print Fatorial(10)       ' 3628800
Print Media(7.5, 8.0, 9.5)  ' 8.333...

Dim p As Integer = 10, q As Integer = 20
Trocar p, q
Print p, q              ' 20  10

Arrays

Declaração, acesso e arrays multidimensionais
' ── Array de tamanho fixo ─────────────────────────────────
Dim notas(5) As Integer   ' índices 0 a 5 (6 elementos)
notas(0) = 90
notas(1) = 85
notas(2) = 72
notas(3) = 95
notas(4) = 88
notas(5) = 78

' Percorrer com FOR
Dim i As Integer
Dim soma As Integer = 0
For i = 0 To 5
    soma = soma + notas(i)
Next i
Print "Média:"; soma / 6

' Funções LBound e UBound
Print LBound(notas)   ' 0 — menor índice
Print UBound(notas)   ' 5 — maior índice

' ── Array dinâmico (REDIM) ───────────────────────────────
Dim dados() As String     ' array sem tamanho definido
ReDim dados(9)            ' redimensionar para 10 elementos (destrói conteúdo)
dados(0) = "Brazil"

ReDim Preserve dados(19)  ' crescer para 20 preservando conteúdo anterior
Print dados(0)            ' "Brazil" — ainda está lá

' ── Array bidimensional (matriz) ─────────────────────────
Dim matriz(2, 2) As Integer   ' 3x3
Dim lin As Integer, col As Integer
For lin = 0 To 2
    For col = 0 To 2
        matriz(lin, col) = lin * 3 + col
    Next col
Next lin

' ── Funções de string ────────────────────────────────────
Dim s As String = "Hello, World!"
Print Len(s)            ' 13  — comprimento
Print Left(s, 5)       ' "Hello"
Print Right(s, 6)      ' "orld!"
Print Mid(s, 8, 5)    ' "World" (início 1-based, comprimento 5)
Print UCase(s)          ' "HELLO, WORLD!"
Print LCase(s)          ' "hello, world!"
Print InStr(s, "World")  ' 8   — posição da substring
Print Str(42)           ' " 42" — número para string
Print Val("3.14")      ' 3.14 — string para número

Entrada e Saída

PRINT, INPUT, WRITE e arquivos em disco
' ── Saída para console ─────────────────────────────────────
Print "Texto simples"
Print "a ="; 42           ' ; sem nova linha entre itens
Print "x =", 10          ' , imprime com tab entre itens
Print                     ' linha em branco

' Formatar número
Print Using "###.##"; 3.14159    ' "  3.14"
Print Using "$$###.##"; 1234.5  ' "$1234.50"

' ── Entrada do usuário ─────────────────────────────────────
Dim nome As String
Dim idade As Integer
Input "Seu nome: ", nome
Input "Sua idade: ", idade
Print "Olá, " & nome & ", você tem " & idade & " anos."

' LINE INPUT — lê linha inteira sem separar por vírgulas
Dim linha As String
Line Input "Digite uma frase: ", linha

' ── Arquivos ────────────────────────────────────────────────
Dim arq As Integer

' Escrever arquivo texto
arq = FreeFile()                ' obter handle disponível
Open "dados.txt" For Output As #arq
Print #arq, "Linha 1"
Print #arq, "Linha 2"
Close #arq

' Ler arquivo linha a linha
arq = FreeFile()
Open "dados.txt" For Input As #arq
Dim buf As String
Do While Not EOF(arq)
    Line Input #arq, buf
    Print buf
Loop
Close #arq

' Append (adicionar ao final)
arq = FreeFile()
Open "dados.txt" For Append As #arq
Print #arq, "Nova linha adicionada"
Close #arq
