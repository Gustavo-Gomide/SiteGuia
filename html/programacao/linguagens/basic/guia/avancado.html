BASIC — Avançado

O FreeBASIC moderno vai muito além do dialeto clássico com line numbers. Suporta User Defined Types (UDT) equivalentes a structs, ponteiros e gerenciamento manual de memória, programação orientada a objetos com herança e polimorfismo, integração com bibliotecas C, inline assembly e metaprogramação com macros de pré-processador.

O Visual Basic .NET oferece uma plataforma ainda mais rica: o ecossistema .NET completo com LINQ, generics, async/await, reflexão e interoperabilidade COM — tornando VB.NET uma opção legítima para aplicações empresariais Windows de larga escala.

User Defined Types (UDT)

TYPE — estruturas de dados
' ── UDT simples (struct) ─────────────────────────────────
Type Ponto
    x As Double
    y As Double
End Type

Type Retangulo
    topLeft     As Ponto   ' UDT aninhado
    bottomRight As Ponto
    cor         As Integer
End Type

Dim p As Ponto
p.x = 3.5
p.y = 7.2
Print "Ponto: (" & p.x & ", " & p.y & ")"

Dim r As Retangulo
r.topLeft.x = 0
r.topLeft.y = 0
r.bottomRight.x = 100
r.bottomRight.y = 50

' WITH — acessa campos sem repetir nome
With r.topLeft
    .x = 10
    .y = 20
End With

' Array de UDTs
Dim pontos(4) As Ponto
Dim i As Integer
For i = 0 To 4
    pontos(i).x = i * 10
    pontos(i).y = i * 5
Next i

Ponteiros e Memória Dinâmica

FreeBASIC — ponteiros (equivalente ao C)
' FreeBASIC suporta ponteiros completos como em C

Dim x As Integer = 42
Dim ptr As Integer Ptr    ' ponteiro para Integer
ptr = @x                   ' @ = endereço de (equivalente ao & em C)
Print *ptr                 ' desreferenciar: imprime 42
*ptr = 100                 ' modificar via ponteiro
Print x                    ' 100 — x foi alterado

' ── Alocação dinâmica ────────────────────────────────────
Dim arr As Integer Ptr
arr = Allocate(10 * SizeOf(Integer))   ' malloc equivalente

If arr = 0 Then
    Print "Falha na alocação!"
    End
End If

Dim i As Integer
For i = 0 To 9
    arr[i] = i * i          ' notação de array com ponteiro
Next i

For i = 0 To 9
    Print arr[i]            ' 0, 1, 4, 9, 16...
Next i

DeAllocate arr              ' free equivalente

' ── New/Delete (orientado a objetos) ─────────────────────
Type Node
    valor   As Integer
    proximo As Node Ptr
End Type

Dim n1 As Node Ptr = New Node
n1->valor = 10
n1->proximo = 0            ' NULL

Delete n1                   ' liberar memória

Programação Orientada a Objetos

FreeBASIC — Classes, herança e polimorfismo
' FreeBASIC suporta OOP completa (desde versão 0.18+)

' ── Classe base ──────────────────────────────────────────
Type Animal Extends Object   ' herdar de Object para ter RTTI e Virtual
    Private:
        nome_ As String
    Public:
        Declare Constructor(nome As String)
        Declare Destructor()
        Declare Property Nome() As String
        Declare Virtual Sub Falar()
End Type

Constructor Animal(nome As String)
    nome_ = nome
    Print "Animal criado: " & nome_
End Constructor

Destructor Animal()
    Print "Animal destruído: " & nome_
End Destructor

Property Animal.Nome() As String
    Return nome_
End Property

Virtual Sub Animal.Falar()
    Print nome_ & " faz algum som."
End Sub

' ── Subclasse ────────────────────────────────────────────
Type Cachorro Extends Animal
    Declare Constructor(nome As String)
    Declare Virtual Sub Falar() Override   ' sobrescreve método virtual
End Type

Constructor Cachorro(nome As String)
    Base(nome)               ' chamar construtor da classe base
End Constructor

Virtual Sub Cachorro.Falar() Override
    Print Nome & " diz: Au au!"
End Sub

' ── Usando polimorfismo ───────────────────────────────────
Dim a As Animal Ptr = New Cachorro("Rex")
a->Falar()                   ' "Rex diz: Au au!" — dispatch virtual
Delete a
VB.NET — Classes e herança
' VB.NET — classe com herança e interface
Public Interface IForma
    ReadOnly Property Area() As Double
    ReadOnly Property Perimetro() As Double
End Interface

Public Class Circulo
    Implements IForma
    Private _raio As Double

    Public Sub New(raio As Double)
        _raio = raio
    End Sub

    Public ReadOnly Property Area() As Double Implements IForma.Area
        Get
            Return Math.PI * _raio * _raio
        End Get
    End Property

    Public ReadOnly Property Perimetro() As Double Implements IForma.Perimetro
        Get
            Return 2 * Math.PI * _raio
        End Get
    End Property
End Class

Tratamento de Erros

ON ERROR, TRY/CATCH (VB.NET) e ERR object
' ── BASIC clássico: ON ERROR ─────────────────────────────
On Error GoTo ErrHandler

Dim arq As Integer = FreeFile()
Open "nao_existe.txt" For Input As #arq  ' pode falhar
Close #arq
GoTo Fim

ErrHandler:
    Print "Erro " & Err & ": " & Error(Err)
    Resume Next    ' continuar na linha seguinte ao erro
    ' ou: Resume Fim — pular para rótulo específico

Fim:
End

' ── VB.NET: Try/Catch/Finally ────────────────────────────
Try
    Dim conteudo = System.IO.File.ReadAllText("dados.txt")
    Console.WriteLine(conteudo)
Catch ex As System.IO.FileNotFoundException
    Console.WriteLine("Arquivo não encontrado: " & ex.Message)
Catch ex As System.UnauthorizedAccessException
    Console.WriteLine("Sem permissão: " & ex.Message)
Catch ex As Exception
    Console.WriteLine("Erro genérico: " & ex.Message)
Finally
    Console.WriteLine("Esta linha sempre executa")
End Try

Interop com C e Bibliotecas Nativas

FreeBASIC — declarando e chamando funções de biblioteca C
' FreeBASIC pode chamar qualquer biblioteca C (.so / .dll / .a)

' Declarar funções da libc (já incluídas automaticamente)
Declare Function strlen_c Alias "strlen" (s As ZString Ptr) As Integer
Declare Function sqrt_c  Alias "sqrt"   (x As Double)      As Double
Declare Sub     printf_c Alias "printf"  (fmt As ZString Ptr, ...)

Dim texto As ZString * 100
texto = "Hello from FreeBASIC"
Print strlen_c(@texto)        ' 20
Print sqrt_c(2.0)            ' 1.41421...

' Linkar com biblioteca externa: #inclib
' #inclib "SDL2"              ' linka -lSDL2

' ── Inline Assembly no FreeBASIC ─────────────────────────
Dim resultado As Integer
Asm
    mov eax, 5
    add eax, 3
    mov [resultado], eax
End Asm
Print resultado    ' 8
