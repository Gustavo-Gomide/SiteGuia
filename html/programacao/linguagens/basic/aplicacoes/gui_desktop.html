BASIC — GUI Desktop

O BASIC tem raízes profundas em GUI: o Visual Basic da Microsoft (1991) popularizou o modelo de programação visual drag-and-drop e o conceito de Event-Driven Programming no Windows. Hoje, o FreeBASIC pode criar janelas Win32 nativas, o QB64 tem primitivas gráficas completas de tela SCREEN, e o Gambas oferece uma IDE visual para Linux comparável ao VB clássico.

Para quem precisa de GUI rápida no Windows sem depender do Visual Studio ou .NET, o FreeBASIC + WinAPI é uma alternativa séria: gera executável nativo sem instalação de runtime, com controles Win32 autênticos (janelas, botões, textboxes, menus).

QB64 — Modo Gráfico SCREEN

QB64 — janela gráfica, desenho e events do mouse
' QB64 / FreeBASIC — Modo gráfico com SCREEN

Screen 12         ' modo 640x480, 16 cores (modo VGA clássico)
' Screen 18       — 1024x768, 32-bit color (QB64 Phoenix)
' _NewImage(w,h,32) — janela customizada no QB64

Dim scr As Long
scr = _NewImage(640, 480, 32)   ' janela 640x480 com 32 bits por pixel
Screen scr

' Título da janela (QB64)
_Title "Minha Janela BASIC"

' Desenho básico
Line (10, 10)-(200, 100), _RGB(255, 0, 0)            ' linha vermelha
Circle (320, 240), 80, _RGB(0, 255, 0)               ' círculo verde
Line (50, 200)-(150, 300), _RGB(0, 0, 255), BF        ' retângulo azul preenchido

' Texto no modo gráfico
Color _RGB(255, 255, 255), _RGB(0, 0, 0)
_PrintString (100, 10), "Hello, World!"    ' texto na posição (x,y)

' Loop de eventos com mouse e teclado
Do
    Dim ev As _UNSIGNED Long

    ' Checar tecla
    Dim k As Long
    k = _KeyHit()
    If k = 27 Then Exit Do    ' ESC = sair

    ' Checar mouse
    Dim mx As Integer, my As Integer, mb As Integer
    mx = _MouseX : my = _MouseY : mb = _MouseButton(1)
    If mb Then
        PSet (mx, my), _RGB(255, 255, 0)   ' pintar pixel ao clicar
    End If

    _Limit 60         ' limitar a 60 FPS
    _Display          ' atualizar tela (double-buffering)
Loop

System             ' fechar janela e sair

FreeBASIC + WinAPI — Janelas Nativas

FreeBASIC — Criar janela Win32 com botão
' FreeBASIC — Janela Win32 nativa
' Compilar: fbc -s gui janela.bas

#include "windows.bi"    ' inclui todas as declarações Win32

' IDs dos controles
Const ID_BTN_OK = 101
Const ID_BTN_SAIR = 102

' WndProc — callback de mensagens
Function WndProc(hwnd As HWND, msg As UINT, wp As WPARAM, lp As LPARAM) As LRESULT
    Select Case msg
    Case WM_CREATE
        ' Criar botões quando janela é criada
        CreateWindowEx(0, "BUTTON", "OK", _
            WS_CHILD Or WS_VISIBLE Or BS_PUSHBUTTON, _
            10, 10, 80, 30, hwnd, Cast(HMENU, ID_BTN_OK), GetModuleHandle(0), 0)
        CreateWindowEx(0, "BUTTON", "Sair", _
            WS_CHILD Or WS_VISIBLE Or BS_PUSHBUTTON, _
            100, 10, 80, 30, hwnd, Cast(HMENU, ID_BTN_SAIR), GetModuleHandle(0), 0)

    Case WM_COMMAND
        Select Case LoWord(wp)
        Case ID_BTN_OK
            MessageBox(hwnd, "Você clicou em OK!", "Info", MB_OK Or MB_ICONINFORMATION)
        Case ID_BTN_SAIR
            PostQuitMessage(0)
        End Select

    Case WM_DESTROY
        PostQuitMessage(0)
    Case Else
        WndProc = DefWindowProc(hwnd, msg, wp, lp)
        Exit Function
    End Select
    WndProc = 0
End Function

' Ponto de entrada
Dim wc    As WNDCLASS
Dim hwnd  As HWND
Dim msg   As MSG

wc.lpfnWndProc   = @WndProc
wc.hInstance     = GetModuleHandle(0)
wc.hCursor       = LoadCursor(0, IDC_ARROW)
wc.hbrBackground = Cast(HBRUSH, COLOR_WINDOW + 1)
wc.lpszClassName = @"MinhaJanela"
RegisterClass(@wc)

hwnd = CreateWindowEx(0, "MinhaJanela", "Minha App BASIC", _
    WS_OVERLAPPEDWINDOW, 100, 100, 300, 150, 0, 0, wc.hInstance, 0)

ShowWindow(hwnd, SW_SHOW)
UpdateWindow(hwnd)

Do While GetMessage(@msg, 0, 0, 0)
    TranslateMessage(@msg)
    DispatchMessage(@msg)
Loop

VBA — UserForms (Formulários Personalizados)

VBA — UserForm com controles e eventos
' No Editor VBA: Insert > UserForm
' Adicionar controles pela Toolbox:
'   - Label: lblNome (Caption: "Nome:")
'   - TextBox: txtNome
'   - Label: lblIdade (Caption: "Idade:")
'   - TextBox: txtIdade
'   - CommandButton: btnOK (Caption: "OK")
'   - CommandButton: btnCancelar (Caption: "Cancelar")

' Código do formulário (duplo-clique nos controles)
Private Sub btnOK_Click()
    Dim nome  As String
    Dim idade As Integer

    nome  = Trim(txtNome.Text)
    idade = CInt(txtIdade.Text)

    If nome = "" Then
        MsgBox "Digite um nome!", vbExclamation
        txtNome.SetFocus
        Exit Sub
    End If

    If idade < 0 Or idade > 150 Then
        MsgBox "Idade inválida!", vbExclamation
        txtIdade.SetFocus
        Exit Sub
    End If

    ' Escrever na planilha ativa
    Dim ws As Worksheet
    ws = ActiveSheet
    Dim prox_linha As Long
    prox_linha = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row + 1
    ws.Cells(prox_linha, "A") = nome
    ws.Cells(prox_linha, "B") = idade

    Me.Hide    ' fechar formulário
End Sub

Private Sub btnCancelar_Click()
    Me.Hide
End Sub

' Exibir o formulário de uma macro
Public Sub MostrarFormulario()
    FormCadastro.Show
End Sub

O Visual Basic 1.0 (1991) introduziu ao mainstream o conceito de event-driven programming — código que reage a eventos do usuário (cliques, teclas, timer) em vez de executar sequencialmente. Esse paradigma se tornou o padrão para GUIs em todas as linguagens modernas: JavaScript, Python Tkinter, Java Swing e Qt usam o mesmo modelo de callbacks/event-handlers que o VB popularizou.
