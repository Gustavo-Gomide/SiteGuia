Assembly — Embedded e IoT

Em sistemas embarcados, assembly não é opcional — é às vezes o único caminho. O código de startup que inicializa a RAM, configura o stack pointer e chama main() precisa ser escrito em assembly. As ISRs (Interrupt Service Routines) de tempo-crítico (controle de motor, DAC de áudio, protocolo de comunicação bit-bang) precisam de timing exato em ciclos — impossível garantir com C.

Microcontroladores com poucos kilobytes de Flash não têm luxo para overhead de framework. Um Arduino Uno (ATmega328P) tem apenas 2 KB de RAM e 32 KB de Flash. Código assembly pode fazer em 50 bytes o que C geraria em 200 bytes — a diferença entre um programa que cabe e um que não cabe.

AVR Assembly (Arduino / ATmega)

AVR Assembly — Piscar LED no ATmega328P (Arduino Uno)
; AVR Assembly — ATmega328P (Arduino Uno)
; Assemblar: avr-as -mmcu=atmega328p blink.s -o blink.o
; Linkar:    avr-ld -m avr5 -Tdata 0x800100 blink.o -o blink.elf
; Upload:    avrdude -p m328p -c arduino -P /dev/ttyUSB0 -U flash:w:blink.hex

    .arch atmega328p

; Endereços de registradores de hardware (I/O Memory)
.equ DDRB,  0x24     ; Data Direction Register B
.equ PORTB, 0x25     ; Port B Output Register
.equ PB5,   5        ; bit 5 = pino 13 do Arduino

    .section .vectors
    .org 0x0000
    rjmp reset_handler   ; vetor de reset — primeiro código que executa

    .section .text
reset_handler:
    ; Inicializar stack pointer (topo da SRAM)
    ldi  r28, lo8(RAMEND)   ; YL = 0xFF
    ldi  r29, hi8(RAMEND)   ; YH = 0x08
    out  SPH, r29           ; Stack Pointer High
    out  SPL, r28           ; Stack Pointer Low

    ; Configurar PB5 como saída (bit 5 de DDRB = 1)
    sbi  DDRB, PB5          ; Set Bit in I/O Register (1 instrução = 2 ciclos)

main_loop:
    ; Ligar LED
    sbi  PORTB, PB5         ; PB5 = HIGH
    rcall delay_500ms

    ; Apagar LED
    cbi  PORTB, PB5         ; Clear Bit — PB5 = LOW
    rcall delay_500ms

    rjmp main_loop

;;------------------------------------------------------------
;; delay_500ms: busy-wait de ~500ms a 16 MHz
;; 16 MHz / 5 ciclos_por_iter = 3.2M iterações = ~500ms
;;------------------------------------------------------------
delay_500ms:
    ldi  r16, 41           ; outer = 41
.outer:
    ldi  r17, 0           ; middle = 256
.middle:
    ldi  r18, 0           ; inner = 256
.inner:
    nop                     ; 1 ciclo
    dec  r18                ; 1 ciclo
    brne .inner             ; 2 ciclos (1 se não saltar)
    dec  r17
    brne .middle
    dec  r16
    brne .outer
    ret

ARM Cortex-M Assembly (STM32)

ARM Cortex-M4 GAS — Startup e ISR
@ ARM Cortex-M4 Assembly (GAS, Thumb-2)
@ STM32F4xx — Startup Code
@ Compilar: arm-none-eabi-as -mcpu=cortex-m4 -mthumb startup.s -o startup.o

    .syntax unified
    .cpu cortex-m4
    .thumb

@ ── Vetor de interrupções (Vector Table) ──────────────────
    .section .isr_vector, "a", %progbits
    .word  _estack              @ endereço inicial do SP (topo da SRAM)
    .word  Reset_Handler        @ 1: Reset
    .word  NMI_Handler          @ 2: Non-Maskable Interrupt
    .word  HardFault_Handler    @ 3: Hard Fault
    .word  MemManage_Handler    @ 4: Memory Management Fault
    .word  BusFault_Handler     @ 5: Bus Fault
    .word  UsageFault_Handler   @ 6: Usage Fault
    .word  0, 0, 0, 0           @ reservados
    .word  SVC_Handler          @ 11: SVCall
    .word  DebugMon_Handler     @ 12: Debug Monitor
    .word  0                    @ reservado
    .word  PendSV_Handler       @ 14: PendSV
    .word  SysTick_Handler      @ 15: SysTick
    @ ... IRQs periféricos a partir de 16
    .word  TIM2_IRQHandler      @ 44: Timer 2

@ ── Reset Handler ─────────────────────────────────────────
    .section .text.Reset_Handler
    .weak  Reset_Handler
    .type  Reset_Handler, %function
Reset_Handler:
    @ Copiar .data de Flash para SRAM
    ldr  r0, =_sdata  @ destino: início de .data na SRAM
    ldr  r1, =_edata  @ fim de .data
    ldr  r2, =_sidata @ fonte: .data armazenado em Flash
    movs r3, #0
.copy_data:
    cmp  r0, r1
    bge  .zero_bss
    ldr  r4, [r2, r3]
    str  r4, [r0, r3]
    add  r3, r3, #4
    b    .copy_data

    @ Zerar .bss
.zero_bss:
    ldr  r0, =_sbss
    ldr  r1, =_ebss
    movs r2, #0
.zero_loop:
    cmp  r0, r1
    bge  .call_main
    str  r2, [r0]
    add  r0, r0, #4
    b    .zero_loop

.call_main:
    bl   main         @ chamar main() do projeto C
    b    .            @ travar se main() retornar

@ ── Timer ISR — Cortex-M salva/restaura contexto automaticamente ─
    .section .text.TIM2_IRQHandler
    .weak  TIM2_IRQHandler
    .type  TIM2_IRQHandler, %function
TIM2_IRQHandler:
    @ Cortex-M empurra automaticamente: xPSR, PC, LR, R12, R3, R2, R1, R0
    @ (não precisamos salvar/restaurar esses)
    push  {r4-r7, lr}     @ salvar callee-saved + LR

    @ Limpar flag de interrupção do TIM2 (UIF em TIM2_SR)
    ldr  r0, =0x40000010  @ endereço TIM2_SR
    ldr  r1, [r0]
    bic  r1, r1, #1       @ limpar bit UIF
    str  r1, [r0]

    @ Lógica da ISR (mínima — nunca bloquear em ISR)
    bl   timer2_callback

    pop  {r4-r7, pc}      @ restaurar e retornar (pop LR → PC = BX LR)

Acesso a Registradores de Hardware

ARM Cortex-M — GPIO, UART e SPI em assembly
@ STM32F4 — Acesso direto a periféricos via endereços mapeados em memória

@ ── GPIO: piscar LED PA5 ─────────────────────────────────
@ Endereços base STM32F4:
.equ  RCC_AHB1ENR,  0x40023830  @ Clock enable para periféricos AHB1
.equ  GPIOA_MODER,  0x40020000  @ PA mode register
.equ  GPIOA_ODR,   0x40020014  @ PA output data register
.equ  GPIOA_BSRR,  0x40020018  @ PA bit set/reset (thread-safe, 1 write)

gpio_init_pa5:
    @ 1. Habilitar clock do GPIOA (bit 0 de RCC_AHB1ENR)
    ldr  r0, =RCC_AHB1ENR
    ldr  r1, [r0]
    orr  r1, r1, #(1 << 0)    @ bit 0 = GPIOAEN
    str  r1, [r0]

    @ 2. Configurar PA5 como saída (MODER5[1:0] = 01)
    ldr  r0, =GPIOA_MODER
    ldr  r1, [r0]
    bic  r1, r1, #(3 << 10)   @ limpar bits 11:10
    orr  r1, r1, #(1 << 10)   @ set bit 10 → output mode
    str  r1, [r0]
    bx   lr

gpio_toggle_pa5:
    @ Togglear PA5 via XOR no ODR
    ldr  r0, =GPIOA_ODR
    ldr  r1, [r0]
    eor  r1, r1, #(1 << 5)
    str  r1, [r0]
    bx   lr

gpio_set_pa5:
    @ Set via BSRR (atômico — bits 0-15 = set, bits 16-31 = reset)
    ldr  r0, =GPIOA_BSRR
    mov  r1, #(1 << 5)       @ bit 5 = set PA5
    str  r1, [r0]
    bx   lr

gpio_reset_pa5:
    ldr  r0, =GPIOA_BSRR
    mov  r1, #(1 << 21)      @ bit 16+5 = reset PA5
    str  r1, [r0]
    bx   lr

No Cortex-M, o hardware salva automaticamente R0–R3, R12, LR, PC e xPSR ao entrar em interrupção — por isso a ISR em assembly só precisa salvar registradores adicionais que usa. O EXC_RETURN no LR ao final (pop {pc}) instrui o processador a restaurar o contexto e retornar ao código interrompido, trocando de modo privilegiado para não-privilegiado se necessário.
