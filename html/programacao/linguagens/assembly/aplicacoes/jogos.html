Assembly — Jogos

Assembly e jogos têm uma história profunda: nos anos 1980–90, jogos inteiros eram escritos em assembly para espremer cada ciclo de CPU de máquinas lentas. Hoje, assembly aparece nos hot paths de engines modernas — cálculo matricial SIMD, mistura de áudio, decompress de assets e shaders de software. Entender assembly torna você um programador de jogos muito mais eficiente.

A matemática de jogos 3D — transformações de vértices, quatérnios, AABB tests, ray casting — envolve intensas multiplicações matriciais. Com AVX2, é possível transformar 8 vértices simultaneamente em uma única passagem de loop, o que é o coração dos renderers de alto desempenho como os usados na Unreal Engine e no id Tech.

Era DOS: VGA Mode 13h

NASM x86 16-bit — Plotar pixel em VGA Mode 13h
; VGA Mode 13h: 320×200 pixels, 256 cores paleta, buffer em 0xA000:0000
; 1 byte por pixel, endereço = y*320 + x

    BITS 16
    ORG  0x100         ; programa .COM: carregado em 0x100

section .text
_start:
    ; Ativar modo 13h via BIOS INT 10h
    mov  ax, 0x0013    ; AH=00 (set mode), AL=13h
    int  0x10

    ; Segmento de vídeo
    mov  ax, 0xA000
    mov  es, ax

    ; Preencher tela de branco (cor 15)
    xor  di, di
    mov  cx, 320 * 200
    mov  al, 15        ; cor branca na paleta padrão
    rep  stosb          ; ES:[DI++] = AL, CX vezes

    ; Plotar pixel vermelho (cor 4) em (160, 100)
    mov  bx, 100       ; y
    imul bx, 320       ; y * 320
    add  bx, 160       ; + x
    mov  byte [es:bx], 4

    ; Desenhar linha horizontal azul (cor 1) em y=50
    mov  di, 50 * 320  ; início da linha y=50
    mov  cx, 320
    mov  al, 1
    rep  stosb

    ; Aguardar tecla (INT 16h)
    xor  ah, ah
    int  0x16

    ; Voltar ao modo texto 80×25
    mov  ax, 0x0003
    int  0x10

    ; Sair do programa .COM (INT 21h, AH=4Ch)
    mov  ax, 0x4C00
    int  0x21

SIMD para Matematica de Jogos

NASM x86-64 — Multiplicar vetor 4D por matriz 4x4 com SSE
; vec4_transform: multiplica vetor4 por matriz 4x4 (row-major)
; Entrada: rdi = float4 v[4], rsi = float16 m[4][4]
; Saída:   rdi = resultado (in-place)
; Layout:  v = [x,y,z,w],  m = row-major

vec4_transform:
    movaps  xmm0, [rdi]           ; xmm0 = [x, y, z, w]

    ; Expandir cada componente para todos os 4 raiaspor coluna
    shufps  xmm1, xmm0, xmm0, 0x00  ; xmm1 = [x,x,x,x]
    shufps  xmm2, xmm0, xmm0, 0x55  ; xmm2 = [y,y,y,y]
    shufps  xmm3, xmm0, xmm0, 0xAA  ; xmm3 = [z,z,z,z]
    shufps  xmm4, xmm0, xmm0, 0xFF  ; xmm4 = [w,w,w,w]

    ; Multiplicar por cada linha da matriz e somar
    mulps   xmm1, [rsi]            ; linha 0
    mulps   xmm2, [rsi + 16]      ; linha 1
    mulps   xmm3, [rsi + 32]      ; linha 2
    mulps   xmm4, [rsi + 48]      ; linha 3

    addps   xmm1, xmm2
    addps   xmm3, xmm4
    addps   xmm1, xmm3            ; xmm1 = resultado final

    movaps  [rdi], xmm1           ; salvar resultado
    ret

; Fast inverse square root — clássico do Quake III
; float InvSqrt(float x) — agora fazemos com SSE direto
inv_sqrt_sse:
    ; rdi = ptr para float, resultado em xmm0
    movss   xmm0, [rdi]
    rsqrtss xmm0, xmm0       ; 1/sqrt(x) com ~12 bits de precisão
    ; Passo de Newton-Raphson para aumentar precisão:
    movss   xmm1, [rdi]
    mulss   xmm1, xmm0       ; x * est
    mulss   xmm1, xmm0       ; x * est^2
    movss   xmm2, [half]
    mulss   xmm1, xmm2       ; 0.5 * x * est^2
    movss   xmm3, [three_halves]
    subss   xmm3, xmm1       ; 1.5 - 0.5*x*est^2
    mulss   xmm0, xmm3       ; resultado refinado
    ret

section .rodata
    align 4
    half          dd 0.5
    three_halves  dd 1.5

Audio Mixing Otimizado com SIMD

NASM x86-64 — Mixagem de audio PCM com SSE
; mix_audio_sse: misturar dois buffers de áudio PCM float32
; audio_out[i] = clamp(audio_a[i] + audio_b[i], -1.0, 1.0)
; rdi = out*, rsi = a*, rdx = b*, rcx = sample_count

section .data
    align 16
    clamp_max  dd  1.0, 1.0, 1.0, 1.0
    clamp_min  dd -1.0, -1.0, -1.0, -1.0

section .text
mix_audio_sse:
    movaps  xmm4, [clamp_max]   ; xmm4 = +1.0 em 4 canais
    movaps  xmm5, [clamp_min]   ; xmm5 = -1.0 em 4 canais
    test    rcx, rcx
    jz      .fim
.loop:
    movaps  xmm0, [rsi]         ; 4 samples do canal A
    movaps  xmm1, [rdx]         ; 4 samples do canal B
    addps   xmm0, xmm1          ; somar
    minps   xmm0, xmm4          ; clamp superior: min(x, 1.0)
    maxps   xmm0, xmm5          ; clamp inferior: max(x, -1.0)
    movaps  [rdi], xmm0         ; salvar
    add     rdi, 16
    add     rsi, 16
    add     rdx, 16
    sub     rcx, 4
    jnz     .loop
.fim:
    ret

rsqrtss (recíproca da raiz quadrada) é 5× mais rápida que sqrtss + divss combinados, com precisão de ~12 bits. Para normalizar vetores 3D em motores de jogo — onde erros de 0.01% são invisíveis — é a escolha certa. Um passo de Newton-Raphson dobra a precisão para ~23 bits mantendo o speedup de ~3×.
