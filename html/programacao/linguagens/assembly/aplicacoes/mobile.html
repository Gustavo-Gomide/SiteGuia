Assembly — Mobile e ARM

Smartphones modernos usam processadores ARM AArch64 (ARM64). Todo código mobile — Kotlin, Swift, Java — compila para instruções ARM. Conhecer assembly ARM64 permite otimizar rotinas críticas com NEON SIMD (equivalente ao AVX do x86), depurar crashes de baixo nível, entender o inline assembly de C/C++ no NDK Android, e aproveitar ao máximo os núcleos de eficiência e desempenho dos chips Apple Silicon e Qualcomm Snapdragon.

ARM64 tem 31 registradores de propósito geral (X0–X30), uma ABI elegante onde os 8 primeiros argumentos inteiros passam em X0–X7, e o NEON estende os registradores V0–V31 para 128 bits — permitindo 4 floats de 32-bit ou 2 doubles de 64-bit em paralelo por instrução.

ARM64 GAS — Funcoes basicas e ABI
// ARM64 Assembly (GAS syntax — usado no Android NDK / GCC)
// Montar: as -o prog.o prog.s   ou   clang --target=aarch64-linux-gnu -c prog.s

    .text
    .global soma_arm64
    .type   soma_arm64, %function

// int64_t soma_arm64(int64_t a, int64_t b)
// x0 = a, x1 = b  → retorno em x0
soma_arm64:
    add  x0, x0, x1     // x0 = a + b
    ret                  // retornar (BX LR implícito via ret)
    .size soma_arm64, . - soma_arm64

// Função com frame e callee-saved
    .global fib_arm64
fib_arm64:
    // Prologo: salvar LR (x30) e x19
    stp  x19, x30, [sp, #-16]!   // store pair + pre-decrement SP
    mov  x19, x0                  // n = arg

    cmp  x0, #1
    b.le .base                    // if n<=1: retornar n

    sub  x0, x19, #1
    bl   fib_arm64                // fib(n-1)
    mov  x1, x0                   // salvar resultado

    sub  x0, x19, #2
    bl   fib_arm64                // fib(n-2)
    add  x0, x0, x1
    b    .ret
.base:
    mov  x0, x19
.ret:
    ldp  x19, x30, [sp], #16     // load pair + post-increment SP
    ret

NEON SIMD no ARM64

ARM64 GAS — NEON: operacoes vetoriais (equivalente ao SSE/AVX)
// NEON: registradores V0-V31, cada um de 128 bits
// Notação de tipo: V0.4s = 4×float32, V0.2d = 2×float64
//                 V0.16b = 16×uint8,  V0.8h = 8×int16

    .text
    .global somar_neon
// void somar_neon(float* a, float* b, float* c, int n)
// x0=a, x1=b, x2=c, x3=n
somar_neon:
    cbz  x3, .fim             // if n==0: retornar
.loop:
    ld1  {v0.4s}, [x0], #16  // carregar 4 floats de a, avançar ptr
    ld1  {v1.4s}, [x1], #16  // carregar 4 floats de b
    fadd v2.4s, v0.4s, v1.4s // soma paralela: 4 floats
    st1  {v2.4s}, [x2], #16  // salvar resultado, avançar ptr
    sub  x3, x3, #4           // n -= 4
    cbnz x3, .loop            // loop enquanto n > 0
.fim:
    ret

// Produto escalar de vetores float32 com NEON
    .global dot_product_neon
dot_product_neon:
    // x0=a, x1=b, x2=n; retorna float em s0
    movi  v3.4s, #0            // acumulador = 0.0
    cbz   x2, .dp_fim
.dp_loop:
    ld1   {v0.4s}, [x0], #16
    ld1   {v1.4s}, [x1], #16
    fmla  v3.4s, v0.4s, v1.4s // v3 += v0 * v1 (fused multiply-add)
    sub   x2, x2, #4
    cbnz  x2, .dp_loop
    // somar as 4 raias do acumulador
    faddp v3.4s, v3.4s, v3.4s  // redução horizontal
    faddp s0, v3.2s
.dp_fim:
    ret

Android NDK: Inline Assembly em C

C/NDK — Inline assembly ARM64 e compilando para Android
// ARM64 inline assembly em C (GCC/Clang NDK)
#include <stdint.h>

// Ler contador de ciclos ARM (CNTVCT_EL0)
static inline uint64_t arm_cycles() {
    uint64_t val;
    asm volatile("mrs %0, cntvct_el0" : "=r"(val));
    return val;
}

// Memory barrier (fence) ARM64
static inline void dmb() {
    asm volatile("dmb ish" ::: "memory");  // data memory barrier inner-shareable
}

// Incluir cabeçalho NEON
#include <arm_neon.h>

void somar_neon_c(float* a, float* b, float* c, int n) {
    for (int i = 0; i < n; i += 4) {
        float32x4_t va = vld1q_f32(&a[i]);  // carregar 4 floats
        float32x4_t vb = vld1q_f32(&b[i]);
        float32x4_t vc = vaddq_f32(va, vb); // soma NEON
        vst1q_f32(&c[i], vc);
    }
}

A instrução FMLA (Fused Multiply-Add) do NEON ARM64 faz v3 += v0 * v1 em uma única instrução sem arredondamento intermediário — é mais precisa que fadd + fmul separados e 2× mais rápida. Motores de IA usam FMLA extensivamente para operações de dot-product em redes neurais.
