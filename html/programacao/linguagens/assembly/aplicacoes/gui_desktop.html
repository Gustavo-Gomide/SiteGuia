Assembly — GUI e Desktop

Criar interfaces gráficas em Assembly puro é um exercício revelador de como sistemas operacionais funcionam internamente. No Windows, significa chamar a Win32 API diretamente — CreateWindowExA, ShowWindow, DispatchMessage. No Linux, a API gráfica de baixo nível é o X11, acessado via syscall de socket ou libX11.

GUI em assembly puro é educativo mas raramente prático em produção — a complexidade explodiria. O uso real é: rotinas de rendering SIMD (blitting, filtros de imagem, compositing) chamadas de um framework C/C++, e hacks de performance em engines gráficas como preenchimento de buffer de frame.

Janela Windows com Win32 API em NASM

NASM x64/Win — Janela Win32 completa com message loop
; NASM Windows PE64 — Janela Win32
; nasm -f win64 win32_janela.asm -o win32_janela.obj
; link /subsystem:windows /entry:WinMain win32_janela.obj kernel32.lib user32.lib

extern RegisterClassExA, CreateWindowExA, ShowWindow, UpdateWindow
extern GetMessageA, TranslateMessage, DispatchMessageA
extern DefWindowProcA, PostQuitMessage, ExitProcess
extern GetModuleHandleA, LoadCursorA

WS_OVERLAPPEDWINDOW equ 0xCF0000
SW_SHOW             equ 5
WM_DESTROY          equ 0x0002
IDC_ARROW           equ 32512

section .data
    class_name db "AsmWindow", 0
    win_title  db "Janela Assembly!", 0

section .bss
    hwnd      resq 1        ; handle da janela
    wndclass  resb 80       ; WNDCLASSEX = 80 bytes
    msg       resb 48       ; MSG = 48 bytes

section .text

;;------------------------------------------------------------
;; WndProc: callback de mensagens da janela
;; rcx=hWnd,  rdx=uMsg,  r8=wParam,  r9=lParam
;;------------------------------------------------------------
WndProc:
    sub   rsp, 40
    cmp   edx, WM_DESTROY
    jne   .default
    xor   ecx, ecx
    call  PostQuitMessage
    xor   eax, eax
    add   rsp, 40
    ret
.default:
    add   rsp, 40
    jmp   DefWindowProcA     ; tail call

WinMain:
    sub   rsp, 40

    ; GetModuleHandleA(NULL)
    xor   ecx, ecx
    call  GetModuleHandleA
    mov   [wndclass + 8], rax   ; WNDCLASSEX.hInstance

    ; Preencher WNDCLASSEX
    mov   dword [wndclass], 80          ; cbSize
    mov   dword [wndclass + 4], 3       ; style = CS_HREDRAW|CS_VREDRAW
    lea   rax, [rel WndProc]
    mov   [wndclass + 16], rax          ; lpfnWndProc
    lea   rax, [rel class_name]
    mov   [wndclass + 64], rax          ; lpszClassName

    ; Cursor: IDC_ARROW
    xor   ecx, ecx
    mov   edx, IDC_ARROW
    call  LoadCursorA
    mov   [wndclass + 48], rax          ; hCursor

    ; RegisterClassExA(&wndclass)
    lea   rcx, [rel wndclass]
    call  RegisterClassExA

    ; CreateWindowExA(0, class, title, style, x, y, w, h, parent, menu, inst, param)
    xor   ecx, ecx
    lea   rdx, [rel class_name]
    lea   r8,  [rel win_title]
    mov   r9d, WS_OVERLAPPEDWINDOW
    push  0   push 0              ; lpParam, hMenu
    push  [wndclass + 8]          ; hInstance
    push  480  push 640           ; h, w
    push  100  push 100           ; y, x
    call  CreateWindowExA
    add   rsp, 64                  ; limpar 8 parâmetros de stack
    mov   [hwnd], rax

    ; ShowWindow + UpdateWindow
    mov   rcx, rax
    mov   edx, SW_SHOW
    call  ShowWindow
    mov   rcx, [hwnd]
    call  UpdateWindow

    ; Message loop
.msg_loop:
    lea   rcx, [rel msg]
    xor   edx, edx
    xor   r8d, r8d
    xor   r9d, r9d
    call  GetMessageA
    test  eax, eax
    jz    .sair
    lea   rcx, [rel msg]
    call  TranslateMessage
    lea   rcx, [rel msg]
    call  DispatchMessageA
    jmp   .msg_loop
.sair:
    xor   ecx, ecx
    call  ExitProcess

X11 no Linux: janela via libX11

NASM x86-64/Linux — Janela X11 com libX11
; Janela X11 usando libX11 (protocolo X Window System)
; nasm -f elf64 x11_janela.asm -o x11_janela.o
; gcc x11_janela.o -lX11 -o x11_janela

extern XOpenDisplay, XDefaultScreen, XRootWindow
extern XCreateSimpleWindow, XSelectInput, XMapWindow
extern XNextEvent, XCloseDisplay, XDrawString
extern XDefaultGC, XStoreName

KeyPressMask     equ (1 << 0)
ExposureMask     equ (1 << 15)
Expose           equ 12
KeyPress         equ 2

section .data
    disp_name db ":0", 0
    titulo    db "Janela Assembly X11", 0
    texto     db "Ola do Assembly!"
    .len      equ $ - texto

section .bss
    display   resq 1
    janela    resq 1
    screen    resd 1
    xevent    resb 192      ; XEvent = 192 bytes

global main
section .text
main:
    push  rbp
    mov   rbp, rsp

    ; XOpenDisplay(":0")
    lea   rdi, [rel disp_name]
    call  XOpenDisplay
    mov   [display], rax

    ; screen = XDefaultScreen(display)
    mov   rdi, rax
    call  XDefaultScreen
    mov   [screen], eax

    ; root = XRootWindow(display, screen)
    mov   rdi, [display]
    mov   esi, [screen]
    call  XRootWindow         ; rax = root window

    ; XCreateSimpleWindow(display, root, x, y, w, h, border, bg, fg)
    mov   rdi, [display]
    mov   rsi, rax
    xor   edx, edx            ; x=0
    xor   ecx, ecx            ; y=0
    push  0xFFFFFF            ; background = branco
    push  1                   ; border=1
    push  300                 ; h=300
    push  400                 ; w=400
    call  XCreateSimpleWindow
    add   rsp, 32
    mov   [janela], rax

    ; Título, eventos, mapear
    mov   rdi, [display]
    mov   rsi, [janela]
    lea   rdx, [rel titulo]
    call  XStoreName

    mov   rdi, [display]
    mov   rsi, [janela]
    mov   edx, ExposureMask | KeyPressMask
    call  XSelectInput

    mov   rdi, [display]
    mov   rsi, [janela]
    call  XMapWindow

    ; Event loop
.loop:
    mov   rdi, [display]
    lea   rsi, [rel xevent]
    call  XNextEvent
    mov   eax, [xevent]         ; xevent.type
    cmp   eax, Expose
    jne   .check_key
    ; Desenhar texto
    mov   rdi, [display]
    mov   rsi, [janela]
    mov   rdi, [display]
    mov   rsi, [janela]
    mov   rdx, [display]       ; gc
    mov   ecx, 50              ; x
    mov   r8d, 50              ; y
    lea   r9,  [rel texto]
    push  texto.len
    call  XDrawString
    add   rsp, 8
.check_key:
    cmp   eax, KeyPress
    jne   .loop

    ; Fechar ao pressionar tecla
    mov   rdi, [display]
    call  XCloseDisplay
    pop   rbp
    xor   eax, eax
    ret
