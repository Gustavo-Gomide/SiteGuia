Assembly — Banco de Dados

Motores de banco de dados de alto desempenho como SQLite, PostgreSQL e RocksDB contêm código assembly em seus caminhos críticos: índices B-Tree, comparadores de chave com SIMD, e rotinas de I/O de arquivos mapeadas em memória. Entender como assembladores lêem e escrevem estruturas de dados em disco ajuda a compreender por que essas escolhas de implementação existem.

Assembly oferece acesso direto a mmap (memória mapeada de arquivo), manipulação de offsets com aritmética pura de ponteiros, e otimizações SIMD para comparação em massa de chaves — tarefas que um JIT de banco de dados deve dominar.

Arquivos com mmap (memory-mapped files)

Linux syscall mmap — mapear arquivo em memória
; NASM x86-64 Linux — mmap um arquivo de banco de dados para memória
; O sistema operacional gerencia páginas (lazy loading) — acesso por ponteiro comum

section .data
    filename   db "data.db", 0
    fd         dq 0      ; file descriptor
    map_addr   dq 0      ; ponteiro retornado por mmap
    file_size  dq 0

section .text
    global _start

;;------------------------------------------------------------
;; open(filename, O_RDONLY=0, 0)
;;------------------------------------------------------------
_start:
    mov   rax, 2             ; syscall: open
    lea   rdi, [rel filename]
    xor   rsi, rsi            ; O_RDONLY = 0
    xor   rdx, rdx
    syscall
    test  rax, rax
    js    .error
    mov   [rel fd], rax

;;------------------------------------------------------------
;; fstat(fd, &stat_buf) — obter tamanho do arquivo
;; struct stat.st_size está no offset 48
;;------------------------------------------------------------
    sub   rsp, 144            ; alocar struct stat na stack
    mov   rax, 5             ; syscall: fstat
    mov   rdi, [rel fd]
    mov   rsi, rsp
    syscall
    mov   rax, [rsp + 48]    ; st_size
    mov   [rel file_size], rax
    add   rsp, 144

;;------------------------------------------------------------
;; mmap(NULL, size, PROT_READ, MAP_PRIVATE, fd, 0)
;;------------------------------------------------------------
    mov   rax, 9             ; syscall: mmap
    xor   rdi, rdi            ; addr = NULL (kernel escolhe)
    mov   rsi, [rel file_size]
    mov   rdx, 1             ; PROT_READ
    mov   r10, 2             ; MAP_PRIVATE
    mov   r8,  [rel fd]
    xor   r9,  r9             ; offset = 0
    syscall
    test  rax, rax
    js    .error
    mov   [rel map_addr], rax  ; agora rax aponta para os dados do arquivo

;;------------------------------------------------------------
;; Agora podemos ler o arquivo como memória comum
;; Exemplo: ler os primeiros 8 bytes (magic number / version header)
;;------------------------------------------------------------
    mov   rbx, [rel map_addr]
    mov   rax, [rbx]          ; primeiros 8 bytes do arquivo
    ; rax agora contém o header magic (ex: 0x53514C697465 = "SQLite")

;;------------------------------------------------------------
;; munmap(addr, size) — liberar mapeamento
;;------------------------------------------------------------
    mov   rax, 11            ; syscall: munmap
    mov   rdi, [rel map_addr]
    mov   rsi, [rel file_size]
    syscall

    mov   rax, 60            ; exit(0)
    xor   rdi, rdi
    syscall

.error:
    mov   rax, 60
    mov   rdi, 1
    syscall

Estruturas de Registro (Fixed-Width Records)

Leitura de registros fixos — tabela de banco de dados simples
; Estrutura de registro (como se fosse uma row/linha de tabela):
; struct Record {
;   uint32_t id;        // offset 0,  4 bytes
;   char     name[32];  // offset 4,  32 bytes
;   double   salary;    // offset 36, 8 bytes
;   uint8_t  active;    // offset 44, 1 byte
;   uint8_t  pad[3];    // offset 45, 3 bytes (alinhamento)
; };                    // total: 48 bytes

%define REC_SIZE    48
%define OFF_ID      0
%define OFF_NAME    4
%define OFF_SALARY  36
%define OFF_ACTIVE  44

;;------------------------------------------------------------
;; get_record(base_addr, index) → ponteiro para o registro
;; rdi = base_addr, rsi = index (0-based)
;; retorna em rax
;;------------------------------------------------------------
get_record:
    imul  rsi, rsi, REC_SIZE   ; offset = index * 48
    lea   rax, [rdi + rsi]     ; ponteiro = base + offset
    ret

;;------------------------------------------------------------
;; linear_scan: varrer todos os registros buscando id == alvo
;; rdi = base_addr, rsi = num_records, rdx = target_id
;; retorna em rax (ponteiro ou NULL)
;;------------------------------------------------------------
linear_scan:
    push  rbx
    push  r12
    mov   rbx, rdi             ; rbx = ponteiro corrente
    mov   r12, rdx             ; r12 = target_id
    xor   rcx, rcx             ; rcx = contador

.loop:
    cmp   rcx, rsi
    jge   .not_found

    mov   eax, [rbx + OFF_ID]  ; ler id do registro atual
    cmp   eax, r12d
    je    .found

    add   rbx, REC_SIZE        ; avançar para próximo registro
    inc   rcx
    jmp   .loop

.found:
    mov   rax, rbx             ; retornar ponteiro
    pop   r12
    pop   rbx
    ret

.not_found:
    xor   rax, rax             ; retornar NULL
    pop   r12
    pop   rbx
    ret

;;------------------------------------------------------------
;; cmp_records_by_id: comparador para qsort
;; rdi = *Record a, rsi = *Record b
;; retorna: negativo / zero / positivo
;;------------------------------------------------------------
cmp_records_by_id:
    mov   eax, [rdi + OFF_ID]
    mov   edx, [rsi + OFF_ID]
    sub   eax, edx             ; retorna a.id - b.id
    ret

B-Tree: Traversal em Assembly

Busca binária em nó de B-Tree (chaves inteiras de 8 bytes)
; Nó de B-Tree (formato SQLite-like, simplificado):
; struct BTreeNode {
;   uint16_t num_keys;   // offset 0: número de chaves no nó
;   uint8_t  is_leaf;    // offset 2: 1 = folha, 0 = interno
;   uint8_t  pad;        // offset 3
;   int64_t  keys[MAX];  // offset 4: array de chaves (8 bytes cada)
; };
%define OFF_NUM_KEYS  0
%define OFF_IS_LEAF   2
%define OFF_KEYS      4

;;------------------------------------------------------------
;; btree_node_binary_search
;; rdi = ponteiro para BTreeNode
;; rsi = chave de busca (int64)
;; retorna: rax = índice onde chave foi encontrada (ou -1)
;;          rcx = índice do filho a seguir (para nós internos)
;;------------------------------------------------------------
btree_node_binary_search:
    movzx r8, word [rdi + OFF_NUM_KEYS]  ; r8 = num_keys
    test  r8, r8
    jz    .empty_node

    xor   r9, r9              ; r9 = low = 0
    mov   r10, r8             ; r10 = high = num_keys
    dec   r10

.bsearch_loop:
    cmp   r9, r10
    jg    .not_found_exact

    mov   r11, r9
    add   r11, r10
    shr   r11, 1              ; mid = (low + high) / 2

    ; Ler keys[mid] — cálculo de offset: 4 + mid*8
    lea   rax, [rdi + OFF_KEYS]
    mov   rax, [rax + r11*8]  ; rax = keys[mid]

    cmp   rsi, rax
    je    .found_exact
    jl    .go_left
    ; maior: mover low para mid+1
    lea   r9, [r11 + 1]
    jmp   .bsearch_loop

.go_left:
    lea   r10, [r11 - 1]
    jmp   .bsearch_loop

.found_exact:
    mov   rax, r11            ; índice encontrado
    mov   rcx, r11            ; filho à direita da chave = índice+1
    inc   rcx
    ret

.not_found_exact:
    mov   rax, -1            ; não encontrado
    mov   rcx, r9            ; índice do filho a seguir descendo
    ret

.empty_node:
    mov   rax, -1
    xor   rcx, rcx
    ret

CRC-32 para Integridade de Dados

CRC-32 com instrução SSE4.2 crc32 (hardware)
; Intel SSE4.2 inclui instrução CRC32 CRC32 por hardware (Nehalem+)
; Usada em SQLite páginas de journal, checksums de transação

;;------------------------------------------------------------
;; crc32_sse42(data, len) → uint32 em eax
;; rdi = ponteiro para dados, rsi = comprimento em bytes
;;------------------------------------------------------------
crc32_sse42:
    mov    eax, 0xFFFFFFFF   ; valor inicial
    xor    rcx, rcx

    ; Processar 8 bytes por vez enquanto possível
.loop8:
    sub    rsi, 8
    jl     .tail

    crc32  rax, qword [rdi + rcx]   ; CRC-32C sobre 8 bytes
    add    rcx, 8
    jmp    .loop8

    ; Processar bytes restantes individualmente
.tail:
    add    rsi, 8           ; restaurar contagem restante
.loop1:
    test   rsi, rsi
    jz     .done
    crc32  eax, byte [rdi + rcx]
    inc    rcx
    dec    rsi
    jmp    .loop1

.done:
    not    eax               ; complemento final
    ret

;;------------------------------------------------------------
;; Uso: verificar página de banco de dados (ex: 4KB)
;;------------------------------------------------------------
verify_page:
    ; rdi = ponteiro da página (4096 bytes)
    ; Os primeiros 4 bytes contêm o CRC salvo
    mov   r15d, [rdi]         ; CRC stored header
    mov   [rdi], dword 0     ; zerar campo CRC antes de calcular
    mov   rsi, 4096
    call  crc32_sse42
    mov   [rdi], r15d        ; restaurar campo CRC
    cmp   eax, r15d
    sete  al                  ; al = 1 se válido, 0 se corrompido
    movzx eax, al
    ret

Comparação de Chaves em Massa com AVX2

Varredura vetorizada de índice — localizar chave em 256 inteiros
; AVX2: processar 8 chaves int32 por ciclo
; Usado em bancos colunar (DuckDB, ClickHouse) para predicados WHERE col = val

;;------------------------------------------------------------
;; simd_find_key(int32* keys, int32 target, int count) → index ou -1
;; rdi = keys array, esi = target (int32), edx = count
;;------------------------------------------------------------
simd_find_key:
    push   rbx
    vmovd  xmm0, esi            ; xmm0 = target
    vpbroadcastd ymm0, xmm0    ; ymm0 = [target x8] (broadcast para 256 bits)
    xor    rbx, rbx             ; índice corrente

.loop8:
    cmp    rbx, rdx
    jge    .not_found

    ; Carregar 8 chaves de 32 bits (256 bits)
    vmovdqu  ymm1, [rdi + rbx*4]

    ; Comparar 8 pares em paralelo
    vpcmpeqd ymm2, ymm0, ymm1  ; ymm2[lane] = 0xFFFFFFFF se igual, 0 se diferente

    ; Extrair bitmask dos bytes mais significativos dos 32 resultados
    vpmovmskb ecx, ymm2          ; ecx = 32 bits, 1 bit por byte de ymm2
    test   ecx, ecx
    jz     .next8               ; nenhum match neste bloco de 8

    ; Encontrou: mapear bit para índice
    ; Cada chave int32 ocupa 4 bytes → cada match tem 4 bits consecutivos setados
    bsf    ecx, ecx             ; posição do primeiro bit setado
    shr    ecx, 2              ; dividir por 4 → índice dentro dos 8
    add    rbx, rcx             ; índice absoluto
    mov    rax, rbx
    pop    rbx
    ret

.next8:
    add    rbx, 8

    ; Garantir que não ultrapasse count (loop handle parcial)
    jmp    .loop8

.not_found:
    mov    rax, -1
    pop    rbx
    vzeroupper                  ; limpeza AVX obrigatória antes de SSE
    ret

A instrução CRC32 do Intel SSE4.2 processa um byte (ou 2/4/8 bytes) por ciclo, sem desvios — é um exemplo perfeito de como hardware dedicado supera algoritmos de software. Engines de banco de dados como InnoDB e RocksDB usam CRC32-C (a variante Castagnoli) exatamente por ser acelerado em hardware em praticamente todos os processadores modernos.
