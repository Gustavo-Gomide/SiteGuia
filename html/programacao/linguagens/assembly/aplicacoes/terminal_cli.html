Assembly — Terminal e CLI

Programas de linha de comando em Assembly puro são uma das aplicações mais diretas e educativas da linguagem. Desde ler argc/argv passados pelo kernel no stack, até fazer I/O com syscalls read/write, converter números em strings e implementar parsing de argumentos — tudo sem nenhuma biblioteca externa.

Quando o kernel Linux inicia um processo ELF, o stack já contém: [rsp] = argc, [rsp+8] = argv[0], [rsp+16] = argv[1]... seguidos de envp[]. O código assembly pode lê-los diretamente de rsp antes mesmo de configurar o frame do main.

Lendo argc, argv e envp

NASM x86-64 — _start: acessar argumentos da linha de comando
; Layout do stack ao iniciar _start:
;  [rsp]       = argc     (int64)
;  [rsp+8]     = argv[0]  (char* → nome do executável)
;  [rsp+16]    = argv[1]  (char* → 1° argumento)
;  ...         = argv[argc-1]
;  [rsp+8*(argc+1)] = NULL (fim de argv)
;  [rsp+8*(argc+2)] = envp[0] ...

section .data
    uso_msg  db "Uso: prog <nome>", 10, 0
    ola_pre  db "Ola, ", 0
    newline  db 10

section .text
global _start

;;------------------------------------------------------------
;; strlen_asm: calcula comprimento de string terminada com \0
;; rdi = ptr, retorna rax = comprimento
;;------------------------------------------------------------
strlen_asm:
    xor   rax, rax
.loop:
    cmp   byte [rdi + rax], 0
    je    .fim
    inc   rax
    jmp   .loop
.fim:
    ret

_start:
    mov   rbx, [rsp]           ; rbx = argc
    cmp   rbx, 2               ; precisamos de pelo menos 1 argumento
    jl    .sem_args

    ; argv[1] = [rsp + 16]
    mov   r12, [rsp + 16]      ; r12 = argv[1] (callee-saved = não perde)

    ; escrever "Ola, "
    lea   rdi, [rel ola_pre]
    call  strlen_asm
    mov   rdx, rax              ; tamanho
    lea   rsi, [rel ola_pre]
    mov   edi, 1               ; fd=stdout
    mov   eax, 1               ; sys_write
    syscall

    ; escrever argv[1]
    mov   rdi, r12
    call  strlen_asm
    mov   rdx, rax
    mov   rsi, r12
    mov   edi, 1
    mov   eax, 1
    syscall

    ; escrever newline
    lea   rsi, [rel newline]
    mov   edx, 1
    mov   edi, 1
    mov   eax, 1
    syscall
    jmp   .sair

.sem_args:
    lea   rsi, [rel uso_msg]
    mov   edx, 18
    mov   edi, 2               ; fd=stderr
    mov   eax, 1
    syscall

.sair:
    xor   edi, edi
    mov   eax, 60
    syscall

Leitura de stdin e Conversoes Numericas

NASM x86-64 — ler do stdin e converter string para inteiro
section .bss
    buf     resb 64        ; buffer para leitura

section .data
    prompt  db "Digite um número: ", 0

section .text

;;------------------------------------------------------------
;; itoa64: converte rax (int64) → string em buffer
;; rdi = buffer de saída (mín. 21 bytes), retorna rdi=início str
;;------------------------------------------------------------
itoa64:
    push  rbx
    mov   rbx, rdi
    add   rbx, 20
    mov   byte [rbx], 0        ; terminador nulo
    test  rax, rax
    jns   .positivo
    neg   rax                  ; tornar positivo
    push  rax
    mov   rax, 1               ; flag: negativo
    push  rax
    mov   rax, [rsp + 8]
.positivo:
    mov   rcx, 10
.loop:
    xor   rdx, rdx
    div   rcx                  ; rax = quociente, rdx = resto (0-9)
    add   dl, '0'
    dec   rbx
    mov   [rbx], dl
    test  rax, rax
    jnz   .loop
    mov   rax, rbx             ; retorna ponteiro para início
    pop   rbx
    ret

;;------------------------------------------------------------
;; atoi64: converte string decimal → int64 em rax
;; rdi = ptr string  (suporta sinal negativo)
;;------------------------------------------------------------
atoi64:
    xor   rax, rax
    xor   rcx, rcx            ; sinal: 0=positivo, 1=negativo
    cmp   byte [rdi], '-'
    jne   .digits
    inc   rdi
    inc   rcx
.digits:
    movzx rdx, byte [rdi]
    sub   dl, '0'
    js    .fim                 ; dl < 0 → não é dígito
    cmp   dl, 9
    ja    .fim
    imul  rax, rax, 10
    add   rax, rdx
    inc   rdi
    jmp   .digits
.fim:
    test  rcx, rcx
    jz    .ret
    neg   rax
.ret:
    ret

ler_numero:
    ; read(0, buf, 64)
    xor   edi, edi             ; fd=stdin
    lea   rsi, [rel buf]
    mov   edx, 64
    xor   eax, eax             ; sys_read = 0
    syscall                    ; rax = bytes lidos
    ; substituir '\n' por '\0'
    lea   rdi, [rel buf]
    add   rdi, rax
    dec   rdi
    mov   byte [rdi], 0
    ; converter string → inteiro
    lea   rdi, [rel buf]
    call  atoi64               ; rax = valor numérico
    ret

Usar r12–r15 para guardar ponteiros importantes (como argv[1]) durante chamadas de função é essencial — esses registradores são callee-saved pela System V ABI, o que significa que qualquer função C que você chamar via call preservará seus valores automaticamente.
