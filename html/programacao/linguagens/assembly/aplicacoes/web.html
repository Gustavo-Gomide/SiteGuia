Assembly — Web e WebAssembly

WebAssembly (WASM) é a forma moderna de escrever código de baixo nível para a web. É uma máquina virtual de stack com seu próprio assembly — o WAT (WebAssembly Text Format) — que compila para um binário .wasm executável em qualquer navegador moderno com desempenho próximo ao nativo. E com WASI, os mesmos módulos rodam fora do browser, no servidor.

WASM não é x86 — é uma arquitetura virtual de propósito específico para a web. Mas os mesmos conceitos de Assembly (pilha, memória linear, tipos primitivos) se aplicam. Um módulo WASM pode ser um hot path computacional em C/C++/Rust compilado para .wasm e chamado via JavaScript ou um módulo WAT escrito diretamente.

WAT — WebAssembly Text Format

WAT — Modulo WebAssembly completo
;; WebAssembly Text Format (WAT)
;; Compilar: wat2wasm modulo.wat -o modulo.wasm
;; Instalar: npm install -g wabt   (inclui wat2wasm, wasm2wat)

(module
  ;; ── importar função do host (JS/Node) ──────────────────
  (import "console" "log"
    (func $log (param i32))
  )

  ;; ── memória: 1 página = 64 KB ──────────────────────────
  (memory (export "memory") 1)

  ;; ── dados iniciais na memória ──────────────────────────
  (data (i32.const 0) "Ola WASM!\00")

  ;; ── função: somar dois i32 ──────────────────────────────
  (func (export "soma")
    (param $a i32) (param $b i32)
    (result i32)
    local.get $a
    local.get $b
    i32.add
  )

  ;; ── função: fibonacci(n) ────────────────────────────────
  (func (export "fib")
    (param $n i32) (result i32)
    (local $a i32) (local $b i32) (local $tmp i32)
    (local.set $a (i32.const 0))
    (local.set $b (i32.const 1))
    (block
      (loop
        (br_if 1 (i32.le_s (local.get $n) (i32.const 0)))
        (local.set $tmp (i32.add (local.get $a) (local.get $b)))
        (local.set $a (local.get $b))
        (local.set $b (local.get $tmp))
        (local.set $n (i32.sub (local.get $n) (i32.const 1)))
        br 0
      )
    )
    local.get $a
  )

  ;; ── função: escrever string (chama $log) ────────────────
  (func (export "greet")
    (i32.const 0)   ;; endereço da string na memória
    call $log
  )
)
JavaScript — Carregar e usar modulo WASM
// Carregar modulo .wasm via fetch/WebAssembly API
async function carregarWasm() {
    const importObject = {
        console: {
            log: (ptr) => {
                // ptr = endereço na memória linear do WASM
                const mem = new Uint8Array(instance.exports.memory.buffer);
                let str = "";
                let i = ptr;
                while (mem[i] !== 0) str += String.fromCharCode(mem[i++]);
                console.log(str);
            }
        }
    };

    // Opção 1: fetch + instantiateStreaming (mais eficiente)
    const { instance } = await WebAssembly.instantiateStreaming(
        fetch("modulo.wasm"),
        importObject
    );

    // Chamar funções exportadas
    console.log(instance.exports.soma(3, 4));     // 7
    console.log(instance.exports.fib(10));       // 55
    instance.exports.greet();                   // "Ola WASM!"

    // Acessar memória linear diretamente
    const mem = new Int32Array(instance.exports.memory.buffer);
    mem[0] = 42;   // escrever na posição 0 da memória WASM
}

carregarWasm();

WASI — WebAssembly fora do Browser

Terminal — Compilar C para WASM com WASI e rodar com Wasmtime
# WASI = WebAssembly System Interface — syscalls portáveis
# Permite rodar .wasm no servidor sem Docker, sem OS específico

# Instalar WASI SDK
# https://github.com/WebAssembly/wasi-sdk/releases

# Compilar C → WASM com WASI target
/opt/wasi-sdk/bin/clang \
    --sysroot=/opt/wasi-sdk/share/wasi-sysroot \
    -target wasm32-wasi \
    -O2 \
    programa.c -o programa.wasm

# Rodar com Wasmtime (runtime WebAssembly nativo)
wasmtime programa.wasm           # executar diretamente
wasmtime --dir=. programa.wasm   # dar acesso ao diretório atual

# Instalar Wasmtime
curl https://wasmtime.dev/install.sh -sSf | bash

# Rodar com Node.js (WASI experimental)
node --experimental-wasi-unstable-preview1 run_wasm.js
Node.js — Executar modulo WASM com WASI
import { WASI } from 'node:wasi';
import { readFileSync } from 'node:fs';

const wasi = new WASI({
    version: 'preview1',
    args: process.argv,
    env: process.env,
    preopens: { '/': '/' }   // mapear diretório raiz
});

const bytes = readFileSync('programa.wasm');
const { module, instance } = await WebAssembly.instantiate(
    bytes,
    wasi.getImportObject()
);

wasi.start(instance);   // executa _start do WASM

Compilar C/C++ para WebAssembly com Emscripten

Terminal — Emscripten: compilar para browser
# Instalar Emscripten SDK (emsdk)
git clone https://github.com/emscripten-core/emsdk.git
cd emsdk
./emsdk install latest
./emsdk activate latest
source ./emsdk_env.sh

# Compilar .c → .wasm + .js (glue code automático)
emcc programa.c -o saida.html     # gera saida.html + saida.wasm + saida.js
emcc programa.c -o saida.js       # só o módulo JS
emcc programa.c -o saida.wasm     # só o binário wasm

# Com otimizações
emcc -O3 -flto programa.c -o saida.wasm

# Exportar funções específicas
emcc programa.c \
    -EXPORTED_FUNCTIONS="['_soma', '_fib']" \
    -EXPORTED_RUNTIME_METHODS="['ccall', 'cwrap']" \
    -o saida.js

# Servir localmente (navegadores bloqueiam file:// para WASM)
python3 -m http.server 8080

WebAssembly não substitui JavaScript — ele complementa. Use WASM para hot paths: parsing, criptografia, codecs de vídeo/áudio, simulações físicas, hashing e qualquer operação de CPU-intensiva. Para lógica de UI, DOM, eventos e rede, JavaScript continua sendo a escolha certa. A interop WASM↔JS tem custo — minimize chamadas de fronteira.
