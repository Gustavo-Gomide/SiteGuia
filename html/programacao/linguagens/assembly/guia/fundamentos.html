Assembly — Fundamentos

Os fundamentos do Assembly cobrem o conjunto completo de instruções essenciais: movimentação de dados, aritmética inteira e de ponto flutuante, operações lógicas, flags de status, comparações, saltos condicionais, estruturas de controle, a pilha e a mecânica de chamada de sub-rotinas através de CALL/RET e convenções de chamada.

Em Assembly, não existem if, for ou funções no sentido de linguagens de alto nível. Tudo é construído com CMP + Jcc (jump condicional), CALL e manipulação explícita da pilha via PUSH/POP.

Movimentacao de Dados

NASM x86-64 — MOV, LEA e movimentacao com memoria
; MOV: copia valor de src para dst
mov  rax, 100         ; rax = 100 (imediato)
mov  rbx, rax         ; rbx = rax (reg → reg)
mov  rax, [var]       ; rax = *var (memória → reg)
mov  [var], rbx       ; *var = rbx (reg → memória)
mov  [rbx + rcx*8], rax  ; enderecamento indexado: base + índice*escala

; MOV não pode: mem → mem diretamente  (usar reg como intermediário)
mov  rax, [src]
mov  [dst], rax

; LEA: Load Effective Address — calcula endereço sem acessar memória
lea  rdi, [rbp - 16]   ; rdi = rbp - 16  (endereço de variável local)
lea  rax, [rax*4 + rax] ; rax = rax*5  (truque para multiplicar)
lea  rsi, [rip + msg]   ; RIP-relative: endereço de msg (código independente)

; MOVZX / MOVSX: estender com zero ou sinal
movzx rax, byte [buf]  ; zero-extend: rax = (uint8_t) *buf
movsx rax, dword [n]  ; sign-extend: rax = (int32_t) *n

; XCHG: troca atômica
xchg  rax, rbx         ; troca rax ↔ rbx

; CMOV: move condicional (sem branch)
cmp   rax, rbx
cmovg rcx, rdx         ; if (rax > rbx): rcx = rdx  (sem salto)

Aritmetica Inteira

NASM x86-64 — ADD, SUB, MUL, DIV, operacoes de bit
; Adição e subtração
add   rax, 10      ; rax += 10  (seta CF, OF, SF, ZF, PF)
add   rax, rbx     ; rax += rbx
sub   rcx, 1       ; rcx -= 1
inc   rdx          ; rdx++  (mais eficiente, não seta CF)
dec   rdx          ; rdx--
neg   rax          ; rax = -rax (complemento de dois)
adc   rax, rbx     ; add with carry: rax += rbx + CF  (big-int)
sbb   rax, rbx     ; sub with borrow: rax -= rbx + CF

; Multiplicação sem sinal — RESULTADO em RDX:RAX (128 bits)
mov   rax, 1000
mov   rbx, 2000
mul   rbx           ; rdx:rax = rax * rbx (unsigned)
; IMUL — assinada, versão de 2/3 operandos é mais simples
imul  rax, rbx, 5  ; rax = rbx * 5  (truncado em 64-bit)
imul  rax, rcx      ; rax *= rcx

; Divisão — DIVIDENDO em RDX:RAX, divisor no operando
xor   rdx, rdx      ; limpar RDX (parte alta do dividendo)
mov   rax, 100
mov   rbx, 7
div   rbx           ; rax = 100 / 7 = 14  (quociente), rdx = 100 % 7 = 2 (resto)
idiv  rbx           ; versão assinada — cuidado: rdx deve ter extensao de sinal de rax

; Operações de bit
and   rax, 0xFF     ; rax &= 0xFF  (mascarar byte baixo)
or    rax, 0x80     ; setar bit 7
xor   rax, rax      ; rax = 0  (idioma mais rápido para zerar)
xor   rax, 1        ; toggle bit 0
not   rax           ; bitwise NOT

; Shifts
shl   rax, 1        ; shift left 1 = multiplicar por 2 (rápido)
shr   rax, 3        ; shift right lógico 3 = dividir por 8 (unsigned)
sar   rax, 1        ; shift right ARITMÉTICO = divide por 2 (preserva sinal)
rol   rax, 1        ; rotate left
ror   rbx, 4        ; rotate right (nibble swap)
bswap rax           ; byte swap: big-endian ↔ little-endian

NASM x86-64 — CMP, TEST e instruções de salto condicional
; CMP: subtrai dst - src sem salvar resultado (só atualiza flags)
; TEST: AND dst & src sem salvar (só atualiza SF, ZF, PF)

; ── if (rax == 0) ──────────────────────────────
    test  rax, rax    ; equivalente a cmp rax, 0 (mais rápido)
    jz    .eh_zero    ; jump if zero (ZF=1)
    jnz   .nao_zero   ; jump if not zero (ZF=0)

; ── if (rax == rbx) ────────────────────────────
    cmp   rax, rbx
    je    .iguais     ; jump if equal
    jne   .diferentes ; jump if not equal

; ── Saltos com sinal (signed comparisons) ──────
    cmp   rax, rbx
    jl    .menor      ; jump if less  (SF ≠ OF)
    jle   .menor_ig   ; jump if less or equal
    jg    .maior      ; jump if greater
    jge   .maior_ig   ; jump if greater or equal

; ── Saltos sem sinal (unsigned) ─────────────────
    cmp   rax, rbx
    jb    .abaixo     ; jump if below  (CF=1)
    jbe   .abaixo_ig  ; jump if below or equal
    ja    .acima      ; jump if above  (CF=0 e ZF=0)
    jae   .acima_ig   ; jump if above or equal

; ── if/else em assembly ─────────────────────────
    cmp   rcx, 10
    jge   .else_branch
    ; bloco then: rcx < 10
    imul  rcx, rcx, 2
    jmp   .end_if
.else_branch:
    ; bloco else: rcx >= 10
    sub   rcx, 5
.end_if:

; ── Loop for equivalente a  for(rcx=10; rcx>0; rcx--) ─
    mov   rcx, 10
.loop_inicio:
    dec   rcx
    jnz   .loop_inicio  ; ou usar LOOP (lento em CPUs modernas)

Pilha e Sub-rotinas

NASM x86-64 — PUSH, POP, CALL, RET e prologo/epilogo
; A pilha cresce para BAIXO em x86-64 (RSP decrementa)
; PUSH rax  ≡  sub rsp, 8  ;  mov [rsp], rax
; POP  rax  ≡  mov rax, [rsp]  ;  add rsp, 8

; CALL: empurra endereço de retorno e salta
; RET:  desempilha endereço e salta (retorna)

; ────────────────────────────────────────────────
; Função: soma(a, b) -> rax = a + b
; System V ABI: args em rdi, rsi, rdx, rcx, r8, r9
; ────────────────────────────────────────────────
soma:
    ; Prólogo — salvar frame pointer
    push  rbp
    mov   rbp, rsp
    ; Sub rsp, N para alocar variáveis locais (deve ser alinhado a 16)
    sub   rsp, 16

    ; Corpo
    mov   rax, rdi    ; rax = a (1° arg)
    add   rax, rsi    ; rax += b (2° arg)

    ; Epílogo — restaurar frame e retornar
    mov   rsp, rbp
    pop   rbp
    ret              ; retorna com resultado em rax

; Chamar a função
_start:
    mov   rdi, 3     ; a = 3
    mov   rsi, 4     ; b = 4
    call  soma        ; rax = 7

; Salvar/restaurar registradores callee-saved (rbx, rbp, r12-r15)
minha_funcao:
    push  rbx
    push  r12
    push  r13
    ; ... usa rbx, r12, r13 livremente ...
    pop   r13
    pop   r12
    pop   rbx
    ret

; Alinhamento de pilha: em x86-64, ao entrar em uma função
; (após o CALL que empurrou 8 bytes), RSP deve ser ≡ 0 mod 16.
; O prólogo push rbp adiciona 8 bytes → RSP ≡ 0 mod 16 ✓

Modos de Enderecamento

NASM x86-64 — Todos os modos de enderecamento
; Formato geral: [base + índice*escala + deslocamento]
; escala: 1, 2, 4 ou 8

; 1. Imediato
mov  rax, 42

; 2. Registrador
mov  rax, rbx

; 3. Direto (absolute address)
mov  rax, [0x404000]

; 4. Indireto por registrador
mov  rax, [rsi]           ; rax = *rsi

; 5. Base + deslocamento
mov  rax, [rbp - 8]       ; var local no stack frame
mov  rax, [rdi + 16]      ; campo de struct com offset 16

; 6. Base + índice*escala
mov  rax, [rbx + rcx*8]   ; array de int64: rbx=base, rcx=índice
mov  eax, [rbx + rcx*4]   ; array de int32

; 7. Base + índice*escala + deslocamento
mov  rax, [rbx + rcx*8 + 32]  ; elemento de array com offset

; 8. RIP-relative (código independente de posição, PIC)
lea  rdi, [rel mensagem]   ; endereço relativo ao RIP
mov  rax, [rel variavel]   ; acesso a varíavel global

; Tamanho explícito quando ambíguo:
mov  byte  [rdi], 0    ; escreve 1 byte
mov  word  [rdi], 0    ; escreve 2 bytes
mov  dword [rdi], 0    ; escreve 4 bytes
mov  qword [rdi], 0    ; escreve 8 bytes
