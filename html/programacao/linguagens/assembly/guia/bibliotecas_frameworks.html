Assembly — Bibliotecas e Interop

Assembly raramente existe em isolamento total. Na prática, código assembly chama funções da libc (printf, malloc, memcpy), faz syscalls diretas ao kernel Linux, usa a Win32 API no Windows, acessa BIOS/firmware em código bare-metal e se integra com projetos C/C++ como módulos compilados isolados linkados junto.

A regra de ouro: use syscalls diretas quando precisar de zero dependências (processo de inicialização, bootloader, shellcode), use a libc quando quiser portabilidade entre Linux/macOS/Windows, e use DDLs/SOs externos quando precisar de funcionalidades complexas sem reimplementar tudo do zero.

Usando a libc em Assembly

NASM x86-64 — Chamando printf, malloc e memcpy
; Linkagem: gcc -o prog prog.o   (ou ld com -lc e -dynamic-linker)
; As funções externas são declaradas com "extern"

extern printf
extern malloc
extern free
extern memcpy
extern strlen
extern puts

global main

section .data
    fmt_int   db "Valor: %d\n", 0
    fmt_str   db "String: %s\n", 0
    hello     db "Ola, Mundo!", 0

section .text
main:
    ; ── printf("Valor: %d\n", 42) ──
    lea   rdi, [rel fmt_int]   ; 1° arg: formato
    mov   esi, 42               ; 2° arg: inteiro
    xor   eax, eax              ; RAX=0: nenhum argumento XMM
    call  printf

    ; ── puts("Ola, Mundo!") ──
    lea   rdi, [rel hello]
    call  puts

    ; ── malloc(256) ──
    mov   edi, 256          ; tamanho em bytes
    call  malloc             ; rax = ponteiro ou NULL
    test  rax, rax
    jz    .erro
    push  rax                ; salvar ponteiro

    ; ── memcpy(dest, src, n) ──
    mov   rdi, rax           ; destino = ponteiro alocado
    lea   rsi, [rel hello]   ; fonte = string hello
    mov   edx, 12            ; n = 12 bytes
    call  memcpy

    ; ── free(ptr) ──
    pop   rdi                ; recuperar ponteiro
    call  free

.erro:
    xor   eax, eax
    ret

NASM — fork + execve (criar processo filho)
section .data
    programa  db "/bin/ls", 0
    arg0      db "ls", 0
    arg1      db "-la", 0
    argvec    dq arg0, arg1, 0    ; argv[] = { "ls", "-la", NULL }
    envvec    dq 0                 ; envp[] = { NULL }

section .text
global _start
_start:
    ; fork()
    mov   eax, 57
    syscall
    test  eax, eax
    jz    .filho          ; fork retorna 0 no filho
    jl    .erro
    ; parent: rax = pid do filho
    mov   rdi, rax        ; wait4(pid, NULL, 0, NULL)
    xor   esi, esi
    xor   edx, edx
    xor   r10d, r10d
    mov   eax, 61         ; wait4
    syscall
    jmp   .sair
.filho:
    ; execve("/bin/ls", ["ls","-la",NULL], [NULL])
    lea   rdi, [rel programa]
    lea   rsi, [rel argvec]
    lea   rdx, [rel envvec]
    mov   eax, 59
    syscall
.erro:
.sair:
    xor   edi, edi
    mov   eax, 60
    syscall

Win32 API em NASM (Windows)

NASM x64/Win — MessageBox e ExitProcess via Win32
; NASM Windows PE64
; nasm -f win64 prog.asm -o prog.obj
; link /subsystem:windows /entry:_start prog.obj kernel32.lib user32.lib

extern MessageBoxA
extern ExitProcess
extern GetConsoleWindow

MB_OK        equ 0
MB_ICONINFO  equ 0x40

section .data
    titulo  db "Ola", 0
    msg     db "Assembly Windows!", 0

section .text
global _start
_start:
    ; Windows x64 ABI: RCX=arg1, RDX=arg2, R8=arg3, R9=arg4
    ; OBRIGATÓRIO: 32 bytes de shadow space (sub rsp, 32)
    sub   rsp, 40                 ; 32 shadow + 8 para alinhar a 16

    ; MessageBoxA(NULL, "Assembly Windows!", "Ola", MB_OK|MB_ICONINFO)
    xor   ecx, ecx                ; hWnd = NULL
    lea   rdx, [rel msg]          ; lpText
    lea   r8,  [rel titulo]       ; lpCaption
    mov   r9d, MB_OK              ; uType
    call  MessageBoxA

    add   rsp, 40
    ; ExitProcess(0)
    xor   ecx, ecx
    call  ExitProcess

; ── Janela Console (GetStdHandle + WriteConsoleA) ──
extern GetStdHandle
extern WriteConsoleA
extern WriteFile

STD_OUTPUT_HANDLE equ -11

escrever_console:
    sub   rsp, 40
    mov   ecx, STD_OUTPUT_HANDLE
    call  GetStdHandle             ; rax = handle de stdout
    mov   rcx, rax                ; hStdOut
    lea   rdx, [rel msg]          ; lpBuffer
    mov   r8d, 18                 ; nNumberOfCharsToWrite
    lea   r9,  [rel .written]     ; lpNumberOfCharsWritten
    push  0                       ; lpReserved = NULL (stack)
    call  WriteConsoleA
    add   rsp, 48
    ret
.written dd 0

BIOS Interrupts e UEFI (Bare Metal)

NASM x86 Real Mode — Bootloader com interrupcoes BIOS
; Bootloader x86 real mode (16-bit)
; BIOS carrega os 512 primeiros bytes do disco em 0x7C00
; nasm -f bin bootloader.asm -o bootloader.bin
; dd if=bootloader.bin of=/dev/sdX bs=512 count=1

    BITS 16
    ORG  0x7C00

jmp inicio
times 3-($-$$) db 0

inicio:
    cli
    xor  ax, ax
    mov  ds, ax          ; data segment = 0
    mov  es, ax
    mov  ss, ax
    mov  sp, 0x7C00      ; stack abaixo do bootloader
    sti

    ; INT 10h, AH=0Eh — imprimir caractere em modo texto BIOS
    mov  si, msg
.print_loop:
    lodsb                ; AL = [SI++]
    test al, al
    jz   .done
    mov  ah, 0x0E        ; BIOS teletype output
    mov  bh, 0           ; page 0
    int  0x10
    jmp  .print_loop
.done:
    hlt

    ; INT 13h, AH=02h — ler setores do disco via BIOS
ler_disco:
    mov  ah, 0x02        ; função: read sectors
    mov  al, 1           ; número de setores
    mov  ch, 0           ; cylinder 0
    mov  cl, 2           ; setor 2 (começa em 1)
    mov  dh, 0           ; head 0
    mov  dl, 0x80        ; 1° HD
    mov  bx, 0x8000      ; ES:BX = destino na memória
    int  0x13
    jc   .disk_error     ; carry flag = erro
    ret
.disk_error:
    hlt

msg db "Assembly Bootloader!", 13, 10, 0

; Boot signature: os dois últimos bytes DEVEM ser 0x55, 0xAA
    times 510-($-$$) db 0
    dw    0xAA55
