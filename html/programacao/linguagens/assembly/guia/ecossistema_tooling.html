Assembly — Ecossistema e Tooling

O ecossistema de Assembly inclui assemblers (que traduzem mnemônicos em bytes de máquina), linkers (que combinam objetos em executáveis), debuggers de baixo nível, ferramentas de análise de binários e suporte em IDEs. Conhecer estas ferramentas é indispensável para escrever, depurar e otimizar código assembly de forma produtiva.

O fluxo completo é: código.asm → assembler (NASM/GAS) → objeto.o → linker (ld) → executável ELF/PE → executado pelo sistema operacional. Ferramentas como objdump, strace e gdb permitem inspecionar cada etapa deste processo.

Terminal — Usando NASM (Intel syntax)
# Instalação
sudo apt install nasm      # Ubuntu/Debian
brew install nasm          # macOS

# Compilar para ELF64 (Linux x86-64)
nasm -f elf64 programa.asm -o programa.o

# Com informações de debug (DWARF)
nasm -f elf64 -g -F dwarf programa.asm -o programa.o

# Listar saída (.lst) - mostra bytecode ao lado do asm
nasm -f elf64 -l programa.lst programa.asm -o programa.o

# Linkagem com ld
ld -o programa programa.o

# Linkagem com gcc (acesso à libc)
gcc -o programa programa.o -nostartfiles   # sem crt0
gcc -o programa programa.o                  # com crt0 (main normal)

# macOS: formato Mach-O 64-bit
nasm -f macho64 programa.asm -o programa.o
ld -macosx_version_min 10.15 -o programa programa.o -lSystem
GAS (GNU Assembler) — sintaxe AT&T vs Intel
# GAS AT&T syntax (padrão)
# Ordem operandos: FONTE, DESTINO (oposto ao Intel!)
# Sufixos: b=byte, w=word, l=long(32-bit), q=quad(64-bit)
# Registradores: %rax, %rdi...  Memória: (%rdi), 8(%rsp)
# Imediatos: $42, $0x1F

    .section .data
msg:
    .asciz "Hello\n"           # .asciz = string + \0
    .align 8

    .section .text
    .global _start
_start:
    movq  $1, %rax             # sys_write
    movq  $1, %rdi             # stdout
    leaq  msg(%rip), %rsi      # endereço da string (RIP-relative)
    movq  $6, %rdx             # comprimento
    syscall

    movq  $60, %rax
    xorq  %rdi, %rdi
    syscall

# Usando sintaxe Intel dentro do GAS:
    .intel_syntax noprefix
    mov rax, 1
    mov rdi, 1
    syscall
    .att_syntax prefix           # voltar para AT&T

Linkers: ld, gold, lld

Terminal — Usando linkers ld, gold e lld
# ── ld (BFD linker) — padrão GNU ──────────────────────────
ld -o saida arquivo.o               # simples
ld -o saida arquivo.o -lc -dynamic-linker /lib64/ld-linux-x86-64.so.2
# -lc: ligar com libc.  -dynamic-linker: especificar o loader ELF

ld -T script.ld -o firmware.elf kernel.o init.o  # linker script customizado

# ── Linker Script básico ───────────────────────────────────
/* firmware.ld — bare metal ARM */
OUTPUT_FORMAT(elf32-littlearm)
ENTRY(_start)
SECTIONS {
    . = 0x08000000;          /* Flash address */
    .text : { *(.text*) }
    .data : { *(.data*) }
    .bss  : { *(.bss*)  }
}

# ── lld (LLVM linker) — mais rápido ───────────────────────
clang -fuse-ld=lld -o programa arquivo.o

# ── Analisar seções do objeto ─────────────────────────────
objdump -h programa.o         # exibir seções (.text, .data, .bss...)
size programa.o               # tamanho de each section
nm programa.o                 # tabela de símbolos (funções, variáveis)
readelf -a programa           # tudo: header, seções, símbolos, dinâmicos
file programa                 # tipo do arquivo (ELF 64-bit vs PE)

GDB — Depurando Assembly

Terminal — GDB para codigo assembly x86-64
# Compilar com debug info
nasm -f elf64 -g -F dwarf prog.asm -o prog.o
ld -o prog prog.o

# Iniciar GDB
gdb ./prog

# ── Comandos essenciais dentro do GDB ─────────────────────
(gdb) layout asm           # janela dividida: assembly + registers
(gdb) layout regs          # mostrar registradores
(gdb) tui enable           # interface TUI (terminal)

(gdb) break _start         # breakpoint no label
(gdb) break *0x401000      # breakpoint em endereço
(gdb) run
(gdb) stepi                # si: executar 1 instrução
(gdb) nexti                # ni: próxima instrução (sem entrar em call)
(gdb) continue             # c: continuar até próximo break

# Inspecionar registradores
(gdb) info registers       # todos os registradores
(gdb) info registers rax rsi rdi
(gdb) p/x $rax             # print rax em hexadecimal
(gdb) p/d $rax             # print em decimal

# Inspecionar memória
(gdb) x/10xg $rsp          # 10 × 8 bytes da stack (hex)
(gdb) x/s $rdi             # string apontada por RDI
(gdb) x/4i $rip            # próximas 4 instruções

# Modificar valores
(gdb) set $rax = 42
(gdb) set *(int*)0x601000 = 99

# Desassemblar
(gdb) disas _start
(gdb) disas /r _start,+32  # +32 bytes, mostrar bytes raw

objdump, strace e perf

Terminal — Analise de binarios e syscalls
# ── objdump — disassembler e inspector de binários ─────────
objdump -d programa                    # desassemblar seções de código
objdump -d -M intel programa           # sintaxe Intel em vez de AT&T
objdump -s -j .data programa           # hex dump da seção .data
objdump -d --no-show-raw-insn prog     # sem bytes brutos (mais limpo)

# ── strace — rastrear syscalls em tempo real ───────────────
strace ./programa                      # mostrar todas as syscalls
strace -e trace=read,write ./prog      # filtrar syscalls específicas
strace -c ./programa                   # sumário estatístico ao final
strace -o log.txt ./programa           # salvar em arquivo
strace -p 1234                         # attach a processo existente

# ── perf — profiling de desempenho ─────────────────────────
perf stat ./programa                   # contadores de hardware (cache, etc.)
perf record ./programa                 # gravar amostras de CPU
perf report                            # analisar hot paths
perf list                              # eventos disponíveis

# ── readelf — estrutura ELF completa ───────────────────────
readelf -h programa                    # header ELF
readelf -S programa                    # seções
readelf -l programa                    # program headers (segmentos)
readelf -s programa                    # tabela de símbolos
readelf -d programa                    # dynamic section (libs dependidas)

# ── Extras ─────────────────────────────────────────────────
ldd programa                           # bibliotecas dinâmicas dependidas
strings programa                       # strings ASCII/UTF-8 no binário
hexdump -C programa | head             # dump hexadecimal com ASCII
xxd programa | head                    # alternativa ao hexdump

Makefile para Projetos Assembly

Makefile — Projeto NASM x86-64 completo
# Makefile para projetos NASM x86-64 / Linux

ASM     := nasm
ASFLAGS := -f elf64 -g -F dwarf
LD      := ld
LDFLAGS :=
CC      := gcc
CFLAGS  := -O2 -Wall -g

SRC_DIR := src
OBJ_DIR := build

SRCS := $(wildcard $(SRC_DIR)/*.asm)
OBJS := $(patsubst $(SRC_DIR)/%.asm, $(OBJ_DIR)/%.o, $(SRCS))
TARGET := programa

.PHONY: all clean debug run

all: $(TARGET)

# Compilar cada .asm → .o
$(OBJ_DIR)/%.o: $(SRC_DIR)/%.asm | $(OBJ_DIR)
	$(ASM) $(ASFLAGS) $< -o $@

# Linkar todos os .o
$(TARGET): $(OBJS)
	$(LD) $(LDFLAGS) -o $@ $^

$(OBJ_DIR):
	mkdir -p $@

run: all
	./$(TARGET)

debug: all
	gdb ./$(TARGET)

clean:
	rm -rf $(OBJ_DIR) $(TARGET)
