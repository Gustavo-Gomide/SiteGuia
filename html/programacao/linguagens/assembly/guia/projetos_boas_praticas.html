Assembly — Projetos e Boas Práticas

Escrever assembly de qualidade exige disciplina de nomenclatura, uso consistente de macros para evitar repetição, comentários que explicam por que (não apenas o que), organização de seções e a sabedoria de saber quando não usar Assembly. Projetos reais combinam assembly de forma cirúrgica com C/C++ de alto desempenho.

A melhor prática mais importante em Assembly: comente cada função como se explicasse para alguém que nunca viu código assembly. Diferente de C, não há nomes de variáveis nem tipos que comuniquem intenção — o comentário é toda a documentação disponível.

Estrutura e Organizacao de Projetos

NASM — Estrutura de arquivo assembly bem organizado
;;==============================================================
;; módulo: vetor_ops.asm
;; descrição: Operações sobre vetores de double (IEEE 754 64-bit)
;; ABI: System V AMD64 (Linux/macOS)
;;==============================================================

; ── Exports (funções públicas) ─────────────────────────────
global vec_soma
global vec_produto_escalar
global vec_norma

; ── Imports (dependências externas) ───────────────────────
extern sqrt                   ; da libm

; ── Constantes (symbolic names em vez de magic numbers) ────
DOUBLE_SIZE equ 8
FLOAT_SIZE  equ 4

; ── Seção de dados read-only ───────────────────────────────
section .rodata
    .mask_abs dq 0x7FFFFFFFFFFFFFFF   ; máscara para abs(double)

; ── Seção de dados writable ────────────────────────────────
section .data
    ; (variáveis globais mutáveis aqui — evitar globais quando possível)

; ── Seção não-inicializada ────────────────────────────────
section .bss
    buf_tmp  resq 16           ; reservar 128 bytes temporários

; ── Código ───────────────────────────────────────────────
section .text

;;--------------------------------------------------------------
;; vec_soma: soma vetorial de dois arrays de doubles
;; Entrada:  rdi = ptr array A
;;           rsi = ptr array B
;;           rdx = ptr array output
;;           rcx = n (quantidade de elementos)
;; Retorno:  void
;; Clobbers: rax, rdx, rcx, xmm0, xmm1
;;--------------------------------------------------------------
vec_soma:
    test   rcx, rcx
    jz     .ret
.loop:
    movsd  xmm0, [rdi]           ; xmm0 = A[i]
    addsd  xmm0, [rsi]           ; xmm0 += B[i]
    movsd  [rdx], xmm0           ; output[i] = resultado
    add    rdi, DOUBLE_SIZE
    add    rsi, DOUBLE_SIZE
    add    rdx, DOUBLE_SIZE
    dec    rcx
    jnz    .loop
.ret:
    ret

Macros NASM: %macro, %rep, %if

NASM — Sistema de macros completo
; ── Macro simples: prologue/epilogue ──────────────────────
%macro fn_entra 1                   ; 1 argumento: bytes de stack locais
    push  rbp
    mov   rbp, rsp
    sub   rsp, %1                   ; %1 = primeiro argumento da macro
%endmacro

%macro fn_sai 0
    mov   rsp, rbp
    pop   rbp
    ret
%endmacro

; Uso:
minha_funcao:
    fn_entra 32                     ; reservar 32 bytes locais
    ; ... código da função ...
    fn_sai

; ── Macro para syscall genérica ────────────────────────────
%macro syscall1 2                   ; syscall1 num, arg1
    mov   eax, %1
    mov   rdi, %2
    syscall
%endmacro

%macro syscall3 4                   ; syscall3 num, arg1, arg2, arg3
    mov   eax, %1
    mov   rdi, %2
    mov   rsi, %3
    mov   rdx, %4
    syscall
%endmacro

; Uso:
    syscall3 1, 1, msg, 13          ; write(stdout, msg, 13)
    syscall1 60, 0                  ; exit(0)

; ── Rotulos locais com %% ──────────────────────────────────
%macro max_int 2                    ; arg1=reg_a, arg2=reg_b  → result em rax
    mov   rax, %1
    cmp   rax, %2
    jge   %%fim                      ; %% = local label (único por expansão)
    mov   rax, %2
%%fim:
%endmacro

; ── %rep — gerar código repetitivo ────────────────────────
%assign i 0
%rep 8
    db i * 2                         ; gera: db 0, db 2, db 4... db 14
    %assign i i+1
%endrep

; ── %if — compilação condicional ──────────────────────────
%define DEBUG 1

%if DEBUG
    ; INT3 = breakpoint de software — apenas em builds debug
    %macro debug_break 0
        int 3
    %endmacro
%else
    %macro debug_break 0
    %endmacro          ; no-op em release
%endif

Interop com C/C++

NASM + C — Modulo assembly chamado de C
; --- arquivo: ops.asm ---
; Funções assembly exportadas para uso em C/C++

global checksum_fletcher32
global popcount64

section .text

;;--------------------------------------------------------------
;; uint32_t checksum_fletcher32(const uint16_t* data, size_t n)
;; rdi = ptr, rsi = número de uint16_t
;;--------------------------------------------------------------
checksum_fletcher32:
    xor   eax, eax      ; sum1 = 0
    xor   ecx, ecx      ; sum2 = 0
    test  rsi, rsi
    jz    .done
.loop:
    movzx edx, word [rdi]
    add   eax, edx
    cmp   eax, 0xFFFF
    jb    .no_wrap1
    sub   eax, 0xFFFF
.no_wrap1:
    add   ecx, eax
    cmp   ecx, 0xFFFF
    jb    .no_wrap2
    sub   ecx, 0xFFFF
.no_wrap2:
    add   rdi, 2
    dec   rsi
    jnz   .loop
.done:
    shl   ecx, 16
    or    eax, ecx
    ret

;;--------------------------------------------------------------
;; int popcount64(uint64_t x)  — conta bits 1 em uma palavra 64-bit
;; rdi = x, retorna int em rax
;;--------------------------------------------------------------
popcount64:
    popcnt rax, rdi     ; instrução dedicada (SSE4.2 / Nehalem+)
    ret
C — Header para usar o modulo assembly
// ops.h — declarações C das funções em assembly
#pragma once
#include <stdint.h>
#include <stddef.h>

uint32_t checksum_fletcher32(const uint16_t* data, size_t n);
int      popcount64(uint64_t x);

// main.c — uso
#include "ops.h"
#include <stdio.h>

int main() {
    uint16_t dados[] = {1, 2, 3, 4, 5};
    uint32_t cs = checksum_fletcher32(dados, 5);
    printf("Checksum: 0x%08X\n", cs);
    printf("popcount(255) = %d\n", popcount64(255));
}

// Compilar:
// nasm -f elf64 ops.asm -o ops.o
// gcc -o programa main.c ops.o

Quando Usar Assembly

Use Assembly quando...

  • — Otimizar hot paths críticos que o compilador não vetoriza (SIMD manual)
  • — Escrever bootloaders, firmware ou código bare-metal sem OS
  • — Implementar rotinas de criptografia que precisam de constant-time (sem branch por dado secreto)
  • — Fazer profiling e entender o output do compilador (para retroalimentar o código C)
  • — Escrever shellcode ou analisadores de segurança
  • — Desenvolver ISRs (interrupt service routines) em sistemas embarcados

Evite Assembly quando...

  • — O compilador C/C++ com -O3 já gera código equivalente
  • — O gargalo é I/O (não CPU) — assembly não acelera leitura de disco/rede
  • — A lógica muda frequentemente — manutenção de assembly grande é cara
  • — Precisar de portabilidade entre arquiteturas (x86, ARM, RISC-V)
  • — Intrinsics C (AVX/SSE) já expressam o que você precisa
  • — O time não tem expertise em assembly

Convencoes de Nomenclatura e Comentarios

NASM — Estilo de codigo assembly profissional
; ── Nomenclatura ──────────────────────────────────────────
; GLOBAIS:   snake_case, substantivos ou verbo_objeto
; Locais:    .loop, .fim, .erro, .ret (ponto no inicio = local label NASM)
; Constantes: CAPS_SNAKE_CASE  (SYS_WRITE, BUFFER_SIZE)
; Seções:    sempre .text, .data, .bss, .rodata (minúsculo)

; ── Comentários: explicar INTENÇÃO, não a instrução ────────

; MAU comentário (óbvio demais):
    inc  rax       ; incrementa rax

; BOM comentário (explica por quê):
    inc  rax       ; avança ao próximo índice do array

; BOM comentário (explica convenção):
    xor  eax, eax  ; zero-extend para 64-bit sem REX prefix (menor e mais rápido)

; ── Cabeçalho de função (template) ────────────────────────
;;------------------------------------------------------------
;; nome_funcao: descrição breve
;; Entradas:
;;   rdi = descrição do argumento 1
;;   rsi = descrição do argumento 2
;; Saída:
;;   rax = descrição do retorno
;; Preserva: rbx, rbp, r12-r15 (System V ABI)
;; Destrói:  rcx, rdx, r8, r9 (clobbers)
;;------------------------------------------------------------

; ── Alinhamento de instruções (legibilidade) ──
; Use tabulações consistentes — instrução, operandos, comentário
    mov   rdi, rax        ; align operands at column 10, comments at 40
    test  rax, rax
    jz    .erro
    call  processa

O princípio mais importante de projetos assembly: misture assembly com C com precisão cirúrgica. Escreva toda a lógica de negócio em C/C++, profile com perf, identifique as 3–5 funções que consomem 80% do CPU, e reescreva apenas essas em assembly com SIMD — o resultado é portabilidade + desempenho máximo ao mesmo tempo.
