Assembly — Avançado

Os tópicos avançados de Assembly cobrem SIMD (SSE/AVX) para processamento vetorial de múltiplos dados em paralelo, instruções de string otimizadas, syscalls diretas ao kernel Linux, convenções de chamada System V ABI e Windows x64, inline assembly dentro de C/C++ e técnicas de otimização como branch-free code, loop unrolling e prefetch de memória.

Com SIMD, uma única instrução AVX2 como vaddps ymm0, ymm1, ymm2 soma 8 floats de 32 bits simultaneamente em vez de 1 — viabilizando processamento de áudio, vídeo, criptografia e ML a velocidades impossíveis de atingir em linguagens de alto nível sem estas extensões.

NASM x86-64 — SSE: somar vetores de floats 32-bit
; Somar 4 floats de 32-bit com SSE (XMM — 128 bits = 4 × 32 bits)

section .data
    align 16                     ; SSE exige alinhamento de 16 bytes
    a    dd  1.0, 2.0, 3.0, 4.0  ; 4 floats
    b    dd  5.0, 6.0, 7.0, 8.0
    res  dd  0.0, 0.0, 0.0, 0.0

section .text
somar_sse:
    movaps  xmm0, [a]     ; carrega 4 floats alinhados
    movaps  xmm1, [b]
    addps   xmm0, xmm1    ; xmm0[0..3] += xmm1[0..3] em paralelo
    movaps  [res], xmm0   ; salva resultado
    ret

; Operações ps = packed single (4×float32)
; Operações pd = packed double (2×float64)
; Operações ss = scalar single (1×float32, bits altos ignorados)
; Operações sd = scalar double  (1×float64)

; Produto escalar de 4 floats com SSE4.1
produto_escalar:
    movaps  xmm0, [a]
    movaps  xmm1, [b]
    dpps    xmm0, xmm1, 0xF1  ; dot product: soma a*b[0..3], resultado em xmm0[0]
    ; 0xF1 = mask 1111 0001: multiplicar todos 4, somar, colocar em elem 0
    movss   [resultado], xmm0   ; salva scalar float
    ret
NASM x86-64 — AVX2: multiplicar array de 8 int32 em paralelo
; AVX2 — YMM = 256 bits = 8 × int32
; Verificar suporte: CPUID EAX=7, ECX=0 → EBX bit 5 (AVX2)

section .data
    align 32                              ; AVX precisa alinhar a 32 bytes
    v1   dd  1,2,3,4,5,6,7,8
    v2   dd  2,2,2,2,2,2,2,2
    out  dd  0,0,0,0,0,0,0,0

section .text
multiplicar_avx2:
    vmovdqa  ymm0, [v1]       ; carrega 8 int32 alinhados
    vmovdqa  ymm1, [v2]
    vpmulld  ymm2, ymm0, ymm1 ; multiply 8 packed int32 low (AVX2)
    vmovdqa  [out], ymm2
    vzeroupper                 ; OBRIGATÓRIO: limpa bits 128+ antes de chamar C
    ret

; Loop SIMD: processar N floats com AVX (passo de 8)
simd_loop:
    ; rdi = ponteiro para array, rsi = n elementos
    xor   rcx, rcx
.loop:
    cmp   rcx, rsi
    jge   .fim
    vmovups  ymm0, [rdi + rcx*4]  ; vmovups = não alinhado
    vmulps   ymm0, ymm0, ymm0       ; elevar ao quadrado (8 floats)
    vmovups  [rdi + rcx*4], ymm0
    add   rcx, 8
    jmp   .loop
.fim:
    vzeroupper
    ret

Syscalls Linux e Convencoes de Chamada

NASM x86-64 — Syscalls Linux (tabela e exemplos)
; Linux x86-64 syscall ABI:
;   Número da syscall: RAX
;   Argumentos:        RDI, RSI, RDX, R10, R8, R9
;   Retorno:           RAX  (negativo = -errno)
;   Registradores preservados: RBX, RBP, RSP, R12-R15
;   Clobbers:          RCX, R11, RAX

SYS_read    equ 0
SYS_write   equ 1
SYS_open    equ 2
SYS_close   equ 3
SYS_mmap    equ 9
SYS_munmap  equ 11
SYS_exit    equ 60
SYS_socket  equ 41
SYS_connect equ 42

; open("/etc/hostname", O_RDONLY)
open_arquivo:
    lea   rdi, [rel path]    ; rdi = "/etc/hostname"
    xor   esi, esi           ; flags = O_RDONLY = 0
    xor   edx, edx           ; mode = 0
    mov   eax, SYS_open
    syscall
    ; rax = fd  (ou < 0 em erro)
    test  rax, rax
    js    .erro              ; SF=1 → negativo → erro
    ret

; mmap: alocar memória anônima
alocar_mem:
    ; void* mmap(NULL, size, PROT_READ|PROT_WRITE, MAP_PRIVATE|MAP_ANON, -1, 0)
    xor   edi, edi           ; addr = NULL
    mov   rsi, 4096         ; size = 1 página
    mov   edx, 3            ; PROT_READ(1)|PROT_WRITE(2)
    mov   r10d, 0x22        ; MAP_PRIVATE(2)|MAP_ANONYMOUS(0x20)
    mov   r8d, -1           ; fd = -1
    xor   r9d, r9d           ; offset = 0
    mov   eax, SYS_mmap
    syscall                  ; rax = endereço da memória alocada
    ret
NASM — Convencoes de chamada: System V vs Windows x64
; ╔═══════════════════════════╦══════════════════════╦══════════════════════╗
; ║ Posição                   ║ System V AMD64 (Linux)║ Windows x64          ║
; ╠═══════════════════════════╬══════════════════════╬══════════════════════╣
; ║ 1° argumento inteiro      ║ RDI                  ║ RCX                  ║
; ║ 2° argumento inteiro      ║ RSI                  ║ RDX                  ║
; ║ 3° argumento inteiro      ║ RDX                  ║ R8                   ║
; ║ 4° argumento inteiro      ║ RCX                  ║ R9                   ║
; ║ 5° argumento inteiro      ║ R8                   ║ stack                ║
; ║ 6° argumento inteiro      ║ R9                   ║ stack                ║
; ║ Retorno inteiro           ║ RAX (RDX extensão)   ║ RAX                  ║
; ║ Retorno float             ║ XMM0                 ║ XMM0                 ║
; ║ Callee-saved              ║ RBX,RBP,R12-R15      ║ RBX,RBP,RDI,RSI      ║
; ║ Shadow space              ║ Nao                  ║ 32 bytes obrigatorio ║
; ╚═══════════════════════════╩══════════════════════╩══════════════════════╝

; Windows x64: SEMPRE reservar 32 bytes de shadow space antes de CALL
chamada_windows:
    sub   rsp, 40        ; 32 shadow + 8 alinhar (40 = 8*5, múltiplo de 16)
    mov   rcx, rdi       ; 1° arg: Windows usa RCX
    mov   rdx, rsi       ; 2° arg
    call  funcao_win
    add   rsp, 40
    ret

Inline Assembly em C/C++

C/GCC — GCC inline assembly com constraints
// GCC Extended Inline Assembly:  asm volatile ("instrucoes" : outputs : inputs : clobbers);

// 1. Somar dois inteiros sem usar o operador +
int soma_asm(int a, int b) {
    int result;
    asm volatile (
        "addl %2, %0"          // instrucao: %0=result, %1=a, %2=b
        : "=r" (result)         // output: result em qualquer reg
        : "0" (a), "r" (b)      // inputs: %0 inicializado com 'a', b em reg
        : "cc"                   // clobber: flags (cc)
    );
    return result;
}

// 2. CPUID — detectar funcionalidades do processador
void get_cpuid(uint32_t leaf, uint32_t* eax, uint32_t* ebx,
               uint32_t* ecx, uint32_t* edx) {
    asm volatile (
        "cpuid"
        : "=a"(*eax), "=b"(*ebx), "=c"(*ecx), "=d"(*edx)
        : "a"(leaf)
    );
}

// 3. RDTSC — ler contador de ciclos (benchmarking)
uint64_t rdtsc() {
    uint32_t lo, hi;
    asm volatile (
        "rdtsc"
        : "=a"(lo), "=d"(hi)
    );
    return ((uint64_t)hi << 32) | lo;
}

// 4. Memory barrier (fence)
void memory_fence() {
    asm volatile ("mfence" ::: "memory"); // "memory" = clobber toda memória
}

// 5. SIMD via intrinsics (recomendado sobre inline ASM)
#include <immintrin.h>
void somar_avx(float* a, float* b, float* c, int n) {
    for (int i = 0; i < n; i += 8) {
        __m256 va = _mm256_loadu_ps(&a[i]);  // carregar 8 floats
        __m256 vb = _mm256_loadu_ps(&b[i]);
        __m256 vc = _mm256_add_ps(va, vb);   // soma vetorial
        _mm256_storeu_ps(&c[i], vc);
    }
}

Tecnicas de Otimizacao

NASM x86-64 — Branch-free, loop unrolling, prefetch
; ── Branch-free: min(a, b) sem salto ──────────────
; Usando CMOV (conditional move) — sem branch prediction penalty
min_branchfree:
    ; rdi = a,  rsi = b,  retorna min em rax
    mov   rax, rdi
    cmp   rdi, rsi
    cmovg rax, rsi      ; if (a > b): rax = b
    ret

; ── Absolute value sem branch ──────────────────────
abs_inteiro:
    ; rdi = x, retorna |x| em rax
    mov   rax, rdi
    mov   rcx, rdi
    sar   rcx, 63       ; rcx = 0 se x>=0, -1 (0xFFF...) se x<0
    xor   rax, rcx       ; inverte bits se negativo
    sub   rax, rcx       ; soma 1 se negativo (complemento de dois)
    ret

; ── Loop Unrolling — somar array de 8 int64 (4-way unroll) ─
soma_array:
    ; rdi = ponteiro, rsi = count (múltiplo de 4)
    xor   rax, rax        ; acumulador
    xor   rcx, rcx
.loop:
    cmp   rcx, rsi
    jge   .fim
    add   rax, [rdi + rcx*8]      ; iter 1
    add   rax, [rdi + rcx*8 + 8]  ; iter 2
    add   rax, [rdi + rcx*8 + 16] ; iter 3
    add   rax, [rdi + rcx*8 + 24] ; iter 4 (4 loads por iteração)
    add   rcx, 4
    jmp   .loop
.fim:
    ret

; ── Prefetch — carregar cache antes de precisar ────────
prefetch_example:
    ; PREFETCHT0: carregar em todos os níveis de cache
    ; PREFETCHT1: L2/L3.  PREFETCHNTA: non-temporal (streaming)
    prefetcht0 [rdi + 256]   ; pré-carregar 256 bytes à frente
    mov        rax, [rdi]
    ret
