OCaml — Web com Dream

OCaml — Web com Dream

Dream é um framework web completo para OCaml: roteamento tipado, middleware composável, sessões, WebSockets e templates HTML nativos em um único pacote.

Dream combina a segurança de tipos de OCaml com uma API web inspirada no Express (Node.js) e no Plug (Elixir). Cada handler tem tipo Dream.handler = Dream.request -> Dream.response Lwt.t, permitindo composição segura sem magic implícito.

Roteamento e Parâmetros

main.ml — Rotas, parâmetros e métodos HTTP
open Lwt.Syntax

(* Handlers são funções: request -> response Lwt.t *)
let home _req =
  Dream.html {|
    <html>
    <body>
      <h1>Home</h1>
      <a href="/usuarios">Usuários</a>
    </body>
    </html>
  |}

(* Parâmetros de rota com :nome *)
let get_usuario req =
  let id = Dream.param req "id" in   (* string *)
  match int_of_string_opt id with
  | None   -> Dream.respond ~status:`Bad_Request "ID inválido"
  | Some n ->
      Dream.json
        (Printf.sprintf {|{"id":%d,"nome":"Usuário %d"}|} n n)

(* Query strings *)
let buscar req =
  let query = Dream.query req "q" |> Option.value ~default:"" in
  let pagina = Dream.query req "pagina"
    |> Option.bind int_of_string_opt
    |> Option.value ~default:1 in
  Dream.json
    (Printf.sprintf {|{"query":"%s","pagina":%d}|} query pagina)
(* GET /buscar?q=ocaml&pagina=2 *)

(* Corpo de requisição POST *)
let criar_usuario req =
  let* corpo = Dream.body req in
  (* corpo é uma string JSON *)
  Dream.respond ~status:`Created
    (Printf.sprintf {|{"criado":true,"dados":%s}|} corpo)

(* Headers de resposta customizados *)
let com_cors _req =
  Dream.respond
    ~headers:[
      "Access-Control-Allow-Origin", "*";
      "Content-Type", "application/json";
    ]
    {|{"status":"ok"}|}

let () =
  Dream.run ~port:8080
  @@ Dream.logger
  @@ Dream.router [
    Dream.get  "/"                 home;
    Dream.get  "/usuarios/:id"    get_usuario;
    Dream.post "/usuarios"         criar_usuario;
    Dream.get  "/buscar"           buscar;
    Dream.get  "/cors"             com_cors;
  ]

Middleware — Composição de Handlers

Middlewares em Dream são funções Dream.handler -> Dream.handler. O operador @@ compõe middlewares em ordem: cada um pode interceptar antes e depois do handler interno.

middleware.ml — Middleware customizado, autenticação e rate limiting
open Lwt.Syntax

(* ── Middleware de autenticação JWT simplificado ────────────────────── *)
let autenticar next req =
  match Dream.header req "Authorization" with
  | None ->
      Dream.respond ~status:`Unauthorized
        {|{"erro":"token obrigatório"}|}
  | Some header when String.length header < 8 ->
      Dream.respond ~status:`Unauthorized
        {|{"erro":"token inválido"}|}
  | Some header ->
      let token = String.sub header 7 (String.length header - 7) in
      (* validar JWT aqui... *)
      Dream.set_local req "usuario_token" token;
      next req

(* ── Middleware de tempo de resposta ────────────────────────────────── *)
let medir_tempo next req =
  let t0 = Unix.gettimeofday () in
  let* resp = next req in
  let elapsed = (Unix.gettimeofday () -. t0) *. 1000.0 in
  Printf.printf "[%s] %s %.2fms\n"
    (Dream.method_to_string (Dream.method_ req))
    (Dream.target req)
    elapsed;
  Lwt.return resp

(* ── Sessões com Dream.session ──────────────────────────────────────── *)
let login req =
  let* corpo = Dream.body req in
  (* parse e validar credenciais aqui... *)
  ignore corpo;
  let* () = Dream.invalidate_session req in
  let* () = Dream.set_session_field req "usuario_id" "42" in
  Dream.respond {|{"login":"ok"}|}

let perfil req =
  match Dream.session_field req "usuario_id" with
  | None    -> Dream.respond ~status:`Unauthorized {|{"erro":"não autenticado"}|}
  | Some id -> Dream.json (Printf.sprintf {|{"usuario_id":"%s"}|} id)

let () =
  Dream.run ~port:8080
  @@ Dream.logger
  @@ medir_tempo
  @@ Dream.memory_sessions   (* sessões em memória para dev *)
  @@ Dream.router [
    Dream.post "/login"   login;
    Dream.get  "/perfil"  perfil;
    Dream.scope "/api" [autenticar] [
      Dream.get  "/dados" (fun _r -> Dream.json {|{"dados":[1,2,3]}|});
    ];
  ]

Yojson — Processamento de JSON

json_exemplo.ml — Yojson e ppx_deriving_yojson
(* opam install yojson ppx_deriving_yojson *)
(* dune: (libraries yojson) (preprocess (pps ppx_deriving_yojson)) *)

(* ── Manual parsing com Yojson.Basic ────────────────────────────────── *)
open Yojson.Basic.Util

let json_str = {|
  {
    "nome": "Alice",
    "idade": 30,
    "ativo": true,
    "enderecos": ["brasil", "eua"]
  }
|}

let () =
  let json = Yojson.Basic.from_string json_str in
  let nome   = json |> member "nome"  |> to_string in
  let idade  = json |> member "idade" |> to_int in
  let ativo  = json |> member "ativo" |> to_bool in
  let ends   = json |> member "enderecos" |> to_list |> List.map to_string in
  Printf.printf "%s, %d, %b, [%s]\n"
    nome idade ativo (String.concat "; " ends)

(* ── Construir JSON ─────────────────────────────────────────────────── *)
let usuario_para_json nome idade =
  `Assoc [
    "nome",  `String nome;
    "idade", `Int    idade;
    "ativo", `Bool   true;
  ]
  |> Yojson.Basic.to_string

(* ── ppx_deriving_yojson — serialização automática ─────────────────── *)
type usuario = {
  id    : int;
  nome  : string;
  email : string option;   (* campo opcional *)
} [@@deriving yojson]
(* Gera automaticamente:
   usuario_to_yojson : usuario -> Yojson.Safe.t
   usuario_of_yojson : Yojson.Safe.t -> (usuario, string) result *)

let alice = { id = 1; nome = "Alice"; email = Some "alice@ex.com" }
let json  = usuario_to_yojson alice |> Yojson.Safe.to_string
(* {"id":1,"nome":"Alice","email":"alice@ex.com"} *)

let () =
  match Yojson.Safe.from_string json |> usuario_of_yojson with
  | Ok u  -> Printf.printf "Nome: %s\n" u.nome
  | Error e -> Printf.printf "Erro: %s\n" e

Cohttp — Cliente HTTP

http_client.ml — Cohttp cliente assíncrono
(* opam install cohttp-lwt-unix *)
(* dune: (libraries cohttp-lwt-unix lwt lwt.unix) *)

open Lwt.Syntax
module Client = Cohttp_lwt_unix.Client

(* GET request *)
let get_json url =
  let* resp, corpo = Client.get (Uri.of_string url) in
  let status = Cohttp.Response.status resp |> Cohttp.Code.code_of_status in
  let* body_str = Cohttp_lwt.Body.to_string corpo in
  Lwt.return (status, body_str)

(* POST request com JSON *)
let post_json url json_body =
  let headers = Cohttp.Header.of_list
    ["Content-Type", "application/json"] in
  let body = Cohttp_lwt.Body.of_string json_body in
  let* resp, corpo = Client.post ~headers ~body (Uri.of_string url) in
  let status = Cohttp.Response.status resp |> Cohttp.Code.code_of_status in
  let* body_str = Cohttp_lwt.Body.to_string corpo in
  Lwt.return (status, body_str)

let () =
  Lwt_main.run (
    let* status, corpo = get_json "https://httpbin.org/get" in
    Printf.printf "Status: %d\nCorpo: %s\n" status corpo;
    Lwt.return ()
  )

Dream + htmx — Server-Side Rendering

Dream integra excelentemente com htmx: o servidor retorna fragmentos HTML diretamente, sem precisar de um SPA JavaScript. Isso mantém a lógica de negócio inteiramente em OCaml com segurança de tipos completa.

main.ml — Dream + htmx (SSR interativo)
open Lwt.Syntax

(* Estado simples em memória — em produção use banco de dados *)
let tarefas : (int * string * bool) list ref =
  ref [(1, "Aprender OCaml", true); (2, "Usar Dream", false)]

let render_tarefa (id, texto, feita) =
  Printf.sprintf {|
    <li id="tarefa-%d">
      <input type="checkbox" %s
        hx-patch="/tarefas/%d/toggle"
        hx-target="#tarefa-%d"
        hx-swap="outerHTML">
      %s
      <button hx-delete="/tarefas/%d"
              hx-target="#tarefa-%d"
              hx-swap="outerHTML">X</button>
    </li>
  |} id (if feita then "checked" else "") id id texto id id

let render_lista () =
  let itens = List.map render_tarefa !tarefas |> String.concat "" in
  Printf.sprintf {|<ul id="lista-tarefas">%s</ul>|} itens

let pagina_completa () = Printf.sprintf {|
<!DOCTYPE html>
<html>
<head>
  <title>To-do OCaml</title>
  <script src="https://unpkg.com/htmx.org@2"></script>
</head>
<body>
  <h1>To-do List (OCaml + htmx)</h1>
  %s
  <form hx-post="/tarefas" hx-target="#lista-tarefas" hx-swap="outerHTML">
    <input name="texto" placeholder="Nova tarefa">
    <button type="submit">Adicionar</button>
  </form>
</body>
</html>
|} (render_lista ())

let proximo_id = ref 3

let () =
  Dream.run ~port:8080
  @@ Dream.logger
  @@ Dream.router [
    Dream.get "/" (fun _r -> Dream.html (pagina_completa ()));

    Dream.post "/tarefas" (fun req ->
      let* form = Dream.form req in
      match List.assoc_opt "texto" form with
      | Some texto when texto <> "" ->
          let id = !proximo_id in
          incr proximo_id;
          tarefas := !tarefas @ [(id, texto, false)];
          Dream.html (render_lista ())
      | _ -> Dream.respond ~status:`Bad_Request "texto obrigatório"
    );

    Dream.delete "/tarefas/:id" (fun req ->
      let id = Dream.param req "id" |> int_of_string in
      tarefas := List.filter (fun (i,_,_) -> i <> id) !tarefas;
      Dream.respond ""
    );
  ]

Deploy com Docker

Dockerfile — multistage build para OCaml
# Stage 1: Build
FROM ocaml/opam:ubuntu-22.04-ocaml-5.2 AS builder

WORKDIR /app

# Instalar dependências do sistema
RUN sudo apt-get update && sudo apt-get install -y \
    libssl-dev pkg-config

# Copiar arquivos do projeto
COPY --chown=opam:opam . .

# Instalar dependências OCaml
RUN opam install . --deps-only --yes

# Compilar
RUN eval $(opam env) && dune build bin/main.exe

# Stage 2: Runtime (imagem mínima)
FROM ubuntu:22.04

RUN apt-get update && apt-get install -y \
    libssl3 ca-certificates && \
    rm -rf /var/lib/apt/lists/*

COPY --from=builder /app/_build/default/bin/main.exe /usr/local/bin/app

EXPOSE 8080

CMD ["/usr/local/bin/app"]
bash — Build e deploy
# Build da imagem
docker build -t meu-app-ocaml .

# Rodar localmente
docker run -p 8080:8080 meu-app-ocaml

# Com variáveis de ambiente
docker run -p 8080:8080 \
  -e DATABASE_URL="postgres://..." \
  -e SECRET_KEY="chave-secreta" \
  meu-app-ocaml

# Deploy no Fly.io (plataforma amigável para OCaml)
fly launch --dockerfile Dockerfile
fly deploy
