OCaml — Terminal & CLI

Construa ferramentas de linha de comando robustas, com parsing de argumentos tipado, executáveis standalone e integração nativa com o sistema operacional.

OCaml compila para executáveis nativos pequenos e rápidos — ideais para ferramentas CLI. A biblioteca Cmdliner oferece parsing de argumentos declarativo e geração automática de páginas de manual (manpages). Ferramentas como opam, dune e odig são escritas em OCaml.

Argumentos Simples — Sys.argv

Para ferramentas simples sem subcomandos, Sys.argv é o ponto de entrada direto. É um array de strings onde Sys.argv.(0) é o nome do programa.

bin/main.ml — Argumentos básicos com Sys.argv
(* Sys.argv : string array *)
let () =
  let argc = Array.length Sys.argv in
  if argc < 2 then (
    Printf.eprintf "Uso: %s <arquivo>\n" Sys.argv.(0);
    exit 1
  );
  let arquivo = Sys.argv.(1) in
  if not (Sys.file_exists arquivo) then (
    Printf.eprintf "Erro: arquivo '%s' não encontrado\n" arquivo;
    exit 2
  );
  Printf.printf "Processando: %s\n" arquivo

(* Processar flags manualmente *)
let parse_args () =
  let verbose = ref false in
  let output  = ref "saida.txt" in
  let inputs  = ref [] in
  let i = ref 1 in
  while !i < Array.length Sys.argv do
    match Sys.argv.(!i) with
    | "-v" | "--verbose" ->
        verbose := true;
        incr i
    | "-o" | "--output" ->
        incr i;
        if !i >= Array.length Sys.argv then
          (Printf.eprintf "-o requer argumento\n"; exit 1);
        output := Sys.argv.(!i);
        incr i
    | arg ->
        inputs := arg :: !inputs;
        incr i
  done;
  (!verbose, !output, List.rev !inputs)

Cmdliner — Parsing Declarativo de Argumentos

Cmdliner é a biblioteca padrão para CLIs profissionais em OCaml. Define argumentos de forma declarativa usando tipos OCaml — sem parsing manual, com validação automática, mensagens de ajuda geradas automaticamente e suporte completo a subcomandos.

bin/main.ml — Cmdliner completo com subcomandos
(* opam install cmdliner *)
(* dune: (libraries cmdliner) *)

open Cmdliner

(* ── Tipos dos argumentos ─────────────────────────────────────────── *)

(* Flag booleana *)
let verbose =
  Arg.(value & flag & info ["v"; "verbose"]
    ~doc:"Modo verboso: imprime informações extras")

(* Argumento com valor *)
let output =
  Arg.(value
    & opt string "saida.txt"
    & info ["o"; "output"]
      ~docv:"ARQUIVO"
      ~doc:"Arquivo de saída. Padrão: $(docv).")

(* Argumento posicional obrigatório *)
let entrada =
  Arg.(required
    & pos 0 (some string) None
    & info [] ~docv:"ARQUIVO_ENTRADA"
      ~doc:"Arquivo de entrada a processar.")

(* Lista de arquivos posicionais *)
let arquivos =
  Arg.(non_empty
    & pos_all string []
    & info [] ~docv:"ARQUIVOS")

(* ── Implementação dos comandos ────────────────────────────────────── *)

let cmd_converter verbose output entrada =
  if verbose then
    Printf.printf "[VERBOSE] Convertendo %s -> %s\n" entrada output;
  (* lógica real de conversão aqui *)
  Printf.printf "Convertido: %s\n" entrada;
  Cmd.Exit.ok

let cmd_listar verbose arquivos =
  if verbose then
    Printf.printf "Listando %d arquivos...\n" (List.length arquivos);
  List.iter (Printf.printf "- %s\n") arquivos;
  Cmd.Exit.ok

(* ── Definição dos termos (subcomandos) ────────────────────────────── *)

let converter_cmd =
  let doc = "Converte um arquivo para o formato de saída" in
  Cmd.v (Cmd.info "converter" ~doc) @@
  Term.(const cmd_converter $ verbose $ output $ entrada)

let listar_cmd =
  let doc = "Lista os arquivos fornecidos" in
  Cmd.v (Cmd.info "listar" ~doc) @@
  Term.(const cmd_listar $ verbose $ arquivos)

(* ── Grupo principal ───────────────────────────────────────────────── *)

let () =
  let doc = "Ferramenta de processamento de arquivos" in
  let man = [
    `S "DESCRIÇÃO";
    `P "Esta ferramenta converte e lista arquivos.";
    `S "EXEMPLOS";
    `P "$(tname) converter entrada.txt -o saida.txt";
    `P "$(tname) listar -v arquivo1.txt arquivo2.txt";
  ] in
  let info = Cmd.info "minha-ferramenta" ~version:"1.0.0" ~doc ~man in
  exit @@ Cmd.eval @@ Cmd.group info [converter_cmd; listar_cmd]

(* Uso:
   ./minha-ferramenta --help
   ./minha-ferramenta converter --help
   ./minha-ferramenta converter entrada.txt -o saida.txt -v
   ./minha-ferramenta listar *.ml *)

I/O de Arquivos e stdin/stdout

io.ml — Leitura e escrita de arquivos
(* ── Leitura de arquivo inteiro ─────────────────────────────────────── *)
let ler_arquivo caminho =
  let ic = open_in caminho in
  let n = in_channel_length ic in
  let s = Bytes.create n in
  really_input ic s 0 n;
  close_in ic;
  Bytes.to_string s

(* ── Leitura linha por linha ────────────────────────────────────────── *)
let ler_linhas caminho =
  let ic = open_in caminho in
  let linhas = ref [] in
  (try
    while true do
      linhas := input_line ic :: !linhas
    done
  with End_of_file -> ());
  close_in ic;
  List.rev !linhas

(* Alternativa idiomática com Seq (OCaml 4.11+) *)
let linhas_seq ic =
  Seq.of_dispenser (fun () ->
    try Some (input_line ic)
    with End_of_file -> None)

(* ── Escrita de arquivo ─────────────────────────────────────────────── *)
let escrever_arquivo caminho conteudo =
  let oc = open_out caminho in
  output_string oc conteudo;
  close_out oc

(* Com tratamento de erro seguro *)
let com_arquivo modo caminho f =
  let canal = match modo with
    | `Leitura -> In_channel.open_text caminho
    | `Escrita -> failwith "use Out_channel"
  in
  let resultado = try f canal with exn ->
    In_channel.close canal;
    raise exn
  in
  In_channel.close canal;
  resultado

(* ── Leitura de stdin ───────────────────────────────────────────────── *)
let () =
  try
    while true do
      let linha = input_line stdin in
      Printf.printf "Lido: %s\n" linha
    done
  with End_of_file -> ()

(* Pipeline: cat arquivo.txt | ./programa *)

Módulo Unix — Integração com o Sistema

O módulo Unix expõe as syscalls POSIX. Permite criar processos filhos, pipes, sinais, sockets e manipular o sistema de arquivos de forma precisa.

unix_exemplo.ml — Processos, pipes e variáveis de ambiente
(* dune: (libraries unix) *)
(* opam install conf-pkg-config *)

(* ── Executar comandos externos ─────────────────────────────────────── *)

(* Método simples: Sys.command *)
let codigo = Sys.command "ls -la"   (* retorna exit code *)

(* Capturar saída de um comando *)
let capturar_saida cmd =
  let ic = Unix.open_process_in cmd in
  let buf = Buffer.create 256 in
  (try
    while true do
      Buffer.add_string buf (input_line ic ^ "\n")
    done
  with End_of_file -> ());
  ignore (Unix.close_process_in ic);
  Buffer.contents buf

let git_log = capturar_saida "git log --oneline -5"

(* ── Variáveis de ambiente ─────────────────────────────────────────── *)
let path  = Sys.getenv_opt "PATH"      (* string option *)
let home  = Sys.getenv_opt "HOME"      (* None no Windows *)
let ()    = Unix.putenv "MEU_VAR" "valor"

(* ── Fork e exec ────────────────────────────────────────────────────── *)
let spawn_process programa args =
  match Unix.fork () with
  | 0 ->
    (* Processo filho *)
    Unix.execvp programa (Array.of_list (programa :: args));
    exit 127   (* execvp não retorna em caso de sucesso *)
  | pid ->
    (* Processo pai — aguarda filho *)
    match Unix.waitpid [] pid with
    | _, WEXITED n   -> n
    | _, WSIGNALED _ -> -1
    | _, WSTOPPED _  -> -2

(* ── Manipulação de arquivos Unix ──────────────────────────────────── *)
let info_arquivo caminho =
  let stat = Unix.stat caminho in
  Printf.printf "Tamanho: %d bytes\n" stat.Unix.st_size;
  Printf.printf "Permissões: %o\n"  stat.Unix.st_perm;
  match stat.Unix.st_kind with
  | S_REG  -> print_endline "Arquivo regular"
  | S_DIR  -> print_endline "Diretório"
  | S_LNK  -> print_endline "Link simbólico"
  | _      -> print_endline "Outro"

bin/main.ml — Clone do wc
type contagem = {
  linhas   : int;
  palavras : int;
  bytes    : int;
}

let contar_arquivo caminho =
  let ic = open_in_bin caminho in
  let linhas = ref 0 in
  let palavras = ref 0 in
  let bytes = ref 0 in
  let em_palavra = ref false in
  (try
    while true do
      let c = input_char ic in
      incr bytes;
      if c = '\n' then incr linhas;
      if c = ' ' || c = '\t' || c = '\n' then
        em_palavra := false
      else if not !em_palavra then (
        incr palavras;
        em_palavra := true
      )
    done
  with End_of_file -> ());
  close_in ic;
  { linhas = !linhas; palavras = !palavras; bytes = !bytes }

let () =
  if Array.length Sys.argv < 2 then (
    Printf.eprintf "Uso: meu-wc <arquivo...>\n";
    exit 1
  );
  let arquivos = Array.to_list (Array.sub Sys.argv 1 (Array.length Sys.argv - 1)) in
  List.iter (fun f ->
    let c = contar_arquivo f in
    Printf.printf "%8d %8d %8d %s\n" c.linhas c.palavras c.bytes f
  ) arquivos

(* Compilar e usar:
   dune build
   ./_build/default/bin/main.exe arquivo.txt
   dune install
   meu-wc arquivo.txt *)
