OCaml — Avançado

Functors, GADTs, concorrência com Lwt, tipos algébricos polimórficos e o sistema de módulos mais poderoso das linguagens de produção.

O sistema de módulos de OCaml é considerado o mais expressivo entre linguagens de produção. Functors são "funções sobre módulos" — permitem parametrizar implementações por comportamento, de forma verificada pelo compilador, sem overhead em runtime.

Tipos Algébricos — Variants

Tipos algébricos (também chamados de sum types ou variants) são o coração expressivo de OCaml. Permitem modelar dados com estrutura precisa, sem null pointers, sem casting, sem erros de tipo em runtime.

variants.ml — Tipos algébricos completos
(* Enumeração simples *)
type cor = Vermelho | Verde | Azul | Outro of string

let nome_cor = function
  | Vermelho  -> "vermelho"
  | Verde     -> "verde"
  | Azul      -> "azul"
  | Outro s  -> s

(* Tipo com dados associados — expressões de calculadora *)
type expr =
  | Num    of float
  | Var    of string
  | Add    of expr * expr
  | Mul    of expr * expr
  | Neg    of expr
  | Let    of string * expr * expr  (* let x = e1 in e2 *)

(* Avaliador recursivo *)
let rec eval env = function
  | Num n          -> n
  | Var x          -> List.assoc x env
  | Add (e1, e2)   -> eval env e1 +. eval env e2
  | Mul (e1, e2)   -> eval env e1 *. eval env e2
  | Neg e          -> -. (eval env e)
  | Let (x, e1, e2) ->
      let v = eval env e1 in
      eval ((x, v) :: env) e2

(* let x = 3 + 4 in x * 2 → 14.0 *)
let arvore =
  Let ("x", Add (Num 3.0, Num 4.0),
    Mul (Var "x", Num 2.0))

let () = Printf.printf "%.1f\n" (eval [] arvore)  (* 14.0 *)

(* Tipo polimórfico — listas e árvores genéricas *)
type 'a arvore_bin =
  | Folha
  | Nó of 'a arvore_bin * 'a * 'a arvore_bin

let rec inserir v = function
  | Folha          -> Nó (Folha, v, Folha)
  | Nó (l, x, r)  ->
      if      v < x  then Nó (inserir v l, x, r)
      else if v > x  then Nó (l, x, inserir v r)
      else                  Nó (l, x, r)   (* duplicata: ignora *)

let rec em_ordem = function
  | Folha         -> []
  | Nó (l, x, r) -> em_ordem l @ [x] @ em_ordem r

let bst =
  [5; 3; 7; 1; 4]
  |> List.fold_left (fun t v -> inserir v t) Folha
  |> em_ordem
(* [1; 3; 4; 5; 7] *)

Sistema de Módulos — Signature e Functor

O sistema de módulos de OCaml tem três componentes:

  • Structure (struct ... end): implementação — o "corpo" do módulo
  • Signature (sig ... end): interface — o "contrato" do módulo
  • Functor: função que recebe módulos como argumento e retorna novos módulos
modulos_avancados.ml — Signature, hiding e Functor
(* ── Signature (interface/contrato) ────────────────────────────────── *)
module type PILHA = sig
  type 'a t                           (* tipo abstrato *)
  val vazia  : 'a t
  val empilhar : 'a -> 'a t -> 'a t
  val topo   : 'a t -> 'a option
  val desempilhar : 'a t -> 'a t
  val tamanho : 'a t -> int
end

(* ── Implementação (structure) ──────────────────────────────────────── *)
module PilhaLista : PILHA = struct
  type 'a t = 'a list
  let vazia            = []
  let empilhar x xs   = x :: xs
  let topo             = function [] -> None | x :: _ -> Some x
  let desempilhar      = function [] -> [] | _ :: xs -> xs
  let tamanho          = List.length
end

(* A assinatura PILHA oculta 'a t = 'a list — encapsulamento real *)

(* ── Functor — função sobre módulos ─────────────────────────────────── *)
(* Assinatura que qualquer tipo ordenável deve satisfazer *)
module type ORDENAVEL = sig
  type t
  val compare : t -> t -> int
end

(* Functor: recebe módulo ORDENAVEL, retorna módulo com Heap *)
module HeapMin (Elem : ORDENAVEL) = struct
  type heap = Vazia | Nó of Elem.t * heap * heap

  let vazia = Vazia

  let rec inserir x = function
    | Vazia          -> Nó (x, Vazia, Vazia)
    | Nó (r, l, d)  ->
        if Elem.compare x r <= 0
        then Nó (x, inserir r d, l)
        else Nó (r, inserir x d, l)

  let mínimo = function
    | Vazia     -> None
    | Nó(x,_,_) -> Some x
end

(* Instanciar functor com inteiros *)
module HeapInt = HeapMin(struct
  type t = int
  let compare = compare
end)

(* Instanciar com strings — mesmo functor, tipo diferente *)
module HeapStr = HeapMin(String)

let h = [3; 1; 4; 1; 5]
        |> List.fold_left (fun h x -> HeapInt.inserir x h) HeapInt.vazia

let () =
  match HeapInt.mínimo h with
  | Some m -> Printf.printf "Mínimo: %d\n" m  (* 1 *)
  | None   -> print_endline "Heap vazia"

Funções de Ordem Superior Avançadas

ordem_superior.ml — Map, Filter, Fold, combinadores
(* fold_left é a função mais versátil: implementa tudo *)
let minha_map f xs =
  List.fold_left (fun acc x -> acc @ [f x]) [] xs

let meu_filter pred xs =
  List.fold_left (fun acc x -> if pred x then acc @ [x] else acc) [] xs

let meu_flatten xss =
  List.fold_left (@) [] xss

(* Transducer pattern — composição eficiente de map+filter *)
let transforma ops lista =
  List.fold_left
    (fun acc x ->
      List.fold_left
        (fun acc' op -> op acc' x)
        acc
        ops)
    []
    lista

(* Memoização com Hashtbl *)
let memoize f =
  let cache = Hashtbl.create 64 in
  fun x ->
    match Hashtbl.find_opt cache x with
    | Some v -> v
    | None   ->
        let v = f x in
        Hashtbl.add cache x v;
        v

(* Fibonacci com memoização *)
let rec fib_raw n =
  if n <= 1 then n else fib (n - 1) + fib (n - 2)
and fib = memoize fib_raw

(* Continuations (CPS) — evitar stack overflow *)
let rec map_cps f xs k =
  match xs with
  | []      -> k []
  | x :: xs -> map_cps f xs (fun ys -> k (f x :: ys))

(* Uso: map_cps (fun x -> x * 2) [1;2;3] (fun r -> Printf.printf "%d\n" (List.length r)) *)

(* Funções de Array.* — mais eficientes para dados grandes *)
let arr = [|1; 2; 3; 4; 5|]   (* [| |] para arrays *)
let dobrados_arr = Array.map (fun x -> x * 2) arr
let total_arr = Array.fold_left (+) 0 arr
let () = arr.(0) <- 99   (* arrays são mutáveis! *)

Concorrência com Lwt

Lwt (Lightweight Threads) é a biblioteca de I/O assíncrono mais popular em OCaml. Usa promises (tipo 'a Lwt.t) em vez de callbacks, com operadores monádicos para composição. Em OCaml 5, effect handlers oferecem uma alternativa mais nativa.

lwt_exemplo.ml — Concorrência cooperativa
(* opam install lwt lwt_ppx *)
(* No dune: (libraries lwt lwt.unix) + (preprocess (pps lwt_ppx)) *)

open Lwt
open Lwt.Syntax   (* let*, let+, and* *)

(* Lwt.t — promessa de valor futuro *)
let esperar_e_retornar n msg : string Lwt.t =
  let* () = Lwt_unix.sleep (float_of_int n) in   (* não bloqueia thread *)
  Lwt.return msg

(* Executar duas promessas em paralelo *)
let paralelo () =
  let* (a, b) = Lwt.both
    (esperar_e_retornar 1 "primeiro")
    (esperar_e_retornar 2 "segundo")
  in
  Printf.printf "%s e %s\n" a b;
  Lwt.return ()

(* Tratar erros com Lwt.catch *)
let seguro p =
  Lwt.catch
    (fun () -> p)
    (fun exn ->
      Printf.printf "Erro: %s\n" (Printexc.to_string exn);
      Lwt.return ())

(* Ler arquivo de forma assíncrona *)
let ler_arquivo caminho =
  let* ic = Lwt_io.open_file ~mode:Input caminho in
  let* conteudo = Lwt_io.read ic in
  let* () = Lwt_io.close ic in
  Lwt.return conteudo

(* Servidor TCP simples com Lwt *)
let servidor port =
  let socket = Lwt_unix.(socket PF_INET SOCK_STREAM 0) in
  Lwt_unix.setsockopt socket SO_REUSEADDR true;
  Lwt_unix.bind socket
    (ADDR_INET (Unix.inet_addr_loopback, port)) |> ignore;
  Lwt_unix.listen socket 10;
  Printf.printf "Servidor em 127.0.0.1:%d\n%!" port;
  let rec loop () =
    let* (client_sock, _addr) = Lwt_unix.accept socket in
    let _conn = Lwt.async (fun () ->
      let* () = Lwt_unix.close client_sock in
      Lwt.return ()
    ) in
    loop ()
  in
  loop ()

(* Executar o loop de eventos *)
let () = Lwt_main.run (paralelo ())

GADTs — Tipos Algébricos Generalizados

GADTs (Generalized Algebraic Data Types) permitem que cada construtor de um tipo refine o parâmetro de tipo. Isso possibilita representar invariantes complexos diretamente no sistema de tipos, eliminando estados impossíveis em tempo de compilação.

gadts.ml — Tipos que codificam invariantes
(* Exemplo clássico: linguagem com tipos *)
type _ expr =
  | Int   : int                      -> int expr
  | Bool  : bool                     -> bool expr
  | Add   : int expr * int expr      -> int expr
  | If    : bool expr * 'a expr * 'a expr -> 'a expr
  | Eq    : int expr * int expr      -> bool expr

(* eval é totalmente type-safe: retorna o tipo correto *)
let rec eval : type a. a expr -> a = function
  | Int n          -> n
  | Bool b         -> b
  | Add (e1, e2)   -> eval e1 + eval e2
  | If (c, t, f)   -> if eval c then eval t else eval f
  | Eq (e1, e2)    -> eval e1 = eval e2

(* Uso — o compilador rejeita expressões mal-tipadas *)
let prog = If (Eq (Int 5, Int 5), Int 42, Int 0)
let r : int = eval prog   (* 42 *)

(* Add (Int 1, Bool true) — ERRO de compilação! *)
(* Bool na posição que exige int — capturado pelo tipo *)

(* Exemplo 2: listas com tamanho no tipo *)
type _ lista_sz =
  | Nil  : z lista_sz
  | Cons : 'a * 'n lista_sz -> (s * 'n) lista_sz
and z = Z
and 'n s = S

(* head só aceita lista não-vazia — verificado em compilação *)
let head (Cons (x, _) : (_ s * _) lista_sz) = x

Effect Handlers — OCaml 5

OCaml 5 introduziu effect handlers — uma forma de controlar efeitos colaterais (I/O, estado, exceções, corrotinas) de maneira composável. São mais gerais que exceções e mais leves que monads.

effects.ml — OCaml 5 Effect Handlers (experimental)
(* Requer: OCaml >= 5.0 *)
open Effect
open Effect.Deep

(* Declarar um efeito *)
type _ Effect.t += Logging : string -> unit Effect.t

(* Função que usa o efeito *)
let programa () =
  perform (Logging "iniciando...");
  let x = 21 + 21 in
  perform (Logging (Printf.sprintf "resultado: %d" x));
  x

(* Handler: intercepta o efeito Logging e imprime *)
let com_log f =
  match_with f ()
  { retc = (fun x -> x);
    exnc = raise;
    effc = fun (type a) (eff : a t) ->
      match eff with
      | Logging msg ->
          Some (fun (k : (a, _) continuation) ->
            Printf.printf "[LOG] %s\n" msg;
            continue k ()
          )
      | _ -> None
  }

(* Handler alternativo: silencioso *)
let silencioso f =
  match_with f ()
  { retc = (fun x -> x);
    exnc = raise;
    effc = fun (type a) (eff : a t) ->
      match eff with
      | Logging _ -> Some (fun k -> continue k ())
      | _          -> None
  }

let () =
  Printf.printf "Resultado: %d\n" (com_log programa)
  (* [LOG] iniciando...
     [LOG] resultado: 42
     Resultado: 42 *)
