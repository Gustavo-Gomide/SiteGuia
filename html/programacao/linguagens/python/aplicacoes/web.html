FastAPI: API REST Assíncrona Completa

Python — FastAPI: modelos Pydantic, rotas, dependency injection, JWT
from fastapi import FastAPI, Depends, HTTPException, status
from fastapi.security import OAuth2PasswordBearer, OAuth2PasswordRequestForm
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel, EmailStr, Field
from datetime import datetime, timedelta
from typing import Annotated
import jwt, bcrypt

app = FastAPI(title="Loja API", version="1.0.0")

app.add_middleware(
    CORSMiddleware,
    allow_origins=["https://meusite.com"],
    allow_methods=["*"], allow_headers=["*"],
)

SECRET_KEY = "sua-chave-secreta"     # use os.environ em produção
oauth2_scheme = OAuth2PasswordBearer(tokenUrl="token")

# --- Modelos ---
class Produto(BaseModel):
    id:    int
    nome:  str
    preco: float = Field(gt=0)

class ProdutoCreate(BaseModel):
    nome:  str = Field(min_length=1, max_length=200)
    preco: float = Field(gt=0)

class Token(BaseModel):
    access_token: str
    token_type:   str

# --- Dependency injection ---
async def get_usuario_atual(token: Annotated[str, Depends(oauth2_scheme)]) -> str:
    try:
        payload = jwt.decode(token, SECRET_KEY, algorithms=["HS256"])
        sub: str = payload["sub"]
    except jwt.PyJWTError:
        raise HTTPException(status_code=401, detail="Token inválido")
    return sub

# --- Rotas ---
@app.post("/token", response_model=Token)
async def login(form: Annotated[OAuth2PasswordRequestForm, Depends()]):
    # Em produção: verificar hash no banco
    if form.username != "admin" or form.password != "secret":
        raise HTTPException(401, "Credenciais inválidas")
    token = jwt.encode(
        {"sub": form.username, "exp": datetime.utcnow() + timedelta(hours=1)},
        SECRET_KEY, algorithm="HS256"
    )
    return Token(access_token=token, token_type="bearer")

# banco em memória para demonstração
_db: dict[int, ProdutoCreate] = {}
_seq = iter(range(1, 10_000))

@app.get("/produtos", response_model=list[Produto])
async def listar_produtos():
    return [Produto(id=k, **v.model_dump()) for k, v in _db.items()]

@app.post("/produtos", response_model=Produto, status_code=201)
async def criar_produto(
    body: ProdutoCreate,
    _user: Annotated[str, Depends(get_usuario_atual)],
):
    pid = next(_seq)
    _db[pid] = body
    return Produto(id=pid, **body.model_dump())

@app.delete("/produtos/{pid}", status_code=204)
async def deletar_produto(pid: int, _user: Annotated[str, Depends(get_usuario_atual)]):
    if pid not in _db:
        raise HTTPException(404, "Produto não encontrado")
    del _db[pid]

FastAPI com SQLAlchemy Assíncrono

Python — FastAPI + SQLAlchemy async + asyncpg + Alembic
from sqlalchemy.ext.asyncio import AsyncSession, create_async_engine, async_sessionmaker
from sqlalchemy.orm import DeclarativeBase, Mapped, mapped_column
from fastapi import Depends

DATABASE_URL = "postgresql+asyncpg://user:pass@localhost/appdb"

engine = create_async_engine(DATABASE_URL, pool_size=20, max_overflow=0)
AsyncSessionLocal = async_sessionmaker(engine, expire_on_commit=False)

class Base(DeclarativeBase): pass

class Pedido(Base):
    __tablename__ = "pedidos"
    id:     Mapped[int] = mapped_column(primary_key=True)
    valor:  Mapped[float]
    status: Mapped[str] = mapped_column(default="pendente")

async def get_db() -> AsyncSession:
    async with AsyncSessionLocal() as session:
        yield session

@app.get("/pedidos/{pid}")
async def get_pedido(pid: int, db: AsyncSession = Depends(get_db)):
    pedido = await db.get(Pedido, pid)
    if not pedido:
        raise HTTPException(404)
    return pedido

Django: Framework Full-Stack

Python — Django: models, views class-based, serializers DRF
# models.py
from django.db import models
from django.contrib.auth.models import AbstractUser


class UsuarioCustom(AbstractUser):
    bio       = models.TextField(blank=True)
    avatar    = models.ImageField(upload_to="avatars/", null=True)


class Artigo(models.Model):
    titulo   = models.CharField(max_length=200)
    conteudo = models.TextField()
    autor    = models.ForeignKey(UsuarioCustom, on_delete=models.CASCADE, related_name="artigos")
    criado   = models.DateTimeField(auto_now_add=True)
    publicado = models.BooleanField(default=False)

    class Meta:
        ordering = ["-criado"]
        indexes  = [models.Index(fields=["autor", "-criado"])]


# serializers.py (Django REST Framework)
from rest_framework import serializers

class ArtigoSerializer(serializers.ModelSerializer):
    autor_nome = serializers.CharField(source="autor.get_full_name", read_only=True)

    class Meta:
        model  = Artigo
        fields = ["id", "titulo", "autor_nome", "criado", "publicado"]


# views.py (class-based)
from rest_framework import viewsets, permissions

class ArtigoViewSet(viewsets.ModelViewSet):
    serializer_class   = ArtigoSerializer
    permission_classes = [permissions.IsAuthenticatedOrReadOnly]

    def get_queryset(self):
        qs = Artigo.objects.select_related("autor")
        if publicado := self.request.query_params.get("publicado"):
            qs = qs.filter(publicado=publicado.lower() == "true")
        return qs

    def perform_create(self, serializer):
        serializer.save(autor=self.request.user)

Deploy com Gunicorn, Uvicorn e Docker

Docker — Dockerfile multi-stage para FastAPI
FROM python:3.12-slim AS builder
WORKDIR /app
RUN pip install uv
COPY pyproject.toml .
RUN uv pip install --system -e ".[prod]"

FROM python:3.12-slim
WORKDIR /app
COPY --from=builder /usr/local/lib/python3.12/site-packages /usr/local/lib/python3.12/site-packages
COPY src/ ./src/

ENV PYTHONUNBUFFERED=1 \
    PYTHONDONTWRITEBYTECODE=1 \
    PORT=8000

EXPOSE 8000
CMD ["uvicorn", "src.meuapp.api.main:app",
     "--host", "0.0.0.0",
     "--port", "8000",
     "--workers", "4",
     "--access-log"]
