Pandas: DataFrames e Análise de Dados

Python — Pandas: ETL, groupby, merge, window functions
import pandas as pd
import numpy as np

# Leitura otimizada com tipos explícitos
df = pd.read_csv("vendas.csv", parse_dates=["data"],
                  dtype={"produto_id": "Int32", "quantidade": "Int16"})

# Limpeza e transformação
df = df.dropna(subset=["valor"])
df["valor"] = pd.to_numeric(df["valor"], errors="coerce")
df["mes"] = df["data"].dt.to_period("M")
df["desconto"] = np.where(df["quantidade"] > 10, 0.05, 0.0)

# Aggregation + groupby
resumo = (
    df.groupby(["mes", "categoria"])
    .agg(total_venda=("valor", "sum"),
         qtd_pedidos=("pedido_id", "nunique"),
         ticket_medio=("valor", "mean"))
    .reset_index()
    .sort_values("total_venda", ascending=False)
)

# Merge (JOIN)
produtos = pd.read_csv("produtos.csv")
df_full = df.merge(produtos, left_on="produto_id", right_on="id", how="left")

# Window function: média móvel 7 dias
df_full = df_full.sort_values("data")
df_full["media_movel"] = (
    df_full["valor"].rolling(window=7, min_periods=1).mean()
)

# Exportar
resumo.to_parquet("resumo.parquet", index=False)

Scikit-learn: Pipeline de Machine Learning

Python — sklearn: Pipeline, GridSearchCV, cross-validation
from sklearn.pipeline import Pipeline
from sklearn.compose import ColumnTransformer
from sklearn.preprocessing import StandardScaler, OneHotEncoder
from sklearn.ensemble import GradientBoostingClassifier
from sklearn.model_selection import GridSearchCV, cross_val_score, StratifiedKFold
from sklearn.metrics import classification_report
import numpy as np
import joblib

# Pre-processamento por tipo de feature
num_features = ["idade", "salario", "historico"]
cat_features = ["regiao", "categoria"]

preprocessor = ColumnTransformer([
    ("num", StandardScaler(), num_features),
    ("cat", OneHotEncoder(handle_unknown="ignore"), cat_features),
])

pipeline = Pipeline([
    ("pre", preprocessor),
    ("clf", GradientBoostingClassifier(random_state=42)),
])

# Busca de hiperparâmetros
param_grid = {
    "clf__n_estimators": [100, 200],
    "clf__learning_rate": [0.05, 0.1],
    "clf__max_depth": [3, 5],
}

cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)
grid = GridSearchCV(pipeline, param_grid, cv=cv, scoring="f1_macro", n_jobs=-1)
grid.fit(X_train, y_train)

print("Melhor score:", grid.best_score_)
print(classification_report(y_test, grid.predict(X_test)))

# Persistir modelo
joblib.dump(grid.best_estimator_, "modelo_v1.pkl")

PyTorch: Redes Neurais e Autograd

Python — PyTorch: MLP manual + treinamento com DataLoader
import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.data import DataLoader, TensorDataset

# Modelo MLP para classificação
class MLP(nn.Module):
    def __init__(self, n_input: int, n_hidden: int, n_output: int) -> None:
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(n_input, n_hidden),
            nn.BatchNorm1d(n_hidden),
            nn.ReLU(),
            nn.Dropout(p=0.3),
            nn.Linear(n_hidden, n_hidden),
            nn.ReLU(),
            nn.Linear(n_hidden, n_output),
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.net(x)


device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
model  = MLP(20, 128, 2).to(device)
opt    = torch.optim.AdamW(model.parameters(), lr=1e-3, weight_decay=1e-4)
crit   = nn.CrossEntropyLoss()

# Dados sintéticos
X = torch.randn(1000, 20)
y = torch.randint(0, 2, (1000,))
loader = DataLoader(TensorDataset(X, y), batch_size=64, shuffle=True)

# Loop de treinamento
for epoch in range(20):
    model.train()
    total = 0.0
    for xb, yb in loader:
        xb, yb = xb.to(device), yb.to(device)
        opt.zero_grad()
        loss = crit(model(xb), yb)
        loss.backward()      # autograd calcula gradientes
        nn.utils.clip_grad_norm_(model.parameters(), 1.0)
        opt.step()
        total += loss.item()
    print(f"Epoch {epoch+1:02d} loss={total/len(loader):.4f}")

# Exportar para ONNX
dummy = torch.zeros((1, 20)).to(device)
torch.onnx.export(model, dummy, "modelo.onnx",
                   input_names=["input"], output_names=["output"],
                   dynamic_axes={"input": {0: "batch"}})

Hugging Face Transformers

Python — fine-tuning de classificação de texto com Trainer API
from transformers import (AutoTokenizer, AutoModelForSequenceClassification,
                             TrainingArguments, Trainer)
from datasets import load_dataset
import numpy as np
from sklearn.metrics import accuracy_score

model_name = "neuralmind/bert-base-portuguese-cased"
tokenizer  = AutoTokenizer.from_pretrained(model_name)
model      = AutoModelForSequenceClassification.from_pretrained(model_name, num_labels=2)

dataset = load_dataset("csv", data_files={"train": "train.csv", "test": "test.csv"})

def tokenize(batch):
    return tokenizer.__call__(batch["texto"], truncation=True, max_length=128,
                               padding="max_length")

tok_dataset = dataset.map(tokenize, batched=True)

args = TrainingArguments(
    output_dir="./resultado",
    num_train_epochs=3, per_device_train_batch_size=16,
    per_device_eval_batch_size=32, eval_strategy="epoch",
    save_strategy="epoch", load_best_model_at_end=True,
    fp16=True,
)

def metricas(p):
    preds = np.argmax(p.predictions, axis=1)
    return {"accuracy": accuracy_score(p.label_ids, preds)}

trainer = Trainer(model, args, train_dataset=tok_dataset["train"],
                  eval_dataset=tok_dataset["test"], compute_metrics=metricas)
trainer.train()
