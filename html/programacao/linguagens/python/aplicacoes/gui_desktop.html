tkinter: GUI Nativa Embutida

Python — tkinter: app de notas com MVC e ttk widgets
import tkinter as tk
from tkinter import ttk, messagebox, filedialog
from pathlib import Path


class NotasApp:
    def __init__(self, root: tk.Tk) -> None:
        self.root = root
        root.title("Notas")
        root.geometry("800x600")
        root.minsize(400, 300)

        # Variáveis de estado (reactive com trace)
        self.titulo_var    = tk.StringVar(value="Sem título")
        self.modificado    = tk.BooleanVar(value=False)
        self.modificado.trace_add("write", self._atualizar_titulo)

        self._criar_menu()
        self._criar_widgets()

    def _criar_menu(self) -> None:
        mb = tk.Menu(self.root)
        arq = tk.Menu(mb, tearoff=0)
        arq.add_command(label="Abrir...", accelerator="Ctrl+O", command=self._abrir)
        arq.add_command(label="Salvar",  accelerator="Ctrl+S", command=self._salvar)
        arq.add_separator()
        arq.add_command(label="Sair", command=self.root.quit)
        mb.add_cascade(label="Arquivo", menu=arq)
        self.root.config(menu=mb)
        self.root.bind("<Control-s>", lambda e: self._salvar())

    def _criar_widgets(self) -> None:
        # Frame principal com peso responsivo
        frame = ttk.Frame(self.root, padding=8)
        frame.pack(fill=tk.BOTH, expand=True)
        frame.columnconfigure(0, weight=1)
        frame.rowconfigure(1, weight=1)

        # Campo de título
        ttk.Entry(frame, textvariable=self.titulo_var, font=("sans", 16)).grid(
            row=0, column=0, sticky="ew", pady=(0, 4)
        )
        # Área de texto com scrollbar
        self.texto = tk.Text(frame, font=("monospace", 12), wrap=tk.WORD)
        self.texto.grid(row=1, column=0, sticky="nsew")
        scroll = ttk.Scrollbar(frame, command=self.texto.yview)
        scroll.grid(row=1, column=1, sticky="ns")
        self.texto.configure(yscrollcommand=scroll.set)
        self.texto.bind("<<Modified>>", lambda e: self.modificado.set(True))

    def _abrir(self) -> None:
        path = filedialog.askopenfilename(filetypes=[("Texto", "*.txt")])
        if path:
            self.texto.delete("1.0", tk.END)
            self.texto.insert(tk.END, Path(path).read_text())

    def _salvar(self) -> None:
        path = filedialog.asksaveasfilename(defaultextension=".txt")
        if path:
            Path(path).write_text(self.texto.get("1.0", tk.END))
            messagebox.showinfo("Salvo", f"Arquivo salvo em {path}")

    def _atualizar_titulo(self, *_) -> None:
        asterisco = "* " if self.modificado.get() else ""
        self.root.title(f"{asterisco}{self.titulo_var.get()} — Notas")


root = tk.Tk()
app  = NotasApp(root)
root.mainloop()

PyQt6/PySide6: Qt com Signals e Slots

Python — PyQt6: app de contatos com QTableView e Model/View
from PyQt6.QtWidgets import (QApplication, QMainWindow, QTableView,
                               QVBoxLayout, QWidget, QPushButton, QHBoxLayout,
                               QLineEdit, QLabel, QFormLayout, QDialog, QDialogButtonBox)
from PyQt6.QtCore import Qt, QAbstractTableModel, QModelIndex
from dataclasses import dataclass
import sys


@dataclass
class Contato:
    nome: str
    email: str
    fone: str


class ContatosModel(QAbstractTableModel):
    HEADERS = ["Nome", "E-mail", "Telefone"]

    def __init__(self, dados: list[Contato]) -> None:
        super().__init__()
        self._dados = dados

    def rowCount(self, parent=QModelIndex()) -> int:
        return len(self._dados)

    def columnCount(self, parent=QModelIndex()) -> int:
        return 3

    def data(self, index: QModelIndex, role=Qt.ItemDataRole.DisplayRole):
        if not index.isValid(): return None
        contato = self._dados[index.row()]
        if role == Qt.ItemDataRole.DisplayRole:
            return [contato.nome, contato.email, contato.fone][index.column()]

    def headerData(self, section, orientation, role=Qt.ItemDataRole.DisplayRole):
        if orientation == Qt.Orientation.Horizontal and role == Qt.ItemDataRole.DisplayRole:
            return self.HEADERS[section]

    def adicionar(self, c: Contato) -> None:
        row = len(self._dados)
        self.beginInsertRows(QModelIndex(), row, row)
        self._dados.append(c)
        self.endInsertRows()


class MainWindow(QMainWindow):
    def __init__(self) -> None:
        super().__init__()
        self.setWindowTitle("Contatos")
        self.model = ContatosModel([
            Contato("Ana", "ana@ex.com", "(11) 9000-0001"),
        ])
        table = QTableView()
        table.setModel(self.model)
        table.horizontalHeader().setStretchLastSection(True)
        btn = QPushButton("Novo contato")
        btn.clicked.connect(self._novo_contato)  # signal → slot
        w = QWidget()
        lay = QVBoxLayout(w)
        lay.addWidget(table)
        lay.addWidget(btn)
        self.setCentralWidget(w)

    def _novo_contato(self) -> None:
        dlg = QDialog(self)
        form = QFormLayout()
        nome  = QLineEdit(); email = QLineEdit(); fone = QLineEdit()
        form.addRow("Nome:",  nome)
        form.addRow("E-mail:", email)
        form.addRow("Fone:",  fone)
        btns = QDialogButtonBox(QDialogButtonBox.StandardButton.Ok | QDialogButtonBox.StandardButton.Cancel)
        btns.accepted.connect(dlg.accept)
        btns.rejected.connect(dlg.reject)
        lay = QVBoxLayout(); lay.addLayout(form); lay.addWidget(btns)
        dlg.setLayout(lay)
        if dlg.exec():
            self.model.adicionar(Contato(nome.text(), email.text(), fone.text()))

app = QApplication(sys.argv)
win = MainWindow()
win.show()
sys.exit(app.exec())

Dear PyGui: Dashboards Immediate Mode

Python — Dear PyGui: janela, gráfico de linha ao vivo, slider
import dearpygui.dearpygui as dpg
import math, time

dpg.create_context()

dados_x:  list[float] = []
dados_y:  list[float] = []
t_inicio = time.time()

def atualizar():
    t = time.time() - t_inicio
    freq = dpg.get_value("slider_freq")
    dados_x.append(t)
    dados_y.append(math.sin(2 * math.pi * freq * t))
    if len(dados_x) > 300:
        dados_x.pop(0); dados_y.pop(0)
    dpg.set_value("serie", [dados_x, dados_y])

with dpg.window(label="Osciloscopio", width=800, height=500):
    dpg.add_slider_float(tag="slider_freq", label="Frequência (Hz)",
                         default_value=1.0, min_value=0.1, max_value=5.0)
    with dpg.plot(height=350, width=-1):
        dpg.add_plot_axis(dpg.mvXAxis, label="t (s)")
        with dpg.plot_axis(dpg.mvYAxis, label="Amplitude"):
            dpg.add_line_series([], [], label="sin", tag="serie")

dpg.create_viewport(title="Dear PyGui Demo")
dpg.setup_dearpygui()
dpg.show_viewport()

while dpg.is_dearpygui_running():
    atualizar()
    dpg.render_dearpygui_frame()

dpg.destroy_context()
