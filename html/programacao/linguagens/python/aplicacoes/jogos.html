Pygame 2: Game Loop, Sprites e Física

Python — Pygame: jogo Pong completo com delta time e colisão
import pygame
from dataclasses import dataclass, field
import sys

pygame.init()

W, H  = 800, 600
FPS   = 60
WHITE = (255, 255, 255)
BLACK = (0, 0, 0)
GRAY  = (80, 80, 80)

screen = pygame.display.set_mode((W, H))
pygame.display.set_caption("Pong")
clock  = pygame.time.Clock()
font   = pygame.font.SysFont("monospace", 36)


@dataclass
class Raquete:
    x: float
    y: float
    w: int = 15
    h: int = 90
    vel: float = 400.0

    @property
    def rect(self) -> pygame.Rect:
        return pygame.Rect(self.x, self.y, self.w, self.h)

    def mover(self, dy: float, dt: float) -> None:
        self.y += dy * self.vel * dt
        self.y  = max(0, min(H - self.h, self.y))


@dataclass
class Bola:
    x: float = W / 2
    y: float = H / 2
    r: int   = 10
    vx: float = 350.0
    vy: float = 250.0

    def update(self, dt: float, r1: Raquete, r2: Raquete) -> bool:
        # bool: True = ponto marcado
        self.x += self.vx * dt
        self.y += self.vy * dt
        if self.y - self.r < 0 or self.y + self.r > H:
            self.vy = -self.vy
        # colisão AABB
        bola_rect = pygame.Rect(self.x - self.r, self.y - self.r, self.r*2, self.r*2)
        if bola_rect.colliderect(r1.rect) or bola_rect.colliderect(r2.rect):
            self.vx = -self.vx * 1.05
        if self.x < 0 or self.x > W:
            self.x, self.y = W/2, H/2
            self.vx, self.vy = 350.0, 250.0
            return True
        return False


r1 = Raquete(30, H/2 - 45)
r2 = Raquete(W - 45, H/2 - 45)
bola = Bola()
placar = [0, 0]

while True:
    dt = clock.tick(FPS) / 1000.0    # delta time em segundos
    for ev in pygame.event.get():
        if ev.type == pygame.QUIT:
            pygame.quit(); sys.exit()

    keys = pygame.key.get_pressed()
    r1.mover(keys[pygame.K_s] - keys[pygame.K_w], dt)
    r2.mover(keys[pygame.K_DOWN] - keys[pygame.K_UP], dt)

    if bola.update(dt, r1, r2):
        placar[0 if bola.x < W/2 else 1] += 1

    screen.fill(BLACK)
    pygame.draw.rect(screen, WHITE, r1.rect)
    pygame.draw.rect(screen, WHITE, r2.rect)
    pygame.draw.circle(screen, WHITE, (int(bola.x), int(bola.y)), bola.r)
    pygame.draw.aaline(screen, GRAY, (W//2, 0), (W//2, H))
    txt = font.render(f"{placar[0]}  {placar[1]}", True, WHITE)
    screen.blit(txt, (W//2 - txt.get_width()//2, 10))
    pygame.display.flip()

Pygame: Sprite Groups e Animação

Python — Pygame: Sprite, Group, animação por spritesheet
import pygame

class Personagem(pygame.sprite.Sprite):
    FRAMES       = 4
    FRAME_WIDTH  = 64
    FRAME_HEIGHT = 64
    FPS_ANIM     = 8     # quadros de animação por segundo

    def __init__(self, sheet_path: str, x: int, y: int) -> None:
        super().__init__()
        sheet = pygame.image.load(sheet_path).convert_alpha()
        self.frames = [
            sheet.subsurface(pygame.Rect(i * self.FRAME_WIDTH, 0,
                                           self.FRAME_WIDTH, self.FRAME_HEIGHT))
            for i in range(self.FRAMES)
        ]
        self._frame_atual = 0
        self._tempo_anim  = 0.0
        self.image  = self.frames[0]
        self.rect   = self.image.get_rect(topleft=(x, y))
        self.vel_x  = 200.0

    def update(self, dt: float) -> None:
        # movimento
        keys = pygame.key.get_pressed()
        if keys[pygame.K_RIGHT]: self.rect.x += int(self.vel_x * dt)
        if keys[pygame.K_LEFT]:  self.rect.x -= int(self.vel_x * dt)
        # animação
        self._tempo_anim += dt
        if self._tempo_anim >= 1 / self.FPS_ANIM:
            self._tempo_anim = 0
            self._frame_atual = (self._frame_atual + 1) % self.FRAMES
            self.image = self.frames[self._frame_atual]

Arcade: Biblioteca Moderna com OpenGL

Python — arcade: Window, PhysicsEnginePlatformer, Tilemap
import arcade

TILE_SIZE    = 64
SCREEN_W, SCREEN_H = 1280, 720
GRAVIDADE    = 1500
VEL_PULO     = 700
VEL_ANDAR    = 250


class JogoPlatform(arcade.Window):
    def __init__(self) -> None:
        super().__init__(SCREEN_W, SCREEN_H, "Platform Demo")
        self.jogador : arcade.Sprite                 = None
        self.plataformas: arcade.SpriteList           = None
        self.fisica: arcade.PhysicsEnginePlatformer  = None
        self.camera_jogador: arcade.Camera           = None

    def setup(self) -> None:
        self.jogador = arcade.Sprite(":resources:images/animated_characters/female_adventurer/femaleAdventurer_idle.png")
        self.jogador.center_x = 128
        self.jogador.center_y = 200

        self.plataformas = arcade.SpriteList(use_spatial_hash=True)
        for x in range(0, 4096, TILE_SIZE):
            tile = arcade.SpriteSolidColor(TILE_SIZE, TILE_SIZE, arcade.color.BROWN)
            tile.center_x = x + TILE_SIZE // 2
            tile.center_y = TILE_SIZE // 2
            self.plataformas.append(tile)

        self.fisica = arcade.PhysicsEnginePlatformer(
            self.jogador, gravity_constant=GRAVIDADE, walls=self.plataformas
        )
        self.camera_jogador = arcade.Camera(SCREEN_W, SCREEN_H)

    def on_draw(self) -> None:
        self.clear()
        self.camera_jogador.use()
        self.plataformas.draw()
        self.jogador.draw()

    def on_update(self, dt: float) -> None:
        self.fisica.update()
        self.camera_jogador.move_to(
            (self.jogador.center_x - SCREEN_W / 2, 0), speed=0.15
        )

    def on_key_press(self, key, mod) -> None:
        if key == arcade.key.RIGHT: self.jogador.change_x =  VEL_ANDAR
        if key == arcade.key.LEFT:  self.jogador.change_x = -VEL_ANDAR
        if key == arcade.key.UP and self.fisica.can_jump():
            self.jogador.change_y = VEL_PULO

    def on_key_release(self, key, mod) -> None:
        if key in (arcade.key.RIGHT, arcade.key.LEFT):
            self.jogador.change_x = 0


if __name__ == "__main__":
    jogo = JogoPlatform()
    jogo.setup()
    arcade.run()
