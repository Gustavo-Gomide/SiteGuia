Kivy: Apps Cross-Platform com OpenGL ES2

Python + Kivy — App completo com KV language e navegação entre telas
# app.py — instale: pip install kivy
from kivy.app import App
from kivy.uix.screenmanager import ScreenManager, Screen, SlideTransition
from kivy.uix.boxlayout import BoxLayout
from kivy.properties import StringProperty, NumericProperty, ObjectProperty
from kivy.lang import Builder

KV = """
<MenuScreen>:
    name: 'menu'
    BoxLayout:
        orientation: 'vertical'
        padding: 24
        spacing: 12
        Label:
            text: 'Contador Kivy'
            font_size: '28sp'
            bold: True

        Label:
            text: app.mensagem   # reativo: atualiza automaticamente
            id: lbl_msg

        Button:
            text: 'Abrir Contador'
            size_hint_y: None
            height: '48dp'
            on_press: root.ir_para_contador()

<ContadorScreen>:
    name: 'contador'
    btn_menos: btn_menos_id
    BoxLayout:
        orientation: 'vertical'
        padding: 24
        spacing: 8
        Label:
            text: str(root.contagem)
            font_size: '64sp'
        BoxLayout:
            size_hint_y: None
            height: '56dp'
            Button:
                id: btn_menos_id
                text: '-'
                on_press: root.decrementar()
            Button:
                text: '+'
                on_press: root.incrementar()
        Button:
            text: 'Voltar'
            size_hint_y: None
            height: '48dp'
            on_press: app.root.current = 'menu'
"""

Builder.load_string(KV)


class MenuScreen(Screen):
    def ir_para_contador(self):
        self.manager.transition = SlideTransition(direction="left")
        self.manager.current = "contador"


class ContadorScreen(Screen):
    contagem = NumericProperty(0)
    btn_menos = ObjectProperty(None)

    def incrementar(self):
        self.contagem += 1
        self.btn_menos.disabled = False

    def decrementar(self):
        if self.contagem > 0:
            self.contagem -= 1
        self.btn_menos.disabled = (self.contagem == 0)


class MeuApp(App):
    mensagem = StringProperty("Bem-vindo!")

    def build(self):
        sm = ScreenManager()
        sm.add_widget(MenuScreen())
        sm.add_widget(ContadorScreen())
        return sm


if __name__ == "__main__":
    MeuApp().run()

BeeWare / Toga: Widgets Nativos

Python + Toga (BeeWare) — App nativo com briefcase
# src/meuprojeto/app.py
import toga
from toga.style import Pack
from toga.style.pack import COLUMN, ROW


class MeuApp(toga.App):
    def startup(self):
        # Caixa principal
        box = toga.Box(style=Pack(direction=COLUMN, padding=20))

        self._entrada = toga.TextInput(
            placeholder="Digite seu nome",
            style=Pack(flex=1)
        )
        self._resultado = toga.Label(
            text="", style=Pack(font_size=18)
        )

        linha = toga.Box(style=Pack(direction=ROW, padding_bottom=10))
        linha.add(self._entrada)
        linha.add(
            toga.Button("Saudar", on_press=self._saudar,
                         style=Pack(padding_left=8))
        )

        box.add(linha)
        box.add(self._resultado)

        self.main_window = toga.MainWindow(title=self.formal_name)
        self.main_window.content = box
        self.main_window.show()

    def _saudar(self, widget):
        nome = self._entrada.value.strip() or "Mundo"
        self._resultado.text = f"Olá, {nome}!"


def main():
    return MeuApp(
        formal_name="Meu App",
        app_id="br.com.exemplo.meuapp",
    )
Terminal — Briefcase: criar, testar e publicar Android
# Instalar ferramentas
pip install briefcase

# Criar projeto novo (interativo)
briefcase new

# Executar no desktop (dev mode, lento para debugar)
briefcase dev

# Criar artefatos Android (requer JDK 17+ e Android SDK)
briefcase create android
briefcase build android

# Rodar direto no emulador ou dispositivo USB
briefcase run android

# Gerar APK de release assinado
briefcase build android --clean --release

Buildozer: APK Android com Kivy

buildozer.spec — Configuração completa para Kivy Android APK
[app]
title         = Meu App Kivy
package.name  = meuappkivy
package.domain= br.com.exemplo
source.dir    = .
source.include_exts = py,png,jpg,kv,atlas,json

# Versão do SDK Android
android.api        = 33
android.minapi     = 21
android.ndk        = 25b
android.archs      = arm64-v8a, armeabi-v7a

# Permissões necessárias
android.permissions = INTERNET, CAMERA, ACCESS_FINE_LOCATION

# Requirements: nome exato do PyPI + versão
requirements = python3,kivy==2.3.0,paho-mqtt==1.6.1,requests

# Orientação padrão
orientation = portrait

# Ícone e splash
icon.filename     = %(source.dir)s/imagens/icon.png
presplash.filename= %(source.dir)s/imagens/splash.png

[buildozer]
log_level = 2
Terminal — Buildozer: compilar APK debug e release
# Inicializar buildozer.spec (uma vez por projeto)
buildozer init

# Build APK debug (primeiro build baixa toolchain: > 1 GB, > 10 min)
buildozer -v android debug

# Instalar direto no dispositivo USB com ADB
buildozer -v android debug deploy run

# APK release (assinar com keystore)
export P4A_RELEASE_KEYSTORE=~/.android/minha-chave.keystore
export P4A_RELEASE_KEYSTORE_PASSWD=senha
export P4A_RELEASE_KEYALIAS=meuapp
buildozer android release
