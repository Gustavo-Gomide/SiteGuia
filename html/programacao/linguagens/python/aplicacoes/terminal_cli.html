argparse: Biblioteca Nativa

Python — argparse: subcomandos, tipos, metavar e grupos
import argparse
import sys


def cmd_processar(args: argparse.Namespace) -> None:
    print(f"Processando {args.arquivo} com workers={args.workers}")

def cmd_limpar(args: argparse.Namespace) -> None:
    print(f"Limpando diretório: {args.diretorio}")


def criar_parser() -> argparse.ArgumentParser:
    parser = argparse.ArgumentParser(
        prog="meutool", description="Ferramenta de processamento de dados",
        formatter_class=argparse.RawDescriptionHelpFormatter,
    )
    parser.add_argument("--versao", action="version", version="%(prog)s 1.0.0")
    parser.add_argument("-v", "--verbose", action="count", default=0,
                        help="Aumentar verbosidade (-v, -vv, -vvv)")

    sub = parser.add_subparsers(dest="comando", required=True)

    # subcomando: processar
    p1 = sub.add_parser("processar", help="Processa um arquivo")
    p1.add_argument("arquivo", type=argparse.FileType("r"))
    p1.add_argument("-w", "--workers", type=int, default=4,
                    choices=[1, 2, 4, 8], metavar="N")
    p1.set_defaults(func=cmd_processar)

    # subcomando: limpar
    p2 = sub.add_parser("limpar", help="Limpa arquivos temporários")
    p2.add_argument("diretorio", nargs="?", default="/tmp")
    p2.set_defaults(func=cmd_limpar)

    return parser


if __name__ == "__main__":
    parser = criar_parser()
    args   = parser.parse_args()
    args.func(args)

Click: CLI Elegante com Decorators

Python — Click: groups, options, callbacks, progress
import click
from pathlib import Path


@click.group()
@click.version_option("1.0.0")
def cli(): """Ferramenta de sincronização de arquivos."""


@cli.command()
@click.argument("origem", type=click.Path(exists=True, path_type=Path))
@click.argument("destino", type=click.Path(path_type=Path))
@click.option("--dry-run", is_flag=True, help="Simula sem copiar nada")
@click.option("--pattern", default="*", show_default=True)
def sync(origem: Path, destino: Path, dry_run: bool, pattern: str):
    """Sincroniza ORIGEM → DESTINO."""
    arquivos = list(origem.glob(pattern))
    with click.progressbar(arquivos, label="Sincronizando") as bar:
        for f in bar:
            dest = destino / f.relative_to(origem)
            if dry_run:
                click.echo(click.style(f"[DRY] {f} → {dest}", fg="yellow"))
            else:
                dest.parent.mkdir(parents=True, exist_ok=True)
                import shutil
                shutil.copy2(f, dest)
    click.secho(f"\n{len(arquivos)} arquivo(s) sincronizado(s).", fg="green")


@cli.command()
@click.option("--formato", type=click.Choice(["json", "csv", "yaml"]))
def relatorio(formato: str):
    """Gera relatório de estatísticas."""
    click.echo(f"Gerando relatório em formato {formato}")


if __name__ == "__main__":
    cli()

Rich: Output Profissional no Terminal

Python — Rich: tabelas, syntax highlight, live e logging
from rich import print as rprint
from rich.table import Table
from rich.console import Console
from rich.progress import Progress, SpinnerColumn, TimeElapsedColumn
from rich.syntax import Syntax
from rich.logging import RichHandler
import logging, time

console = Console()

# Logging com Rich
logging.basicConfig(level=logging.INFO, handlers=[RichHandler()])
logger = logging.getLogger()
logger.info("Aplicação iniciada")

# Tabela formatada
tabela = Table(title="Servidores", show_lines=True)
tabela.add_column("Host", style="cyan")
tabela.add_column("Status", justify="center")
tabela.add_column("Latência", justify="right")
tabela.add_row("web-1.prod", "[green]OK[/]", "12ms")
tabela.add_row("db-1.prod", "[red]ERRO[/]", "—")
tabela.add_row("cache.prod", "[green]OK[/]", "2ms")
console.print(tabela)

# Syntax highlighted code
codigo = """
def fib(n):
    return n if n < 2 else fib(n-1) + fib(n-2)
"""
console.print(Syntax(codigo, "python", theme="monokai", line_numbers=True))

# Progress bar com tasks múltiplas
with Progress(SpinnerColumn(), "[progress.description]{task.description}",
               TimeElapsedColumn()) as prog:
    t1 = prog.add_task("Download...", total=100)
    t2 = prog.add_task("Processando...", total=50)
    while not prog.finished:
        prog.update(t1, advance=2)
        prog.update(t2, advance=1)
        time.sleep(0.05)

Typer: CLI com Type Hints Puro

Python — Typer: app completo com tipos, enums e callback
import typer
from enum import Enum
from pathlib import Path
from rich import print as rprint

app = typer.Typer(name="migrador", help="Ferramenta de migração de dados")

class Formato(str, Enum):
    json = "json"
    csv  = "csv"
    sql  = "sql"

@app.command()
def exportar(
    tabela:  str            = typer.Argument(..., help="Nome da tabela"),
    saida:   Path           = typer.Option(..., "--saida", "-o", help="Arquivo de saída"),
    formato: Formato        = typer.Option(Formato.json),
    limite:  int            = typer.Option(1000, help="Máx. de linhas"),
    verbose: bool           = typer.Option(False, "--verbose/--quiet"),
):
    """Exporta uma tabela para arquivo."""
    if verbose:
        rprint(f"[cyan]Exportando {tabela} → {saida} ({formato.value})[/]")
    rprint(f"[green]✓ {limite} linhas exportadas para {saida}[/]")


@app.command()
def importar(entrada: Path = typer.Argument(..., exists=True)):
    """Importa dados de um arquivo."""
    rprint(f"Importando de [bold]{entrada}[/]")


if __name__ == "__main__":
    app.__call__()
