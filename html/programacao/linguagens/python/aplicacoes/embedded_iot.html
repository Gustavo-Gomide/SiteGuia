Python para Embedded e IoT

MicroPython no ESP32/RP2040, CircuitPython da Adafruit, RPi.GPIO e gpiozero no Raspberry Pi, MQTT com paho e automação de hardware com Python.

Python chegou ao hardware embarcado via MicroPython e CircuitPython, rodando em microcontroladores com apenas 256 KB de RAM. No Raspberry Pi, Python controla GPIOs, I2C, SPI e UART com bibliotecas maduras usadas em projetos industriais e hobbystas.

MicroPython: Python em Microcontroladores

ESP32

Dual-core Xtensa 240 MHz, 520 KB RAM, Wi-Fi + BT. Suporte nativo MicroPython com módulo network.

RP2040 (Pico)

Dual-core ARM Cortex-M0+ 133 MHz, 264 KB SRAM. MicroPython oficial da Raspberry Pi Foundation.

ESP8266

80 MHz, 80 KB RAM, Wi-Fi. Ideal para sensores IoT simples. Menor subconjunto do MicroPython.

STM32 / pyboard

Plataforma original do MicroPython. STM32F4 com USB nativo, SD card, acelerômetro.

MicroPython (ESP32) — Wi-Fi, MQTT, DHT22, NeoPixel
# main.py — roda ao inicializar o ESP32
import network, time, ujson
from machine import Pin, I2C
import dht
from umqtt.simple import MQTTClient
import neopixel

# ----- Wi-Fi -----
wlan = network.WLAN(network.STA_IF)
wlan.active(True)
wlan.connect("MinhaRede", "MinhaSenh@")
while not wlan.isconnected():
    time.sleep_ms(100)
print("IP:", wlan.ifconfig()[0])

# ----- Sensores -----
sensor = dht.DHT22(Pin(4))
pixel  = neopixel.NeoPixel(Pin(48), 1)   # RGB LED embutido

# ----- MQTT -----
client = MQTTClient(
    client_id=b"esp32-01",
    server=b"192.168.1.10",
    user=b"iot", password=b"senha_mqtt"
)
client.connect()

while True:
    try:
        sensor.measure()
        payload = ujson.dumps({
            "temp": sensor.temperature(),
            "umid": sensor.humidity(),
        })
        client.publish(b"sensores/esp32-01", payload.encode())
        # LED verde: OK
        pixel[0] = (0, 20, 0)
        pixel.write()
    except OSError:
        # LED vermelho: erro
        pixel[0] = (20, 0, 0)
        pixel.write()
    time.sleep(5)

Raspberry Pi: gpiozero e RPi.GPIO

Python (Raspberry Pi) — gpiozero: LED, Button, PWM, distância HC-SR04
from gpiozero import LED, Button, PWMLED, DistanceSensor, Device
from gpiozero.pins.pigpio import PiGPIOFactory
from signal import pause
import time

# Usar pigpio para PWM de hardware preciso (sudo pigpiod para iniciar)
Device.pin_factory = PiGPIOFactory()

led    = LED(17)
botao  = Button(27, pull_up=True, bounce_time=0.05)
led_pw = PWMLED(18)    # PWM no pino 18 (hardware)

# Callbacks de evento
def ao_pressionar():
    led.on()
    print("Botão pressionado")

def ao_soltar():
    led.off()

botao.when_pressed  = ao_pressionar
botao.when_released = ao_soltar

# Sensor ultrassônico HC-SR04
sensor = DistanceSensor(echo=24, trigger=23, max_distance=2)

def monitorar():
    while True:
        dist = sensor.distance * 100    # metros → cm
        print(f"Distância: {dist:.1f} cm")
        # LED mais brilhante conforme mais perto
        led_pw.value = max(0, 1 - dist / 50)
        time.sleep(0.1)

try:
    monitorar()
except KeyboardInterrupt:
    led.off();  led_pw.off()

CircuitPython: Adafruit e USB Drive

CircuitPython (RP2040 / Feather) — I2C + Display OLED + BME280
# code.py — CircuitPython: salve no CIRCUITPY drive via USB
import board, busio, time
import adafruit_bme280.basic as adafruit_bme280
import adafruit_displayio_ssd1306
import displayio, terminalio
from adafruit_display_text import label

displayio.release_displays()

# --- I2C e sensores ---
i2c = busio.I2C(board.SCL, board.SDA)
bme = adafruit_bme280.Adafruit_BME280_I2C(i2c, address=0x76)
bme.sea_level_pressure = 1013.25

# --- Display OLED 128×64 ---
display_bus = busio.I2C(board.SCL, board.SDA)
disp = adafruit_displayio_ssd1306.SSD1306(display_bus, width=128, height=64)
splash = displayio.Group()
disp.root_group = splash

txt = label.Label(terminalio.FONT, color=0xFFFFFF, x=0, y=8)
splash.append(txt)

while True:
    T = bme.temperature
    H = bme.humidity
    P = bme.pressure
    txt.text = f"T:{T:.1f}C H:{H:.0f}%\nP:{P:.0f}hPa"
    time.sleep(2)

MQTT com paho: IoT na Nuvem

Python (host) — paho-mqtt: publicar e subscrever com TLS
import paho.mqtt.client as mqtt
import json, ssl, threading

BROKER = "broker.hivemq.com"
PORT   = 8883   # TLS
TOPIC  = "sensores/#"


def on_connect(client, userdata, flags, rc, props):
    print(f"Conectado: {rc}")
    client.subscribe(TOPIC, qos=1)

def on_message(client, userdata, msg):
    dados = json.loads(msg.payload)
    print(f"[{msg.topic}] T={dados['temp']:.1f}°C U={dados['umid']:.0f}%")

client = mqtt.Client(mqtt.CallbackAPIVersion.VERSION2)
client.tls_set(cert_reqs=ssl.CERT_REQUIRED)    # TLS obrigatório
client.on_connect = on_connect
client.on_message = on_message
client.connect(BROKER, PORT, keepalive=60)

# Publicar de outro thread
def publicar_periodicamente():
    import time, random
    while True:
        dados = {"temp": 20 + random.uniform(-2, 2), "umid": 55.0}
        client.publish("sensores/simulado/01", json.dumps(dados), qos=1)
        time.sleep(5)

threading.Thread(target=publicar_periodicamente, daemon=True).start()
client.loop_forever()
