sqlite3: Banco Embutido

Python — sqlite3: CRUD completo com Row factory e context manager
import sqlite3
from contextlib import contextmanager
from typing import Iterator

@contextmanager
def get_conn(path: str) -> Iterator[sqlite3.Connection]:
    conn = sqlite3.connect(path, detect_types=sqlite3.PARSE_DECLTYPES)
    conn.row_factory = sqlite3.Row   # acesso por col[nome]
    conn.execute("PRAGMA journal_mode=WAL")
    conn.execute("PRAGMA foreign_keys=ON")
    try:
        yield conn
        conn.commit()
    except:
        conn.rollback()
        raise
    finally:
        conn.close()


DDL = """
CREATE TABLE IF NOT EXISTS produtos (
    id      INTEGER PRIMARY KEY AUTOINCREMENT,
    nome    TEXT    NOT NULL,
    preco   REAL    NOT NULL CHECK (preco > 0),
    estoque INTEGER NOT NULL DEFAULT 0
)
"""

with get_conn("loja.db") as conn:
    conn.executescript(DDL)

    # INSERT com parâmetros nomeados (previne SQL injection)
    conn.execute(
        "INSERT INTO produtos (nome, preco, estoque) VALUES (:nome, :preco, :estoque)",
        {"nome": "Produto A", "preco": 29.90, "estoque": 100},
    )
    conn.executemany(
        "INSERT INTO produtos (nome, preco, estoque) VALUES (?, ?, ?)",
        [("B", 9.90, 50), ("C", 49.90, 20)],
    )

with get_conn("loja.db") as conn:
    for row in conn.execute("SELECT * FROM produtos WHERE preco < ?", (50,)):
        print(row["nome"], row["preco"])

SQLAlchemy 2.0 ORM — Relacionamentos e Queries

Python — SQLAlchemy: eager loading, transactions, bulk insert
from sqlalchemy import create_engine, select, func, text
from sqlalchemy.orm import Session, selectinload

# Reutilizando modelos Autor/Livro do guia
engine = create_engine("postgresql+psycopg2://user:pass@localhost/app",
                         pool_size=10, pool_pre_ping=True)

with Session(engine) as session:

    # Eager loading evita N+1 queries
    stmt = (
        select(Autor)
        .options(selectinload(Autor.livros))
        .where(Autor.nome.ilike("%silva%"))
    )
    autores = session.scalars(stmt).all()

    # Aggregation
    media = session.scalar(
        select(func.avg(Livro.paginas))
    )

    # Bulk insert (muito mais rápido que add() em loop)
    session.execute(
        text("INSERT INTO livros (titulo, autor_id) VALUES (:t, :a)"),
        [{"t": f"Livro {i}", "a": 1} for i in range(1000)],
    )
    session.commit()


# Transação com rollback automático
from sqlalchemy.exc import IntegrityError

with Session(engine) as session:
    with session.begin():
        try:
            session.add(Autor(nome="Novo Autor"))
        except IntegrityError:
            session.rollback()
            print("Autor já existe")

asyncpg: PostgreSQL Assíncrono Direto

Python — asyncpg: pool de conexões, prepared statements, copy
import asyncio
import asyncpg


async def main() -> None:
    pool = await asyncpg.create_pool(
        dsn="postgresql://user:pass@localhost/app",
        min_size=5, max_size=20,
    )
    async with pool.acquire() as conn:
        # Query com parâmetros posicionais ($1, $2...)
        rows = await conn.fetch(
            "SELECT id, nome FROM produtos WHERE preco < $1 LIMIT $2",
            50.0, 100
        )
        for row in rows:
            print(row["id"], row["nome"])

        # Prepared statement (reutilizável)
        stmt = await conn.prepare("SELECT * FROM pedidos WHERE usuario_id = $1")
        pedidos = await stmt.fetch(42)

        # COPY para bulk insert ultra-rápido
        dados = [(i, f"Prod {i}", i * 9.9) for i in range(10_000)]
        await conn.copy_records_to_table(
            "produtos", records=dados, columns=["id", "nome", "preco"]
        )

asyncio.run(main())

Redis: Cache, Filas e Pub/Sub

Python — redis-py: cache, rate limiting, pub/sub
import redis, json, functools, time
from typing import Callable, TypeVar
from redis.asyncio import Redis as AsyncRedis

# Cache decorator com Redis
r = redis.Redis(host="localhost", port=6379, decode_responses=True)

def cache_redis(ttl: int = 300):
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            chave = f"{func.__name__}:{args}:{kwargs}"
            cached = r.get(chave)
            if cached:
                return json.loads(cached)
            resultado = func(*args, **kwargs)
            r.setex(chave, ttl, json.dumps(resultado))
            return resultado
        return wrapper
    return decorator


@cache_redis(ttl=60)
def buscar_produto(produto_id: int) -> dict:
    # query ao banco real
    return {"id": produto_id, "nome": "Produto X"}


# Rate limiting com sliding window
def checar_rate_limit(usuario_id: str, limite: int = 100, janela: int = 60) -> bool:
    chave = f"rate:{usuario_id}"
    agora = time.time()
    pipe = r.pipeline()
    pipe.zremrangebyscore(chave, 0, agora - janela)
    pipe.zadd(chave, {str(agora): agora})
    pipe.zcard(chave)
    pipe.expire(chave, janela)
    _, _, contagem, _ = pipe.execute()
    return contagem <= limite


# Pub/Sub assíncrono
async def subscriber() -> None:
    ar = await AsyncRedis.from_url("redis://localhost")
    ps = ar.pubsub()
    await ps.subscribe("eventos")
    async for mensagem in ps.listen():
        if mensagem["type"] == "message":
            print("Evento:", mensagem["data"])

Alembic: Migrations Versionadas

Shell + Python — Alembic: setup e migration manual
# Inicializar
alembic init alembic

# alembic/env.py — apontar para seus modelos
# from app.models import Base
# target_metadata = Base.metadata

# Gerar migration automática (detecta diff do ORM)
alembic revision --autogenerate -m "adicionar coluna telefone em usuarios"

# Aplicar
alembic upgrade head

# Reverter uma migration
alembic downgrade -1
