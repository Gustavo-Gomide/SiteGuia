Python — stdlib: pathlib, logging, subprocess, multiprocessing
from pathlib import Path
import json, logging, subprocess, multiprocessing

# pathlib — navegar e manipular arquivos sem concatenar strings
base = Path.home() / "projetos" / "dados"
base.mkdir(parents=True, exist_ok=True)

config = base / "config.json"
config.write_text(json.dumps({"versao": "1.0"}, indent=2))

for p in base.glob("**/*.json"):
    print(p.name, p.stat().st_size, "bytes")

# logging — configuração profissional
logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s %(levelname)-8s %(name)s: %(message)s",
    datefmt="%H:%M:%S",
)
logger = logging.getLogger("meu_app")
logger.info("Servidor iniciado na porta %d", 8080)
logger.warning("Conexão lenta detectada")

# subprocess — rodar um comando externo com segurança
result = subprocess.run(
    ["git", "log", "--oneline", "-5"],
    capture_output=True, text=True, check=True
)
print(result.stdout)

# multiprocessing — CPU-bound: sem GIL entre processos
def calcular(n: int) -> int:
    return sum(i * i for i in range(n))

if __name__ == "__main__":
    with multiprocessing.Pool() as pool:
        resultados = pool.map(calcular, [1_000_000, 2_000_000, 3_000_000])
    print(resultados)

HTTP: requests e httpx

Python — requests (sync) e httpx (sync + async)
import requests
from requests.adapters import HTTPAdapter
from urllib3.util.retry import Retry

# Session com retry automático e timeout
session = requests.Session()
retry_policy = Retry(total=3, backoff_factor=1,
                      status_forcelist=[429, 500, 502, 503])
adapter = HTTPAdapter(max_retries=retry_policy)
session.mount("https://", adapter)
session.headers.update({"Accept": "application/json"})

resp = session.get("https://httpbin.org/json", timeout=10)
resp.raise_for_status()  # levanta HTTPError em 4xx/5xx
dados = resp.json()

# Upload de arquivo
with open("logo.png", "rb") as f:
    resp = session.post("https://api.exemplo.com/upload", files={"file": f})


# httpx: mesma API + async nativo
import httpx, asyncio

async def main() -> None:
    limits = httpx.Limits(max_connections=100, max_keepalive_connections=20)
    async with httpx.AsyncClient(limits=limits, timeout=10) as client:
        respostas = await asyncio.gather(
            client.get("https://httpbin.org/get"),
            client.get("https://httpbin.org/uuid"),
        )
    for r in respostas:
        print(r.status_code, r.json())

asyncio.run(main())

Pydantic: Validação de Dados

Python — Pydantic v2: BaseModel, validators, Settings
from pydantic import BaseModel, Field, field_validator, model_validator, EmailStr
from pydantic_settings import BaseSettings
from typing import Annotated
from datetime import datetime


class Endereco(BaseModel):
    rua:   str
    cep:   str
    cidade: str

    @field_validator("cep")
    @classmethod
    def validar_cep(cls, v: str) -> str:
        v = v.replace("-", "")
        if not v.isdigit() or len(v) != 8:
            raise ValueError("CEP deve ter 8 dígitos")
        return v


class Usuario(BaseModel):
    nome:     str = Field(min_length=2, max_length=100)
    email:    EmailStr
    idade:    int = Field(ge=0, le=150)
    endereco: Endereco
    criado_em: datetime = Field(default_factory=datetime.now)

    @model_validator(mode="after")
    def verificar_maior_de_idade(self) -> "Usuario":
        if self.idade < 18:
            raise ValueError("Usuário deve ser maior de idade")
        return self


# Cria instância sendo validada
u = Usuario(
    nome="Maria Silva", email="maria@exemplo.com",
    idade=30, endereco={"rua": "Av. Paulista, 1000",
                         "cep": "01311-000", "cidade": "São Paulo"}
)
print(u.model_dump_json(indent=2))


# BaseSettings: carrega ENV + .env file
class Settings(BaseSettings):
    database_url: str
    secret_key:   str
    debug:        bool = False
    max_workers:  int = 4

    class Config:
        env_file = ".env"

settings = Settings()  # lê DATABASE_URL, SECRET_KEY, etc. do ambiente

SQLAlchemy: ORM e Core

Python — SQLAlchemy 2.0 ORM com typed declarative
from sqlalchemy import create_engine, String, ForeignKey, select
from sqlalchemy.orm import (DeclarativeBase, Mapped, mapped_column,
                              relationship, Session)
from typing import Optional

engine = create_engine("sqlite:///app.db", echo=True)

class Base(DeclarativeBase): pass

class Autor(Base):
    __tablename__ = "autores"
    id:   Mapped[int]   = mapped_column(primary_key=True)
    nome: Mapped[str]   = mapped_column(String(200), nullable=False)
    livros: Mapped[list["Livro"]] = relationship(back_populates="autor")

class Livro(Base):
    __tablename__ = "livros"
    id:         Mapped[int]            = mapped_column(primary_key=True)
    titulo:     Mapped[str]            = mapped_column(String(300))
    paginas:    Mapped[Optional[int]]  = mapped_column(default=None)
    autor_id:   Mapped[int]            = mapped_column(ForeignKey("autores.id"))
    autor:      Mapped[Autor]          = relationship(back_populates="livros")

Base.metadata.create_all(engine)

with Session(engine) as session:
    autor = Autor(nome="Guido van Rossum", livros=[
        Livro(titulo="Computer Programming for Everybody", paginas=320),
    ])
    session.add(autor)
    session.commit()

    # Query type-safe com select()
    stmt = select(Livro).join(Livro.autor).where(Autor.nome.contains("Guido"))
    livros = session.scalars(stmt).all()
    for l in livros:
        print(f"{l.titulo} ({l.paginas}p)")
