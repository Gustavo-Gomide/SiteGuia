Python — Fundamentos

Fundamentos do Python

Tipos built-in, variáveis, operadores, controle de fluxo, funções, coleções (list, dict, set, tuple), strings e os princípios básicos de OOP em Python.

Python tem tipagem dinâmica forte — a linguagem infere o tipo em tempo de execução mas não faz conversões implícitas silenciosas. Tudo em Python é um objeto, incluindo funções, classes e módulos.

Python — tipos, inferência e verificação
# Inferência dinâmica — sem declaração explícita
x = 42          # int
y = 3.14        # float
nome = "Python" # str
ativo = True    # bool

print(type(x))    # <class 'int'>
print(type(nome))  # <class 'str'>

# Verificação de tipo
print(isinstance(x, int))    # True
print(isinstance(y, (int, float)))  # True (aceita tupla)

# Inteiros têm precisão ARBITRÁRIA
gigante = 2 ** 100
print(gigante)  # 1267650600228229401496703205376

# Divisão: / sempre retorna float; // retorna int
print(7 / 2)   # 3.5
print(7 // 2)  # 3
print(7 % 2)   # 1  (módulo)
print(2 ** 10) # 1024 (potência)

# Conversão explícita
print(int("42"))     # 42
print(float("3.14")) # 3.14
print(str(100))       # "100"
print(bool(0))       # False  (0, "", [], {}, None → False)

Python — list: operações, comprehensions, métodos
# List literals e operações
numeros = [3, 1, 4, 1, 5, 9, 2, 6]
numeros.append(5)             # adicionar ao final
numeros.insert(0, 0)          # inserir no índice 0
numeros.remove(1)             # remove PRIMEIRO 1 encontrado
ultimo = numeros.pop()         # remove e retorna último
numeros.sort()                 # ordena in-place
numeros.sort(reverse=True)    # decrescente
copia = numeros.copy()         # cópia rasa (shallow)

# List comprehensions — sintaxe [expr for item in iterável if condição]
quadrados     = [x**2 for x in range(10)]
pares         = [x    for x in range(20) if x % 2 == 0]
maiusculas    = [s.upper() for s in ["a", "b", "c"]]
matriz_flat   = [val for row in [[1,2],[3,4]]
                     for val in row]  # [1, 2, 3, 4]

# enumerate e zip
frutas = ["maçã", "banana", "uva"]
for i, fruta in enumerate(frutas, start=1):
    print(f"{i}. {fruta}")

precos = [1.50, 0.80, 2.30]
for fruta, preco in zip(frutas, precos):
    print(f"{fruta}: R${preco:.2f}")

Python — if/elif/else, for, while, match (3.10+)
# if / elif / else
nota = 8.5
if nota >= 9.0:
    conceito = "A"
elif nota >= 7.0:
    conceito = "B"
elif nota >= 5.0:
    conceito = "C"
else:
    conceito = "F"

# Operador ternário
resultado = "aprovado" if nota >= 5.0 else "reprovado"

# for com break, continue, else
primos = []
for n in range(2, 30):
    for d in range(2, int(n**0.5) + 1):
        if n % d == 0:
            break
    else:            # for...else: ejecutado se loop não sofreu break
        primos.append(n)

# while com walrus operator (:=) — 3.8+
import random
tentativas = 0
while (n := random.randint(1, 10)) != 7:
    tentativas += 1
print(f"Achou 7 em {tentativas} tentativas")

# match/case — 3.10+ (Pattern Matching)
comando = "mover norte 3".split()

match comando:
    case ["sair"]:
        print("Encerrando")
    case ["mover", direcao, str(passos)]:
        print(f"Mover {direcao} por {passos} passos")
    case ["atacar", alvo] if alvo in ["dragão", "troll"]:
        print(f"Atacando {alvo}!")
    case _:
        print("Comando desconhecido")

Funções: Parâmetros, *args, **kwargs e Anotações

Python — todas as formas de parâmetros de função
# Parâmetros padrão
def saudar(nome: str, saudacao: str = "Olá") -> str:
    return f"{saudacao}, {nome}!"

print(saudar("Guido"))             # Olá, Guido!
print(saudar("Guido", "Oi"))       # Oi, Guido!
print(saudar(saudacao="Hey", nome="Tim"))  # parâmetros nomeados

# *args: número variável de posicionais
def somar(*args: float) -> float:
    return sum(args)

print(somar(1, 2, 3, 4, 5))    # 15.0

# **kwargs: número variável de nomeados
def registrar(evento: str, **meta) -> None:
    print(f"Evento: {evento}")
    for k, v in meta.items():
        print(f"  {k}={v}")

registrar("login", usuario="ada", ip="127.0.0.1", sucesso=True)

# /  e  *  para forçar posicional/nomeado
def dividir(numerador: float, /, denominador: float, *, arredondar: bool = False) -> float:
    # numerador: somente posicional (/ separa)
    # arredondar: somente nomeado (* separa)
    r = numerador / denominador
    return round(r) if arredondar else r

print(dividir(10, 3, arredondar=True))   # 3

# Unpacking na chamada
args_lista = [1, 2, 3]
print(somar(*args_lista))     # 6.0

kwargs_dict = {"nome": "Turing", "saudacao": "Bom dia"}
print(saudar(**kwargs_dict))  # Bom dia, Turing!

Exceções e Tratamento de Erros

Python — try/except/else/finally, raise, hierarquia
import sys
from pathlib import Path

def ler_inteiro(caminho: str) -> int:
    try:
        conteudo = Path(caminho).read_text(encoding="utf-8").strip()
        valor = int(conteudo)
    except FileNotFoundError:
        print(f"Arquivo não encontrado: {caminho}", file=sys.stderr)
        raise                    # re-raise — não silencie
    except (ValueError, TypeError) as e:
        raise ValueError(f"Conteúdo inválido em {caminho}: {e}") from e
    else:
        print(f"Lido com sucesso: {valor}")  # só executa se sem exceção
        return valor
    finally:
        print("Tentativa de leitura concluída")  # sempre executa


# Exceções personalizadas
class SaldoInsuficienteError(ValueError):
    def __init__(self, saldo: float, tentativa: float):
        self.saldo = saldo
        self.tentativa = tentativa
        super().__init__(
            f"Saldo insuficiente: R${saldo:.2f} < R${tentativa:.2f}"
        )

def sacar(saldo: float, valor: float) -> float:
    if valor > saldo:
        raise SaldoInsuficienteError(saldo, valor)
    return saldo - valor

try:
    saldo = sacar(100.0, 150.0)
except SaldoInsuficienteError as e:
    print(f"Erro: {e}")

OOP Básico: Classes, __init__ e Herança

Python — classe, herança, dunder methods e propriedades
from dataclasses import dataclass, field
from typing import ClassVar

class Conta:
    """Conta bancária simples."""
    taxa_juros: ClassVar[float] = 0.01  # atributo de classe

    def __init__(self, titular: str, saldo: float = 0.0) -> None:
        self.titular = titular
        self._saldo  = saldo  # convenção de protegido

    @property
    def saldo(self) -> float:
        return self._saldo

    def depositar(self, valor: float) -> None:
        if valor <= 0:
            raise ValueError("Valor deve ser positivo")
        self._saldo += valor

    def sacar(self, valor: float) -> None:
        if valor > self._saldo:
            raise ValueError("Saldo insuficiente")
        self._saldo -= valor

    def __repr__(self) -> str:
        return f"Conta({self.titular!r}, saldo={self._saldo:.2f})"

    def __str__(self) -> str:
        return f"Conta de {self.titular}: R${self._saldo:.2f}"


class ContaPoupanca(Conta):
    """Conta poupança com rendimento mensal."""

    def render(self) -> None:
        self._saldo *= (1 + Conta.taxa_juros)
        print(f"Rendimento aplicado. Novo saldo: R${self._saldo:.2f}")


# Uso
c = Conta("Ada", 500.0)
c.depositar(200.0)
print(c)          # Conta de Ada: R$700.00
print(repr(c))    # Conta('Ada', saldo=700.00)

p = ContaPoupanca("Turing", 1000.0)
p.render()        # Rendimento aplicado. Novo saldo: R$1010.00

# dataclass — boilerplate zero
@dataclass
class Ponto:
    x: float
    y: float
    z: float = 0.0

    def distancia(self, outro: "Ponto") -> float:
        import math
        return math.sqrt((self.x-outro.x)**2 + (self.y-outro.y)**2 + (self.z-outro.z)**2)

p1, p2 = Ponto(0, 0), Ponto(3, 4)
print(p1.distancia(p2))  # 5.0
