TypeScript — Introducao

Introducao ao TypeScript

TypeScript e JavaScript com tipagem estatica opcional. Desenvolvido pela Microsoft, compila para JS puro e roda em qualquer ambiente que execute JavaScript.

TypeScript foi lancado em 2012 por Anders Hejlsberg (criador do C#). A versao 5.0 (2023) trouxe decorators TC39, const type parameters e multiplos arquivos tsconfig. O sistema de tipos e estrutural (duck typing) em vez de nominal, permitindo grande flexibilidade com seguranca.

Tipos Basicos e Inferencia

TypeScript — Anotacoes, inferencia e strict mode
// Tipos primitivos (raramente anotar — TS infere)
const nome:    string  = "Alice";
const idade:   number  = 30;
const ativo:   boolean = true;
const nulo:    null    = null;       // so null
const indet:   undefined = undefined;
const qualquer: unknown = obterDadosExternos(); // prefira unknown a any
const nunca:   never   = (() => { throw new Error(); })();

// Arrays e tuplas
const nums:  number[]        = [1, 2, 3];
const pares: [string, number] = ["Alice", 30]; // tupla

// Inferencia — TS sabe os tipos sem anotacao
const x = 42;           // inferido: number
const msg = "ola";      // inferido: string
const arr = [1, "dois"]; // inferido: (number | string)[]

// Type aliases vs interfaces
type Ponto = { x: number; y: number };

interface Forma {
  area(): number;
  perimetro(): number;
}

// Funcoes tipadas
function somar(a: number, b: number): number {
  return a + b;
}

const multiplicar = (a: number, b: number): number => a * b;

// Parametros opcionais e default
function cumprimentar(nome: string, saudacao = "Ola"): string {
  return `${saudacao}, ${nome}!`;
}

// strict mode no tsconfig bloqueia:
// - any implicito
// - acesso a possivelmente undefined/null sem verificacao
// - this implicito em funcoes

function comprimento(s: string | null): number {
  if (s === null) return 0;
  return s.length; // TS sabe que s e string aqui
}

tsconfig.json Essencial

tsconfig.json — Configuracao moderna para Node.js e browser
// tsconfig.json — configuracao recomendada para Node.js 20+
{
  "compilerOptions": {
    "target":            "ES2022",
    "module":            "NodeNext",  // ou "ESNext" para bundler
    "moduleResolution":  "NodeNext",
    "lib":               ["ES2023"],

    // Strict suite completo
    "strict":                      true,
    "noUncheckedIndexedAccess":    true,  // arr[i] pode ser undefined
    "exactOptionalPropertyTypes":  true,  // { a?: string } != { a: string | undefined }
    "noImplicitOverride":          true,

    // Saida
    "outDir":    "./dist",
    "rootDir":   "./src",
    "declaration":         true,  // gerar .d.ts
    "declarationMap":      true,
    "sourceMap":           true,

    // Interoperabilidade
    "esModuleInterop":     true,
    "forceConsistentCasingInFileNames": true,
    "skipLibCheck":        true
  },
  "include": ["src"],
  "exclude": ["node_modules", "dist"]
}

Primeiro Programa

TypeScript — HTTP server tipado com Node.js nativo
// src/index.ts
import { createServer, IncomingMessage, ServerResponse } from "node:http";

interface Rota {
  metodo: "GET" | "POST" | "PUT" | "DELETE";
  path:   string;
  handler(req: IncomingMessage, res: ServerResponse): void;
}

const rotas: Rota[] = [
  {
    metodo: "GET",
    path:   "/",
    handler(_req, res) {
      res.writeHead(200, { "Content-Type": "application/json" });
      res.end(JSON.stringify({ ok: true, mensagem: "TypeScript API" }));
    },
  },
];

const server = createServer((req, res) => {
  const rota = rotas.find(
    (r) => r.metodo === req.method && r.path === req.url
  );
  if (rota) {
    rota.handler(req, res);
  } else {
    res.writeHead(404);
    res.end("Nao encontrado");
  }
});

server.listen(3000, () => console.log("Servidor em http://localhost:3000"));

// Executar: npx tsx src/index.ts
// Compilar: npx tsc --build
// Executar JS compilado: node dist/index.js
