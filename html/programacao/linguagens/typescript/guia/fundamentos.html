TypeScript — Fundamentos

Fundamentos do TypeScript

Union types, narrowing, generics, classes orientadas a objetos, enums, interfaces e os utility types que cobrem 80% dos casos de uso diario.

O sistema de tipos estrutural do TypeScript significa que dois tipos sao compativeis se tiverem a mesma forma (shape) — nao precisam herdar do mesmo pai. O narrowing via type guards deixa o compilador rastrear que tipo e possivel em cada ramo do codigo.

Union Types e Narrowing

TypeScript — Union, discriminated union e type guards
// Union simples
type Id = string | number;

function formatarId(id: Id): string {
  if (typeof id === "string") {
    return id.toUpperCase();  // TS sabe: string
  }
  return id.toFixed(0);        // TS sabe: number
}

// Discriminated union — padrao mais importante do TS
type Resultado<T> =
  | { ok: true;  dado: T      }
  | { ok: false; erro: string };

function processar<T>(res: Resultado<T>): T | null {
  if (res.ok) {
    return res.dado;  // TS sabe que res.dado existe aqui
  }
  console.error(res.erro);
  return null;
}

// Type guard customizado (is)
function ehString(v: unknown): v is string {
  return typeof v === "string";
}

// Assertion function (asserts)
function assertNonNull<T>(val: T | null): asserts val is T {
  if (val === null) throw new Error("Valor nulo inesperado");
}

// in narrowing — estreitar por propriedade
interface Gato  { tipo: "gato";  miar(): void }
interface Cachorro { tipo: "cachorro"; latir(): void }
type Animal = Gato | Cachorro;

function fazerSom(a: Animal) {
  switch (a.tipo) {
    case "gato":     a.miar(); break;
    case "cachorro": a.latir(); break;
    default: {
      const _esgotado: never = a; // erro se faltar um case
    }
  }
}

Generics

TypeScript — Generics com constraints, defaults e inferencia
// Funcao generica simples
function primeiro<T>(arr: T[]): T | undefined {
  return arr[0];
}
primeiro([1, 2, 3]);  // T inferido como number
primeiro(["a", "b"]); // T inferido como string

// Constraint: T deve ter length
function maior<T extends { length: number }>(a: T, b: T): T {
  return a.length >= b.length ? a : b;
}
maior("hello", "hi");       // "hello"
maior([1, 2, 3], [4]);    // [1, 2, 3]

// Interface generica + default
interface Repositorio<T, ID = number> {
  buscarPorId(id: ID): Promise<T | null>;
  listar(): Promise<T[]>;
  salvar(entidade: T): Promise<T>;
  deletar(id: ID): Promise<void>;
}

// keyof + typeof para tipos derivados
const cores = { vermelho: "#f00", verde: "#0f0", azul: "#00f" } as const;
type NomeCor = keyof typeof cores;   // "vermelho" | "verde" | "azul"
type ValorCor = (typeof cores)[NomeCor]; // "#f00" | "#0f0" | "#00f"

// Mapped type
type Opcional<T> = { [K in keyof T]?: T[K] };         // como Partial<T>
type SoLeitura<T> = { readonly [K in keyof T]: T[K] }; // como Readonly<T>

// Conditional type
type Achatar<T> = T extends Array<infer U> ? U : T;
type A = Achatar<number[]>;   // number
type B = Achatar<string>;     // string

Classes e OOP

TypeScript — Classes com visibilidade, abstract e implements
abstract class Forma {
  constructor(protected readonly cor: string) {}

  abstract area(): number;
  abstract perimetro(): number;

  descrever(): string {
    return `${this.cor}: area=${this.area().toFixed(2)}`;
  }
}

class Circulo extends Forma {
  constructor(
    private readonly raio: number,
    cor: string
  ) {
    super(cor);
  }

  override area(): number      { return Math.PI * this.raio ** 2; }
  override perimetro(): number { return 2 * Math.PI * this.raio; }
}

class Retangulo extends Forma {
  constructor(
    private largura: number,
    private altura:  number,
    cor: string
  ) {
    super(cor);
  }

  override area(): number      { return this.largura * this.altura; }
  override perimetro(): number { return 2 * (this.largura + this.altura); }
}

// Interface implementada por classe
interface Serializavel<T> {
  toJSON():  string;
  static fromJSON(json: string): T;
}

// Uso polimorfismo
const formas: Forma[] = [
  new Circulo(5, "vermelho"),
  new Retangulo(4, 6, "azul"),
];
formas.forEach(f => console.log(f.descrever()));

Utility Types Essenciais

TypeScript — Partial, Required, Pick, Omit, Record, ReturnType
interface Usuario {
  id:    number;
  nome:  string;
  email: string;
  senha: string;
  admin: boolean;
}

// Partial — todos opcionais (para PATCH)
type AtualizarUsuario = Partial<Usuario>;

// Required — todos obrigatorios
type UsuarioCompleto = Required<Usuario>;

// Pick — selecionar propriedades
type UsuarioPublico = Pick<Usuario, "id" | "nome" | "email">;

// Omit — remover propriedades
type CriarUsuario = Omit<Usuario, "id">; // sem id (gerado pelo DB)

// Record — dicionario tipado
type Permissoes = Record<"ler" | "criar" | "deletar", boolean>;
const adminPerms: Permissoes = { ler: true, criar: true, deletar: true };

// Readonly — imutavel
type Config = Readonly<{
  host: string;
  port: number;
}>;

// ReturnType — tipo de retorno de funcao
function criarSessao() {
  return { token: "abc", expira: new Date() };
}
type Sessao = ReturnType<typeof criarSessao>;
// { token: string; expira: Date }

// Parameters — tipos dos parametros
function enviarEmail(para: string, assunto: string, corpo: string): void {}
type ParamsEmail = Parameters<typeof enviarEmail>;
// [para: string, assunto: string, corpo: string]

// Awaited — desembrulhar Promise
async function buscarUsuario(id: number): Promise<Usuario> { return {} as Usuario; }
type UsuarioResp = Awaited<ReturnType<typeof buscarUsuario>>;
// Usuario (sem Promise)
