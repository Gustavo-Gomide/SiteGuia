Template Literal Types

TypeScript — Template literal types para rotas e eventos
// Construir tipos de string combinando unions
type Metodo  = "GET" | "POST" | "PUT" | "DELETE";
type Recurso = "usuarios" | "posts" | "comentarios";

type Rota = `/${Recurso}` | `/${Recurso}/${string}`;
// "/usuarios" | "/posts" | "/comentarios" | "/usuarios/${string}" | ...

// Tipos de evento derivados de modelo
type EventosCRUD<T extends string> =
  | `${T}:criado`
  | `${T}:atualizado`
  | `${T}:deletado`;

type Eventos = EventosCRUD<"usuario"> | EventosCRUD<"post">;
// "usuario:criado" | "usuario:atualizado" | "usuario:deletado"
// | "post:criado" | "post:atualizado" | "post:deletado"

// Utility types de strings embutidos
type U = Uppercase<"hello">;         // "HELLO"
type L = Lowercase<"WORLD">;         // "world"
type C = Capitalize<"hello">;        // "Hello"
type UC = Uncapitalize<"Hello">;    // "hello"

// Construir getter tipado a partir de objeto
type Getters<T> = {
  [K in keyof T as `get${Capitalize<string & K>}`]: () => T[K];
};

interface Pessoa { nome: string; idade: number }
type PessoaGetters = Getters<Pessoa>;
// { getNome: () => string; getIdade: () => number }

Branded / Opaque Types

TypeScript — Branded types para evitar confusao de primitivos
// Sem branded types: facil confundir IDs!
function transferir(de: number, para: number, valor: number) {}
transferir(1001, 2002, 500); // OK, mas e se inverter de/para?

// Com branded types: o compilador impede confusao
declare const __brand: unique symbol;
type Brand<T, B> = T & { readonly [__brand]: B };

type ContaId = Brand<number, "ContaId">;
type Centavos = Brand<number, "Centavos">;

function contaId(id: number): ContaId  { return id as ContaId; }
function centavos(v: number): Centavos { return v as Centavos; }

function transferirSeguro(de: ContaId, para: ContaId, valor: Centavos) {}

const contaA = contaId(1001);
const contaB = contaId(2002);
const valor  = centavos(50000);

transferirSeguro(contaA, contaB, valor); // OK
// transferirSeguro(1001, 2002, 500); // ERRO: number nao e ContaId
// transferirSeguro(contaA, valor, contaB); // ERRO: ordem errada

// Outro uso: URL validada
type UrlSegura = Brand<string, "UrlSegura">;

function validarUrl(input: string): UrlSegura {
  const url = new URL(input); // lanca se invalida
  if (!url.href.startsWith("https://")) throw new Error("Precisa ser HTTPS");
  return input as UrlSegura;
}

function fetch(url: UrlSegura) { /* aceita apenas URLs validadas */ }

Decorators TC39 (TS 5.0+)

TypeScript — Decorators de classe, metodo e accessor
// tsconfig: "experimentalDecorators": false (nuevo TC39)
// Decorators TC39 nao precisam de experimentalDecorators

// Decorator de classe — registrar num mapa
const servicosRegistrados = new Map<string, new (...args: unknown[]) => unknown>();

function Servico(nome: string) {
  return function<T extends new (...args: unknown[]) => unknown>(
    alvo: T, ctx: ClassDecoratorContext
  ) {
    servicosRegistrados.set(nome, alvo);
    return alvo;
  };
}

// Decorator de metodo — medidor de tempo
function Medido<T>(
  metodo:  (this: T, ...args: unknown[]) => unknown,
  ctx:     ClassMethodDecoratorContext
) {
  return function(this: T, ...args: unknown[]) {
    const inicio = performance.now();
    const res    = metodo.apply(this, args);
    console.log(`${String(ctx.name)}: ${(performance.now() - inicio).toFixed(2)}ms`);
    return res;
  };
}

// Decorator de campo — validacao
function NaoVazio<T>(
  _alvo: undefined,
  ctx: ClassFieldDecoratorContext<T, string>
) {
  return function(this: T, valor: string): string {
    if (!valor.trim()) throw new Error(`${String(ctx.name)} nao pode ser vazio`);
    return valor;
  };
}

@Servico("UsuarioService")
class UsuarioService {
  @NaoVazio
  nome = "Alice";

  @Medido
  buscarTodos() {
    return [{ id: 1, nome: "Alice" }];
  }
}

satisfies Operator e const type params

TypeScript 5.x — satisfies e const type parameters
// satisfies: valida o tipo sem alargar (widening)
type Config = { host: string; port: number; debug?: boolean };

const cfg = {
  host:  "localhost",
  port:  3000,
  debug: true,
} satisfies Config;

// cfg.host e string (nao string | undefined)
console.log(cfg.host.toUpperCase()); // OK, TS sabe que host e string

// Sem satisfies: seria widened para Config e host seria string
// Com as Config: perdemos acesso ao literal "localhost"
// satisfies e o meio-termo perfeito

// const type parameters (TS 5.0) — preservar literais sem as const
function primeiroOu<const T extends readonly unknown[], D>(arr: T, def: D) {
  return arr.length > 0 ? (arr[0] as T[0]) : def;
}
const resultado = primeiroOu(["a", "b"], "z" as const);
// resultado: "a" (literal, nao string)

// NoInfer (TS 5.4) — evitar inferencia indesejada num parametro
function criarEstado<T>(inicial: T, valido: NoInfer<T>[]): T {
  if (!valido.includes(inicial))
    throw new Error("Estado inicial invalido");
  return inicial;
}
criarEstado("ativo", ["ativo", "inativo", "suspense"]);
// T inferido de "ativo" (1o param) — NoInfer impede re-inferencia de valido[]
