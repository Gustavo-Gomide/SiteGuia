O maior beneficio de TypeScript em React Native e a tipagem de rotas de navegacao — impossivel navegar para uma tela que nao existe ou passar o tipo errado de parametro. RootParamList e a fonte de verdade compartilhada.

TypeScript + React Navigation — RootParamList e hooks tipados
// src/navigation/types.ts — mapa de rotas global
import type { NativeStackScreenProps } from "@react-navigation/native-stack";

export type RootStackParamList = {
  Inicio:     undefined;
  Produtos:   { categoria?: string; filtro?: string };
  Produto:    { id: string; nome: string };
  Carrinho:   undefined;
  Checkout:   { itens: Array<{ id: string; quantidade: number }> };
  Perfil:     { usuarioId: string };
  Autenticar: { redirecionarPara?: keyof RootStackParamList };
};

// Props de tela tipadas
export type ProdutoScreenProps = NativeStackScreenProps<RootStackParamList, "Produto">;
export type CheckoutScreenProps = NativeStackScreenProps<RootStackParamList, "Checkout">;

// Hooks tipados globalmente (definir uma vez, usar em qualquer lugar)
import {
  useNavigation, useRoute, type NavigationProp, type RouteProp,
} from "@react-navigation/native";

export function useAppNavigation() {
  return useNavigation<NavigationProp<RootStackParamList>>();
}

export function useAppRoute<K extends keyof RootStackParamList>() {
  return useRoute<RouteProp<RootStackParamList, K>>();
}
TypeScript + React Native — Telas tipadas com StyleSheet e FlatList
import React, { useState, useCallback } from "react";
import {
  View, Text, FlatList, TouchableOpacity, StyleSheet,
  ActivityIndicator, RefreshControl, type ListRenderItem,
} from "react-native";
import type { ProdutoScreenProps } from "../navigation/types";
import { useAppNavigation } from "../navigation/types";

interface Produto {
  id:       string;
  nome:     string;
  preco:    number;
  estoque:  number;
  imagem:   string;
}

export function ProdutosScreen({ route }: ProdutoScreenProps): React.JSX.Element {
  const navigation = useAppNavigation();
  const [produtos, setProdutos]    = useState<Produto[]>([]);
  const [carregando, setCarregando] = useState(true);
  const [refresh, setRefresh]       = useState(false);

  const carregarProdutos = useCallback(async () => {
    try {
      const resp = await fetch(`/api/produtos?categoria=${route.params?.categoria ?? ""}`);
      const dados: Produto[] = await resp.json();
      setProdutos(dados);
    } finally {
      setCarregando(false);
      setRefresh(false);
    }
  }, [route.params?.categoria]);

  // renderItem tipado automaticamente como (info: ListRenderItemInfo<Produto>) => ...
  const renderProduto: ListRenderItem<Produto> = useCallback(({ item }) => (
    <TouchableOpacity
      style={styles.card}
      onPress={() => navigation.navigate("Produto", { id: item.id, nome: item.nome })}
    >
      <Text style={styles.nome}>{item.nome}</Text>
      <Text style={styles.preco}>R$ {item.preco.toFixed(2)}</Text>
      {item.estoque === 0 && <Text style={styles.semEstoque}>Sem estoque</Text>}
    </TouchableOpacity>
  ), [navigation]);

  if (carregando) return <ActivityIndicator style={styles.flex1} size="large" />;

  return (
    <FlatList<Produto>
      data={produtos}
      keyExtractor={(item) => item.id}
      renderItem={renderProduto}
      numColumns={2}
      refreshControl={
        <RefreshControl
          refreshing={refresh}
          onRefresh={() => { setRefresh(true); carregarProdutos(); }}
        />
      }
    />
  );
}

const styles = StyleSheet.create({
  flex1:      { flex: 1, justifyContent: "center", alignItems: "center" },
  card:       { flex: 1, margin: 8, padding: 12, borderRadius: 8, backgroundColor: "#1e1e1e" },
  nome:       { fontSize: 14, fontWeight: "600", color: "#fff" },
  preco:      { fontSize: 16, fontWeight: "bold", color: "#4ade80", marginTop: 4 },
  semEstoque: { fontSize: 12, color: "#f87171", marginTop: 2 },
});

TypeScript + Expo — Camera, Location e Notifications tipados
import { CameraView, useCameraPermissions, type CameraType, type CameraCapturedResult } from "expo-camera";
import * as Location from "expo-location";
import * as Notifications from "expo-notifications";
import { useRef, useState } from "react";

interface CoordenadaGPS {
  latitude:  number;
  longitude: number;
  altitude:  number | null;
  precisao:  number;
}

// ---- Camera com tipos ----
export function CameraScreen() {
  const [permissao,     requisitarPermissao] = useCameraPermissions();
  const [frente,        setFrente]            = useState<CameraType>("back");
  const [foto,          setFoto]              = useState<CameraCapturedResult | null>(null);
  const cameraRef = useRef<CameraView>(null);

  async function tirarFoto(): Promise<void> {
    const resultado = await cameraRef.current?.takePictureAsync({
      quality:   0.8,
      base64:    false,
      exif:      true,
      skipProcessing: false,
    });
    if (resultado) setFoto(resultado);
    // resultado.uri — string tipada
    // resultado.width/height — numbers tipados
    // resultado.exif — ExifData | null
  }

  if (!permissao?.granted) {
    return <>{/* Botao para requisitar permissao */}</>;
  }
  return <CameraView ref={cameraRef} facing={frente} />;
}

// ---- Location tipado ----
async function obterLocalizacao(): Promise<CoordenadaGPS> {
  const { status } = await Location.requestForegroundPermissionsAsync();
  if (status !== "granted") throw new Error("Permissao negada");

  const local = await Location.getCurrentPositionAsync({
    accuracy: Location.Accuracy.High,
  });
  return {
    latitude:  local.coords.latitude,
    longitude: local.coords.longitude,
    altitude:  local.coords.altitude,
    precisao:  local.coords.accuracy ?? 0,
  };
}

// ---- Notifications tipadas ----
interface ConfigNotificacao {
  titulo:   string;
  corpo:    string;
  dados?:   Record<string, unknown>;
  atrasoS?: number;
}

async function agendarNotificacao(config: ConfigNotificacao): Promise<string> {
  const { status } = await Notifications.requestPermissionsAsync();
  if (status !== "granted") throw new Error("Permissao de notificacao negada");

  const id = await Notifications.scheduleNotificationAsync({
    content: {
      title: config.titulo,
      body:  config.corpo,
      data:  config.dados ?? {},
      sound: true,
    },
    trigger: config.atrasoS != null
      ? { type: Notifications.SchedulableTriggerInputTypes.TIME_INTERVAL, seconds: config.atrasoS }
      : null,
  });
  return id; // string — ID da notificacao agendada
}
