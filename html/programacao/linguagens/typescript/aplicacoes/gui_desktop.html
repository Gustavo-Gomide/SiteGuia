Electron com canais IPC tipados, contextBridge type-safe e preload com tipos exportados. Tauri com generics de invoke/listen e backend Rust com comandos fortemente tipados.

IPC tipado e a maior vantagem de usar TypeScript no Electron. Definir os canais como tipos literais garante que main e renderer concordem no contrato sem documentacao externa.

Electron com IPC Tipado

TypeScript + Electron — Canais IPC como tipos literais
// src/shared/ipc-channels.ts — compartilhado entre main e renderer
export interface IpcChannels {
  // main -> renderer
  "progresso-download": { porcentagem: number; arquivo: string };
  "download-completo":   { caminho: string };
  "erro-app":            { mensagem: string; codigo: number };

  // renderer -> main (invoke/handle)
  "abrir-arquivo":       [entrada: { filtros: Electron.FileFilter[] }, saida: string | null];
  "salvar-arquivo":      [entrada: { conteudo: string; caminho: string }, saida: boolean];
  "obter-versao":        [entrada: void, saida: string];
}

// src/main/index.ts — processo principal
import { app, BrowserWindow, ipcMain, dialog } from "electron";
import { readFile, writeFile } from "node:fs/promises";
import type { IpcChannels } from "../shared/ipc-channels.js";

// Helper tipado para ipcMain.handle
function handle<K extends keyof IpcChannels>(
  canal:    K,
  callback: (_evt: Electron.IpcMainInvokeEvent, arg: IpcChannels[K] extends [infer I, ...any[]] ? I : never)
            => Promise<IpcChannels[K] extends [any, infer O] ? O : never>,
) { ipcMain.handle(canal as string, callback); }

handle("abrir-arquivo", async (_evt, { filtros }) => {
  const resultado = await dialog.showOpenDialog({ filters: filtros, properties: ["openFile"] });
  return resultado.canceled ? null : resultado.filePaths[0] ?? null;
});

handle("salvar-arquivo", async (_evt, { conteudo, caminho }) => {
  try {
    await writeFile(caminho, conteudo, "utf-8");
    return true;
  } catch { return false; }
});

handle("obter-versao", async () => app.getVersion());
TypeScript + Electron — Preload com contextBridge e renderer tipado
// src/preload/index.ts
import { contextBridge, ipcRenderer } from "electron";
import type { IpcChannels } from "../shared/ipc-channels.js";

// API exposta ao renderer — completamente tipada
const api = {
  invoke<K extends keyof IpcChannels>(
    canal: K,
    arg:   IpcChannels[K] extends [infer I, any] ? I : never,
  ): Promise<IpcChannels[K] extends [any, infer O] ? O : never> {
    return ipcRenderer.invoke(canal as string, arg);
  },
  on<K extends keyof IpcChannels>(
    canal:   K,
    handler: (dados: IpcChannels[K]) => void,
  ): () => void {
    const wrapped = (_evt: unknown, dados: IpcChannels[K]) => handler(dados);
    ipcRenderer.on(canal as string, wrapped);
    return () => ipcRenderer.removeListener(canal as string, wrapped);
  },
};

contextBridge.exposeInMainWorld("api", api);

Tauri com Invoke Tipado

TypeScript + Tauri — Comandos tipados entre frontend e Rust
// src/tauri-commands.ts — tipos dos comandos Rust
import { invoke, listen } from "@tauri-apps/api/tauri";
import type { UnlistenFn } from "@tauri-apps/api/event";

// Interfaces espelham as structs Rust exatamente
interface InfoSistema {
  os:        string;
  arch:      string;
  memoriaGb: number;
  cpuNome:   string;
}

interface ResultadoArquivo {
  caminho:  string;
  conteudo: string;
  tamanho:  number;
}

interface ProgressoScan {
  arquivo:    string;
  processado: number;
  total:      number;
}

// Wrapper tipado em torno de invoke
export const comandos = {
  async obterInfoSistema(): Promise<InfoSistema> {
    return invoke<InfoSistema>("obter_info_sistema");
  },

  async lerArquivo(caminho: string): Promise<ResultadoArquivo> {
    return invoke<ResultadoArquivo>("ler_arquivo", { caminho });
  },

  async escreverArquivo(caminho: string, conteudo: string): Promise<void> {
    return invoke("escrever_arquivo", { caminho, conteudo });
  },

  async escanearDiretorio(
    raiz:     string,
    callback: (p: ProgressoScan) => void,
  ): Promise<string[]> {
    const unlisten: UnlistenFn = await listen<ProgressoScan>(
      "progresso-scan", (evt) => callback(evt.payload),
    );
    try {
      return await invoke<string[]>("escanear_diretorio", { raiz });
    } finally {
      unlisten();
    }
  },
} as const;
