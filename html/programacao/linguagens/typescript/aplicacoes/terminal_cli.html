Commander com Tipos e Subcomandos

TypeScript + Commander — Options tipadas com interface
#!/usr/bin/env node
import { Command, Option, InvalidArgumentError } from "commander";
import { createWriteStream } from "node:fs";

// Interfaces para as opcoes de cada comando
interface OpcoesBuild {
  saida:    string;
  formato:  "esm" | "cjs" | "iife";
  minify:   boolean;
  dts:      boolean;
  watch:    boolean;
  external: string[];
}

interface OpcoesPublish {
  tag:    string;
  dry:    boolean;
  access: "public" | "restricted";
}

function parsearInteiro(valor: string): number {
  const n = Number(valor);
  if (!Number.isInteger(n)) throw new InvalidArgumentError("Deve ser inteiro");
  return n;
}

const programa = new Command()
  .name("minha-cli")
  .description("Ferramenta de build e publicacao")
  .version("1.0.0");

programa
  .command("build <entrada>")
  .description("Compila o projeto")
  .option("-o, --saida <dir>", "Diretorio de saida", "dist")
  .addOption(
    new Option("-f, --formato <fmt>", "Formato de saida")
      .choices(["esm", "cjs", "iife"])
      .default("esm"),
  )
  .option("--minify",              "Minificar saida")
  .option("--dts",                "Gerar declaracoes .d.ts")
  .option("-w, --watch",          "Modo watch")
  .option("-e, --external <mods>", "Modulos externos (CSV)", (v: string) => v.split(","), [])
  .action(async (entrada: string, opts: OpcoesBuild) => {
    console.log(`Compilando ${entrada} -> ${opts.saida} (${opts.formato})`);
    if (opts.dts)    console.log("Gerando .d.ts...");
    if (opts.minify) console.log("Minificando...");
    if (opts.watch)  console.log("Modo watch ativo");
  });

programa
  .command("publish")
  .description("Publica no npm")
  .option("--tag <tag>",    "Tag npm",     "latest")
  .option("--dry",        "Dry-run")
  .addOption(
    new Option("--access <nivel>").choices(["public", "restricted"]).default("public"),
  )
  .action((opts: OpcoesPublish) => {
    if (opts.dry) { console.log("[dry-run] publicaria com tag:", opts.tag); return; }
    console.log(`Publicando com tag ${opts.tag} (${opts.access})...`);
  });

programa.parse();

Inquirer com Generics Tipados

TypeScript + Inquirer — Wizard de configuracao completamente tipado
import { input, select, checkbox, confirm, password } from "@inquirer/prompts";
import { writeFile } from "node:fs/promises";

type Banco      = "postgresql" | "mysql" | "sqlite";
type Framework  = "express" | "fastify" | "hono";
type Extra      = "prisma" | "vitest" | "eslint" | "docker";

interface ConfigProjeto {
  nome:      string;
  framework: Framework;
  banco:     Banco;
  extras:    Extra[];
  porta:     number;
  dbUrl:     string;
}

async function wizardCriarProjeto(): Promise<ConfigProjeto> {
  const nome = await input({
    message: "Nome do projeto:",
    validate(v) {
      if (!/^[a-z][a-z0-9-]*$/.test(v))
        return "Use letras minusculas, numeros e hifens";
      return true;
    },
  });

  const framework = await select<Framework>({
    message: "Framework HTTP:",
    choices: [
      { value: "express",  name: "Express — maduro, ecossistema grande" },
      { value: "fastify",  name: "Fastify — rapido, TypeBox nativo" },
      { value: "hono",     name: "Hono — edge-ready, RPC client" },
    ],
  });

  const banco = await select<Banco>({
    message: "Banco de dados:",
    choices: [
      { value: "postgresql", name: "PostgreSQL" },
      { value: "mysql",      name: "MySQL" },
      { value: "sqlite",     name: "SQLite (dev/embed)" },
    ],
  });

  const extras = await checkbox<Extra>({
    message: "Extras:",
    choices: [
      { value: "prisma",    name: "Prisma ORM",   checked: true },
      { value: "vitest",    name: "Vitest",        checked: true },
      { value: "eslint",    name: "ESLint + TS",   checked: false },
      { value: "docker",    name: "Dockerfile",    checked: false },
    ],
  });

  const porta = await input({
    message: "Porta:",
    default: "3000",
    validate(v) {
      const n = parseInt(v, 10);
      if (isNaN(n) || n < 1 || n > 65535) return "Porta invalida";
      return true;
    },
  });

  const dbUrl = await password({
    message: "DATABASE_URL:",
    validate(v) {
      if (!v.startsWith("postgres://") && !v.startsWith("mysql://") && v !== ":memory:")
        return "URL invalida";
      return true;
    },
  });

  const config: ConfigProjeto = {
    nome, framework, banco, extras, porta: parseInt(porta, 10), dbUrl,
  };

  const confirmar = await confirm({
    message: `Criar projeto "${nome}" com ${framework} + ${banco}?`,
    default: true,
  });
  if (!confirmar) process.exit(0);

  await writeFile(".projeto-config.json", JSON.stringify(config, null, 2));
  return config;
}
