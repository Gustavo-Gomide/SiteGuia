Johnny-Five com Tipos

TypeScript + Johnny-Five — Sensores e atuadores tipados
import j5, { Board, Led, Button, Sensor, Servo, Thermometer } from "johnny-five";
// npm i johnny-five @types/johnny-five

interface LeituraTemperatura {
  celsius:    number;
  fahrenheit: number;
  kelvin:     number;
  timestamp:  number;
}

interface EstadoSistema {
  alertaAtivo:    boolean;
  temperatura:    LeituraTemperatura | null;
  posicaoServo:   number;
  luzLigada:      boolean;
}

const estado: EstadoSistema = {
  alertaAtivo:  false,
  temperatura:  null,
  posicaoServo: 90,
  luzLigada:    false,
};

const board = new Board({ port: process.env.ARDUINO_PORT ?? "/dev/ttyUSB0" });

board.on("ready", () => {
  console.log("Board pronta");

  // LED de alerta — tipo Led e completamente tipado
  const ledAlerta = new Led(13);

  // Botao com debounce
  const botaoReset = new Button({ pin: 2, isPullup: true });
  botaoReset.on("press", () => {
    console.log("Reset pressionado");
    estado.alertaAtivo = false;
    ledAlerta.stop().off();
  });

  // Sensor analogico (potenciometro)
  const potenciometro = new Sensor({ pin: "A0", freq: 250 });
  potenciometro.on("change", function(this: Sensor) {
    // scaleTo normaliza 0..1023 para 0..180
    estado.posicaoServo = this.scaleTo(0, 180);
    servo.to(estado.posicaoServo);
  });

  // Servo motor tipado
  const servo = new Servo({ pin: 9, startAt: 90, range: [0, 180] });

  // Termometro LM35
  const termometro = new Thermometer({
    controller: "LM35",
    pin:        "A1",
    freq:       1000,
  });

  termometro.on("change", function(this: Thermometer) {
    const leitura: LeituraTemperatura = {
      celsius:    this.celsius,
      fahrenheit: this.fahrenheit,
      kelvin:     this.kelvin,
      timestamp:  Date.now(),
    };
    estado.temperatura = leitura;

    if (leitura.celsius > 35 && !estado.alertaAtivo) {
      estado.alertaAtivo = true;
      ledAlerta.blink(250);
      console.error(`ALERTA: Temperatura alta ${leitura.celsius.toFixed(1)}°C`);
    }
  });
});

MQTT.js com Tipos de Evento

TypeScript + MQTT.js — Topicos e payloads tipados com Zod
import mqtt from "mqtt";
import { z } from "zod";
import { readFileSync } from "node:fs";

// ---- Schemas de mensagens IoT ----
const MsgTemperaturaSchema = z.object({
  deviceId:  z.string().uuid(),
  celsius:   z.number().min(-273.15).max(200),
  umidade:   z.number().min(0).max(100),
  timestamp: z.number().int().positive(),
});

const MsgComandoSchema = z.discriminatedUnion("tipo", [
  z.object({ tipo: z.literal("ligar"),    dispositivo: z.string() }),
  z.object({ tipo: z.literal("desligar"), dispositivo: z.string() }),
  z.object({ tipo: z.literal("config"),   parametros: z.record(z.unknown()) }),
]);

type MsgTemperatura = z.infer<typeof MsgTemperaturaSchema>;
type MsgComando     = z.infer<typeof MsgComandoSchema>;

// ---- Mapa de topicos para schemas ----
const topicosSchema = {
  "sensores/+/temperatura": MsgTemperaturaSchema,
  "comandos/+":             MsgComandoSchema,
} as const;

type Topico = keyof typeof topicosSchema;

// ---- Cliente MQTT tipado ----
const client = mqtt.connect("mqtts://broker.interno:8883", {
  clientId:  `gateway-${crypto.randomUUID()}`,
  ca:        readFileSync("./certs/ca.crt"),
  cert:      readFileSync("./certs/client.crt"),
  key:       readFileSync("./certs/client.key"),
  will: {
    topic:   "status/gateway",
    payload: JSON.stringify({ status: "offline", timestamp: Date.now() }),
    qos:     1,
    retain:  true,
  },
});

const handlers: Partial<{
  [K in Topico]: (msg: z.infer<(typeof topicosSchema)[K]>) => void;
}> = {
  "sensores/+/temperatura": (msg: MsgTemperatura) => {
    console.log(`Sensor ${msg.deviceId}: ${msg.celsius.toFixed(1)}°C, ${msg.umidade.toFixed(0)}% UR`);
  },
  "comandos/+": (msg: MsgComando) => {
    if (msg.tipo === "ligar")    console.log(`Ligar: ${msg.dispositivo}`);
    if (msg.tipo === "desligar") console.log(`Desligar: ${msg.dispositivo}`);
  },
};

client.on("connect", () => {
  for (const topico of Object.keys(topicosSchema)) {
    client.subscribe(topico, { qos: 1 });
  }
});

client.on("message", (topico: string, payload: Buffer) => {
  try {
    const dados = JSON.parse(payload.toString());
    for (const [padrao, schema] of Object.entries(topicosSchema)) {
      if (mqtt.match(padrao, topico)) {
        const resultado = schema.safeParse(dados);
        if (resultado.success) {
          handlers[padrao as Topico]?.(resultado.data as never);
        } else {
          console.warn(`Mensagem invalida em ${topico}`, resultado.error.flatten());
        }
      }
    }
  } catch { console.error("Payload JSON invalido"); }
});
