TypeScript para Dados e IA

TensorFlow.js com tipos de Tensor, Transformers.js para pipeline de NLP tipado, e construcao de pipelines de dados com interfaces explícitas para cada etapa.

IA em TypeScript garante que os shapes de tensores, os formatos de entrada dos modelos e as estruturas de saida sejam verificados em compilacao. Erros de dimensao de tensor e incompatibilidade de tipos sao pegos antes de executar o modelo.

TensorFlow.js com Tipos

TypeScript + TensorFlow.js — Modelo tipado com Tensor e LayersModel
import * as tf from "@tensorflow/tfjs-node";
import type { Tensor, Tensor2D, LayersModel, ModelFitArgs } from "@tensorflow/tfjs";

// ---- Tipos de dados tipados para treinamento ----
interface DadosTreinamento {
  entradas:  Tensor2D;   // shape [N, features]
  rotulos:   Tensor2D;   // shape [N, classes]
  validacao: [Tensor2D, Tensor2D];
}

interface ConfigModelo {
  unidadesOcultas: number[];
  taxaAprendizado: number;
  dropout:         number;
  epocas:          number;
  tamanhoBatch:    number;
}

function construirClassificador(
  nEntradas: number,
  nClasses:  number,
  config:    ConfigModelo,
): LayersModel {
  const modelo = tf.sequential();

  // Camada de entrada
  modelo.add(tf.layers.dense({
    inputShape: [nEntradas],
    units:      config.unidadesOcultas[0]!,
    activation: "relu",
    kernelRegularizer: tf.regularizers.l2({ l2: 0.01 }),
  }));
  modelo.add(tf.layers.batchNormalization());
  modelo.add(tf.layers.dropout({ rate: config.dropout }));

  // Camadas ocultas
  for (const units of config.unidadesOcultas.slice(1)) {
    modelo.add(tf.layers.dense({ units, activation: "relu" }));
    modelo.add(tf.layers.dropout({ rate: config.dropout }));
  }

  // Camada de saida
  modelo.add(tf.layers.dense({ units: nClasses, activation: "softmax" }));

  modelo.compile({
    optimizer: tf.train.adam(config.taxaAprendizado),
    loss:      "categoricalCrossentropy",
    metrics:   ["accuracy"],
  });
  return modelo;
}

async function treinar(
  modelo:  LayersModel,
  dados:   DadosTreinamento,
  config:  ConfigModelo,
): Promise<void> {
  const args: ModelFitArgs = {
    epochs:          config.epocas,
    batchSize:       config.tamanhoBatch,
    validationData:  dados.validacao,
    callbacks: tf.node.tensorBoard("./logs"),
  };

  const historico = await modelo.fit(dados.entradas, dados.rotulos, args);

  const ultimaAcc = historico.history["val_accuracy"]?.at(-1) as number;
  console.log(`Acuracia validacao final: ${(ultimaAcc * 100).toFixed(2)}%`);

  await modelo.save("file://./modelo-salvo");
}

// Inferencia tipada
async function classificar(
  modeloPath: string,
  entrada:    number[],
): Promise<{ classe: number; confianca: number }> {
  const modelo = await tf.loadLayersModel(`file://${modeloPath}/model.json`);
  const tensor = tf.tensor2d([entrada]);
  const saida  = modelo.predict(tensor) as Tensor2D;
  const probs  = await saida.data();
  const classe = Array.from(probs).reduce(
    (iMax, v, i, arr) => v > arr[iMax]! ? i : iMax, 0,
  );
  tensor.dispose();
  saida.dispose();
  return { classe, confianca: probs[classe]! };
}

Transformers.js com Tipos

TypeScript + Transformers.js — Pipeline NLP tipado
import { pipeline } from "@xenova/transformers";
import type { TextClassificationOutput, FeatureExtractionOutput } from "@xenova/transformers";

// ---- Classificacao de sentimento ----
interface ResultadoSentimento {
  texto:      string;
  sentimento: "POSITIVE" | "NEGATIVE" | "NEUTRAL";
  score:      number;
}

async function analisarSentimentos(
  textos: string[],
): Promise<ResultadoSentimento[]> {
  const classificador = await pipeline(
    "text-classification",
    "Xenova/distilbert-base-uncased-finetuned-sst-2-english",
  );

  const saidas = await classificador(textos, { topk: 1 }) as TextClassificationOutput[];

  return textos.map((texto, i) => {
    const [pred] = saidas[i] as [{ label: string; score: number }];
    return {
      texto,
      sentimento: pred.label as ResultadoSentimento["sentimento"],
      score:      pred.score,
    };
  });
}

// ---- Embeddings + similaridade de cosseno tipados ----
type Vetor = Float32Array | number[];

function similaridadeCosseno(a: Vetor, b: Vetor): number {
  let dot = 0, normA = 0, normB = 0;
  for (let i = 0; i < a.length; i++) {
    dot   += a[i]! * b[i]!;
    normA += a[i]! ** 2;
    normB += b[i]! ** 2;
  }
  return dot / (Math.sqrt(normA) * Math.sqrt(normB));
}

interface DocumentoComEmbedding {
  id:        string;
  texto:     string;
  embedding: Float32Array;
}

async function buscarMaisSimilares(
  consulta:    string,
  documentos:  DocumentoComEmbedding[],
  topK:        number = 5,
): Promise<Array<DocumentoComEmbedding & { score: number }>> {
  const extractor = await pipeline(
    "feature-extraction",
    "Xenova/all-MiniLM-L6-v2",
  );

  const resultado = await extractor([consulta], {
    pooling:   "mean",
    normalize: true,
  }) as FeatureExtractionOutput;

  const embConsulta = new Float32Array(resultado.data as number[]);

  return documentos
    .map((doc) => ({ ...doc, score: similaridadeCosseno(embConsulta, doc.embedding) }))
    .sort((a, b) => b.score - a.score)
    .slice(0, topK);
}

Pipeline de Dados Tipado

TypeScript — Pipeline ETL com tipos para cada etapa
// Pipeline ETL com tipos explicitos por etapa
interface VendaBruta {
  data:       string;   // "2024-01-15"
  produto:    string;
  quantidade: string;   // vem como string do CSV
  preco:      string;
  regiao:     string;
}

interface VendaNormalizada {
  data:       Date;
  produto:    string;
  quantidade: number;
  preco:      number;
  regiao:     "norte" | "sul" | "leste" | "oeste" | "centro";
  total:      number;
}

interface AgregacaoRegional {
  regiao:      string;
  totalVendas: number;
  receita:     number;
  produtoTop:  string;
}

// Validador Zod para etapa de extract
import { z } from "zod";

const VendaBrutaSchema = z.object({
  data:       z.string().regex(/^\d{4}-\d{2}-\d{2}$/),
  produto:    z.string().min(1),
  quantidade: z.string().regex(/^\d+$/),
  preco:      z.string().regex(/^\d+(\.\d{1,2})?$/),
  regiao:     z.enum(["norte", "sul", "leste", "oeste", "centro"]),
});

function transformar(bruta: VendaBruta): VendaNormalizada {
  const validada = VendaBrutaSchema.parse(bruta);
  const qtd   = parseInt(validada.quantidade, 10);
  const preco = parseFloat(validada.preco);
  return {
    data:       new Date(validada.data),
    produto:    validada.produto,
    quantidade: qtd,
    preco,
    regiao:     validada.regiao,
    total:      qtd * preco,
  };
}

function agregarPorRegiao(vendas: VendaNormalizada[]): AgregacaoRegional[] {
  const porRegiao = new Map<string, VendaNormalizada[]>();
  for (const v of vendas) {
    if (!porRegiao.has(v.regiao)) porRegiao.set(v.regiao, []);
    porRegiao.get(v.regiao)!.push(v);
  }
  return Array.from(porRegiao.entries()).map(([regiao, gVendas]) => {
    const freqProduto = new Map<string, number>();
    for (const v of gVendas) freqProduto.set(v.produto, (freqProduto.get(v.produto) ?? 0) + 1);
    const produtoTop = [...freqProduto.entries()].sort((a, b) => b[1] - a[1])[0]![0];
    return {
      regiao,
      totalVendas: gVendas.reduce((s, v) => s + v.quantidade, 0),
      receita:     gVendas.reduce((s, v) => s + v.total, 0),
      produtoTop,
    };
  });
}
