TypeScript para Web

Express com generics de Request/Response, Fastify com TypeBox para validacao em runtime + tipos automaticos e Hono para edge computing com tipagem completa de rotas.

TypeScript transforma APIs Node.js de contratos impliciticos em interfaces explicitas verificadas em compilacao. Com Fastify + TypeBox ou Hono, schema de validacao e tipo sao a mesma fonte de verdade.

Express com @types/express

TypeScript + Express — Request e Response tipados com generics
import express, { Application } from "express";
import type {
  Request, Response, NextFunction, RequestHandler,
} from "express";
import { z } from "zod";

// Generics do Request: <Params, ResBody, ReqBody, Query>
const ProdutoIdSchema = z.object({ id: z.string().uuid() });
const CriarProdutoSchema = z.object({
  nome:  z.string().min(2),
  preco: z.number().positive(),
  sku:   z.string().regex(/^[A-Z]{3}-\d{4}$/),
});

type ProdutoId     = z.infer<typeof ProdutoIdSchema>;
type CriarProduto  = z.infer<typeof CriarProdutoSchema>;
type Produto       = CriarProduto & { id: string; criadoEm: Date };

// Handler com params e body tipados
const buscarProduto: RequestHandler<ProdutoId, Produto> = async (req, res) => {
  const { id } = ProdutoIdSchema.parse(req.params);
  const produto = await buscarNoBanco(id);
  res.json(produto);
};

const criarProduto: RequestHandler<{}, Produto, CriarProduto> = async (req, res) => {
  const dados = CriarProdutoSchema.parse(req.body);
  const produto = await inserirNoBanco(dados);
  res.status(201).json(produto);
};

// Middleware tipado com augmentation de Request
declare global {
  namespace Express {
    interface Request {
      usuarioId?: string;
      papel?:     "admin" | "usuario";
    }
  }
}

const autenticar: RequestHandler = (req, res, next) => {
  const token = req.headers.authorization?.split(" ")[1];
  if (!token) { res.status(401).json({ erro: "Nao autorizado" }); return; }
  req.usuarioId = verificarToken(token);
  next();
};

const app: Application = express();
app.use(express.json());
app.get("/produtos/:id", autenticar, buscarProduto);
app.post("/produtos", autenticar, criarProduto);
app.listen(3000);

Fastify com TypeBox

TypeScript + Fastify — TypeBox schema como fonte de verdade
import Fastify from "fastify";
import { TypeBoxTypeProvider } from "@fastify/type-provider-typebox";
import { Type, type Static } from "@sinclair/typebox";

// TypeBox: schema JSON Schema + tipo TypeScript na mesma declaracao
const UsuarioParamsSchema = Type.Object({
  id: Type.String({ format: "uuid" }),
});

const CriarUsuarioBodySchema = Type.Object({
  nome:  Type.String({ minLength: 2, maxLength: 100 }),
  email: Type.String({ format: "email" }),
  idade: Type.Optional(Type.Integer({ minimum: 0, maximum: 150 })),
});

const UsuarioResponseSchema = Type.Object({
  id:       Type.String({ format: "uuid" }),
  nome:     Type.String(),
  email:    Type.String({ format: "email" }),
  criadoEm: Type.String({ format: "date-time" }),
});

type CriarUsuarioBody  = Static<typeof CriarUsuarioBodySchema>;
type UsuarioResponse   = Static<typeof UsuarioResponseSchema>;

const fastify = Fastify().withTypeProvider<TypeBoxTypeProvider>();

fastify.get("/usuarios/:id", {
  schema: {
    params:   UsuarioParamsSchema,
    response: { 200: UsuarioResponseSchema },
  },
}, async (req, reply) => {
  // req.params.id e string (tipado automaticamente)
  const usuario = await buscarUsuario(req.params.id);
  return reply.send(usuario);
});

fastify.post<{ Body: CriarUsuarioBody; Reply: UsuarioResponse }>("/usuarios", {
  schema: {
    body:     CriarUsuarioBodySchema,
    response: { 201: UsuarioResponseSchema },
  },
}, async (req, reply) => {
  const novoUsuario = await criarUsuario(req.body);
  return reply.status(201).send(novoUsuario);
});

await fastify.listen({ port: 3000, host: "0.0.0.0" });

Hono Edge-Ready com Tipagem de Rotas

TypeScript + Hono — zValidator, RPC client e deploy edge
import { Hono } from "hono";
import { zValidator } from "@hono/zod-validator";
import { jwt } from "hono/jwt";
import { cors } from "hono/cors";
import { z } from "zod";

type Env = { Variables: { usuarioId: string; papel: "admin" | "user" } };

const app = new Hono<Env>();

app.use("*", cors({ origin: ["https://meusite.com"] }));
app.use("/api/*", jwt({ secret: process.env.JWT_SECRET! }));

const ProdutoSchema = z.object({
  nome:     z.string().min(1),
  preco:    z.number().positive(),
  estoque:  z.number().int().nonnegative(),
  categoria: z.enum(["eletronico", "vestuario", "alimento"]),
});

const produto = new Hono<Env>()
  .get("/", async (c) => {
    const produtos = await listarProdutos();
    return c.json(produtos);
  })
  .post("/", zValidator("json", ProdutoSchema), async (c) => {
    const body = c.req.valid("json"); // tipado como z.infer<ProdutoSchema>
    const novo = await criarProduto(body);
    return c.json(novo, 201);
  })
  .get("/:id", async (c) => {
    const id = c.req.param("id");
    const p = await buscarProduto(id);
    if (!p) return c.json({ erro: "nao encontrado" }, 404);
    return c.json(p);
  });

app.route("/api/produtos", produto);

// Exportar tipo para RPC client (Hono RPC)
export type AppType = typeof app;

// Deploy edge: Cloudflare Workers, Vercel Edge, Deno Deploy
export default app;
