Phaser 3 com Scene tipada e GameObjects, Three.js com @types/three para cenas 3D e arquitetura ECS type-safe para logica de entidades e componentes.

Motores de jogo com TypeScript dao autocomplete em metodos de GameObjects, animacoes e physics. Com ECS tipado, sistemas so recebem as entidades que possuem os componentes corretos — verificado em compilacao.

Phaser 3 com Scene Tipada

TypeScript + Phaser 3 — Scene como classe com tipos completos
import Phaser from "phaser";

// Dados tipados entre cenas
interface DadosCena {
  nivel:    number;
  pontuacao:number;
  vidas:    number;
}

interface TeclasJogo {
  cima:    Phaser.Input.Keyboard.Key;
  baixo:   Phaser.Input.Keyboard.Key;
  esquerda:Phaser.Input.Keyboard.Key;
  direita: Phaser.Input.Keyboard.Key;
  acao:    Phaser.Input.Keyboard.Key;
}

class JogoScene extends Phaser.Scene {
  private jogador!:   Phaser.Physics.Arcade.Sprite;
  private inimigos!:  Phaser.Physics.Arcade.Group;
  private moedas!:    Phaser.Physics.Arcade.StaticGroup;
  private teclas!:    TeclasJogo;
  private texPontos!: Phaser.GameObjects.Text;
  private dados:      DadosCena = { nivel: 1, pontuacao: 0, vidas: 3 };

  constructor() { super({ key: "Jogo" }); }

  preload(): void {
    this.load.tilemapTiledJSON("mapa", "assets/mapa.json");
    this.load.image("tiles", "assets/tiles.png");
    this.load.spritesheet("jogador", "assets/jogador.png", {
      frameWidth: 32, frameHeight: 48,
    });
    this.load.spritesheet("inimigo", "assets/inimigo.png", {
      frameWidth: 32, frameHeight: 32,
    });
  }

  create(dados?: DadosCena): void {
    if (dados) this.dados = dados;

    // Tilemap
    const mapa = this.make.tilemap({ key: "mapa" });
    const tileset = mapa.addTilesetImage("terreno", "tiles")!;
    const chao    = mapa.createLayer("Chao", tileset, 0, 0)!;
    chao.setCollisionByProperty({ colide: true });

    // Jogador
    this.jogador = this.physics.add.sprite(100, 400, "jogador");
    this.jogador.setBounce(0.1);
    this.jogador.setCollideWorldBounds(true);

    // Animacoes
    this.anims.create({
      key: "andar",
      frames: this.anims.generateFrameNumbers("jogador", { start: 0, end: 7 }),
      frameRate: 10,
      repeat: -1,
    });
    this.anims.create({
      key: "parado",
      frames: [{ key: "jogador", frame: 0 }],
      frameRate: 20,
    });

    // Colisoes
    this.physics.add.collider(this.jogador, chao);

    // Teclas
    const kb = this.input.keyboard!;
    this.teclas = {
      cima:     kb.addKey(Phaser.Input.Keyboard.KeyCodes.W),
      baixo:    kb.addKey(Phaser.Input.Keyboard.KeyCodes.S),
      esquerda: kb.addKey(Phaser.Input.Keyboard.KeyCodes.A),
      direita:  kb.addKey(Phaser.Input.Keyboard.KeyCodes.D),
      acao:     kb.addKey(Phaser.Input.Keyboard.KeyCodes.SPACE),
    };

    this.texPontos = this.add.text(16, 16, "Pontos: 0", {
      fontSize: "18px", color: "#fff",
    }).setScrollFactor(0);
  }

  update(_time: number, _delta: number): void {
    const vel = 200;
    const body = this.jogador.body as Phaser.Physics.Arcade.Body;

    if (Phaser.Input.Keyboard.JustDown(this.teclas.esquerda)) {
      this.jogador.setVelocityX(-vel);
      this.jogador.setFlipX(true);
      this.jogador.anims.play("andar", true);
    } else if (Phaser.Input.Keyboard.JustDown(this.teclas.direita)) {
      this.jogador.setVelocityX(vel);
      this.jogador.setFlipX(false);
      this.jogador.anims.play("andar", true);
    } else {
      this.jogador.setVelocityX(0);
      this.jogador.anims.play("parado", true);
    }

    if (this.teclas.cima.isDown && body.blocked.down) {
      this.jogador.setVelocityY(-400);
    }
  }
}

const config: Phaser.Types.Core.GameConfig = {
  type:   Phaser.AUTO,
  width:  800,
  height: 600,
  physics: { default: "arcade", arcade: { gravity: { x: 0, y: 400 }, debug: false } },
  scene:  [JogoScene],
};
new Phaser.Game(config);

Three.js com @types/three

TypeScript + Three.js — Cena 3D PBR tipada
import * as THREE from "three";
import { OrbitControls } from "three/examples/jsm/controls/OrbitControls.js";
import { GLTFLoader, type GLTF } from "three/examples/jsm/loaders/GLTFLoader.js";
import { RGBELoader } from "three/examples/jsm/loaders/RGBELoader.js";

interface EntidadeAnimada {
  mesh:   THREE.Object3D;
  mixer:  THREE.AnimationMixer;
  acao:   THREE.AnimationAction;
}

class RendererPBR {
  private readonly renderer: THREE.WebGLRenderer;
  private readonly camera:   THREE.PerspectiveCamera;
  private readonly scene:    THREE.Scene;
  private readonly controls: OrbitControls;
  private readonly clock =   new THREE.Clock();
  private entidades: EntidadeAnimada[] = [];

  constructor(container: HTMLElement) {
    this.renderer = new THREE.WebGLRenderer({ antialias: true });
    this.renderer.setPixelRatio(devicePixelRatio);
    this.renderer.shadowMap.enabled = true;
    this.renderer.toneMapping = THREE.ACESFilmicToneMapping;
    this.renderer.toneMappingExposure = 1.2;
    container.appendChild(this.renderer.domElement);

    this.camera = new THREE.PerspectiveCamera(75, innerWidth / innerHeight, 0.1, 1000);
    this.camera.position.set(0, 2, 5);

    this.scene = new THREE.Scene();
    this.controls = new OrbitControls(this.camera, this.renderer.domElement);
    this.controls.enableDamping = true;

    this.configurarLuzes();
    this.configurarRedimensionar();
  }

  private configurarLuzes(): void {
    const ambiente = new THREE.AmbientLight(0xffffff, 0.4);
    const sol      = new THREE.DirectionalLight(0xffeedd, 1.5);
    sol.position.set(5, 10, 5);
    sol.castShadow = true;
    sol.shadow.mapSize.set(2048, 2048);
    this.scene.add(ambiente, sol);
  }

  async carregarPersonagem(url: string): Promise<EntidadeAnimada> {
    const loader = new GLTFLoader();
    const gltf: GLTF = await loader.loadAsync(url);
    const mesh   = gltf.scene;
    mesh.traverse((filho) => {
      if ((filho as THREE.Mesh).isMesh) filho.castShadow = true;
    });

    const mixer = new THREE.AnimationMixer(mesh);
    const clip  = THREE.AnimationClip.findByName(gltf.animations, "Idle");
    const acao  = mixer.clipAction(clip);
    acao.play();

    this.scene.add(mesh);
    const entidade: EntidadeAnimada = { mesh, mixer, acao };
    this.entidades.push(entidade);
    return entidade;
  }

  iniciar(): void {
    const loop = () => {
      requestAnimationFrame(loop);
      const delta = this.clock.getDelta();
      for (const e of this.entidades) e.mixer.update(delta);
      this.controls.update();
      this.renderer.render(this.scene, this.camera);
    };
    loop();
  }

  private configurarRedimensionar(): void {
    addEventListener("resize", () => {
      this.camera.aspect = innerWidth / innerHeight;
      this.camera.updateProjectionMatrix();
      this.renderer.setSize(innerWidth, innerHeight);
    });
    this.renderer.setSize(innerWidth, innerHeight);
  }
}
