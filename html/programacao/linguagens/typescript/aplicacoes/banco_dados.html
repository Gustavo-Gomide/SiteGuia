TypeScript com Banco de Dados

Prisma com tipos gerados em compilacao, tratamento tipado de erros de banco, transacoes type-safe e Drizzle ORM com inferencia de schema direto para queries.

ORMs com TypeScript fecham o gap entre schema de banco e tipos da aplicacao. Prisma gera um cliente completamente tipado a partir do schema.prisma. Drizzle e code-first: o schema TypeScript e o schema de banco.

Prisma — Cliente Totalmente Tipado

schema.prisma — Modelo de dados
// prisma/schema.prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

model Usuario {
  id        String    @id @default(cuid())
  nome      String
  email     String    @unique
  papel     Papel     @default(USER)
  criadoEm  DateTime  @default(now())
  pedidos   Pedido[]
}

enum Papel { USER ADMIN }

model Pedido {
  id         String    @id @default(cuid())
  total      Decimal
  status     StatusPedido @default(PENDENTE)
  usuarioId  String
  usuario    Usuario   @relation(fields: [usuarioId], references: [id])
  itens      ItemPedido[]
}

enum StatusPedido { PENDENTE CONFIRMADO ENVIADO ENTREGUE CANCELADO }
TypeScript + Prisma — Queries tipadas, erros e transacoes
import { PrismaClient, Prisma } from "@prisma/client";
import type { Usuario, Pedido, StatusPedido } from "@prisma/client";

const prisma = new PrismaClient({
  log: [{ emit: "event", level: "query" }],
});
prisma.$on("query", (e) => console.log(e.query, e.duration + "ms"));

// ---- Tipos utilitarios do Prisma ----
type UsuarioComPedidos = Prisma.UsuarioGetPayload<{
  include: { pedidos: { include: { itens: true } } }
}>;

type CriarUsuarioInput = Prisma.UsuarioCreateInput;
type AtualizarUsuario  = Prisma.UsuarioUpdateInput;

// ---- Queries com tipos inferidos ----
async function buscarUsuarioComPedidos(id: string): Promise<UsuarioComPedidos | null> {
  return prisma.usuario.findUnique({
    where: { id },
    include: { pedidos: { include: { itens: true } } },
  });
}

// ---- Tratamento de erros do Prisma ----
async function criarUsuario(dados: CriarUsuarioInput): Promise<Usuario> {
  try {
    return await prisma.usuario.create({ data: dados });
  } catch (e) {
    if (e instanceof Prisma.PrismaClientKnownRequestError) {
      if (e.code === "P2002") {
        const campo = (e.meta?.target as string[])?.join(", ");
        throw new Error(`Conflito de unicidade no campo: ${campo}`);
      }
      if (e.code === "P2025") throw new Error("Registro nao encontrado");
    }
    throw e;
  }
}

// ---- Transacao com $transaction ----
async function processarPedido(
  usuarioId: string,
  itens: Array<{ produtoId: string; quantidade: number; preco: number }>,
): Promise<Pedido> {
  return prisma.$transaction(async (tx) => {
    // Verificar estoque (dentro da transacao)
    for (const item of itens) {
      const produto = await tx.produto.findUniqueOrThrow({
        where: { id: item.produtoId },
      });
      if (produto.estoque < item.quantidade)
        throw new Error(`Estoque insuficiente: ${produto.nome}`);
    }

    const total = itens.reduce((s, i) => s + i.preco * i.quantidade, 0);
    const pedido = await tx.pedido.create({
      data: {
        usuarioId,
        total,
        itens: { create: itens },
      },
    });

    // Decrementar estoque atomicamente
    await Promise.all(
      itens.map((i) =>
        tx.produto.update({
          where: { id: i.produtoId },
          data:  { estoque: { decrement: i.quantidade } },
        })
      )
    );
    return pedido;
  });
}

Drizzle ORM — Schema como Tipos

TypeScript + Drizzle — Schema, queries e inferencia de tipos
import { drizzle } from "drizzle-orm/node-postgres";
import { Pool } from "pg";
import {
  pgTable, pgEnum, text, numeric, integer, timestamp, uuid,
} from "drizzle-orm/pg-core";
import { eq, lt, gte, and, sql } from "drizzle-orm";
import type { InferSelectModel, InferInsertModel } from "drizzle-orm";

// ---- Schema code-first ----
export const papelEnum = pgEnum("papel", ["user", "admin"]);

export const usuarios = pgTable("usuarios", {
  id:       uuid("id").primaryKey().defaultRandom(),
  nome:     text("nome").notNull(),
  email:    text("email").notNull().unique(),
  papel:    papelEnum("papel").default("user").notNull(),
  criadoEm: timestamp("criado_em").defaultNow().notNull(),
});

export const produtos = pgTable("produtos", {
  id:      uuid("id").primaryKey().defaultRandom(),
  nome:    text("nome").notNull(),
  preco:   numeric("preco", { precision: 10, scale: 2 }).notNull(),
  estoque: integer("estoque").notNull().default(0),
});

// ---- Tipos inferidos do schema ----
export type Usuario         = InferSelectModel<typeof usuarios>;
export type NovoUsuario     = InferInsertModel<typeof usuarios>;
export type Produto         = InferSelectModel<typeof produtos>;
export type NovoProduto     = InferInsertModel<typeof produtos>;

// ---- Conexao e queries ----
const pool = new Pool({ connectionString: process.env.DATABASE_URL });
const db = drizzle(pool, { schema: { usuarios, produtos } });

async function listarUsuarios(papel?: "user" | "admin"): Promise<Usuario[]> {
  if (papel) {
    return db.select().from(usuarios).where(eq(usuarios.papel, papel));
  }
  return db.select().from(usuarios);
}

async function inserirUsuario(dados: NovoUsuario): Promise<Usuario> {
  const [criado] = await db.insert(usuarios).values(dados).returning();
  return criado;
}

// Query com filtros compostos
async function produtosBaratosEmEstoque(maxPreco: string): Promise<Produto[]> {
  return db.select().from(produtos).where(
    and(
      lt(produtos.preco, maxPreco),
      gte(produtos.estoque, 1),
    )
  ).orderBy(produtos.preco);
}
