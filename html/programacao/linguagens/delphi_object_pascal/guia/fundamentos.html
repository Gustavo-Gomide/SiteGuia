Delphi / Object Pascal — Fundamentos

Object Pascal tem um sistema de tipos rico e seguro: tipos ordinais (Integer, Boolean, Char, enumerações), tipos reais (Double, Currency para dinheiro), strings Unicode gerenciadas automaticamente, arrays estáticos e dinâmicos, records (structs), e sets — um tipo específico para conjuntos de enumerações.

A maior diferença de Pascal para C: não existe ponteiro implícito. Strings são contadas por referência automaticamente, arrays dinâmicos são gerenciados pelo runtime, e erros de acesso fora de bounds são detectados em debug mode. Pascal é difícil de quebrar por acidente.

Strings

Operações com strings
uses System.SysUtils;

var
  s1, s2, resultado: string;
  n: Integer;

begin
  s1 := 'Hello';
  s2 := 'World';

  // Concatenação
  resultado := s1 + ' ' + s2;        // 'Hello World'

  // Comprimento
  n := Length(resultado);            // 11

  // Funções do SysUtils
  UpperCase(s1);                     // 'HELLO'
  LowerCase(s1);                     // 'hello'
  Trim('  texto  ');                 // 'texto'
  TrimLeft('  texto  ');              // 'texto  '
  Pos('World', resultado);           // 7 (1-based!)
  Copy(resultado, 1, 5);             // 'Hello'
  StringReplace(s1, 'l', 'L', [rfReplaceAll]);

  // Format — como sprintf
  resultado := Format('Valor: %.2f, Nome: %s', [3.14, 'Pi']);

  // Conversões
  IntToStr(42);                      // '42'
  StrToInt('42');                    // 42 — lança exceção se inválido
  TryStrToInt('abc', n);             // False — seguro
  FloatToStr(3.14);                  // '3.14'
  StrToFloat('3.14');               // 3.14

  // Acesso por índice (1-based em Delphi!)
  resultado[1];                      // 'H' — índice começa em 1
end;

Arrays e Records

Arrays estáticos, dinâmicos e multidimensionais
var
  // Array estático — tamanho fixo em compile time
  notas:    array[1..5] of Double;
  matriz:   array[0..2, 0..2] of Integer;

  // Array dinâmico — tamanho em runtime
  numeros:  array of Integer;
  nomes:    array of string;

begin
  // Estático
  notas[1] := 9.5;
  notas[2] := 8.0;
  Low(notas);    // 1 — índice mínimo
  High(notas);   // 5 — índice máximo

  // Dinâmico
  SetLength(numeros, 10);    // alocar 10 elementos (0..9)
  numeros[0] := 100;
  Length(numeros);            // 10

  SetLength(numeros, 20);    // aumentar — preserva dados

  // Copiar array
  var copia := Copy(numeros, 0, 5);  // 5 elementos a partir do 0

  // for..in (Delphi 2005+)
  for var n in numeros do
    Writeln(n);
end;
Record — struct tipada com métodos
type
  TPonto = record
    X, Y: Double;
  end;

  TPessoa = record
    Nome:  string;
    Idade: Integer;
    CPF:   string;
    function EhMaiorDeIdade: Boolean;   // record pode ter métodos
    class function Criar(const ANome: string; AIdade: Integer): TPessoa; static;
  end;

function TPessoa.EhMaiorDeIdade: Boolean;
begin
  Result := Idade >= 18;
end;

class function TPessoa.Criar(const ANome: string; AIdade: Integer): TPessoa;
begin
  Result.Nome  := ANome;
  Result.Idade := AIdade;
end;

var
  p:  TPessoa;
  pt: TPonto;
begin
  p := TPessoa.Criar('João', 25);
  Writeln(p.EhMaiorDeIdade);   // True

  pt := TPonto(X: 3.0; Y: 4.0);  // record initializer (Delphi 10.3+)
end;

Controle de Fluxo

if, case, for, while, repeat
var i, x: Integer; dia: string;

// ── if / else if / else ──────────────────────────────────────
if x > 0 then
  Writeln('positivo')
else if x < 0 then
  Writeln('negativo')
else
  Writeln('zero');

// ── case of — switch ─────────────────────────────────────────
case i of
  1:       dia := 'Segunda';
  2:       dia := 'Terça';
  3..5:    dia := 'Qua-Sex';
  6, 7:    dia := 'Fim de semana';
else       dia := 'Inválido';
end;

// ── for — contagem ───────────────────────────────────────────
for i := 1 to 10 do
  Writeln(i);

for i := 10 downto 1 do
  Writeln(i);

// ── while — verificação no início ────────────────────────────
i := 0;
while i < 10 do
begin
  Writeln(i);
  Inc(i);   // Inc é mais rápido que i := i + 1
end;

// ── repeat..until — verificação no final ─────────────────────
i := 0;
repeat
  Writeln(i);
  Inc(i);
until i >= 10;

// ── Break, Continue ──────────────────────────────────────────
for i := 0 to 100 do
begin
  if i = 50 then Break;     // sair do loop
  if Odd(i)  then Continue;  // próxima iteração
  Writeln(i);
end;

Procedures e Functions

Procedures, functions, parâmetros e Result
// Procedure — sem valor de retorno
procedure Saudar(const Nome: string);
begin
  Writeln('Olá, ' + Nome + '!');
end;

// Function — com valor de retorno via Result
function Fatorial(N: Integer): Int64;
begin
  if N <= 1 then
    Result := 1
  else
    Result := N * Fatorial(N - 1);
end;

// Parâmetro por referência — var
procedure Trocar(var A, B: Integer);
var Temp: Integer;
begin
  Temp := A;  A := B;  B := Temp;
end;

// Parâmetro de saída — out (não lê o valor inicial)
procedure Dividir(A, B: Double; out Resultado: Double; out Erro: Boolean);
begin
  if B = 0 then
  begin
    Erro := True;
    Exit;
  end;
  Resultado := A / B;
  Erro := False;
end;

// const — passagem eficiente sem cópia (read-only)
function ContarVogais(const S: string): Integer;
var C: Char;
begin
  Result := 0;
  for C in S do
    if C in ['a','e','i','o','u','A','E','I','O','U'] then
      Inc(Result);
end;

Exceções e Gerenciamento de Recursos

try/except, try/finally e raise
uses System.SysUtils, System.IOUtils;

// try/except — capturar erros
procedure LerArquivo(const Caminho: string);
var Conteudo: string;
begin
  try
    Conteudo := TFile.ReadAllText(Caminho);
    Writeln(Conteudo);
  except
    on E: EFileNotFoundException do
      Writeln('Arquivo não encontrado: ', E.Message);
    on E: EAccessViolation do
      Writeln('Sem permissão: ', E.Message);
    on E: Exception do
      Writeln('Erro genérico: ', E.ClassName, ': ', E.Message);
  end;
end;

// try/finally — garantir limpeza (sempre executa)
procedure ProcessarObjeto;
var Obj: TStringList;
begin
  Obj := TStringList.Create;
  try
    Obj.Add('linha 1');
    Obj.Add('linha 2');
    // ... processamento ...
  finally
    Obj.Free;   // SEMPRE executado — sem vazamento de memória
  end;
end;

// Raise — lançar exceção
function Dividir(A, B: Double): Double;
begin
  if B = 0 then
    raise EDivByZero.Create('Divisor não pode ser zero');
  Result := A / B;
end;

Enumerações e Sets

enum, set of — tipos especiais Pascal
type
  TDiaSemana = (dsSegunda, dsTerca, dsQuarta, dsQuinta,
                dsSexta, dsSabado, dsDomingo);

  TDiasSemana = set of TDiaSemana;    // conjunto de dias

  TPermissao  = (pLeitura, pEscrita, pExecucao, pAdmin);
  TPermissoes = set of TPermissao;

var
  hoje:    TDiaSemana  := dsQuarta;
  uteis:   TDiasSemana := [dsSegunda..dsSexta];
  perms:   TPermissoes := [pLeitura, pEscrita];

begin
  // Operações com set
  if hoje in uteis then
    Writeln('Dia útil');

  Include(perms, pExecucao);     // adicionar elemento
  Exclude(perms, pEscrita);      // remover elemento
  pAdmin in perms;               // False — não tem

  // União, interseção, diferença
  perms := perms + [pAdmin];                   // união
  perms := perms * [pLeitura, pExecucao];      // interseção
  perms := perms - [pAdmin];                   // diferença

  // Ord e Chr — converter enum ↔ integer
  Ord(dsSegunda);   // 0
  Ord(dsTerca);     // 1
  TDiaSemana(2);    // dsQuarta
end;
