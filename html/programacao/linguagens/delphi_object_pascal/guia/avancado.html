Object Pascal — Avançado

O nível avançado de Object Pascal abrange o sistema OOP completo — herança, polimorfismo, interfaces e abstract; os genéricos introduzidos no Delphi 2009; os anonymous methods (closures) e TProc/TFunc; a RTTI estendida com RTTI.pas; e o modelo de threads com TThread, TTask e o PPL (Parallel Programming Library).

A herança única com implementação múltipla de interfaces cobre a maioria dos casos de design. Para inversão de dependência: declare interfaces, injete implementações. Para código genérico reutilizável: TGenericList<T>. Para callbacks funcionais: TProc<T> e closures.

Classes e Herança

Herança, override, virtual, abstract
type
  // Classe base
  TForma = class
  protected
    FCor: string;
  public
    constructor Create(const ACor: string); virtual;
    function  Area:      Double; virtual; abstract;  // deve sobrescrever
    function  Perimetro: Double; virtual; abstract;
    procedure Desenhar;           virtual;           // pode sobrescrever
    property  Cor: string read FCor write FCor;
  end;

  TCirculo = class(TForma)
  private
    FRaio: Double;
  public
    constructor Create(ARaio: Double; const ACor: string = 'azul'); override;
    function Area:      Double; override;
    function Perimetro: Double; override;
    property Raio: Double read FRaio;
  end;

constructor TForma.Create(const ACor: string);
begin
  FCor := ACor;
end;

constructor TCirculo.Create(ARaio: Double; const ACor: string);
begin
  inherited Create(ACor);   // chamar construtor pai
  FRaio := ARaio;
end;

function TCirculo.Area: Double;
begin
  Result := PI * FRaio * FRaio;
end;

function TCirculo.Perimetro: Double;
begin
  Result := 2 * PI * FRaio;
end;

// Polimorfismo
var
  formas: array of TForma;
  f: TForma;
begin
  SetLength(formas, 3);
  formas[0] := TCirculo.Create(5);
  formas[1] := TCirculo.Create(3, 'vermelho');

  for f in formas do
    Writeln(f.Area:0:2);   // dispatch dinâmico — chama o Área correto

  // is / as — verificação e cast seguro
  if formas[0] is TCirculo then
    Writeln((formas[0] as TCirculo).Raio);
end;

Interfaces

Interface, implementação múltipla e GUID
type
  ISerializer = interface
  ['{A1B2C3D4-E5F6-7890-ABCD-EF1234567890}']   // GUID obrigatório para suporte COM
    function Serialize(const Obj: TObject): string;
    function Deserialize(const Data: string): TObject;
  end;

  ILogger = interface
  ['{B2C3D4E5-F6A7-8901-BCDE-F12345678901}']
    procedure Log(const Msg: string);
  end;

  // Implementação múltipla de interfaces
  TJsonSerializer = class(TInterfacedObject, ISerializer, ILogger)
  public
    function  Serialize(const Obj: TObject): string;
    function  Deserialize(const Data: string): TObject;
    procedure Log(const Msg: string);
  end;

// Uso via interface — contagem de ref automática
var
  ser:    ISerializer;
  logger: ILogger;
begin
  ser := TJsonSerializer.Create;    // ref counting automático
  ser.Serialize(minhaClasse);
  // sem Free necessário — quando ser sair de escopo, libera

  // Suporte à interface — Supports
  if Supports(ser, ILogger, logger) then
    logger.Log('Serialização concluída');
end;

Genéricos

Genéricos — classes e funções parametrizadas por tipo
uses
  System.Generics.Collections,   // TList<T>, TDictionary<K,V>, TStack<T>, TQueue<T>
  System.Generics.Defaults;      // IComparer<T>, IEqualityComparer<T>

var
  lista:  TList<Integer>;
  dict:   TDictionary<string, TPessoa>;
  fila:   TQueue<string>;

begin
  lista  := TList<Integer>.Create;
  try
    lista.Add(10);
    lista.Add(20);
    lista.Add(30);
    lista.Sort;
    for var v in lista do
      Writeln(v);
  finally
    lista.Free;
  end;

  // Dicionário
  dict := TDictionary<string, TPessoa>.Create;
  try
    dict.Add('001', TPessoa.Criar('Maria', 30));
    if dict.TryGetValue('001', pessoa) then
      Writeln(pessoa.Nome);
    dict.ContainsKey('002');   // False
  finally
    dict.Free;
  end;
end;

// Classe genérica própria
type
  TPar<T, U> = class
  public
    Primeiro:  T;
    Segundo:   U;
    constructor Create(A: T; B: U);
  end;

// Função genérica com constraint
function Max<T: record>(A, B: T): T;
begin
  if TComparer<T>.Default.Compare(A, B) >= 0 then
    Result := A
  else
    Result := B;
end;

Anonymous Methods e Closures

TProc, TFunc, closures — programação funcional em Pascal
uses System.SysUtils;

// Tipos funcionais padrão
// TProc         = procedure
// TProc<T>      = procedure(Arg: T)
// TFunc<T,R>    = function(Arg: T): R
// TPredicate<T> = function(Arg: T): Boolean

var
  somar:    TFunc<Integer, Integer, Integer>;
  filtrar:  TPredicate<Integer>;
  multiplicador: Integer;
  escalar:  TFunc<Integer, Integer>;

begin
  // Anonymous method — closure
  somar := function(A, B: Integer): Integer
           begin
             Result := A + B;
           end;

  Writeln(somar(3, 4));   // 7

  // Closure captura variável do escopo externo
  multiplicador := 3;
  escalar := function(X: Integer): Integer
             begin
               Result := X * multiplicador;   // captura 'multiplicador'
             end;

  Writeln(escalar(5));   // 15
  multiplicador := 10;
  Writeln(escalar(5));   // 50 — referência capturada por ref

  // TList<T> com anonymous methods
  lista.Sort(TComparer<Integer>.Construct(
    function(const A, B: Integer): Integer
    begin
      Result := B - A;   // decrescente
    end));
end;

Threads e PPL

TThread, TTask e Parallel.For
uses
  System.Threading,    // TTask, IFuture, TParallel
  System.SyncObjs;     // TCriticalSection, TMonitor, TEvent

// ── TTask — tarefa assíncrona simples ────────────────────────
var
  tarefa: ITask;
begin
  tarefa := TTask.Run(
    procedure
    begin
      // roda em thread do pool
      Writeln('Processando em background...');
    end);

  tarefa.Wait;   // aguardar conclusão
end;

// ── IFuture — tarefa com resultado ───────────────────────────
var
  futuro: IFuture<Integer>;
begin
  futuro := TTask.Future<Integer>(
    function: Integer
    begin
      Result := CalcPesada;
    end);

  FazOutraCoisaAqui;
  Writeln(futuro.Value);  // bloqueia se não terminou
end;

// ── TParallel.For — loop paralelizado ────────────────────────
TParallel.For(0, High(dados),
  procedure(i: Integer)
  begin
    dados[i] := Processar(dados[i]);
  end);

// ── Sincronização — TCriticalSection ─────────────────────────
var
  cs: TCriticalSection;
  total: Integer := 0;
begin
  cs := TCriticalSection.Create;
  try
    TParallel.For(0, 999,
      procedure(i: Integer)
      begin
        cs.Acquire;
        try
          Inc(total);
        finally
          cs.Release;
        end;
      end);
  finally
    cs.Free;
  end;
  Writeln('Total: ', total);  // 1000
end;
