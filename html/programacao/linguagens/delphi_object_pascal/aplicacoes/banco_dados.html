Delphi — Banco de Dados

Delphi tem um dos modelos de acesso a banco mais maduros do mundo: o FireDAC (antigo AnyDAC) suporta mais de 20 SGBDs com uma API consistente. A arquitetura DataSet → DataSource → controles visuais permite uma grade de dados completamente funcional sem uma linha de código. Para acesso programático, TFDQuery com parâmetros previne injeção SQL.

A conexão data-aware do Delphi: um TFDQuery conectado a um TDataSource, que está ligado a um TDBGrid — ao mudar o SQL ou os dados, a grade atualiza automaticamente. É MVVM aplicado a componentes visuais, décadas antes do React.

FireDAC — Conexão e Queries

Conectar ao banco e executar queries
uses
  FireDAC.Stan.Intf, FireDAC.Stan.Option,
  FireDAC.Phys.PG,              // driver PostgreSQL
  FireDAC.Phys.SQLite,          // driver SQLite
  FireDAC.Comp.Client;          // TFDConnection, TFDQuery

var
  conn:   TFDConnection;
  query:  TFDQuery;

begin
  conn := TFDConnection.Create(nil);
  try
    // PostgreSQL
    conn.DriverName := 'PG';
    conn.Params.Values['Server']   := 'localhost';
    conn.Params.Values['Port']     := '5432';
    conn.Params.Values['Database'] := 'meubd';
    conn.Params.Values['User_Name']:= 'user';
    conn.Params.Values['Password'] := 'senha';
    conn.Open;

    // Query com parâmetro — previne SQL Injection
    query := TFDQuery.Create(nil);
    try
      query.Connection := conn;
      query.SQL.Text := 'SELECT * FROM clientes WHERE cidade = :cidade';
      query.ParamByName('cidade').AsString := 'São Paulo';
      query.Open;   // SELECT → Open

      while not query.EOF do
      begin
        Writeln(query.FieldByName('nome').AsString);
        query.Next;
      end;
    finally
      query.Free;
    end;
  finally
    conn.Free;
  end;
end;
INSERT, UPDATE, DELETE e transações
// INSERT com parâmetros
procedure InserirCliente(AConn: TFDConnection; const ANome, AEmail: string);
var
  qry: TFDQuery;
begin
  qry := TFDQuery.Create(nil);
  try
    qry.Connection := AConn;
    qry.SQL.Text :=
      'INSERT INTO clientes (nome, email, criado_em) VALUES (:nome, :email, NOW())';
    qry.ParamByName('nome').AsString  := ANome;
    qry.ParamByName('email').AsString := AEmail;
    qry.ExecSQL;   // DML → ExecSQL (não Open)
  finally
    qry.Free;
  end;
end;

// Transação
procedure TransferirSaldo(AConn: TFDConnection; DeId, ParaId: Integer; AValor: Currency);
begin
  AConn.StartTransaction;
  try
    ExecutarSQL(AConn,
      'UPDATE contas SET saldo = saldo - :v WHERE id = :id',
      ['v', 'id'], [AValor, DeId]);

    ExecutarSQL(AConn,
      'UPDATE contas SET saldo = saldo + :v WHERE id = :id',
      ['v', 'id'], [AValor, ParaId]);

    AConn.Commit;
  except
    AConn.Rollback;
    raise;
  end;
end;

SQLite Embutido

SQLite — banco embutido perfeito para desktop/mobile
uses
  FireDAC.Phys.SQLite,
  FireDAC.Comp.Client;

function CriarConexaoSQLite(const ACaminho: string): TFDConnection;
begin
  Result := TFDConnection.Create(nil);
  Result.DriverName := 'SQLite';
  Result.Params.Values['Database'] := ACaminho;
  Result.Params.Values['OpenMode'] := 'CreateUTF8';  // criar se não existir
  Result.Open;
end;

// Criar tabela e migração
procedure MigrarBanco(AConn: TFDConnection);
begin
  AConn.ExecSQL(
    'CREATE TABLE IF NOT EXISTS clientes (' +
    '  id      INTEGER PRIMARY KEY AUTOINCREMENT,' +
    '  nome    TEXT NOT NULL,' +
    '  email   TEXT UNIQUE NOT NULL,' +
    '  ativo   INTEGER DEFAULT 1' +
    ')');

  // Índice para busca por email
  AConn.ExecSQL(
    'CREATE INDEX IF NOT EXISTS idx_email ON clientes(email)');
end;

DataSet, DataSource e Componentes Data-Aware

TDataModule com componentes data-aware
// DmClientes.pas — DataModule com componentes configurados no designer
unit DmClientes;

interface
uses
  System.Classes,
  FireDAC.Comp.Client,
  Data.DB;               // TDataSource

type
  TDmClientes = class(TDataModule)
    FDConn:        TFDConnection;
    QryClientes:   TFDQuery;
    DsClientes:    TDataSource;   // DataSource conecta Query ↔ controles visuais
  private
    procedure ConfigurarConexao;
  public
    procedure Abrir;
    procedure Buscar(const ANome: string);
  end;

implementation

procedure TDmClientes.ConfigurarConexao;
begin
  // No designer visual, isso está em propriedades
  FDConn.DriverName := 'SQLite';
  FDConn.Params.Values['Database'] := 'clientes.db';
  QryClientes.Connection := FDConn;
  DsClientes.DataSet := QryClientes;
end;

procedure TDmClientes.Abrir;
begin
  FDConn.Open;
  QryClientes.SQL.Text := 'SELECT * FROM clientes ORDER BY nome';
  QryClientes.Open;
end;

// No Form: TDBGrid.DataSource = DmClientes.DsClientes
// A grade aparece automaticamente com os dados — zero código no form!

Sempre use parâmetros nas queries: WHERE nome = :nome com ParamByName('nome').AsString := valor — NUNCA concatene strings. Além de prevenir injeção SQL, os parâmetros são pré-compilados pelo SGBD, acelerando execuções repetidas. E com FireDAC, o tipo do parâmetro é inferido automaticamente — sem cast manual.
