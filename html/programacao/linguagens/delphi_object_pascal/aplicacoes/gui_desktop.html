Delphi — GUI Desktop com VCL

A VCL (Visual Component Library) é a biblioteca gráfica nativa do Delphi para Windows. Com mais de 30 anos de evolução, tem centenas de componentes: controles básicos, grids de dados, árvores, listas, diálogos, gráficos, impressão e muito mais. Tudo integrado ao Form Designer do IDE — arrastar, configurar, conectar.

O modelo de eventos VCL: cada componente expõe eventos como OnClick, OnChange, OnKeyPress. Dê duplo clique no componente no designer e o IDE cria o método handler automaticamente no code-behind. Programação orientada a eventos em seu estado mais produtivo.

Forms e Componentes Básicos

Code-behind do Form — eventos e lógica
unit FrmCadastro;

interface
uses
  Vcl.Forms, Vcl.StdCtrls, Vcl.ExtCtrls,
  Vcl.Controls, Vcl.Dialogs,
  System.Classes;

type
  TFrmCadastro = class(TForm)
    // Componentes declarados — criados no designer (.dfm)
    edtNome:   TEdit;
    edtEmail:  TEdit;
    edtIdade:  TEdit;
    btnSalvar: TButton;
    btnLimpar: TButton;
    lblStatus: TLabel;
    pnlTopo:   TPanel;
    mmoLog:    TMemo;

    // Eventos — gerados pelo IDE ao dar duplo clique
    procedure btnSalvarClick(Sender: TObject);
    procedure btnLimparClick(Sender: TObject);
    procedure edtNomeKeyPress(Sender: TObject; var Key: Char);
    procedure FormCreate(Sender: TObject);
    procedure FormDestroy(Sender: TObject);
  private
    FServico: TClienteServico;
    procedure ValidarCampos;
    procedure LimparFormulario;
  end;

var
  FrmCadastro: TFrmCadastro;

implementation

{$R *.dfm}   // inclui o arquivo de form (.dfm) com os componentes

procedure TFrmCadastro.FormCreate(Sender: TObject);
begin
  FServico := TClienteServico.Create(TClienteRepoFireDAC.Create);
  Caption := 'Cadastro de Clientes — v1.0';
end;

procedure TFrmCadastro.FormDestroy(Sender: TObject);
begin
  FServico.Free;
end;

procedure TFrmCadastro.btnSalvarClick(Sender: TObject);
begin
  ValidarCampos;
  try
    FServico.Registrar(edtNome.Text, edtEmail.Text);
    lblStatus.Caption := 'Salvo com sucesso!';
    mmoLog.Lines.Add(Format('[%s] Cliente %s salvo.',
                            [TimeToStr(Now), edtNome.Text]));
    LimparFormulario;
  except
    on E: Exception do
    begin
      lblStatus.Caption := 'Erro: ' + E.Message;
      ShowMessage('Falha ao salvar: ' + E.Message);
    end;
  end;
end;

procedure TFrmCadastro.edtNomeKeyPress(Sender: TObject; var Key: Char);
begin
  if Key = #13 then   // Enter → mover para próximo campo
    edtEmail.SetFocus;
end;

TListView e TTreeView

TListView com colunas e itens por código
// Configurar TListView por código (ou no designer)
procedure TFrmPrincipal.ConfigurarListView;
begin
  with lvClientes do
  begin
    ViewStyle  := vsReport;
    GridLines  := True;
    FullRowSelect := True;
    ReadOnly   := True;

    // Adicionar colunas
    with Columns.Add do
    begin
      Caption := 'Nome';
      Width   := 200;
    end;
    with Columns.Add do
    begin
      Caption := 'Email';
      Width   := 300;
    end;
  end;
end;

// Popular ListView com dados
procedure TFrmPrincipal.CarregarClientes;
var
  item: TListItem;
  c: TCliente;
begin
  lvClientes.Items.BeginUpdate;   // performance — suspende repaint
  try
    lvClientes.Items.Clear;
    for c in FServico.ListarTodos do
    begin
      item := lvClientes.Items.Add;
      item.Caption := c.Nome;
      item.SubItems.Add(c.Email);
      item.Data := Pointer(c.Id);   // guardar ID na item.Data
    end;
  finally
    lvClientes.Items.EndUpdate;
  end;
end;

TActionList — Actions e Atalhos

TActionList — centralizar lógica e atalhos de teclado
// TActionList no DataModule ou Form
// Cada TAction tem: Caption, ShortCut, Hint, OnExecute, OnUpdate

procedure TFrmPrincipal.actNovoClienteExecute(Sender: TObject);
begin
  TFrmCadastro.Create(Self).ShowModal;
  CarregarClientes;
end;

procedure TFrmPrincipal.actExcluirExecute(Sender: TObject);
begin
  if lvClientes.Selected = nil then Exit;
  if MessageDlg('Confirma exclusão?', mtConfirmation, [mbYes, mbNo], 0) = mrYes then
  begin
    FServico.Excluir(Integer(lvClientes.Selected.Data));
    CarregarClientes;
  end;
end;

// OnUpdate — habilitar/desabilitar automaticamente
procedure TFrmPrincipal.actExcluirUpdate(Sender: TObject);
begin
  actExcluir.Enabled := lvClientes.Selected <> nil;
  // Botão "Excluir" e item de menu ficam desabilitados automaticamente
end;

// Uma TAction pode estar conectada a:
//   TButton, TMenuItem, TToolButton, TSpeedButton
// — todos se habilitam/desabilitam juntos via OnUpdate

Deploy — Distribuição de Aplicações

Criar instalador e distribuir sem dependências
// Vantagem Delphi: executável standalone
// Release Win64 típico: 5-15 MB sem instalação de runtime

// 1. Build Release
//    Project → Build Configurations → Release
//    Compila com otimizações e sem informação de debug

// 2. DLLs necessárias (FireDAC drivers):
//    midas.dll           — apenas se usar ClientDataSet
//    libsqlite3.dll      — se usar SQLite
//    libpq.dll           — se usar PostgreSQL
//    Redistribuível MSVC — se usar componentes que precisam

// 3. Inno Setup — criador de instalador gratuito
[Setup]
AppName=Meu Sistema
AppVersion=1.0
DefaultDirName={pf}\MeuSistema
DefaultGroupName=Meu Sistema
OutputBaseFilename=MeuSistema_setup
Compression=lzma2
SolidCompression=yes

[Files]
Source: "Win64\Release\MeuSistema.exe"; DestDir: "{app}"
Source: "Win64\Release\libsqlite3.dll";  DestDir: "{app}"

[Icons]
Name: "{group}\Meu Sistema"; Filename: "{app}\MeuSistema.exe"

// 4. Verificar dependências com Dependency Walker (depends.exe)
//    ou com Process Monitor da Sysinternals

lvClientes.Items.BeginUpdate / EndUpdate é essencial ao popular listas com muitos itens. Sem isso, cada Items.Add causa um repaint — 1000 itens causam 1000 repaints. Com BeginUpdate, o controle suspende a renderização até EndUpdate, reduzindo de segundos para milissegundos. Aplique o mesmo princípio para TTreeView, TComboBox e TStringGrid.
