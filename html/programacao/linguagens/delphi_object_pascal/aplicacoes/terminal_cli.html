Exemplo de CLI

Leitura de argumentos
// utilitario simples: saudacao por parametro
program SaudadorCLI;
begin
  if ParamCount = 0 then
    Writeln('Uso: saudador nome')
  else
    Writeln('Ola, ' + ParamStr(1) + '!');
end.
