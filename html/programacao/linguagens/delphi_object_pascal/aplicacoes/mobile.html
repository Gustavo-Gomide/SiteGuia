Delphi — Mobile com FireMonkey

FireMonkey (FMX) é o framework cross-platform do Delphi que compila o mesmo código Object Pascal para iOS, Android, Windows e macOS. A UI é vetorial e renderizada via GPU (Metal / OpenGL ES), garantindo aparência nativa em cada plataforma. A lógica de negócio é 100% compartilhada; apenas personalizações visuais são específicas por plataforma.

A promessa do FMX: escreva uma vez, compile para quatro plataformas. A realidade: escreva a lógica uma vez (60-80% do código) e ajuste a UI para cada plataforma (20-40%). Ainda assim, muito mais eficiente que manter quatro bases de código separadas.

Projeto FireMonkey — Estrutura

Form FMX — diferenças em relação ao VCL
unit FrmPrincipal;

interface
uses
  System.SysUtils, System.Classes,
  FMX.Types,         // TControl, TComponent FMX
  FMX.Controls,
  FMX.Forms,         // TForm — FMX version
  FMX.StdCtrls,      // TButton, TLabel, TEdit
  FMX.Layouts,       // TLayout, TGridLayout, TScrollBox
  FMX.ListView,      // TListView FMX
  FMX.Graphics;      // TBitmap

type
  TFrmPrincipal = class(TForm)
    Layout1:   TLayout;     // container sem bordas visíveis
    btnOla:    TButton;
    edtNome:   TEdit;
    lblTitulo: TLabel;
    procedure btnOlaClick(Sender: TObject);
  end;

implementation
{$R *.fmx}   // arquivo .fmx em vez de .dfm

procedure TFrmPrincipal.btnOlaClick(Sender: TObject);
begin
  lblTitulo.Text := 'Olá, ' + edtNome.Text + '!';
  // FMX: .Text em vez de .Caption para controles
end;

uses
{$IF DEFINED(ANDROID)}
  Androidapi.JNI.GraphicsContentViewText,
  Androidapi.Helpers,
{$ELSEIF DEFINED(IOS)}
  iOSapi.Foundation,
  iOSapi.UIKit,
{$ELSEIF DEFINED(MSWINDOWS)}
  Winapi.Windows,
{$ENDIF}
  System.SysUtils;

procedure AbriUrlExterno(const AURL: string);
begin
{$IF DEFINED(ANDROID)}
  TAndroidHelper.Activity.startActivity(
    TJIntent.JavaClass.init(TJIntent.JavaClass.ACTION_VIEW,
    TURI.Create(AURL)));
{$ELSEIF DEFINED(IOS)}
  SharedApplication.openURL(NSURL.URLWithString(StrToNSStr(AURL)));
{$ELSEIF DEFINED(MSWINDOWS)}
  ShellExecute(0, 'open', PChar(AURL), nil, nil, SW_SHOWNORMAL);
{$ENDIF}
end;

procedure TFrmMapa.ativarGPS;
begin
  LocationSensor.Active := True;

procedure TFrmMapa.LocationSensorLocationChanged(Sender: TObject;
  const OldLocation, NewLocation: TLocationCoord2D);
begin
  lblLat.Text  := Format('Lat: %.6f', [NewLocation.Latitude]);
  lblLong.Text := Format('Long: %.6f', [NewLocation.Longitude]);
end;

// ── Câmera — tirar foto ──────────────────────────────────────
procedure TFrmCamera.tirarFoto;
begin
  CameraComponent.Active := True;
  CameraComponent.CaptureImage(
    procedure(AImage: TBitmap)
    begin
      imgFoto.Bitmap.Assign(AImage);
      SalvarFotoLocal(AImage);
    end);
end;

  [JStringToString(TJManifest_permission.JavaClass.ACCESS_FINE_LOCATION)],
  procedure(const APermissions: TClassicStringDynArray;

      LocationSensor.Active := True;
  end);

// TListView FMX com ItemAppearance = Custom
// Configurado no designer: cada item tem Text, Detail, Accessory, Image

procedure TFrmLista.PopularLista(const AProdutos: TArray<TProduto>);
var
  item:   TListViewItem;
  p:      TProduto;
begin
  lvProdutos.BeginUpdate;
  try
    lvProdutos.Items.Clear;
    for p in AProdutos do
    begin
      item := lvProdutos.Items.Add;
      item.Text   := p.Nome;
      item.Detail := Format('R$ %.2f — Estoque: %d', [p.Preco, p.Estoque]);
      item.Tag    := p.Id;
      // item.Bitmap := p.Thumbnail;  // imagem opcional
    end;
  finally
    lvProdutos.EndUpdate;
  end;
end;

procedure TFrmLista.lvProdutosItemClick(const Sender: TObject;
  const AItem: TListViewItem);
begin
  // Navegar para tela de detalhe passando o ID
  TFrmDetalhe.ShowProduto(AItem.Tag);
end;

A estratégia de compartilhamento de código FMX: coloque toda a lógica de negócio em units sem referência a FMX.*. As units de lógica compilam em qualquer plataforma. Apenas os forms e unidades de UI têm uses FMX.*. Dessa forma, 70% do código é testável em console, independente de plataforma, e a migração futura para nova UI é simples.
